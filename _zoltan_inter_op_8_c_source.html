108 // *****************************************************************************
109 // Partition mesh using Zoltan2 with a geometric partitioner, such as RCB, RIB
110 //! \param[in] algorithm Partitioning algorithm type
111 //! \param[in] centroid Mesh element coordinates
112 //! \param[in] elemid Global mesh element ids
113 //! \parampin] nelem Number of elements in mesh (on this MPI rank)
114 //! \param[in] npart Number of desired graph partitions
115 //! \return Array of chare ownership IDs mapping graph points to concurrent
116 //! async chares
117 //! \details This function uses Zoltan to partition the mesh graph in parallel.
118 //! It assumes that the mesh graph is distributed among all the MPI ranks.
119 //! \author J. Bakosi
120 // *****************************************************************************
121 {
122  // Set Zoltan parameters
123  Teuchos::ParameterList params( "Zoltan parameters" );
124  params.set( "algorithm", tk::ctr::PartitioningAlgorithm().param(algorithm) );
125  params.set( "num_global_parts", std::to_string(npart) );
126  params.set( "objects_to_partition", "mesh_elements" );
127 
128  // Define types for Zoltan2
129  // * 1st argument, 'scalar': the data type for element values, weights and
130  // coordinates
131  // * 2nd argument, 'lno' (local number): the integral data type used by
132  // the application, i.e, quinoa, and by Zoltan2 for local indices and local
133  // counts
134  // * 3rd argument 'gno' (global number): is the integral data type used by
135  // the application, i.e., quinoa, and Zoltan2 to represent global
136  // identifiers and global counts
137  // See also
138  // <tpl>/src/trilinos/packages/zoltan2/src/input/Zoltan2_InputTraits.hpp
139  using ZoltanTypes =
140  Zoltan2::BasicUserTypes< tk::real, std::size_t, std::size_t >;
141 
142  // Create mesh adapter for Zoltan for mesh element partitioning
143  using InciterZoltanAdapter = GeometricMeshElemAdapter< ZoltanTypes >;
144  InciterZoltanAdapter adapter( nelem, centroid, elemid );
145 
146  // Create Zoltan2 partitioning problem using our mesh input adapter
147  Zoltan2::PartitioningProblem< InciterZoltanAdapter >
148  partitioner( &adapter, &params );
149 
150  // Perform partitioning using Zoltan
151  partitioner.solve();
152 
153  // Copy over array of chare IDs corresponding to the ownership of elements
154  // in our chunk of the mesh graph, i.e., the coloring or chare ids for the
155  // mesh elements we operated on
156  auto partlist = partitioner.getSolution().getPartListView();
157  std::vector< std::size_t > chare( nelem );
158  for (std::size_t p=0; p<nelem; ++p )
159  chare[p] = static_cast< std::size_t >( partlist[p] );
160 
161  return chare;
162 }
