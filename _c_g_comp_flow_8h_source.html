53  //! \param[in] coord Mesh node coordinates

75  //! Compute the left hand side sparse matrix
76  //! \param[in] coord Mesh node coordinates
77  //! \param[in] inpoel Mesh element connectivity
78  //! \param[in] psup Linked lists storing IDs of points surrounding points
79  //! \param[in,out] lhsd Diagonal of the sparse matrix storing nonzeros
80  //! \param[in,out] lhso Off-diagonal of the sparse matrix storing nonzeros
81  //! \details Sparse matrix storing the nonzero matrix values at rows and
82  //! columns given by psup. The format is similar to compressed row
83  //! storage, but the diagonal and off-diagonal data are stored in separate
84  //! vectors. For the off-diagonal data the local row and column indices,
85  //! at which values are nonzero, are stored by psup (psup1 and psup2,
86  //! where psup2 holds the indices at which psup1 holds the point ids
87  //! surrounding points, see also tk::genPsup()). Note that the number of
88  //! mesh points (our chunk) npoin = psup.second.size()-1.
89  void lhs( const std::array< std::vector< tk::real >, 3 >& coord,
90  const std::vector< std::size_t >& inpoel,
91  const std::pair< std::vector< std::size_t >,
92  std::vector< std::size_t > >& psup,
93  tk::Fields& lhsd,
94  tk::Fields& lhso ) const
95  {
96  Assert( psup.second.size()-1 == coord[0].size(),
97  "Number of mesh points and number of global IDs unequal" );
98  Assert( lhsd.nunk() == psup.second.size()-1, "Number of unknowns in "
99  "diagonal sparse matrix storage incorrect" );
100  Assert( lhso.nunk() == psup.first.size(), "Number of unknowns in "
101  "off-diagonal sparse matrix storage incorrect" );
102 
103  // Lambda to compute the sparse matrix vector index for row and column
104  // indices. Used only for off-diagonal entries.
105  auto spidx = [ &psup ]( std::size_t r, std::size_t c ) -> std::size_t {
106  Assert( r != c, "Only for computing the off-diagonal indices" );
107  for (auto i=psup.second[r]+1; i<=psup.second[r+1]; ++i)
108  if (c == psup.first[i]) return i;
109  Throw( "Cannot find row, column: " + std::to_string(r) + ',' +
110  std::to_string(c) + " in sparse matrix" );
111  };
112 
113  const auto& x = coord[0];
114  const auto& y = coord[1];
115  const auto& z = coord[2];
116 
117  // Zero matrix for all components
118  for (ncomp_t c=0; c<5; ++c) {
119  lhsd.fill( c, m_offset, 0.0 );
120  lhso.fill( c, m_offset, 0.0 );
121  }
122 
123  for (std::size_t e=0; e<inpoel.size()/4; ++e) {
124  const auto A = inpoel[e*4+0];
125  const auto B = inpoel[e*4+1];
126  const auto C = inpoel[e*4+2];
127  const auto D = inpoel[e*4+3];
128  std::array< tk::real, 3 > ba{{ x[B]-x[A], y[B]-y[A], z[B]-z[A] }},
129  ca{{ x[C]-x[A], y[C]-y[A], z[C]-z[A] }},
130  da{{ x[D]-x[A], y[D]-y[A], z[D]-z[A] }};
131  const auto J = tk::triple( ba, ca, da ) / 120.0;
132  Assert( J > 0, "Element Jacobian non-positive" );
133 
134  for (ncomp_t c=0; c<5; ++c) {
135  const auto r = lhsd.cptr( c, m_offset );
136  lhsd.var( r, A ) += 2.0 * J;
137  lhsd.var( r, B ) += 2.0 * J;
138  lhsd.var( r, C ) += 2.0 * J;
139  lhsd.var( r, D ) += 2.0 * J;
140 
141  const auto s = lhso.cptr( c, m_offset );
142  lhso.var( s, spidx(A,B) ) += J;
143  lhso.var( s, spidx(A,C) ) += J;
144  lhso.var( s, spidx(A,D) ) += J;
145 
146  lhso.var( s, spidx(B,A) ) += J;
147  lhso.var( s, spidx(B,C) ) += J;
148  lhso.var( s, spidx(B,D) ) += J;
149 
150  lhso.var( s, spidx(C,A) ) += J;
151  lhso.var( s, spidx(C,B) ) += J;
152  lhso.var( s, spidx(C,D) ) += J;
153 
154  lhso.var( s, spidx(D,A) ) += J;
155  lhso.var( s, spidx(D,B) ) += J;
156  lhso.var( s, spidx(D,C) ) += J;
157  }
158  }
159  }

161  //! Compute right hand side
162  //! \param[in] t Physical time
163  //! \param[in] deltat Size of time step
164  //! \param[in] coord Mesh node coordinates
165  //! \param[in] inpoel Mesh element connectivity
166  //! \param[in] U Solution vector at recent time step
167  //! \param[in,out] R Right-hand side vector computed
168  void rhs( tk::real t,
169  tk::real deltat,
170  const std::array< std::vector< tk::real >, 3 >& coord,
171  const std::vector< std::size_t >& inpoel,
172  const tk::Fields& U,
173  tk::Fields& Ue,
174  tk::Fields& R ) const
175  {
176  Assert( U.nunk() == coord[0].size(), "Number of unknowns in solution "
177  "vector at recent time step incorrect" );
178  Assert( R.nunk() == coord[0].size(),
179  "Number of unknowns and/or number of components in right-hand "
180  "side vector incorrect" );
181 
182  const auto& x = coord[0];
183  const auto& y = coord[1];
184  const auto& z = coord[2];
185 
186  // ratio of specific heats
187  auto g = g_inputdeck.get< tag::param, tag::compflow, tag::gamma >()[0];
188 
189  // 1st stage: update element values from node values (gather-add)
190  for (std::size_t e=0; e<inpoel.size()/4; ++e) {
191 
192  // access node IDs
193  const std::array< std::size_t, 4 > N{{ inpoel[e*4+0], inpoel[e*4+1],
194  inpoel[e*4+2], inpoel[e*4+3] }};
195  // compute element Jacobi determinant
196  const std::array< tk::real, 3 >
197  ba{{ x[N[1]]-x[N[0]], y[N[1]]-y[N[0]], z[N[1]]-z[N[0]] }},
198  ca{{ x[N[2]]-x[N[0]], y[N[2]]-y[N[0]], z[N[2]]-z[N[0]] }},
199  da{{ x[N[3]]-x[N[0]], y[N[3]]-y[N[0]], z[N[3]]-z[N[0]] }};
200  const auto J = tk::triple( ba, ca, da ); // J = 6V
201  Assert( J > 0, "Element Jacobian non-positive" );
202 
203  // shape function derivatives, nnode*ndim [4][3]
204  std::array< std::array< tk::real, 3 >, 4 > grad;
205  grad[1] = tk::crossdiv( ca, da, J );
206  grad[2] = tk::crossdiv( da, ba, J );
207  grad[3] = tk::crossdiv( ba, ca, J );
208  for (std::size_t i=0; i<3; ++i)
209  grad[0][i] = -grad[1][i]-grad[2][i]-grad[3][i];
210 
211  // access solution at element nodes
212  std::array< std::array< tk::real, 4 >, 5 > u;
213  for (ncomp_t c=0; c<5; ++c) u[c] = U.extract( c, m_offset, N );
214  // access solution at elements
215  std::array< const tk::real*, 5 > ue;
216  for (ncomp_t c=0; c<5; ++c) ue[c] = Ue.cptr( c, m_offset );
217 
218  // pressure
219  std::array< tk::real, 4 > p;
220  for (std::size_t a=0; a<4; ++a)
221  p[a] = (g-1.0)*(u[4][a] - (u[1][a]*u[1][a] +
222  u[2][a]*u[2][a] +
223  u[3][a]*u[3][a])/2.0/u[0][a]);
224 
225  // sum nodal averages to element
226  for (ncomp_t c=0; c<5; ++c) {
227  Ue.var(ue[c],e) = 0.0;
228  for (std::size_t a=0; a<4; ++a)
229  Ue.var(ue[c],e) += u[c][a]/4.0;
230  }
231 
232  // sum flux contributions to element
233  tk::real d = deltat/2.0;
234  for (std::size_t j=0; j<3; ++j)
235  for (std::size_t a=0; a<4; ++a) {
236  // mass: advection
237  Ue.var(ue[0],e) -= d * grad[a][j] * u[j+1][a];
238  // momentum: advection
239  for (std::size_t i=0; i<3; ++i)
240  Ue.var(ue[i+1],e) -= d * grad[a][j] * u[j+1][a]*u[i+1][a]/u[0][a];
241  // momentum: pressure
242  Ue.var(ue[j+1],e) -= d * grad[a][j] * p[a];
243  // energy: advection and pressure
244  Ue.var(ue[4],e) -= d * grad[a][j] *
245  (u[4][a] + p[a]) * u[j+1][a]/u[0][a];
246  }
247 
248  // add (optional) source to all equations
249  std::array< std::array< tk::real, 5 >, 4 > s{{
250  Problem::src( m_c, x[N[0]], y[N[0]], z[N[0]], t ),
251  Problem::src( m_c, x[N[1]], y[N[1]], z[N[1]], t ),
252  Problem::src( m_c, x[N[2]], y[N[2]], z[N[2]], t ),
253  Problem::src( m_c, x[N[3]], y[N[3]], z[N[3]], t ) }};
254  for (std::size_t c=0; c<5; ++c)
255  for (std::size_t a=0; a<4; ++a)
256  Ue.var(ue[c],e) += d/4.0 * s[a][c];
257 
258  }
259 
260 
261  // zero right hand side for all components
262  for (ncomp_t c=0; c<5; ++c) R.fill( c, m_offset, 0.0 );
263 
264  // 2nd stage: form rhs from element values (scatter-add)
265  for (std::size_t e=0; e<inpoel.size()/4; ++e) {
266 
267  // access node IDs
268  const std::array< std::size_t, 4 > N{{ inpoel[e*4+0], inpoel[e*4+1],
269  inpoel[e*4+2], inpoel[e*4+3] }};
270  // compute element Jacobi determinant
271  const std::array< tk::real, 3 >
272  ba{{ x[N[1]]-x[N[0]], y[N[1]]-y[N[0]], z[N[1]]-z[N[0]] }},
273  ca{{ x[N[2]]-x[N[0]], y[N[2]]-y[N[0]], z[N[2]]-z[N[0]] }},
274  da{{ x[N[3]]-x[N[0]], y[N[3]]-y[N[0]], z[N[3]]-z[N[0]] }};
275  const auto J = tk::triple( ba, ca, da ); // J = 6V
276  Assert( J > 0, "Element Jacobian non-positive" );
277 
278  // shape function derivatives, nnode*ndim [4][3]
279  std::array< std::array< tk::real, 3 >, 4 > grad;
280  grad[1] = tk::crossdiv( ca, da, J );
281  grad[2] = tk::crossdiv( da, ba, J );
282  grad[3] = tk::crossdiv( ba, ca, J );
283  for (std::size_t i=0; i<3; ++i)
284  grad[0][i] = -grad[1][i]-grad[2][i]-grad[3][i];
285 
286  // access solution at elements
287  std::array< tk::real, 5 > ue;
288  for (ncomp_t c=0; c<5; ++c) ue[c] = Ue( e, c, m_offset );
289  // access pointer to right hand side at component and offset
290  std::array< const tk::real*, 5 > r;
291  for (ncomp_t c=0; c<5; ++c) r[c] = R.cptr( c, m_offset );
292 
293  // pressure
294  auto p = (g-1.0)*(ue[4] -
295  (ue[1]*ue[1] + ue[2]*ue[2] + ue[3]*ue[3])/2.0/ue[0]);
296 
297  // scatter-add flux contributions to rhs at nodes
298  tk::real d = deltat * J/6.0;
299  for (std::size_t j=0; j<3; ++j)
300  for (std::size_t a=0; a<4; ++a) {
301  // mass: advection
302  R.var(r[0],N[a]) += d * grad[a][j] * ue[j+1];
303  // momentum: advection
304  for (std::size_t i=0; i<3; ++i)
305  R.var(r[i+1],N[a]) += d * grad[a][j] * ue[j+1]*ue[i+1]/ue[0];
306  // momentum: pressure
307  R.var(r[j+1],N[a]) += d * grad[a][j] * p;
308  // energy: advection and pressure
309  R.var(r[4],N[a]) += d * grad[a][j] * (ue[4] + p) * ue[j+1]/ue[0];
310  }
311 
312  // add (optional) source to all equations
313  auto xc = (x[N[0]] + x[N[1]] + x[N[2]] + x[N[3]]) / 4.0;
314  auto yc = (y[N[0]] + y[N[1]] + y[N[2]] + y[N[3]]) / 4.0;
315  auto zc = (z[N[0]] + z[N[1]] + z[N[2]] + z[N[3]]) / 4.0;
316  auto s = Problem::src( 0, xc, yc, zc, t+deltat/2 );
317  for (std::size_t c=0; c<5; ++c)
318  for (std::size_t a=0; a<4; ++a)
319  R.var(r[c],N[a]) += d/4.0 * s[c];
320 
321  }

323 // Physics::viscousRhs( deltat, J, N, grad, u, r, R );
324 // // add heat conduction contribution to energy rhs
325 // Physics::conductRhs( deltat, J, N, grad, u, r, R );
326  }

328  //! Compute the minimum time step size
329  //! \param[in] U Solution vector at recent time step
330  //! \param[in] coord Mesh node coordinates
331  //! \param[in] inpoel Mesh element connectivity
332  //! \return Minimum time step size
333  tk::real dt( const std::array< std::vector< tk::real >, 3 >& coord,
334  const std::vector< std::size_t >& inpoel,
335  const tk::Fields& U ) const
336  {
337  Assert( U.nunk() == coord[0].size(), "Number of unknowns in solution "
338  "vector at recent time step incorrect" );
339  const auto& x = coord[0];
340  const auto& y = coord[1];
341  const auto& z = coord[2];
342  // ratio of specific heats
343  auto g = g_inputdeck.get< tag::param, tag::compflow, tag::gamma >()[0];
344  // compute the minimum dt across all elements we own
345  tk::real mindt = std::numeric_limits< tk::real >::max();
346  for (std::size_t e=0; e<inpoel.size()/4; ++e) {
347  const std::array< std::size_t, 4 > N{{ inpoel[e*4+0], inpoel[e*4+1],
348  inpoel[e*4+2], inpoel[e*4+3] }};
349  // compute cubic root of element volume as the characteristic length
350  const std::array< tk::real, 3 >
351  ba{{ x[N[1]]-x[N[0]], y[N[1]]-y[N[0]], z[N[1]]-z[N[0]] }},
352  ca{{ x[N[2]]-x[N[0]], y[N[2]]-y[N[0]], z[N[2]]-z[N[0]] }},
353  da{{ x[N[3]]-x[N[0]], y[N[3]]-y[N[0]], z[N[3]]-z[N[0]] }};
354  const auto L = std::cbrt( tk::triple( ba, ca, da ) / 6.0 );
355  // access solution at element nodes at recent time step
356  std::array< std::array< tk::real, 4 >, 5 > u;
357  for (ncomp_t c=0; c<5; ++c) u[c] = U.extract( c, m_offset, N );
358  // compute the maximum length of the characteristic velocity (fluid
359  // velocity + sound velocity) across the four element nodes
360  tk::real maxvel = 0.0;
361  for (std::size_t j=0; j<4; ++j) {
362  auto& r = u[0][j]; // rho
363  auto& ru = u[1][j]; // rho * u
364  auto& rv = u[2][j]; // rho * v
365  auto& rw = u[3][j]; // rho * w
366  auto& re = u[4][j]; // rho * e
367  auto p = (g-1.0)*(re - (ru*ru + rv*rv + rw*rw)/2.0/r); // pressure
368  if (p < 0) p = 0.0;
369  auto c = std::sqrt(g*p/r); // sound speed
370  auto v = std::sqrt((ru*ru + rv*rv + rw*rw)/r/r) + c; // char. velocity
371  if (v > maxvel) maxvel = v;
372  }
373  // compute element dt for the Euler equations
374  auto euler_dt = L / maxvel;
375  // compute element dt based on the viscous force
376  auto viscous_dt = Physics::viscous_dt( L, u );
377  // compute element dt based on thermal diffusion
378  auto conduct_dt = Physics::conduct_dt( L, u );
379  // compute minimum element dt
380  auto elemdt = std::min( euler_dt, std::min( viscous_dt, conduct_dt ) );
381  // find minimum dt across all elements
382  if (elemdt < mindt) mindt = elemdt;
383  }
384  return mindt;
385  }

387  //! Extract the velocity field at cell nodes
388  //! \param[in] U Solution vector at recent time step
389  //! \param[in] N Element node indices
390  //! \return Array of the four values of the velocity field
391  std::array< std::array< tk::real, 4 >, 3 >
392  velocity( const tk::Fields& U,
393  const std::array< std::vector< tk::real >, 3 >&,
394  const std::array< std::size_t, 4 >& N ) const
395  {
396  std::array< std::array< tk::real, 4 >, 3 > v;
397  v[0] = U.extract( 1, m_offset, N );
398  v[1] = U.extract( 2, m_offset, N );
399  v[2] = U.extract( 3, m_offset, N );
400  auto r = U.extract( 0, m_offset, N );
401  std::transform( r.begin(), r.end(), v[0].begin(), v[0].begin(),
402  []( tk::real s, tk::real& d ){ return d /= s; } );
403  std::transform( r.begin(), r.end(), v[1].begin(), v[1].begin(),
404  []( tk::real s, tk::real& d ){ return d /= s; } );
405  std::transform( r.begin(), r.end(), v[2].begin(), v[2].begin(),
406  []( tk::real s, tk::real& d ){ return d /= s; } );
407  return v;
408  }

416  //! \brief Query Dirichlet boundary condition value on a given side set for
417  //! all components in this PDE system
418  //! \param[in] t Physical time
419  //! \param[in] deltat Time step size
420  //! \param[in] ss Pair of side set ID and (local) node IDs on the side set
421  //! \param[in] coord Mesh node coordinates
422  //! \return Vector of pairs of bool and boundary condition value associated
423  //! to mesh node IDs at which Dirichlet boundary conditions are set. Note
424  //! that instead of the actual boundary condition value, we return the
425  //! increment between t+dt and t, since that is what the solution requires
426  //! as we solve for the soution increments and not the solution itself.
427  std::map< std::size_t, std::vector< std::pair<bool,tk::real> > >
428  dirbc( tk::real t,
429  tk::real deltat,
430  const std::pair< const int, std::vector< std::size_t > >& ss,
431  const std::array< std::vector< tk::real >, 3 >& coord ) const
432  {
433  using tag::param; using tag::compflow; using tag::bcdir;
434  using NodeBC = std::vector< std::pair< bool, tk::real > >;
435  std::map< std::size_t, NodeBC > bc;
436  const auto& ubc = g_inputdeck.get< param, compflow, bcdir >();
437  if (!ubc.empty()) {
438  Assert( ubc.size() > 0, "Indexing out of Dirichlet BC eq-vector" );
439  const auto& x = coord[0];
440  const auto& y = coord[1];
441  const auto& z = coord[2];
442  for (const auto& b : ubc[0])
443  if (std::stoi(b) == ss.first)
444  for (auto n : ss.second) {
445  Assert( x.size() > n, "Indexing out of coordinate array" );
446  auto s = Problem::solinc( 0, x[n], y[n], z[n], t, deltat );
447  bc[n] = {{ {true,s[0]}, {true,s[1]}, {true,s[2]}, {true,s[3]},
448  {true,s[4]} }};
449  }
450  }
451  return bc;
452  }
