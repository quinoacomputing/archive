49  //! For testing boost::value_factory (runtime polymorphism using value
50  //! semantics). The struct below uses runtime polymorphism without client-side
51  //! inheritance: inheritance is confined to the internals of the struct below,
52  //! invisible to client-code. The struct exclusively deals with ownership
53  //! enabling client-side value semantics. Credit goes to Sean Parent at Adobe:
54  //! https://github.com/sean-parent/sean-parent.github.com/wiki/
55  //! Papers-and-Presentations
56  struct VBase {
57  //! Constructor taking an object modeling Concept (see below). The object
58  //! of class T was pre-constructed.
59  template< typename T >
60  explicit VBase( T x ) :
61  self( tk::make_unique< Model<T> >( std::move(x) ) ),
62  ctor( "val" ),
63  assg() {}
64 
65  //! Constructor taking a function pointer to a constructor of an object
66  //! modeling Concept (see below). Passing std::function allows late
67  //! execution of the constructor, i.e., as late as inside this struct's
68  //! constructor, and thus usage from a factory.
69  template< typename T >
70  explicit VBase( std::function<T()> x ) :
71  self( tk::make_unique< Model<T> >( std::move(x()) ) ),
72  ctor( "fun" ),
73  assg() {}
74 
75  //! Public interface to querying the child constructor type invoked
76  std::string Type() const { return self->Type(); }
77  //! Copy assignment
78  VBase& operator=( const VBase& x )
79  { VBase tmp(x); *this = std::move(tmp); assg = "cpy"; return *this; }
80  //! Copy constructor
81  VBase( const VBase& x ) : self( x.self->copy() ), ctor( "cpy" ), assg() {}
82  //! Move assignment: could be default
83  VBase& operator=( VBase&& x ) noexcept {
84  self = std::move( x.self );
85  ctor = std::move( x.ctor );
86  assg = "mov";
87  return *this;
88  }
89  //! Move constructor: could be default, but in terms of move assignment
90  VBase( VBase&& x ) noexcept { *this = std::move(x); ctor = "mov"; }
91 
92  //! Accessor to ctor type
93  std::string ctortype() const { return ctor; }
94  //! Accessor to assignment type
95  std::string assgntype() const { return assg; }
96 
97  //! Concept is a pure virtual base struct specifying the requirements of
98  //! polymorphic objects deriving from it
99  struct Concept {
100  Concept() = default;
101  Concept( const Concept& ) = default;
102  virtual ~Concept() = default;
103  virtual Concept* copy() const = 0;
104  virtual std::string Type() const = 0;
105  };
106 
107  //! Model models the Concept above by deriving from it and overriding the
108  //! the virtual functions required by Concept
109  template< typename T >
110  struct Model : Concept {
111  Model( T x ) : data( std::move(x) ) {}
112  Concept* copy() const override { return new Model( *this ); }
113  std::string Type() const override { return data.Type(); }
114  T data;
115  };
116 
117  std::unique_ptr< Concept > self; //!< VBase pointer used polymorphically
118  std::string ctor;
119  std::string assg;
120  };

363 //! For testing boost::value_factory (runtime polymorphism using value
364 //! semantics) with Charm++ children. Not in Factory_common so Charm++ can find
365 //! it. The struct below uses runtime polymorphism without client-side
366 //! inheritance: inheritance is confined to the internals of the struct below,
367 //! invisible to client-code. The struct exclusively deals with ownership
368 //! enabling client-side value semantics. Credit goes to Sean Parent at Adobe:
369 //! https://github.com/sean-parent/sean-parent.github.com/wiki/
370 //! Papers-and-Presentations
371 struct VBase {
372  //! Constructor taking a function pointer to a constructor of an object
373  //! modeling Concept (see below). Passing std::function allows late
374  //! execution of the constructor of T, i.e., at some future time, and thus
375  //! usage from a factory. Note that the value of the first function
376  //! argument, std::function<T()>, is not used here, but its constructor
377  //! type, T, is used to enable the compiler to deduce the model constructor
378  //! type, used to create its Charm proxy, defined by T::Proxy. The actual
379  //! constructor of T is not called here but at some future time by the
380  //! Charm++ runtime system, here only an asynchrounous ckNew() is called,
381  //! i.e., a message (or request) for a future call to T's constructor. This
382  //! overload is only enabled for Charm++ chare objects defining typedef
383  //! 'Proxy', which must define the Charm++ proxy. All optional constructor
384  //! arguments are forwarded to ckNew() and thus to T's constructor. If
385  //! it was somehow possible to obtain all bound arguments' types and values
386  //! from an already-bound std::function, we could use those instead of
387  //! having to explicitly forward the model constructor arguments via this
388  //! host constructor. See also tk::recordCharmModel().
389  template< typename T, typename... ConstrArgs,
390  typename std::enable_if< tk::HasTypedefProxy<T>::value, int >::type = 0 >
391  explicit VBase( std::function<T()> c, ConstrArgs... args ) :
392  self( tk::make_unique< Model< typename T::Proxy > >
393  (std::move(T::Proxy::ckNew(std::forward<ConstrArgs>(args)...))) ) {
394  Assert( c == nullptr, "std::function argument to VBase Charm "
395  "constructor must be nullptr" );
396  #ifdef NDEBUG
397  IGNORE(c);
398  #endif
399  }
400  //! Copy assignment
401  VBase& operator=( const VBase& x )
402  { VBase tmp(x); *this = std::move(tmp); return *this; }
403  //! Copy constructor
404  VBase( const VBase& x ) : self( x.self->copy() ) {}
405  //! Move assignment
406  VBase& operator=( VBase&& ) noexcept = default;
407  //! Move constructor
408  VBase( VBase&& ) noexcept = default;
409  //! Concept is a pure virtual base struct specifying the requirements of
410  //! polymorphic objects deriving from it
411  struct Concept {
412  Concept() = default;
413  Concept( const Concept& ) = default;
414  virtual ~Concept() = default;
415  virtual Concept* copy() const = 0;
416  };
417  //! Model models the Concept above by deriving from it and overriding the
418  //! the virtual functions required by Concept
419  template< typename T >
420  struct Model : Concept {
421  Model( T x ) : data( std::move(x) ) {}
422  Concept* copy() const { return new Model( *this ); }
423  T data;
424  };
425  std::unique_ptr< Concept > self; //!< VBase pointer used polymorphically
426 };
