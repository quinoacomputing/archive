21 //! Serialize std::unordered_map to raw memory stream
22 //! \param[in] m Hash map to serialize
23 //! \return Pair of the length and the raw stream containing the serialized map
24 template< class Key,
25  class T,
26  class Hash = std::hash< Key >,
27  class KeyEqual = std::equal_to< Key > >
28 std::pair< int, std::unique_ptr<char[]> >
29 serialize( const std::unordered_map< Key, T, Hash, KeyEqual >& m ) {
30  // Prepare for serializing map to a raw binary stream, compute size
31  PUP::sizer sizer;
32  sizer | const_cast< std::unordered_map< Key, T, Hash, KeyEqual >& >( m );
33 
34  // Create raw character stream to store the serialized map
35  std::unique_ptr<char[]> flatData = tk::make_unique<char[]>( sizer.size() );
36 
37  // Serialize map, each message will contain a map
38  PUP::toMem packer( flatData.get() );
39  packer | const_cast< std::unordered_map< Key, T, Hash, KeyEqual >& >( m );
40 
41  // Return size of and raw stream
42  return { sizer.size(), std::move(flatData) };
43 }

45 //! \brief Charm++ custom reducer for merging std::unordered_maps during
46 //! reduction across PEs
47 //! \param[in] nmsg Number of messages in msgs
48 //! \param[in] msgs Charm++ reduction message containing the serialized maps
49 //! \return Aggregated std::unordered_maps built for further aggregation if
50 //! needed
51 //! \details During aggregation the map keys are inserted, i.e., the keys remain
52 //! unique and the mapped values, assuming containers defining begin() and
53 //! end() iterators() are concatenated.
54 //! \note The mapped type must be a container, i.e., must provide iterators
55 //! begin() and end().
56 template< class Key,
57  class T,
58  class Hash = std::hash< Key >,
59  class KeyEqual = std::equal_to< Key > >
60 CkReductionMsg*
61 mergeHashMap( int nmsg, CkReductionMsg **msgs ) {
62  // Will store deserialized map
63  std::unordered_map< Key, T, Hash, KeyEqual > p;
64 
65  // Create PUP deserializer based on message passed in
66  PUP::fromMem creator( msgs[0]->getData() );
67 
68  // Deserialize map from raw stream
69  creator | p;
70 
71  for (int m=1; m<nmsg; ++m) {
72  // Unpack map
73  std::unordered_map< Key, T, Hash, KeyEqual > u;
74  PUP::fromMem curCreator( msgs[m]->getData() );
75  curCreator | u;
76  // Concatenate maps
77  for (const auto& c : u) {
78  auto& b = p[ c.first ];
79  b.insert( end(b), begin(c.second), end(c.second) );
80  }
81  }
82 
83  // Serialize concatenated maps to raw stream
84  auto stream = tk::serialize( p );
85 
86  // Forward serialized hash map
87  return CkReductionMsg::buildNew( stream.first, stream.second.get() );
88 }
