32 // *****************************************************************************
33 // Compute point-based communication maps
34 //! \param[in] graphsize Size of unstructured mesh graph object
35 //! \param[in] chp Array of chare ownership IDs mapping graph points to
36 //! concurrent async chares
37 //! \param[in] tetinpoel Tetrahedra element connectivity
38 //! \param[in] nchare Number of work units (Charm++ chares)
39 //! \param[in] toofine Error message to print triggered for too large
40 //! overdecomposition
41 //! \return Point-based communication map for all chares
42 //! \details This is a _point-based_ export map, because it stores the global
43 //! ids of the mesh points that chares need to export to fellow chares
44 //! computed based on which chare owns a mesh point. This is for algorithms
45 //! that work by computing data on the mesh by looping over mesh points and
46 //! their surrounding points, e.g., edge-based algorithms. The communication
47 //! map computed here is stored in a vector of maps associating a vector of
48 //! global mesh point ids to chare ids that the points need to be exported to.
49 //! Using the map produced here amounts to each chare taking its own map
50 //! indexing the outermost vector with its chare index. The map computed here
51 //! is an an export map from which the import map can be computed if needed.
52 //!
53 //! In the MPI paradigm, these chare export maps correspond to the export
54 //! lists, i.e., lists of global ids exported by a given rank to a set of
55 //! receiver ranks and their associated mesh points sent at which data are to
56 //! be sent (exported). This is as opposed to import maps which are lists of
57 //! global ids imported by a given rank to a set of sender ranks and their
58 //! associated mesh points sent at which data are to be received (imported).
59 //! For example, in Zoltan, this export/import roughly corresponds to the
60 //! "exported" and "imported" mesh node ids after partitioning. Actually,
61 //! Zoltan_LB_Partition() already returns this information. However, if the
62 //! partitioning with Zoltan is done using less MPI ranks than the number of
63 //! desired mesh partitions, i.e., overdecomposition (as is the case here),
64 //! there are more mesh partitions than MPI ranks and thus the arrays returned
65 //! by Zoltan are not sufficient to determine the export and import maps for
66 //! all the chares. Thus we compute the export mapping here.
67 //! \note This function is only supposed to operate on MPI rank 0.
68 //! \note This function does not and should not modify global-scope data.
69 // *****************************************************************************
70 {
71  // Map to associate a chare id to a map of receiver chare ids associated to
72  // unique global point ids sent (export map)
73  std::map< std::size_t,
74  std::map< std::size_t, std::set< std::size_t > > > comm;
75 
76  // Generate points surrounding points
77  auto psup = tk::genPsup( tetinpoel, 4, tk::genEsup(tetinpoel,4) );
78 
79  // Construct point-based export maps
80  for (std::size_t p=0; p<graphsize; ++p) // for all mesh points
81  for (auto i=psup.second[p]+1; i<=psup.second[p+1]; ++i) {
82  auto q = psup.first[i];
83  if (chp[p] != chp[q]) // if the point-colors differ, store global id
84  comm[ chp[p] ][ chp[q] ].insert( p );
85  }
86 
87  // This check should always be done, as it can result from incorrect user
88  // input compared to the mesh size and not due to programmer error.
89  ErrChk( comm.size() == nchare, std::move(toofine) );
90 
91  #ifndef NDEBUG
92  std::size_t c = 0;
93  for (const auto& e : comm)
94  Assert( e.first == c++,
95  "Export/import maps should not be missing for chare id " +
96  std::to_string(c-1) );
97  #endif
98 
99  // Construct final product: a vector of export maps associating receiver
100  // chare ids to unique communicated global point ids for all chare ids, and
101  // store it in global scope so that the Charm++ chares can access it
102  std::vector< std::map< std::size_t, std::vector< std::size_t > > >
103  pcomm( nchare );
104  for (const auto& e : comm) {
105  for (const auto& x : e.second)
106  for (auto p : x.second)
107  pcomm[ e.first ][ x.first ].push_back( p );
108  }
109 
110  Assert( pcomm.size() == nchare,
111  "Number of export/import maps must equal the number of chares" );
112

114 // for (const auto& m : pcomm) {
115 // std::cout << h++ << " p-> ";
116 // for (const auto& x : m) {
117 // std::cout << x.first << ": ";
118 // for (auto p : x.second)
119 // std::cout << p << " ";
120 // }
121 // std::cout << '\n';
122 // }
123 
124  return pcomm;
125 }
