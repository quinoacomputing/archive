24 std::size_t
25 shiftToZero( std::vector< std::size_t >& inpoel )
26 // *****************************************************************************
27 // Shift node IDs to start with zero in element connectivity
28 //! \param[inout] inpoel Inteconnectivity of points and elements
29 //! \return Amount shifted
30 //! \details This function implements a simple reordering of the node ids of the
31 //! element connectivity in inpoel by shifting the node ids so that the
32 //! smallest is zero.
33 //! \note It is okay to call this function with an empty container; it will
34 //! simply return without throwing an exception.
35 //! \author J. Bakosi
36 // *****************************************************************************
37 {
38  if (inpoel.empty()) return 0;
39 
40  // find smallest node id
41  auto minId = *std::min_element( begin(inpoel), end(inpoel) );
42 
43  // shift node ids to start from zero
44  for (auto& n : inpoel) n -= minId;
45 
46  return minId;
47 }

50 remap( std::vector< std::size_t >& id, const std::vector< std::size_t >& newid )
51 // *****************************************************************************
52 // Reorder mesh points given a new order, i.e., index map
53 //! \param[inout] id Vector of point ids
54 //! \param[in] newid Array of indices creating a new order
55 //! \details This function implements a simple reordering (or remapping) of the
56 //! node ids of the vector passed in using the vector newid. Thus the vector
57 //! in newid is thought of as a mapping between the array index to its value.
58 //! The function overwrites every value, n, of vector id with newid[n].
59 //! \note It is okay to call this function with either of the containers empty;
60 //! it will simply return without throwing an exception.
61 //! \author J. Bakosi
62 // *****************************************************************************
63 {
64  if (id.empty() || newid.empty()) return;
65 
66  Assert( *max_element( begin(id), end(id) ) < newid.size(),
67  "attempt to index out of node id bounds using newid" );
68 
69  // remap node ids in vector id
70  for (auto& n : id) n = newid[n];
71 }

76 // *****************************************************************************
77 // Reorder mesh points with the advancing front technique
78 //! \param[in] psup Points surrounding points
79 //! \return Pair of maps between old->new and new->old order
80 //! \author J. Bakosi
81 // *****************************************************************************
82 {
83  // Find out number of nodes in graph
84  auto npoin = psup.second.size()-1;
85 
86  // Construct mapping using advancing front
87  std::vector< int > hpoin( npoin, -1 ), lpoin( npoin, 0 );
88  std::vector< std::size_t > mapvec( npoin, 0 );
89  hpoin[0] = 0;
90  lpoin[0] = 1;
91  std::size_t num = 1;
92  while (num < npoin) {
93  std::size_t cnt = 0;
94  std::size_t i = 0;
95  std::vector< int > kpoin( npoin, -1 );
96  int p;
97  while ((p = hpoin[i]) != -1) {
98  ++i;
99  auto P = static_cast< std::size_t >( p );
100  for (auto j=psup.second[P]+1; j<=psup.second[P+1]; ++j) {
101  auto q = psup.first[j];
102  if (lpoin[q] != 1) { // consider points not yet counted
103  mapvec[q] = num++;
104  kpoin[cnt] = static_cast< int >( q ); // register point as counted
105  lpoin[q] = 1; // register the point as counted
106  ++cnt;
107  }
108  }
109  }
110  hpoin = kpoin;
111  }
112 
113  // Construct new->old id map
114  std::size_t i = 0;
115  std::vector< std::size_t > oldmap( npoin );
116  for (auto n : mapvec) oldmap[n] = i++;
117 
118  // Return old->new and new->old maps
119  return { mapvec, oldmap };
120 }

142 // Generate element connectivity of local node IDs from connectivity of global
143 // node IDs also returning the mapping between local to global IDs
144 //! \param[in] ginpoel Element connectivity with global node IDs
145 //! \return Tuple of (1) element connectivity with local node IDs, (2) the
