22 std::size_t
23 shiftToZero( std::vector< std::size_t >& inpoel )
24 // *****************************************************************************
25 // Shift node IDs to start with zero in element connectivity
26 //! \param[inout] inpoel Inteconnectivity of points and elements
27 //! \return Amount shifted
28 //! \details This function implements a simple reordering of the node ids of the
29 //! element connectivity in inpoel by shifting the node ids so that the
30 //! smallest is zero.
31 //! \note It is okay to call this function with an empty container; it will
32 //! simply return without throwing an exception.
33 // *****************************************************************************
34 {
35  if (inpoel.empty()) return 0;
36 
37  // find smallest node id
38  auto minId = *std::min_element( begin(inpoel), end(inpoel) );
39 
40  // shift node ids to start from zero
41  for (auto& n : inpoel) n -= minId;
42 
43  return minId;
44 }

47 remap( std::vector< std::size_t >& id, const std::vector< std::size_t >& map )
48 // *****************************************************************************
49 // Apply new maping to vector of indices
50 //! \param[inout] id Vector of integer IDs to remap
51 //! \param[in] map Array of indices creating a new order
52 //! \details This function applies a mapping (reordering) to the integer IDs
53 //! passed in using the map passed in. The mapping is expressed between the
54 //! array index ands its value. The function overwrites every value, i, of
55 //! vector id with map[i].
56 //! \note The sizes of id and map need not equal. Only the maximum index in id
57 //! must be lower than the size of map.
58 //! \note It is okay to call this function with either of the containers empty;
59 //! it will simply return without throwing an exception.
60 // *****************************************************************************
61 {
62  if (id.empty() || map.empty()) return;
63 
64  Assert( *max_element( begin(id), end(id) ) < map.size(),
65  "Indexing out of bounds" );
66 
67  // remap integer IDs in vector id
68  for (auto& i : id) i = map[i];
69 }

73 // *****************************************************************************
74 // Apply new maping to vector of real numbers
75 //! \param[inout] r Vector of real numbers to remap
76 //! \param[in] map Array of indices creating a new order
77 //! \details This function applies a mapping (reordering) to the real values
78 //! passed in using the map passed in. The mapping is expressed between the
79 //! array index ands its value. The function moves every value r[i] to
80 //! r[ map[i] ].
81 //! \note The sizes of r and map must be equal and the maximum index in map must
82 //! be lower than the size of map.
83 //! \note It is okay to call this function with either of the containers empty;
84 //! it will simply return without throwing an exception.
85 // *****************************************************************************
86 {
87  if (r.empty() || map.empty()) return;
88 
89  Assert( r.size() == map.size(), "Size mismatch" );
90  Assert( *max_element( begin(map), end(map) ) < map.size(),
91  "Indexing out of bounds" );
92 
93  // remap real numbers in vector
94  auto m = r;
95  for (std::size_t i=0; i<map.size(); ++i) r[ map[i] ] = m[i];
96 }

101 // *****************************************************************************
102 // Reorder mesh points with the advancing front technique
103 //! \param[in] psup Points surrounding points
104 //! \return Mapping created by renumbering (reordering)
105 // *****************************************************************************
106 {
107  // Find out number of nodes in graph
108  auto npoin = psup.second.size()-1;
109 
110  // Construct mapping using advancing front
111  std::vector< int > hpoin( npoin, -1 ), lpoin( npoin, 0 );
112  std::vector< std::size_t > map( npoin, 0 );
113  hpoin[0] = 0;
114  lpoin[0] = 1;
115  std::size_t num = 1;
116  while (num < npoin) {
117  std::size_t cnt = 0;
118  std::size_t i = 0;
119  std::vector< int > kpoin( npoin, -1 );
120  int p;
121  while ((p = hpoin[i]) != -1) {
122  ++i;
123  auto P = static_cast< std::size_t >( p );
124  for (auto j=psup.second[P]+1; j<=psup.second[P+1]; ++j) {
125  auto q = psup.first[j];
126  if (lpoin[q] != 1) { // consider points not yet counted
127  map[q] = num++;
128  kpoin[cnt] = static_cast< int >( q ); // register point as counted
129  lpoin[q] = 1; // register the point as counted
130  ++cnt;
131  }
132  }
133  }
134  hpoin = kpoin;
135  }

164 // *****************************************************************************
165 // Generate element connectivity of local node IDs from connectivity of global
166 // node IDs also returning the mapping between local to global IDs
167 //! \param[in] ginpoel Element connectivity with global node IDs
168 //! \return Tuple of (1) element connectivity with local node IDs, (2) the
169 //! vector of unique global node IDs (i.e., the mapping between local to
170 //! global node IDs), and (3) mapping between global to local node IDs.
171 // *****************************************************************************
172 {
173  // Make a copy of the element connectivity with global node ids
174  auto gid = ginpoel;
175 
176  // Generate a vector that holds only the unique global mesh node ids
177  tk::unique( gid );
178 
179  // Assign local node ids to global node ids
180  const auto lid = tk::assignLid( gid );
181 
182  // Generate element connectivity using local node ids
183  std::vector< std::size_t > inpoel;
184  for (auto p : ginpoel) inpoel.push_back( tk::cref_find( lid, p ) );
185 
186  // Return element connectivty with local node IDs
