7  \brief Initialization policies
8  \details This file defines initialization policy classes. As opposed to
9  coefficients policies, see, e.g., DiffEq/BetaCoeffPolicy.h, initialization
10  policies are not SDE-specific -- at least at this time.
11 
12  General requirements on initialization policy classes:
13 
14  - Must define the member function _init_, which is used to do the
15  initialization. Required signature:
16  \code{.cpp}
17  template< class eq >
18  static void init( const ctr::InputDeck& deck,
19  const tk::RNG& rng,
20  int stream,
21  tk::Particles& particles,
22  tk::ctr::ncomp_type e,
23  tk::ctr::ncomp_type ncomp,
24  tk::ctr::ncomp_type offset );
25  \endcode
26  where _deck_ is the input deck from which configuration is read, _rng_ is
27  a reference to a random number generator to use, _stream_ is the thread
28  (or stream) id, _particles_ denotes the particle properties array to be
29  initialized, _e_ is the component index selecting which equation is to be
30  initialized in the system, _ncomp_ is the total number of equations in the
31  system, and _offset_ is the offset in the particle array at which
32  initialization should be done.
33 
34  - Must define the static function _type()_, returning the enum value of the
35  policy option. Example:
36  \code{.cpp}
37  static ctr::InitPolicyType type() noexcept {
38  return ctr::InitPolicyType::RAW;
39  }
40  \endcode
41  which returns the enum value of the option from the underlying option
42  class, collecting all possible options for initialization policies.
43 */

113 //! Delta initialization policy: put in delta-spikes as the joint PDF
114 struct InitDelta {
115 
116  //! Initialize particle properties
117  template< class eq >
118  static void init( const ctr::InputDeck& deck,
119  const tk::RNG& rng,
120  int stream,
121  tk::Particles& particles,
122  tk::ctr::ncomp_type e,
123  tk::ctr::ncomp_type ncomp,
124  tk::ctr::ncomp_type offset )
125  {
126  IGNORE( rng );
127  IGNORE( stream );
128  using ncomp_t = kw::ncomp::info::expect::type;
129 
130  const auto& spike = deck.template get< tag::param, eq, tag::spike >().at(e);
131 
132  // use only the first ncomp spikes if there are more than the equation is
133  // configured for
134  const ncomp_t size = std::min( ncomp, spike.size() );
135 
136  for (ncomp_t c=0; c<size; ++c) {
137  const auto& sc = spike[c]; // vector of spikes for component c
138 
139  ncomp_t i = 0;
140  for (ncomp_t s=0; s<sc.size(); s+=2) {
141  // compute number of samples to be set at relative probability height
142  const auto npar =
143  static_cast< ncomp_t >(
144  static_cast< tk::real >( particles.nunk() ) * sc[s+1] );
145  // assign sample values
146  for (ncomp_t p=0; p<npar; ++p) particles( i+p, c, offset ) = sc[s];
147  i += npar;
148  }
149  }
150  }
151 
152  static ctr::InitPolicyType type() noexcept
153  { return ctr::InitPolicyType::JOINTDELTA; }
154 };

156 //! Beta initialization policy: generate samples from a joint beta PDF
157 struct InitBeta {
158 
159  //! Initialize particle properties (zero)
160  template< class eq >
161  static void init( const ctr::InputDeck& deck,
162  const tk::RNG& rng,
163  int stream,
164  tk::Particles& particles,
165  tk::ctr::ncomp_type e,
166  tk::ctr::ncomp_type ncomp,
167  tk::ctr::ncomp_type offset )
168  {
169  using ncomp_t = kw::ncomp::info::expect::type;
170 
171  const auto& betapdf =
172  deck.template get< tag::param, eq, tag::betapdf >().at(e);
173 
174  // use only the first ncomp betapdfs if there are more than the equation is
175  // configured for
176  const ncomp_t size = std::min( ncomp, betapdf.size() );
177 
178  for (ncomp_t c=0; c<size; ++c) {
179  // get vector of betapdf parameters for component c
180  const auto& bc = betapdf[c];
181 
182  for (ncomp_t s=0; s<bc.size(); s+=4) {
183  // generate beta random numbers for all particles using parameters in bc
184  for (ncomp_t p=0; p<particles.nunk(); ++p)
185  rng.beta( stream, 1, bc[s], bc[s+1], bc[s+2], bc[s+3],
186  &particles( p, c, offset ) );
187  }
188  }
189 
190  }
191 
192  static ctr::InitPolicyType type() noexcept
193  { return ctr::InitPolicyType::JOINTBETA; }
194 };
