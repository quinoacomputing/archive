28 //! Polymorphic mesh reader class for connecting to various mesh readers
29 //! \details This class uses runtime polymorphism without client-side
30 //! inheritance: inheritance is confined to the internals of the this class,
31 //! invisible to client-code. The class exclusively deals with ownership
32 //! enabling client-side value semantics. Credit goes to Sean Parent at Adobe.
33 //! \see http://sean-parent.stlab.cc/papers-and-presentations/#value-semantics-and-concept-based-polymorphism.
34 //! \see For example client code that models a MeshReader, see
35 //! tk::ExodusIIMeshReader or tk::Omega_h_MeshReader.

39  //! Constructor
40  //! \param[in] filename Input mesh filename
41  //! \details Dispatch constructor call to various low level mesh readers by
42  //! creating child class and assigning to base to be used in polymorphic
43  //! fashion.
44  explicit MeshReader( const std::string& filename ) {
45  auto meshtype = detectInput( filename );
46  if (meshtype == MeshReaderType::EXODUSII) {
47  using R = ExodusIIMeshReader;
48  self = make_unique< Model<R> >( R(filename) );
49  #ifdef HAS_OMEGA_H
50  } else if (meshtype == MeshReaderType::OMEGA_H) {
51  using R = Omega_h_MeshReader;
52  self = make_unique< Model<R> >( R(filename) );
53  #endif
54  } else Throw( "Mesh type not implemented or not supported" );
55  }
