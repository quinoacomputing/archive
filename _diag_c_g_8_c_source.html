5  \brief DiagCG for a PDE system with continuous Galerkin without a matrix
6  \details DiagCG advances a system of partial differential equations (PDEs)
7  using continuous Galerkin (CG) finite element (FE) spatial discretization
8  (using linear shapefunctions on tetrahedron elements) combined with a time
9  stepping scheme that is equivalent to the Lax-Wendroff (LW) scheme within
10  the unstructured-mesh FE context and treats discontinuities with
11  flux-corrected transport (FCT). Only the diagonal entries of the left-hand
12  side matrix are non-zero thus it does not need a mtrix-based linear solver.
13  \see The documentation in DiagCG.h.
14 */

142 void
143 DiagCG::start()
144 // *****************************************************************************
145 // Start time stepping
146 // *****************************************************************************
147 {
148  auto d = Disc();
149 
150  // Start timer measuring time stepping wall clock time
151  d->Timer().zero();
152 
153  // Combine own and communicated contributions to LHS and ICs
154  for (const auto& b : d->Bid()) {
155  auto lid = tk::cref_find( d->Lid(), b.first );
156  const auto& blhsc = m_lhsc[ b.second ];
157  for (ncomp_t c=0; c<m_lhs.nprop(); ++c) m_lhs(lid,c,0) += blhsc[c];
158  }
159 
160  // Zero communication buffers for first time step (rhs, mass diffusion rhs)
161  for (auto& b : m_rhsc) std::fill( begin(b), end(b), 0.0 );
162  for (auto& b : m_difc) std::fill( begin(b), end(b), 0.0 );
163 
164  dt();
165 }

167 void
168 DiagCG::lhs()
169 // *****************************************************************************
170 // Compute the left-hand side of transport equations
171 // *****************************************************************************
172 {
173  auto d = Disc();
174 
175  // Compute lumped mass lhs required for both high and low order solutions
176  m_lhs = d->FCT()->lump( *d );
177 
178  if (d->Msum().empty())
179  comlhs_complete();
180  else // send contributions of lhs to chare-boundary nodes to fellow chares
181  for (const auto& n : d->Msum()) {
182  std::vector< std::vector< tk::real > > l;
183  for (auto i : n.second) l.push_back( m_lhs[ tk::cref_find(d->Lid(),i) ] );
184  thisProxy[ n.first ].comlhs( n.second, l );
185  }
186 
187  ownlhs_complete();

191 DiagCG::comlhs( const std::vector< std::size_t >& gid,
192  const std::vector< std::vector< tk::real > >& L )
193 // *****************************************************************************
194 // Receive contributions to left-hand side diagonal matrix on chare-boundaries
195 //! \param[in] gid Global mesh node IDs at which we receive LHS contributions
196 //! \param[in] L Partial contributions of LHS to chare-boundary nodes
197 //! \details This function receives contributions to m_lhs, which stores the
198 //! diagonal (lumped) mass matrix at mesh nodes. While m_lhs stores
199 //! own contributions, m_lhsc collects the neighbor chare contributions during
200 //! communication. This way work on m_lhs and m_lhsc is overlapped. The two
201 //! are combined in start().
202 // *****************************************************************************
203 {
204  Assert( L.size() == gid.size(), "Size mismatch" );
205 
206  using tk::operator+=;
207 
208  auto d = Disc();
209 
210  for (std::size_t i=0; i<gid.size(); ++i) {
211  auto bid = tk::cref_find( d->Bid(), gid[i] );
212  Assert( bid < m_lhsc.size(), "Indexing out of bounds" );
213  m_lhsc[ bid ] += L[i];
214  }
215 
216  if (++m_nlhs == d->Msum().size()) {
217  m_nlhs = 0;
218  comlhs_complete();
219  }

222 void
223 DiagCG::dt()
224 // *****************************************************************************
225 // Comppute time step size
226 // *****************************************************************************
227 {
228  tk::real mindt = std::numeric_limits< tk::real >::max();
229 
230  auto const_dt = g_inputdeck.get< tag::discr, tag::dt >();
231  auto def_const_dt = g_inputdeck_defaults.get< tag::discr, tag::dt >();
232  auto eps = std::numeric_limits< tk::real >::epsilon();
233 
234  auto d = Disc();
235 
236  // use constant dt if configured
237  if (std::abs(const_dt - def_const_dt) > eps) {
238 
239  mindt = const_dt;
240 
241  } else { // compute dt based on CFL
242 
243  // find the minimum dt across all PDEs integrated
244  for (const auto& eq : g_cgpde) {
245  auto eqdt = eq.dt( d->Coord(), d->Inpoel(), m_u );
246  if (eqdt < mindt) mindt = eqdt;
247  }
248 
249  // Scale smallest dt with CFL coefficient
250  mindt *= g_inputdeck.get< tag::discr, tag::cfl >();
251 
252  }
253 
254  // Contribute to minimum dt across all chares the advance to next step
255  contribute( sizeof(tk::real), &mindt, CkReduction::min_double,
256  CkCallback(CkReductionTarget(DiagCG,advance), thisProxy) );
257 }

259 void
260 DiagCG::rhs()
261 // *****************************************************************************
262 // Compute right-hand side of transport equations
263 // *****************************************************************************
264 {
265  auto d = Disc();
266 
267  // Compute right-hand side and query Dirichlet BCs for all equations
268  for (const auto& eq : g_cgpde)
269  eq.rhs( d->T(), d->Dt(), d->Coord(), d->Inpoel(), m_u, m_ue, m_rhs );
270 
271  // Query and match user-specified boundary conditions to side sets
272  bc();
273 
274  if (d->Msum().empty())
275  comrhs_complete();
276  else // send contributions of rhs to chare-boundary nodes to fellow chares
277  for (const auto& n : d->Msum()) {
278  std::vector< std::vector< tk::real > > r;
279  for (auto i : n.second) r.push_back( m_rhs[ tk::cref_find(d->Lid(),i) ] );
280  thisProxy[ n.first ].comrhs( n.second, r );
281  }
282 
283  ownrhs_complete();
284 
285  // Compute mass diffusion rhs contribution required for the low order solution
286  m_dif = d->FCT()->diff( *d, m_u );
287 
288  if (d->Msum().empty())
289  comdif_complete();
290  else // send contributions of diff to chare-boundary nodes to fellow chares
291  for (const auto& n : d->Msum()) {
292  std::vector< std::vector< tk::real > > D;
293  for (auto i : n.second) D.push_back( m_dif[ tk::cref_find(d->Lid(),i) ] );
294  thisProxy[ n.first ].comdif( n.second, D );
295  }
296 
297  owndif_complete();
298 }

377 void
378 DiagCG::solve()
379 // *****************************************************************************
380 // Solve low and high order diagonal systems
381 // *****************************************************************************
382 {
383  const auto ncomp = m_rhs.nprop();
384 
385  auto d = Disc();
386 
387  // Combine own and communicated contributions to rhs and mass diffusion
388  for (const auto& b : d->Bid()) {
389  auto lid = tk::cref_find( d->Lid(), b.first );
390  const auto& brhsc = m_rhsc[ b.second ];
391  for (ncomp_t c=0; c<ncomp; ++c) m_rhs(lid,c,0) += brhsc[c];
392  const auto& bdifc = m_difc[ b.second ];
393  for (ncomp_t c=0; c<ncomp; ++c) m_dif(lid,c,0) += bdifc[c];
394  }
395 
396  // Zero communication buffers for next time step (rhs, mass diffusion rhs)
397  for (auto& b : m_rhsc) std::fill( begin(b), end(b), 0.0 );
398  for (auto& b : m_difc) std::fill( begin(b), end(b), 0.0 );
399 
400  // Set Dirichlet BCs for lhs and both low and high order rhs vectors. Note
401  // that the low order rhs (more prcisely the mass-diffusion term) is set to
402  // zero instead of the solution increment at Dirichlet BCs, because for the
403  // low order solution the right hand side is the sum of the high order right
404  // hand side and mass diffusion so the low order system is L = R + D, where L
405  // is the lumped mass matrix, R is the high order RHS, and D is
406  // mass diffusion, and R already will have the Dirichlet BC set.
407  for (const auto& n : m_bc) {
408  auto b = d->Lid().find( n.first );
409  if (b != end(d->Lid())) {
410  auto id = b->second;
411  for (ncomp_t c=0; c<ncomp; ++c)
412  if (n.second[c].first) {
413  m_lhs( id, c, 0 ) = 1.0;
414  m_rhs( id, c, 0 ) = n.second[c].second;
415  m_dif( id, c, 0 ) = 0.0;
416  }
417  }
418  }
419 
420  // Solve low and high order diagonal systems and update low order solution
421  m_dul = (m_rhs + m_dif) / m_lhs;
422  m_ul = m_u + m_dul;
423  m_du = m_rhs / m_lhs;
424 
425  // Continue with FCT
426  d->FCT()->aec( *d, m_du, m_u, m_bc );
427  d->FCT()->alw( m_u, m_ul, m_dul, thisProxy );
428 }

430 bool
431 DiagCG::correctBC( const tk::Fields& a )
432 // *****************************************************************************
433 // Verify that the change in the solution at those nodes where Dirichlet
434 // boundary conditions are set is exactly the amount the BCs prescribe
435 //! \param[in] a Limited antidiffusive element contributions
436 //! \return True if the solution is correct at Dirichlet boundary condition
437 //! nodes
438 // *****************************************************************************
439 {
440  if (m_bc.empty()) return true;
441 
442  auto d = Disc();
443 
444  // We loop through the map that associates a vector of local node IDs to side
445  // set IDs for all side sets read from mesh file. Then for each side set for
446  // all mesh nodes on a given side set we attempt to find the global node ID in
447  // m_bc, which stores only those nodes (and BC settings) at which the user has
448  // configured Dirichlet BCs to be set. Then for all scalar components of all
449  // system of systems of PDEs integrated if a BC is to be set for a given
450  // component, we compute the low order solution increment + the anti-diffusive
451  // element contributions, which is the current solution increment (to be used
452  // to update the solution at time n) at that node. This solution increment
453  // must equal the BC prescribed at the given node as we solve for solution
454  // increments. If not, the BCs are not set correctly, which is an error.
455  for (const auto& s : m_side)
456  for (auto i : s.second) {
457  auto u = m_bc.find( d->Gid()[i] );
458  if (u != end(m_bc)) {
459  const auto& b = u->second;
460  Assert( b.size() == m_u.nprop(), "Size mismatch" );
461  for (std::size_t c=0; c<b.size(); ++c)
462  if ( b[c].first &&
463  std::abs( m_dul(i,c,0) + a(i,c,0) - b[c].second ) >
464  std::numeric_limits< tk::real >::epsilon() ) {
465  return false;
466  }
467  }
468  }
469 
470  return true;
471 }

473 void
474 DiagCG::writeFields( tk::real time )
475 // *****************************************************************************
476 // Output mesh-based fields to file
477 //! \param[in] time Physical time
478 // *****************************************************************************
479 {
480  auto d = Disc();
481 
482  // Only write if the last time is different than the current one
483  if (std::abs(d->LastFieldWriteTime() - time) <
484  std::numeric_limits< tk::real >::epsilon() )
485  return;
486 
487  // Save time stamp at which the last field write happened
488  d->LastFieldWriteTime() = time;
489 
490  // Increase field output iteration count
491  ++m_itf;
492 
493  // Lambda to collect node fields output from all PDEs
494  auto nodefields = [&]() {
495  auto u = m_u; // make a copy as eq::output() may overwrite its arg
496  std::vector< std::vector< tk::real > > output;
497  for (const auto& eq : g_cgpde) {
498  auto o = eq.fieldOutput( time, m_vol, d->Coord(), d->V(), u );
499  output.insert( end(output), begin(o), end(o) );
500  }
501  return output;
502  };
503 
504  #ifdef HAS_ROOT
505  auto filetype = g_inputdeck.get< tag::selected, tag::filetype >();
506 
507  if (filetype == tk::ctr::FieldFileType::ROOT) {
508 
509  // Create Root writer
510  tk::RootMeshWriter rmw( d->OutFilename(), 1 );
511  // Write time stamp
512  rmw.writeTimeStamp( m_itf, time );
513  // Write node fields to file
514  d->writeNodeSolution( rmw, m_itf, nodefields() );
515 
516  } else
517  #endif
518  {
519 
520  // Create ExodusII writer
521  tk::ExodusIIMeshWriter ew( d->OutFilename(), tk::ExoWriter::OPEN );
522  // Write time stamp
523  ew.writeTimeStamp( m_itf, time );
524  // Write node fields to file
525  d->writeNodeSolution( ew, m_itf, nodefields() );
526 
527  }

580 // *****************************************************************************
581 // Prepare for next step
582 //! \param[in] a Limited antidiffusive element contributions
583 // *****************************************************************************
584 {

586  if (g_inputdeck.get< tag::discr, tag::fct >())

606 void
607 DiagCG::eval()
608 // *****************************************************************************
609 // Evaluate whether to continue with next step
610 // *****************************************************************************
611 {
612  auto d = Disc();
613 
614  const auto term = g_inputdeck.get< tag::discr, tag::term >();
615  const auto nstep = g_inputdeck.get< tag::discr, tag::nstep >();
616  const auto eps = std::numeric_limits< tk::real >::epsilon();
617 
618  // If neither max iterations nor max time reached, continue, otherwise finish
619  if (std::fabs(d->T()-term) > eps && d->It() < nstep)
620  dt();
621  else
622  contribute( CkCallback( CkReductionTarget(Transporter,finish), d->Tr() ) );
