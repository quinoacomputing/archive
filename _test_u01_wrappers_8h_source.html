25 // Global-scope TestU01 uniform RNG wrapper
26 //! \details This functions is used as an external uniform random number
27 //! generator (i.e., external to TestU01) that is called by the TestU01
28 //! library, hence the required signature and hence the global scope. It is
29 //! templated on a unique integer corresponding to the RNG type enum defined
30 //! by tk::ctr::RNGType. Templating on the id enables the compiler to generate
31 //! a different wrapper for a different RNG facilitating simultaneous calls to
32 //! any or all wrappers as they are unique functions.
33 //! \return Random number generated as a double-precision floating point value
34 // *****************************************************************************
35 {
36  double r = 0.0;
37  const auto rng = g_rng.find( id );
38  if (rng != end(g_rng))
39  rng->second.uniform( CkMyPe(), 1, &r );
40  else
41  Throw( "RNG not found" );

43 }
