51 CG::CG( const CProxy_Discretization& disc, const tk::CProxy_Solver& solver ) :
52  m_itf( 0 ),
53  m_nhsol( 0 ),
54  m_nlsol( 0 ),
55  m_naec( 0 ),
56  m_nalw( 0 ),
57  m_nlim( 0 ),
58  m_disc( disc ),
59  m_solver( solver ),
60  m_side(),
61  m_fluxcorrector( m_disc[thisIndex].ckLocal()->Inpoel().size() ),
62  m_u( m_disc[thisIndex].ckLocal()->Gid().size(),
63  g_inputdeck.get< tag::component >().nprop() ),
64  m_ul( m_u.nunk(), m_u.nprop() ),
65  m_du( m_u.nunk(), m_u.nprop() ),
66  m_dul( m_u.nunk(), m_u.nprop() ),
67  m_ue( m_disc[thisIndex].ckLocal()->Inpoel().size()/4, m_u.nprop() ),
68  m_p( m_u.nunk(), m_u.nprop()*2 ),
69  m_q( m_u.nunk(), m_u.nprop()*2 ),
70  m_a( m_u.nunk(), m_u.nprop() ),
71  m_lhsd( m_disc[thisIndex].ckLocal()->Psup().second.size()-1, m_u.nprop() ),
72  m_lhso( m_disc[thisIndex].ckLocal()->Psup().first.size(), m_u.nprop() ),
73  m_pc(),
74  m_qc(),
75  m_ac(),
76  m_vol( 0.0 )
77 // *****************************************************************************
78 // Constructor
79 //! \param[in] transporter Host (Transporter) proxy
80 //! \param[in] disc Discretization proxy
81 //! \param[in] solver Linear system solver (Solver) proxy
82 //! \param[in] filenodes Map associating old node IDs (as in file) to new node
83 //! IDs (as in producing contiguous-row-id linear system contributions)
84 //! \details "Contiguous-row-id" here means that the numbering of the mesh nodes
85 //! (which corresponds to rows in the linear system) are (approximately)
86 //! contiguous (as much as this can be done with an unstructured mesh) as the
87 //! problem is distirbuted across PEs, held by Solver objects. This ordering
88 //! is in start contrast with "as-in-file" ordering, which is the ordering of
89 //! the mesh nodes as it is stored in the file from which the mesh is read in.
90 //! The as-in-file ordering is highly non-contiguous across the distributed
91 //! problem.
92 // *****************************************************************************
93 {
94  auto d = m_disc[ thisIndex ].ckLocal();
95  Assert( d!=nullptr, "Discretization proxy's ckLocal() null" );
96 
97  // Allocate receive buffers for FCT
98  m_pc.resize( d->Bid().size() );
99  for (auto& b : m_pc) b.resize( m_u.nprop()*2 );
100  m_qc.resize( d->Bid().size() );
101  for (auto& b : m_qc) b.resize( m_u.nprop()*2 );
102  m_ac.resize( d->Bid().size() );
103  for (auto& b : m_ac) b.resize( m_u.nprop() );
104 
105  // Invert file-node map, a map associating old node IDs (as in file) to new
106  // node IDs (as in producing contiguous-row-id linear system contributions),
107  // so we can search more efficiently for old node IDs.
108  std::unordered_map< std::size_t, std::size_t > linnodes;
109  for (const auto& i : d->Filenodes()) {
110  auto n = tk::cref_find(d->Lid(),i.first);
111  Assert( n < d->Gid().size(),
112  "Local IDs must be lower than the local number of grid points" );
113  linnodes[ i.second ] = n;
114  }
115 
116  // Access all side sets and their old node IDs (as in file) from Solver
117  auto& oldside = m_solver.ckLocalBranch()->side();
118 
119  // Create map that assigns the local mesh node IDs mapped to side set ids,
120  // storing only those nodes for a given side set that are part of our chunk of
121  // the mesh.
122  for (const auto& s : oldside) {
123  auto& n = m_side[ s.first ];
124  for (auto o : s.second) {
125  auto it = linnodes.find( o );
126  if (it != end(linnodes))
127  n.push_back( it->second );
128  }
129  }
130 
131  // Send off global row IDs to linear system solver
132  m_solver.ckLocalBranch()->charecom( thisProxy, thisIndex, d->Gid() );
133 }

136 CG::setup( tk::real v )
137 // *****************************************************************************
138 // Setup rows, query boundary conditions, output mesh, etc.
139 //! \param[in] v Total mesh volume
140 // *****************************************************************************
141 {
142  auto d = m_disc[ thisIndex ].ckLocal();
143  Assert( d!=nullptr, "Discretization proxy's ckLocal() null" );
144 
145  m_solver.ckLocalBranch()->comfinal();
146 
147  // Store total mesh volume
148  m_vol = v;
149  // Output chare mesh to file
150  d->writeMesh();
151  // Output fields metadata to output file
152  d->writeMeta();
153 
154  // Compute left-hand side of PDEs
155  lhs();
156 
157  // zero initial solution vector
158  m_du.fill( 0.0 );
159 
160  // Set initial conditions for all PDEs
161  for (const auto& eq : g_pdes)
162  eq.initialize( d->Coord(), m_u, d->T(), d->Gid() );
163 
164  // Send off initial guess for assembly
165  m_solver.ckLocalBranch()->charesol( thisIndex, d->Gid(), m_du );
166 
167  // Output initial conditions to file (regardless of whether it was requested)
168  if ( !g_inputdeck.get< tag::cmd, tag::benchmark >() ) writeFields( d->T() );
169 
170  // Start time stepping
171  contribute( CkCallback( CkReductionTarget(Transporter,start), d->Tr()) );
172 }

174 void
175 CG::dt()
176 // *****************************************************************************
177 // Comppute time step size
178 // *****************************************************************************
179 {
180  tk::real mindt = std::numeric_limits< tk::real >::max();
181 
182  auto const_dt = g_inputdeck.get< tag::discr, tag::dt >();
183  auto def_const_dt = g_inputdeck_defaults.get< tag::discr, tag::dt >();
184  auto eps = std::numeric_limits< tk::real >::epsilon();
185 
186  auto d = m_disc[ thisIndex ].ckLocal();
187  Assert( d!=nullptr, "Discretization proxy's ckLocal() null" );
188 
189  // use constant dt if configured
190  if (std::abs(const_dt - def_const_dt) > eps) {
191 
192  mindt = const_dt;
193 
194  } else { // compute dt based on CFL
195 
196  // find the minimum dt across all PDEs integrated
197  for (const auto& eq : g_pdes) {
198  auto eqdt = eq.dt( d->Coord(), d->Inpoel(), m_u );
199  if (eqdt < mindt) mindt = eqdt;
200  }
201 
202  // Scale smallest dt with CFL coefficient
203  mindt *= g_inputdeck.get< tag::discr, tag::cfl >();
204 
205  }
206 
207  // Contribute to minimum dt across all chares the advance to next step
208  contribute( sizeof(tk::real), &mindt, CkReduction::min_double,
209  CkCallback(CkReductionTarget(CG,advance), thisProxy) );
210 }

212 void
213 CG::lhs()
214 // *****************************************************************************
215 // Compute left-hand side of transport equations
216 // *****************************************************************************
217 {
218  auto d = m_disc[ thisIndex ].ckLocal();
219  Assert( d!=nullptr, "Discretization proxy's ckLocal() null" );
220 
221  // Compute left-hand side matrix for all equations
222  for (const auto& eq : g_pdes)
223  eq.lhs( d->Coord(), d->Inpoel(), d->Psup(), m_lhsd, m_lhso );
224 
225  // Send off left hand side for assembly
226  m_solver.ckLocalBranch()->
227  charelhs( thisIndex, d->Gid(), d->Psup(), m_lhsd, m_lhso );
228 
229  // Compute lumped mass lhs required for the low order solution
230  auto lump = m_fluxcorrector.lump( d->Coord(), d->Inpoel() );
231  // Send off lumped mass lhs for assembly
232  m_solver.ckLocalBranch()->charelowlhs( thisIndex, d->Gid(), lump );
233 }

235 void
236 CG::rhs()
237 // *****************************************************************************
238 // Compute right-hand side of transport equations
239 // *****************************************************************************
240 {
241  // Initialize FCT data structures for new time step
242  m_p.fill( 0.0 );
243  m_a.fill( 0.0 );
244  for (std::size_t p=0; p<m_u.nunk(); ++p)
245  for (ncomp_t c=0; c<g_inputdeck.get<tag::component>().nprop(); ++c) {
246  m_q(p,c*2+0,0) = -std::numeric_limits< tk::real >::max();
247  m_q(p,c*2+1,0) = std::numeric_limits< tk::real >::max();
248  }
249 
250  for (auto& b : m_pc) std::fill( begin(b), end(b), 0.0 );
251  for (auto& b : m_ac) std::fill( begin(b), end(b), 0.0 );
252  for (auto& b : m_qc)
253  for (ncomp_t c=0; c<m_u.nprop(); ++c) {
254  b[c*2+0] = -std::numeric_limits< tk::real >::max();
255  b[c*2+1] = std::numeric_limits< tk::real >::max();
256  }
257 
258  auto d = m_disc[ thisIndex ].ckLocal();
259  Assert( d!=nullptr, "Discretization proxy's ckLocal() null" );
260 
261  // Compute right-hand side and query Dirichlet BCs for all equations
262  tk::Fields r( d->Gid().size(), g_inputdeck.get< tag::component >().nprop() );
263  for (const auto& eq : g_pdes)
264  eq.rhs( d->T(), d->Dt(), d->Coord(), d->Inpoel(), m_u, m_ue, r );
265  // Query Dirichlet BCs and send to linear system solver
266  bc();
267  // Send off right-hand sides for assembly
268  m_solver.ckLocalBranch()->charerhs( thisIndex, d->Gid(), r );
269 
270  // Compute mass diffusion rhs contribution required for the low order solution
271  auto diff = m_fluxcorrector.diff( d->Coord(), d->Inpoel(), m_u );
272  // Send off mass diffusion rhs contribution for assembly
273  m_solver.ckLocalBranch()->charelowrhs( thisIndex, d->Gid(), diff );
274 }

276 void
277 CG::bc()
278 // *****************************************************************************
279 // Extract node IDs from side set node lists and match to user-specified
280 // boundary conditions
281 //! \details Boundary conditions (BC), mathematically speaking, are applied on
282 //! finite surfaces. These finite surfaces are given by element sets (i.e., a
283 //! list of elements). This function queries Dirichlet boundary condition
284 //! values from all PDEs in the system of PDEs integrated at the node lists
285 //! associated to side set IDs (previously read from file). As a response to
286 //! this query, each PDE system returns a BC data structure which is then
287 //! sent to the linear system solver which needs to know about this to apply
288 //! BCs before a linear solve. Note that the BC mesh nodes that this function
289 //! results in, stored in dirbc and sent to the linear system solver, only
290 //! contains those nodes that this chare contributes to, i.e., it does not
291 //! contain those BC nodes at which other chares enforce Dirichlet BCs. The
292 //! linear system solver then collects these and communicates to other PEs so
293 //! that BC data held in Solver::m_bc are the same on all PEs. That is
294 // *****************************************************************************
295 {
296  using NodeBC = std::vector< std::pair< bool, tk::real > >;
297 
298  // Vector of pairs of bool and boundary condition value associated to mesh
299  // node IDs at which the user has set Dirichlet boundary conditions for all
300  // PDEs integrated
301  std::unordered_map< std::size_t, NodeBC > dirbc;
302 
303  // Query Dirichlet boundary conditions for all PDEs integrated and assign to
304  // nodes. This is where the individual system of PDEs are queried for boundary
305  // conditions. The outer loop goes through all sides sets that exists in the
306  // input file and passes the map's value_type (a pair of the side set id and a
307  // vector of local node IDs) to PDE::dirbc(). PDE::dirbc() returns a new map
308  // that associates a vector of pairs associated to local node IDs. (The pair
309  // is a pair of bool and real value, the former is the fact that the BC is to
310  // be set while the latter is the value if it is to be set). The length of
311  // this NodeBC vector, returning from each system of PDEs equals to the number
312  // of scalar components the given PDE integrates. Here then we contatenate
313  // this map for all PDEs integrated. If there are multiple BCs set at a mesh
314  // node (dirbc::key), either because (1) in the same PDE system the user
315  // prescribed BCs on side sets that share nodes or (2) because more than a
316  // single PDE system assigns BCs to a given node (on different variables), the
317  // NodeBC vector must be correctly stored. "Correctly" here means that the
318  // size of the NodeBC vectors must all be the same and qual to the sum of all
319  // scalar components integrated by all PDE systems integrated. Example:
320  // single-phase compressible flow (density, momentum, energy = 5) +
321  // transported scalars of 10 variables -> NodeBC vector length = 15. Note that
322  // in case (1) above a new node encountered must "overwrite" the already
323  // existing space for the NodeBC vector. "Overwrite" here means that it should
324  // keep the existing BCs and add the new ones yielding the union the two
325  // prescription for BCs but in the same space that already exist in the NodeBC
326  // vector. In case (2), however, the NodeBC pairs must go to the location in
327  // the vector assigned to the given PDE system, i.e., using the above example
328  // BCs for the 10 (or less) scalars should go in the positions starting at 5,
329  // leaving the first 5 false, indicating no BCs for the flow variables.
330  //
331  // TODO: Note that the logic described above is only partially implemented at
332  // this point. What works is the correct insertion of multiple BCs for nodes
333  // shared among multiple side sets, e.g., corners, originating from the same
334  // PDE system. What is not yet implemented is the case when there are no BCs
335  // set for flow variables but there are BCs for transport, the else branch
336  // below will incorrectly NOT skip the space for the flow variables. In other
337  // words, this only works for a single PDE system and a sytem of systems. This
338  // machinery is only tested with a single system of PDEs at this point.
339 
340  auto d = m_disc[ thisIndex ].ckLocal();
341  Assert( d!=nullptr, "Discretization proxy's ckLocal() null" );
342 
343  for (const auto& s : m_side) {
344  std::size_t c = 0;
345  for (std::size_t eq=0; eq<g_pdes.size(); ++eq) {
346  auto eqbc = g_pdes[eq].dirbc( d->T(), d->Dt(), s, d->Coord() );
347  for (const auto& n : eqbc) {
348  auto id = n.first; // BC node ID
349  const auto& bcs = n.second; // BCs
350  auto& nodebc = dirbc[ d->Gid()[id] ]; // BCs to be set for node
351  if (nodebc.size() > c) { // node already has BCs from this PDE
352  Assert( nodebc.size() == c+bcs.size(), "Size mismatch" );
353  for (std::size_t i=0; i<bcs.size(); i++)
354  if (bcs[i].first)
355  nodebc[c+i] = bcs[i];
356  } else { // node does not yet have BCs from this PDE
357  // This branch needs to be completed for system of systems of PDEs.
358  // See note above.
359  nodebc.insert( end(nodebc), begin(bcs), end(bcs) );
360  }
361  }
362  if (!eqbc.empty()) c += eqbc.cbegin()->second.size();
363  }
364  }
365 
366  // Verify the size of each NodeBC vectors. They must have the same lengths and
367  // equal to the total number of scalar components for all systems of PDEs
368  // integrated. This is intentional, because this way the linear system solver
369  // does not have to (and does not) know about individual equation systems.
370  for (const auto& n : dirbc) {
371  IGNORE(n);
372  Assert( n.second.size() == m_u.nprop(), "Size of NodeBC vector incorrect" );
373  }
374 
375  // Send off list of owned node IDs mapped to side sets to Solver
376  m_solver.ckLocalBranch()->charebc( dirbc );
377 }

379 void
380 CG::aec()
381 // *****************************************************************************
382 // Compute and sum antidiffusive element contributions (AEC) to mesh nodes
383 //! \details This function computes and starts communicating m_p, which stores
384 //! the sum of all positive (negative) antidiffusive element contributions to
385 //! nodes (Lohner: P^{+,-}_i), see also FluxCorrector::aec().
386 // *****************************************************************************
387 {
388  auto d = m_disc[ thisIndex ].ckLocal();
389  Assert( d!=nullptr, "Discretization proxy's ckLocal() null" );
390 
391  // Compute and sum antidiffusive element contributions to mesh nodes. Note
392  // that the sums are complete on nodes that are not shared with other chares
393  // and only partial sums on chare-boundary nodes.
394  auto& dbc = m_solver.ckLocalBranch()->dirbc();
395  m_fluxcorrector.aec( d->Coord(), d->Inpoel(), d->Vol(), dbc, d->Gid(), m_du,
396  m_u, m_p );
397 
398  if (d->Msum().empty())
399  comaec_complete();
400  else // send contributions to chare-boundary nodes to fellow chares
401  for (const auto& n : d->Msum()) {
402  std::vector< std::vector< tk::real > > p;
403  for (auto i : n.second) p.push_back( m_p[ tk::cref_find(d->Lid(),i) ] );
404  thisProxy[ n.first ].comaec( n.second, p );
405  }
406 
407  ownaec_complete();
408  #ifndef NDEBUG
409  ownaec_complete();
410  #endif
411 }

413 void
414 CG::comaec( const std::vector< std::size_t >& gid,
415  const std::vector< std::vector< tk::real > >& P )
416 // *****************************************************************************
417 // Receive sums of antidiffusive element contributions on chare-boundaries
418 //! \param[in] gid Global mesh node IDs at which we receive AEC contributions
419 //! \param[in] P Partial sums of positive (negative) antidiffusive element
420 //! contributions to chare-boundary nodes
421 //! \details This function receives contributions to m_p, which stores the
422 //! sum of all positive (negative) antidiffusive element contributions to
423 //! nodes (Lohner: P^{+,-}_i), see also FluxCorrector::aec(). While m_p stores
424 //! own contributions, m_pc collects the neighbor chare contributions during
425 //! communication. This way work on m_p and m_pc is overlapped. The two are
426 //! combined in lim().
427 // *****************************************************************************
428 {
429  Assert( P.size() == gid.size(), "Size mismatch" );
430 
431  using tk::operator+=;
432 
433  auto d = m_disc[ thisIndex ].ckLocal();
434  Assert( d!=nullptr, "Discretization proxy's ckLocal() null" );
435 
436  for (std::size_t i=0; i<gid.size(); ++i) {
437  auto bid = tk::cref_find( d->Bid(), gid[i] );
438  Assert( bid < m_pc.size(), "Indexing out of bounds" );
439  m_pc[ bid ] += P[i];
440  }
441 
442  if (++m_naec == d->Msum().size()) {
443  m_naec = 0;
444  comaec_complete();
445  }

481 void
482 CG::comalw( const std::vector< std::size_t >& gid,
483  const std::vector< std::vector< tk::real > >& Q )
484 // *****************************************************************************
485 // Receive contributions to the maxima and minima of unknowns of all elements
486 // surrounding mesh nodes on chare-boundaries
487 //! \param[in] gid Global mesh node IDs at which we receive contributions
488 //! \param[in] Q Partial contributions to maximum and minimum unknowns of all
489 //! elements surrounding nodes to chare-boundary nodes
490 //! \details This function receives contributions to m_q, which stores the
491 //! maximum and mimimum unknowns of all elements surrounding each node
492 //! (Lohner: u^{max,min}_i), see also FluxCorrector::alw(). While m_q stores
493 //! own contributions, m_qc collects the neighbor chare contributions during
494 //! communication. This way work on m_q and m_qc is overlapped. The two are
495 //! combined in lim().
496 // *****************************************************************************
497 {
498  Assert( Q.size() == gid.size(), "Size mismatch" );
499 
500  auto d = m_disc[ thisIndex ].ckLocal();
501  Assert( d!=nullptr, "Discretization proxy's ckLocal() null" );
502 
503  for (std::size_t i=0; i<gid.size(); ++i) {
504  auto bid = tk::cref_find( d->Bid(), gid[i] );
505  Assert( bid < m_qc.size(), "Indexing out of bounds" );
506  auto& o = m_qc[ bid ];
507  const auto& q = Q[i];
508  for (ncomp_t c=0; c<m_u.nprop(); ++c) {
509  if (q[c*2+0] > o[c*2+0]) o[c*2+0] = q[c*2+0];
510  if (q[c*2+1] < o[c*2+1]) o[c*2+1] = q[c*2+1];
511  }
512  }
513 
514  if (++m_nalw == d->Msum().size()) {
515  m_nalw = 0;
516  comalw_complete();
517  }
518 }

520 void
521 CG::lim()
522 // *****************************************************************************
523 // Compute the limited antidiffusive element contributions
524 //! \details This function computes and starts communicating m_a, which stores
525 //! the limited antidiffusive element contributions assembled to nodes
526 //! (Lohner: AEC^c), see also FluxCorrector::limit().
527 // *****************************************************************************
528 {
529  auto d = m_disc[ thisIndex ].ckLocal();
530  Assert( d!=nullptr, "Discretization proxy's ckLocal() null" );
531 
532  // Combine own and communicated contributions to P and Q
533  for (const auto& b : d->Bid()) {
534  auto lid = tk::cref_find( d->Lid(), b.first );
535  const auto& bpc = m_pc[ b.second ];
536  const auto& bqc = m_qc[ b.second ];
537  for (ncomp_t c=0; c<m_p.nprop()/2; ++c) {
538  m_p(lid,c*2+0,0) += bpc[c*2+0];
539  m_p(lid,c*2+1,0) += bpc[c*2+1];

677 void
678 CG::diagnostics()
679 // *****************************************************************************
680 // Compute diagnostics, e.g., residuals
681 // *****************************************************************************
682 {
683  auto d = m_disc[ thisIndex ].ckLocal();
684  Assert( d!=nullptr, "Discretization proxy's ckLocal() null" );
685 
686  // Optionally collect analytical solutions and send both the latest analytical
687  // and numerical solutions to Solver for computing and outputing diagnostics
688  if ( !(d->It() % g_inputdeck.get< tag::interval, tag::diag >()) ) {
689 
690  // Collect analytical solutions (if available) from all PDEs. Note that
691  // calling the polymorphic PDE::initialize() is assumed to evaluate the
692  // analytical solution for a PDE. For those PDE problems that have
693  // analytical solutions, this is the same as used for setting the initial
694  // conditions, since if the analytical solution is available for a problem
695  // it is (so far anyway) always initialized to its analytical solution and
696  // that is done by calling PDE::initialize(). If the analytical solution is
697  // a function of time, that is already incorporated in setting initial
698  // conditions. For those PDEs where the analytical solution is not
699  // available, initialize() returns the initial conditions (obviously), and
700  // thus the "error", defined between this "analytical" and numerical
701  // solution will be a measure of the "distance" between the initial
702  // condition and the current numerical solution. This is not necessarily
703  // useful, but simplies the logic because all PDEs can be treated as baing
704  // able to compute an error based on some "analytical" solution, which
705  // really the initial condition.
706  auto a = m_u;
707  for (const auto& eq : g_pdes)
708  eq.initialize( d->Coord(), a, d->T()+d->Dt(), d->Gid() );
709 
710  // Prepare for computing diagnostics. Diagnostics are defined as the L2 norm
711  // of a quantity, computed in mesh nodes, A, as ||A||_2 = sqrt[ sum_i(A_i)^2
712  // V_i ], where the sum is taken over all mesh nodes and V_i is the nodal
713  // volume. We send two sets of quantities to the host for aggregation across
714  // the whole mesh: (1) the numerical solutions of all components of all
715  // PDEs, and their error, defined as A_i = (a_i - n_i), where a_i and n_i
716  // are the analytical and numerical solutions at node i, respectively. We
717  // send these to Solver and the final aggregated solution will end up in
718  // Transporter::diagnostics(). Note that we weigh/multiply all data here by
719  // sqrt(V_i), so that the nodal volumes do not have to be communicated
720  // separately. In Solver::diagnostics(), where we collect all contributions
721  // from chares on a PE, all data is squared. Solver::diagnostics() is where
722  // the sums are computed, then the sums are summed accross the whole problem
723  // in Transporter::diagnostics(), where the final square-root of the L2
724  // norm, defined above, is taken.
725 
726  // Send both numerical and analytical solutions to solver
727  m_solver.ckLocalBranch()->charediag( thisIndex, d->Gid(), m_u, a, d->V() );
728  }
729 }

731 bool
732 CG::correctBC()
733 // *****************************************************************************
734 // Verify that the change in the solution at those nodes where Dirichlet
735 // boundary conditions are set is exactly the amount the BCs prescribe
736 //! \return True if the solution is correct at Dirichlet boundary condition
737 //! nodes
738 // *****************************************************************************
739 {
740  auto& dirbc = m_solver.ckLocalBranch()->dirbc();
741 
742  if (dirbc.empty()) return true;
743 
744  auto d = m_disc[ thisIndex ].ckLocal();
745  Assert( d!=nullptr, "Discretization proxy's ckLocal() null" );
746 
747  // We loop through the map that associates a vector of local node IDs to side
748  // set IDs for all side sets read from mesh file. Then for each side set for
749  // all mesh nodes on a given side set we attempt to find the global node ID
750  // in dirbc, which stores only those nodes (and BC settings) at which the
751  // user has configured Dirichlet BCs to be set. Then for all scalar
752  // components of all system of systems of PDEs integrated if a BC is to be
753  // set for a given component, we compute the low order solution increment +
754  // the anti-diffusive element contributions, which is the current solution
755  // increment (to be used to update the solution at time n) at that node. This
756  // solution increment must equal the BC prescribed at the given node as we
757  // solve for solution increments. If not, the BCs are not set correctly,
758  // which is an error.
759  for (const auto& s : m_side)
760  for (auto i : s.second) {
761  auto u = dirbc.find( d->Gid()[i] );
762  if (u != end(dirbc)) {
763  const auto& b = u->second;
764  Assert( b.size() == m_u.nprop(), "Size mismatch" );
765  for (std::size_t c=0; c<b.size(); ++c)
766  if ( b[c].first &&
767  std::abs( m_dul(i,c,0) + m_a(i,c,0) - b[c].second ) >
768  std::numeric_limits< tk::real >::epsilon() ) {
769  return false;
770  }
771  }
772  }
773 
774  return true;
775 }
