108 void
109 DistFCT::next()
110 // *****************************************************************************
111 // Prepare for next time step stage
112 // *****************************************************************************
113 {
114  #ifndef NDEBUG
115  wait4ver();
116  #endif
117  wait4fct();
118  wait4app();
119 
120  // Initialize FCT data structures for new time step
121  m_p.fill( 0.0 );
122  m_a.fill( 0.0 );
123  for (std::size_t p=0; p<m_a.nunk(); ++p)
124  for (ncomp_t c=0; c<g_inputdeck.get<tag::component>().nprop(); ++c) {
125  m_q(p,c*2+0,0) = -std::numeric_limits< tk::real >::max();
126  m_q(p,c*2+1,0) = std::numeric_limits< tk::real >::max();
127  }
128 
129  for (auto& b : m_pc) std::fill( begin(b), end(b), 0.0 );
130  for (auto& b : m_ac) std::fill( begin(b), end(b), 0.0 );
131  for (auto& b : m_qc)
132  for (ncomp_t c=0; c<m_a.nprop(); ++c) {
133  b[c*2+0] = -std::numeric_limits< tk::real >::max();
134  b[c*2+1] = std::numeric_limits< tk::real >::max();
135  }
136 }

138 void
139 DistFCT::aec( const Discretization& d,
140  const tk::Fields& dUh,
141  const tk::Fields& Un,
142  const std::unordered_map< std::size_t,
143  std::vector< std::pair< bool, tk::real > > >& bc )
144 // *****************************************************************************
145 // Compute and sum antidiffusive element contributions (AEC) to mesh nodes
146 //! \param[in] d Discretization proxy to read mesh data from
147 //! \param[in] dUh Increment of the high order solution
148 //! \param[in] Un Solution at the previous time step
149 //! \param[in] bc Vector of pairs of bool and boundary condition value
150 //! associated to mesh node IDs at which to set Dirichlet boundary conditions.
151 //! Note that this BC data structure must include boundary conditions set
152 //! across all PEs, not just the ones need to be set on this PE.
153 //! \details This function computes and starts communicating m_p, which stores
154 //! the sum of all positive (negative) antidiffusive element contributions to
155 //! nodes (Lohner: P^{+,-}_i), see also FluxCorrector::aec().
156 // *****************************************************************************
157 {
158  // Store a copy of the high order solution increment for later
159  m_du = dUh;
160 
161  // Compute and sum antidiffusive element contributions to mesh nodes. Note
162  // that the sums are complete on nodes that are not shared with other chares
163  // and only partial sums on chare-boundary nodes.
164  m_fluxcorrector.aec(d.Coord(), m_inpoel, d.Vol(), bc, d.Gid(), dUh, Un, m_p);
165 
166  if (d.Msum().empty())
167  comaec_complete();
168  else // send contributions to chare-boundary nodes to fellow chares
169  for (const auto& n : d.Msum()) {
170  std::vector< std::vector< tk::real > > p;
171  for (auto i : n.second) p.push_back( m_p[ tk::cref_find(m_lid,i) ] );
172  thisProxy[ n.first ].comaec( n.second, p );
173  }
174 
175  ownaec_complete();
176  #ifndef NDEBUG
177  ownaec_complete();
178  #endif
179 }

181 void
182 DistFCT::comaec( const std::vector< std::size_t >& gid,
183  const std::vector< std::vector< tk::real > >& P )
184 // *****************************************************************************
185 // Receive sums of antidiffusive element contributions on chare-boundaries
186 //! \param[in] gid Global mesh node IDs at which we receive AEC contributions
187 //! \param[in] P Partial sums of positive (negative) antidiffusive element
188 //! contributions to chare-boundary nodes
189 //! \details This function receives contributions to m_p, which stores the
190 //! sum of all positive (negative) antidiffusive element contributions to
191 //! nodes (Lohner: P^{+,-}_i), see also FluxCorrector::aec(). While m_p stores
192 //! own contributions, m_pc collects the neighbor chare contributions during
193 //! communication. This way work on m_p and m_pc is overlapped. The two are
194 //! combined in lim().
195 // *****************************************************************************
196 {
197  Assert( P.size() == gid.size(), "Size mismatch" );
198 
199  using tk::operator+=;
200 
201  for (std::size_t i=0; i<gid.size(); ++i) {
202  auto bid = tk::cref_find( m_bid, gid[i] );
203  Assert( bid < m_pc.size(), "Indexing out of bounds" );
204  m_pc[ bid ] += P[i];
205  }
206 
207  if (++m_naec == m_msum.size()) {
208  m_naec = 0;
209  comaec_complete();
210  }
211 }

213 void
214 DistFCT::alw( const tk::Fields& Un,
215  const tk::Fields& Ul,
216  const tk::Fields& dUl,
217  const SchemeProxy& scheme )
218 // *****************************************************************************
219 // Compute the maximum and minimum unknowns of elements surrounding nodes
220 //! \param[in] Un Solution at the previous time step
221 //! \param[in] Ul Low order solution
222 //! \param[in] dUl Low order solution increment
223 //! \param[in] scheme Discretization scheme Charm++ proxy we interoperate with
224 //! \details This function computes and starts communicating m_q, which stores
225 //! the maximum and mimimum unknowns of all elements surrounding each node
226 //! (Lohner: u^{max,min}_i), see also FluxCorrector::alw().
227 // *****************************************************************************
228 {
229  // Store a copy of the low order solution vector and its increment for later
230  m_ul = Ul;
231  m_dul = dUl;
232 
233  // Store discretization scheme proxy
234  m_scheme = scheme;
235 
236  // Compute the maximum and minimum unknowns of all elements surrounding nodes
237  // Note that the maximum and minimum unknowns are complete on nodes that are
238  // not shared with other chares and only partially complete on chare-boundary
239  // nodes.
240  m_fluxcorrector.alw( m_inpoel, Un, Ul, m_q );
241 
242  if (m_msum.empty())
243  comalw_complete();
244  else // send contributions at chare-boundary nodes to fellow chares
245  for (const auto& n : m_msum) {
246  std::vector< std::vector< tk::real > > q;
247  for (auto i : n.second) q.push_back( m_q[ tk::cref_find(m_lid,i) ] );
248  thisProxy[ n.first ].comalw( n.second, q );
249  }
250 
251  ownalw_complete();
252  #ifndef NDEBUG
253  ownalw_complete();
254  #endif
255 }

257 void
258 DistFCT::comalw( const std::vector< std::size_t >& gid,
259  const std::vector< std::vector< tk::real > >& Q )
260 // *****************************************************************************
261 // Receive contributions to the maxima and minima of unknowns of all elements
262 // surrounding mesh nodes on chare-boundaries
263 //! \param[in] gid Global mesh node IDs at which we receive contributions
264 //! \param[in] Q Partial contributions to maximum and minimum unknowns of all
265 //! elements surrounding nodes to chare-boundary nodes
266 //! \details This function receives contributions to m_q, which stores the
267 //! maximum and mimimum unknowns of all elements surrounding each node
268 //! (Lohner: u^{max,min}_i), see also FluxCorrector::alw(). While m_q stores
269 //! own contributions, m_qc collects the neighbor chare contributions during
270 //! communication. This way work on m_q and m_qc is overlapped. The two are
271 //! combined in lim().
272 // *****************************************************************************
273 {
274  Assert( Q.size() == gid.size(), "Size mismatch" );
275 
276  for (std::size_t i=0; i<gid.size(); ++i) {
277  auto bid = tk::cref_find( m_bid, gid[i] );
278  Assert( bid < m_qc.size(), "Indexing out of bounds" );
279  auto& o = m_qc[ bid ];
280  const auto& q = Q[i];
281  for (ncomp_t c=0; c<m_a.nprop(); ++c) {
282  if (q[c*2+0] > o[c*2+0]) o[c*2+0] = q[c*2+0];
283  if (q[c*2+1] < o[c*2+1]) o[c*2+1] = q[c*2+1];
284  }
285  }
286 
287  if (++m_nalw == m_msum.size()) {
288  m_nalw = 0;
289  comalw_complete();
290  }
291 }

293 void
294 DistFCT::lim()
295 // *****************************************************************************
296 // Compute the limited antidiffusive element contributions
297 //! \details This function computes and starts communicating m_a, which stores
298 //! the limited antidiffusive element contributions assembled to nodes
299 //! (Lohner: AEC^c), see also FluxCorrector::limit().
300 // *****************************************************************************
301 {
302  // Combine own and communicated contributions to P and Q
303  for (const auto& b : m_bid) {
304  auto lid = tk::cref_find( m_lid, b.first );
305  const auto& bpc = m_pc[ b.second ];
306  const auto& bqc = m_qc[ b.second ];
307  for (ncomp_t c=0; c<m_p.nprop()/2; ++c) {
308  m_p(lid,c*2+0,0) += bpc[c*2+0];
309  m_p(lid,c*2+1,0) += bpc[c*2+1];
310  if (bqc[c*2+0] > m_q(lid,c*2+0,0)) m_q(lid,c*2+0,0) = bqc[c*2+0];
311  if (bqc[c*2+1] < m_q(lid,c*2+1,0)) m_q(lid,c*2+1,0) = bqc[c*2+1];
312  }
313  }
314 
315  m_fluxcorrector.lim( m_inpoel, m_p, m_ul, m_q, m_a );
316 
317  if (m_msum.empty())
318  comlim_complete();
319  else // send contributions to chare-boundary nodes to fellow chares
320  for (const auto& n : m_msum) {
321  std::vector< std::vector< tk::real > > a;
322  for (auto i : n.second) a.push_back( m_a[ tk::cref_find(m_lid,i) ] );
323  thisProxy[ n.first ].comlim( n.second, a );
324  }
325 
326  ownlim_complete();
327 }

371 bool
372 DistFCT::verifyBC()
373 // *****************************************************************************
374 // Verify that the change in the solution at those nodes where Dirichlet
375 // boundary conditions are set is exactly the amount the BCs prescribe
376 //! \return True if the solution is correct at Dirichlet boundary condition
377 //! nodes
378 // *****************************************************************************
379 {
380  // Call Scheme.correctBC(). The code below is equivalent to the function call
381  // m_scheme[ thisIndex ].ckLocal()->correctBC( m_a ). The call is done via
382  // a variant to facilitate calling back to chare arrays of different types,
383  // e.g., MatCG or DiagCG. See also DistFCT::SchemeProxy.
384  auto e = tk::element< ProxyElem >( m_scheme, thisIndex );
385  return boost::apply_visitor( correctBC(m_a), e );
386 }

388 void
389 DistFCT::apply()
390 // *****************************************************************************
391 // Apply limited antidiffusive element contributions
392 // *****************************************************************************
393 {
394  // Combine own and communicated contributions to A
395  for (const auto& b : m_bid) {
396  auto lid = tk::cref_find( m_lid, b.first );
397  const auto& bac = m_ac[ b.second ];
398  for (ncomp_t c=0; c<m_a.nprop(); ++c) m_a(lid,c,0) += bac[c];
399  }
400 
401  // Verify that solution values do not change at Dirichlet BC nodes
402  Assert( verifyBC(), "Dirichlet boundary condition incorrect" );
403 
404  // Prepare for next time step. The code below is equivalent to the function
405  // call m_scheme[ thisIndex ].ckLocal()->next( m_a ). The call is done via a
406  // variant to facilitate calling back to chare arrays of different types,
407  // e.g., MatCG or DiagCG. See also DistFCT::SchemeProxy.
408  auto e = tk::element< ProxyElem >( m_scheme, thisIndex );
409  boost::apply_visitor( Next(m_a), e );
410 }
