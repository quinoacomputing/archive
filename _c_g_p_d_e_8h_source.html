53  //! \brief Constructor taking a function pointer to a constructor of an
54  //! object modeling Concept.
55  //! \details Passing std::function allows late execution of the constructor,
56  //! i.e., as late as inside this class' constructor, and thus usage from
57  //! a factory. Note that there are at least two different ways of using
58  //! this constructor:
59  //! - Bind T's constructor arguments and place it in std::function<T()>
60  //! and passing no arguments as args.... This case then instantiates the
61  //! model via its constructor and stores it in here.
62  //! - Bind a single placeholder argument to T's constructor and pass it in
63  //! as host's args..., which then forwards it to model's constructor. This
64  //! allows late binding, i.e., binding the argument only here.
65  //! \see See also the wrapper tk::recordModel() which does the former and
66  //! tk::recordModelLate() which does the latter, both defined in
67  //! src/Base/Factory.h.
68  //! \param[in] x Function pointer to a constructor of an object modeling
69  //! Concept.
70  //! \param[in] args Zero or more constructor arguments
71  template< typename T, typename...Args >
72  explicit CGPDE( std::function<T(Args...)> x, Args&&... args ) :
73  self( tk::make_unique< Model<T> >(
74  std::move( x( std::forward<Args>(args)... ) ) ) ) {}
