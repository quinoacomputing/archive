27 //! Random123-based random number generator used polymorphically with tk::RNG
28 template< class CBRNG >
29 class Random123 {
30 
31  private:
32  static const std::size_t CBRNG_DATA_SIZE = 3;
33  using ncomp_t = kw::ncomp::info::expect::type;
34  using ctr_type = typename CBRNG::ctr_type;
35  using key_type = typename CBRNG::key_type;
36  using value_type = typename CBRNG::ctr_type::value_type;
37  using arg_type = std::vector< std::array< value_type, CBRNG_DATA_SIZE > >;
38 
39  //! Adaptor to use a std distribution with the Random123 generator
40  //! \see C++ concepts: UniformRandomNumberGenerator
41  struct Adaptor {
42  using result_type = unsigned long;
43  Adaptor( CBRNG& r, arg_type& d, int t ) : rng(r), data(d), tid(t) {}
44  static constexpr result_type min() { return 0u; }
45  static constexpr result_type max() {
46  return std::numeric_limits< result_type >::max();
47  }
48  result_type operator()()
49  {
50  auto& d = data[ static_cast< std::size_t >( tid ) ];
51  d[2] = static_cast< result_type >( tid );
52  ctr_type ctr = {{ d[0], d[1] }}; // assemble counter
53  key_type key = {{ d[2] }}; // assemble key
54  auto res = rng( ctr, key ); // generate
55  ctr.incr();
56  d[0] = ctr[0];
57  d[1] = ctr[1];
58  return res[0];
59  }
60  CBRNG& rng;
61  arg_type& data;
62  int tid;
63  };
64 
65  public:
66  //! Constructor
67  //! \param[in] n Initialize RNG using this many independent streams
68  //! \param[in] seed RNG seed
69  explicit Random123( uint64_t n = 1, uint64_t seed = 0 ) {
70  Assert( n > 0, "Need at least one thread" );
71  m_data.resize( n, {{ 0, seed << 32, 0 }} );
72  }

74  //! Uniform RNG: Generate uniform random numbers
75  //! \param[in] tid Thread (or more precisely) stream ID
76  //! \param[in] num Number of RNGs to generate
77  //! \param[in,out] r Pointer to memory to write the random numbers to
78  void uniform( int tid, ncomp_t num, double* r ) const {
79  auto& d = m_data[ static_cast< std::size_t >( tid ) ];
80  for (ncomp_t i=0; i<num; ++i) {
81  d[2] = static_cast< unsigned long >( tid );
82  ctr_type ctr = {{ d[0], d[1] }}; // assemble counter
83  key_type key = {{ d[2] }}; // assemble key
84  auto res = m_rng( ctr, key ); // generate
85  r[i] = r123::u01fixedpt< double, value_type >( res[0] );
86  ctr.incr();
87  d[0] = ctr[0];
88  d[1] = ctr[1];
89  }
90  }
91 
92  //! Gaussian RNG: Generate Gaussian random numbers
93  //! \param[in] tid Thread (or more precisely stream) ID
94  //! \param[in] num Number of RNGs to generate
95  //! \param[in,out] r Pointer to memory to write the random numbers to
96  //! \details Generating Gaussian random numbers is implemented via an
97  //! adaptor, modeling std::UniformRandomNumberGenerator, outsourcing the
98  //! transformation of uniform random numbers to Gaussian ones, to the
99  //! standard library. The adaptor is instantiated here because a standard
100  //! distribution, such as e.g., std::normal_distribution, generates
101  //! numbers using operator() with no arguments, thus the RNG state and the
102  //! thread ID (this latter only known here) must be stored in the adaptor
103  //! functor's state. Even though creating the adaptor seems like a
104  //! potentially costly operation for every call, using the standard
105  //! library implementation is still faster than a hand-coded
106  //! implementation of the Box-Muller algorithm. Note that libc++ uses a
107  //! cache, as Box-Muller, implemented using the polar algorithm generates
108  //! 2 Gaussian numbers for each pair of uniform ones, caching every 2nd.
109  void gaussian( int tid, ncomp_t num, double* r ) const {
110  Adaptor generator( m_rng, m_data, tid );
111  std::normal_distribution<> gauss_dist( 0.0, 1.0 );
112  for (ncomp_t i=0; i<num; ++i) r[i] = gauss_dist( generator );
113  }

115  //! Beta RNG: Generate beta random numbers
116  //! \param[in] tid Thread (or more precisely stream) ID
117  //! \param[in] num Number of RNGs to generate
118  //! \param[in] p First beta shape parameter
119  //! \param[in] q Second beta shape parameter
120  //! \param[in] a Beta displacement parameter
121  //! \param[in] b Beta scale factor
122  //! \param[in,out] r Pointer to memory to write the random numbers to
123  //! \details Generating beta-distributed random numbers is implemented via
124  //! an adaptor, modeling boost::UniformRandomNumberGenerator, outsourcing
125  //! the transformation of uniform random numbers to beta-distributed ones,
126  //! to boost::random. The adaptor is instantiated here because a boost
127  //! random number distribution, such as e.g.,
128  //! boost::random::beta_distribution, generates numbers using operator()
129  //! with no arguments, thus the RNG state and the thread ID (this latter
130  //! only known here) must be stored in the adaptor functor's state.
131  void beta( int tid, ncomp_t num, double p, double q, double a, double b,
132  double* r ) const {
133  Adaptor generator( m_rng, m_data, tid );
134  boost::random::beta_distribution<> beta_dist( p, q );
135  for (ncomp_t i=0; i<num; ++i) r[i] = beta_dist( generator ) * b + a;
136  }
137 
138  //! Accessor to the number of threads we operate on
139  uint64_t nthreads() const noexcept { return m_data.size(); }

141  private:

143  mutable arg_type m_data; //!< RNG arguments
144 };
