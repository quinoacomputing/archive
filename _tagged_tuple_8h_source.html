87 //! Tagged tuple. Client-side interface. Tagged tuple allowing tag-based access.
88 //! This is very much like
89 //! [std::tuple](http://en.cppreference.com/w/cpp/utility/tuple), but instead of
90 //! having to index the elements by integers, it allows access by a tag, which
91 //! can be an empty struct with a unique name. Credit goes to
92 //! ecatmur_at_stackoverflow.com, for more details, see
93 //! http://stackoverflow.com/questions/13065166/c11-tagged-tuple. For tags, see
94 //! Control/Tags.h. Tagged tuples are extensively used for transferring data
95 //! from the parser to an internal data structure in a type-save manner, which
96 //! is a tagged tuple containing a hierarchy of various containers. As an
97 //! example on how tagged tuples are used for parsing an input file, see
98 //! Control/Walker/InputDeck/InputDeck.h. Another way to use a tagged tuple is a
99 //! compile-time associated container between tags and an arbitrary type. As an
100 //! example, see rngtest::TestU01Stack::runner.
101 template<typename... Ts> struct tagged_tuple :
102  tt_impl<extract<2, 0, Ts...>, extract<2, 1, Ts...>> {
103  //! Constructor
104  template<typename... Args> tagged_tuple(Args &&...args) :
105  tt_impl<extract<2, 0, Ts...>, extract<2, 1, Ts...>>(
106  std::forward<Args>(args)...) {}
107  //! Pack/Unpack

109  tt_impl<extract<2, 0, Ts...>, extract<2, 1, Ts...>>::pup(p); }
110  friend void operator|( PUP::er& p, tagged_tuple<Ts...>& t ) { t.pup(p); }
