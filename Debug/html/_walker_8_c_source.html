100 //! Pack/Unpack selected RNGs. This Pack/Unpack method (re-)creates the full RNG
101 //! stack since it needs to (re-)bind function pointers on different processing
102 //! elements. Therefore we circumvent Charm's usual pack/unpack for this type,
103 //! and thus sizing does not make sense: sizing is a no-op. We could initialize
104 //! the stack in RNGTestDriver's constructor and let this function re-create the
105 //! stack only when unpacking, but that leads to repeating the same code twice:
106 //! once in RNGTestDriver's constructor, once here. Another option is to use
107 //! this pack/unpack routine to both initially create (when packing) and to
108 //! re-create (when unpacking) the stack, which eliminates the need for
109 //! pre-creating the object in RNGTestDriver's constructor and therefore
110 //! eliminates the repeated code. This explains the guard for sizing: the code
111 //! below is called for packing only (in serial) and packing and unpacking (in
112 //! parallel).
113 inline
114 void operator|( PUP::er& p, std::map< tk::ctr::RawRNGType, tk::RNG >& rng ) {
115  if (!p.isSizing()) {
116  tk::RNGStack stack(
117  #ifdef HAS_MKL
118  g_inputdeck.get< tag::param, tag::rngmkl >(),
119  #endif
120  #ifdef HAS_RNGSSE2
121  g_inputdeck.get< tag::param, tag::rngsse >(),
122  #endif
123  g_inputdeck.get< tag::param, tag::rng123 >() );
124  rng = stack.selected( g_inputdeck.get< tag::selected, tag::rng >() );
125  }
126 }

148 //! \brief Charm++ main chare for the random walker executable, walker.
149 //! \details Note that this object should not be in a namespace.
150 class Main : public CBase_Main {
151 
152  public:
153  //! \brief Constructor
154  //! \details The main chare constructor is the main entry point of the
155  //! program, called by the Charm++ runtime system. The constructor does
156  //! basic initialization steps, e.g., parser the command-line, prints out
157  //! some useful information to screen (in verbose mode), and instantiates
158  //! a driver. Since Charm++ is fully asynchronous, the constructor
159  //! usually spawns asynchronous objects and immediately exits. Thus in the
160  //! body of the main chare constructor we fire up an 'execute' chare,
161  //! which then calls back to Main::execute(). Finishing the main chare
162  //! constructor the Charm++ runtime system then starts the
163  //! network-migration of all global-scope data (if any). The execute chare
164  //! calling back to Main::execute() signals the end of the migration of
165  //! the global-scope data. Then we are ready to execute the driver. Since
166  //! the random walker is parallel and asynchronous, its driver fires up
167  //! additional Charm++ chare objects which then call back to
168  //! Main::finalize() at some point in the future when all work has been
169  //! finished. finalize() then exits by calling Charm++'s CkExit(),
170  //! shutting down the runtime system.
171  //! \see http://charm.cs.illinois.edu/manuals/html/charm++/manual.html
172  Main( CkArgMsg* msg )
173  try :
174  m_cmdline(),
175  // Parse command line into m_cmdline using default simple pretty printer
176  m_cmdParser( msg->argc, msg->argv, tk::Print(), m_cmdline ),
177  // Create pretty printer initializing output streams based on command line
178  m_print( m_cmdline.get< tag::verbose >() ? std::cout : std::clog ),
179  // Create Walker driver
180  m_driver( tk::Main< walker::WalkerDriver >
181  ( msg->argc, msg->argv,
182  m_cmdline,
183  tk::HeaderType::WALKER,
184  WALKER_EXECUTABLE,
185  m_print ) ),
186  m_timer(1), // start new timer measuring the total runtime
187  m_timestamp()
188  {
189  delete msg;
190  mainProxy = thisProxy;
191  // Fire up an asynchronous execute object, which when created at some
192  // future point in time will call back to this->execute(). This is
193  // necessary so that this->execute() can access already migrated
194  // global-scope data.
195  CProxy_execute::ckNew();
196  // Start new timer measuring the migration of global-scope data
197  m_timer.emplace_back();
198  } catch (...) { tk::processExceptionCharm(); }
199 
200  //! Execute driver created and initialized by constructor
201  void execute() {
202  try {
203  m_timestamp.emplace_back("Migrate global-scope data", m_timer[1].hms());
204  m_driver.execute(); // fires up async chares
205  } catch (...) { tk::processExceptionCharm(); }
206  }
207 
208  //! Normal exit point
209  void finalize() {
210  try {
211  if (!m_timer.empty()) {
212  m_timestamp.emplace_back( "Total runtime", m_timer[0].hms() );
213  m_print.time( "Timers (h:m:s)", m_timestamp );
214  m_print.endpart();
215  }
216  } catch (...) { tk::processExceptionCharm(); }
217  // Tell the Charm++ runtime system to exit
218  CkExit();
219  }
220 
221  //! Add time stamp contributing to final timers output
222  void timestamp( std::string label, tk::real stamp ) {
223  try{
224  m_timestamp.emplace_back( label, tk::hms( stamp ) );
225  } catch (...) { tk::processExceptionCharm(); }
226  }
