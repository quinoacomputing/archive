32 Collector::chareOrd( const std::vector< tk::real >& ord,
33  const std::vector< tk::UniPDF >& updf,
34  const std::vector< tk::BiPDF >& bpdf,
35  const std::vector< tk::TriPDF >& tpdf )
36 // *****************************************************************************
37 // Chares contribute ordinary moments and ordinary PDFs
38 //! \param[in] ord Vector of partial sums for the estimation of ordinary moments
39 //! \param[in] updf Vector of partial sums for the estimation of univariate
40 //! ordinary PDFs
41 //! \param[in] bpdf Vector of partial sums for the estimation of bivariate
42 //! ordinary PDFs
43 //! \param[in] tpdf Vector of partial sums for the estimation of trivariate
44 //! ordinary PDFs
45 //! \note This function does not have to be declared as a Charm++ entry
46 //! method since it is always called by chares on the same PE.
47 // *****************************************************************************
48 {
49  ++m_nord;
50 
51  for (std::size_t i=0; i<m_ordinary.size(); ++i) m_ordinary[i] += ord[i];
52 
53  // Add contribution from worker chares to partial sums on my PE
54  std::size_t i = 0;
55  for (const auto& p : updf) m_ordupdf[i++].addPDF( p );
56  i = 0;
57  for (const auto& p : bpdf) m_ordbpdf[i++].addPDF( p );
58  i = 0;
59  for (const auto& p : tpdf) m_ordtpdf[i++].addPDF( p );
60 
61  // If all chares on my PE have contributed, send partial sums to host
62  if (m_nord == m_nchare) {
63 
64  // Create Charm++ callback function for reduction
65  CkCallback c1( CkReductionTarget( Distributor, estimateOrd ), m_hostproxy );
66 
67  // Contribute partial sums to host via Charm++ reduction
68  contribute( static_cast< int >( m_ordinary.size() * sizeof(tk::real) ),
69  m_ordinary.data(), CkReduction::sum_double, c1 );
70 
71  // Zero counters for next collection operation
72  std::fill( begin(m_ordinary), end(m_ordinary), 0.0 );
73 
74  // Serialize vector of PDFs to raw stream
75  auto stream = tk::serialize( m_ordupdf, m_ordbpdf, m_ordtpdf );
76 
77  // Create Charm++ callback function for reduction.
78  // Distributor::estimateOrdPDF() will be the final target of the reduction
79  // where the results of the reduction will appear.
80  CkCallback c2( CkIndex_Distributor::estimateOrdPDF(nullptr), m_hostproxy );
81 
82  // Contribute serialized PDFs of partial sums to host via Charm++ reduction
83  contribute( stream.first, stream.second.get(), PDFMerger, c2 );
84 
85  // Zero counters for next collection operation
86  for (auto& p : m_ordupdf) p.zero();
87  for (auto& p : m_ordbpdf) p.zero();
88  for (auto& p : m_ordtpdf) p.zero();
89 
90  m_nord = 0;
91  }
92 }
