125  //! \brief Constructor: set defaults
126  //! \param[in] cl Previously parsed and store command line
127  //! \details Anything not set here is initialized by the compiler using the
128  //! default constructor for the corresponding type.
129  //! \author J. Bakosi
130  InputDeck( const CmdLine& cl = {} ) {
131  // Set previously parsed command line
132  set< tag::cmd >( cl );
133  // Default discretization parameters
134  set< tag::discr, tag::nstep >
135  ( std::numeric_limits< kw::nstep::info::expect::type >::max() );
136  set< tag::discr, tag::term >
137  ( std::numeric_limits< kw::term::info::expect::type >::max() );
138  set< tag::discr, tag::t0 >( 0.0 );
139  set< tag::discr, tag::dt >( 0.0 );
140  set< tag::discr, tag::cfl >( 0.0 );
141  set< tag::discr, tag::ctau >( 1.0 );
142  // Default AMR settings
143  set< tag::selected, tag::initialamr >( tk::ctr::InitialAMRType::NONE );
144  // Default txt floating-point output precision in digits
145  set< tag::prec, tag::diag >( std::cout.precision() );
146  // Default intervals
147  set< tag::interval, tag::tty >( 1 );
148  set< tag::interval, tag::field >( 1 );
149  set< tag::interval, tag::diag >( 1 );
150  // Initialize help: fill own keywords
151  const auto& ctrinfoFill = tk::ctr::Info( get< tag::cmd, tag::ctrinfo >() );
152  boost::mpl::for_each< keywords1 >( ctrinfoFill );
153  boost::mpl::for_each< keywords2 >( ctrinfoFill );
154  boost::mpl::for_each< keywords3 >( ctrinfoFill );
155  boost::mpl::for_each< keywords4 >( ctrinfoFill );
156  }
