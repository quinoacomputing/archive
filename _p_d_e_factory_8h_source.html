35 //! \brief Function object for registering a partial differential equation
36 //! into the partial differential equation factory
37 //! \details This functor is repeatedly called by MPL's cartesian_product,
38 //! sweeping all combinations of the partial differential equation
39 //! policies. The purpose of template template is to simplify client code
40 //! as that will not have to specify the template arguments of the
41 //! template argument (the policies of Eq), since we can figure it out
42 //! here. See also http://stackoverflow.com/a/214900. The template
43 //! argument Eq specifies a "child" class that is used polymorphically
44 //! with a "base" class modeling a concept defined in the base. The base
45 //! is given by the template argument PDE. The template argument Factory
46 //! specifies which factory to store the registered and configured child
47 //1 PDE.
48 template< template< class, class > class Eq, class Factory, class PDE >
49 struct registerPDE {
50  //! Need to store the reference to factory we are registering into
51  Factory& factory;
52  //! Need to store which differential equation we are registering
53  const ctr::PDEType type;
54  //! Constructor, also count number of unique equation types registered
55  //! \param[in,out] f Factory into which to register PDE
56  //! \param[in] t Enum selecting PDE type, Control/Inciter/Options/PDE.h
57  //! \param[in] eqTypes Equation type counters
58  explicit registerPDE( Factory& f,
59  ctr::PDEType t,
60  std::set< ctr::PDEType >& eqTypes ) :
61  factory( f ), type( t ) { eqTypes.insert( t ); }
62  //! \brief Function call operator called with tk::cartesian_product for
63  //! each unique sequence of policy combinations
64  template< typename U > void operator()( brigand::type_<U> ) {
65  // Get problem policy: first type of brigand::list U
66  using Physics = typename brigand::front< U >;
67  // Get problem policy: last type of brigand::list U
68  using Problem = typename brigand::back< U >;
69  // Build differential equation key
70  ctr::PDEKey key{ type, Physics::type(), Problem::type() };
71  // Register equation (with policies given by brigand::list U) into factory
72  tk::recordModelLate< PDE, Eq< Physics, Problem > >
73  ( factory, key, static_cast<tk::ctr::ncomp_type>(0) );
74  }
75 };
