7  joint probability density function (PDF) of two scalar variables from an

38  // Hash functor for key_type
39  struct key_hash {
40  std::size_t operator()( const key_type& key ) const {
41  return std::hash< long >()( key[0] ) ^ std::hash< long >()( key[1] );
42  }
43  };

45  //! \brief Joint bivariate PDF
46  //! \details The underlying container type is an unordered_map where the key
47  //! is two bin ids corresponding to the two sample space dimensions, and
48  //! the mapped value is the sample counter. The hasher functor, defined by
49  //! key_hash provides an XORed hash of the two bin ids.
50  using map_type = std::unordered_map< key_type, tk::real, key_hash >;
51 
52  //! Empty constructor for Charm++
53  explicit BiPDF() : m_binsize( {{ 0, 0 }} ), m_nsample( 0 ), m_pdf() {}
54 
55  //! Constructor: Initialize joint bivariate PDF container
56  //! \param[in] bs Sample space bin size in both directions
57  explicit BiPDF( const std::vector< tk::real >& bs ) :
58  m_binsize( {{ bs[0], bs[1] }} ), m_nsample( 0 ), m_pdf() {}
59 
60  //! Accessor to number of samples
61  //! \return Number of samples collected
62  std::size_t nsample() const noexcept { return m_nsample; }
63 
64  //! Add sample to bivariate PDF
65  //! \param[in] sample Sample to add
66  void add( std::array< tk::real, dim > sample ) {
67  ++m_nsample;
68  ++m_pdf[ {{ std::lround( sample[0] / m_binsize[0] ),
69  std::lround( sample[1] / m_binsize[1] ) }} ];
70  }
71 
72  //! Add multiple samples from a PDF
73  //! \param[in] p PDF whose samples to add
74  void addPDF( const BiPDF& p ) {
75  m_binsize = p.binsize();
76  m_nsample += p.nsample();
77  for (const auto& e : p.map()) m_pdf[ e.first ] += e.second;
78  }
79

92  //! Return minimum and maximum bin ids of sample space in both dimensions
93  //! \return {xmin,xmax,ymin,ymax} Minima and maxima of the bin ids in a
94  //! std::array
95  std::array< long, 2*dim > extents() const {
96  auto x = std::minmax_element( begin(m_pdf), end(m_pdf),
97  []( const pair_type& a, const pair_type& b )
98  { return a.first[0] < b.first[0]; } );
99  auto y = std::minmax_element( begin(m_pdf), end(m_pdf),
100  []( const pair_type& a, const pair_type& b )
101  { return a.first[1] < b.first[1]; } );
102  return {{ x.first->first[0], x.second->first[0],
103  y.first->first[1], y.second->first[1] }};
104  }
