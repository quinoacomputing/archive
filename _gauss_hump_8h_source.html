26 //! Transport PDE problem: advection of two-dimensional Gaussian hump
27 class TransportProblemGaussHump {
28 
29  public:
30  //! Evaluate analytical solution at (x,y,t) for all components
31  //! \param[in] ncomp Number of components in this transport equation system
32  //! \param[in] x X coordinate where to evaluate the solution
33  //! \param[in] y Y coordinate where to evaluate the solution
34  //! \param[in] t Time where to evaluate the solution
35  //! \return Values of all components evaluated at (x,y,t)
36  static std::vector< tk::real >
37  solution( ncomp_t, ncomp_t ncomp,
38  tk::real x, tk::real y, tk::real, tk::real t )
39  {
40  const auto vel = prescribedVelocity( x, y, 0.0, ncomp, ncomp );
41 
42  std::vector< tk::real > s( ncomp, 0.0 );
43  for (ncomp_t c=0; c<ncomp; ++c)
44  {
45  // center of the hump
46  auto x0 = 0.25 + vel[c][0]*t;
47  auto y0 = 0.25 + vel[c][1]*t;
48 
49  // hump
50  s[c] = 1.0 * exp( -((x-x0)*(x-x0)
51  + (y-y0)*(y-y0))/(2.0 * 0.005) );
52  }
53  return s;
54  }
55 
56  //! \brief Evaluate the increment from t to t+dt of the analytical solution
57  //! at (x,y,z) for all components
58  //! \param[in] ncomp Number of components in this transport equation system
59  //! \param[in] x X coordinate where to evaluate the solution
60  //! \param[in] y Y coordinate where to evaluate the solution
61  //! \param[in] t Time where to evaluate the solution increment starting from
62  //! \param[in] dt Time increment at which evaluate the solution increment to
63  //! \return Increment in values of all components evaluated at (x,y,t+dt)
64  static std::vector< tk::real >
65  solinc( ncomp_t, ncomp_t ncomp, tk::real x, tk::real y, tk::real,
66  tk::real t, tk::real dt )
67  {
68  auto st1 = solution( 0, ncomp, x, y, 0.0, t );
69  auto st2 = solution( 0, ncomp, x, y, 0.0, t+dt );
70  std::transform( begin(st1), end(st1), begin(st2), begin(st2),
71  []( tk::real s, tk::real& d ){ return d -= s; } );
72  return st2;
73  }
74

76  static void errchk( ncomp_t, ncomp_t ) {}

88  //! Assign prescribed velocity at a point
89  //! \param[in] ncomp Number of components in this transport equation
90  //! \return Velocity assigned to all vertices of a tetrehedron, size:
91  //! ncomp * ndim = [ncomp][3]
92  static std::vector< std::array< tk::real, 3 > >
93  prescribedVelocity( tk::real,

95  tk::real,

103  }
