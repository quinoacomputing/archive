81 void
82 RootMeshWriter::writeNodes( const UnsMesh& mesh )
83 // *****************************************************************************
84 // Write node coordinates to Root file
85 //! \param[in] mesh Unstructured mesh object
86 // *****************************************************************************
87 {
88  // the file requires the vertices and the number of triangles
89  // 4 triangles per tetrahedron and mesh.tetinpoel() stores 4
90  // vertices per tet in the vector (# vertices = # of triangles)
91 
92  m_tree_connect->Branch( "coord", &m_csobject->coordinates, "coordinates/I" );
93  m_tree_connect->Branch( "trian", &m_csobject->triangles, "triangles/I" );
94 
95  m_tree_connect->Branch( "x_coord", &m_csobject->mx_root );
96  m_tree_connect->Branch( "y_coord", &m_csobject->my_root );
97  m_tree_connect->Branch( "z_coord", &m_csobject->mz_root );
98 
99  auto c = static_cast< std::size_t >( m_csobject->coordinates );
100  for (std::size_t i=0 ; i<c; ++i) {
101  m_csobject->mx_root.push_back( mesh.x()[i] );
102  m_csobject->my_root.push_back( mesh.y()[i] );
103  m_csobject->mz_root.push_back( mesh.z()[i] );
104  }
105 }

121 RootMeshWriter::writeElemBlock( int& elclass,
122  const std::vector< std::size_t >& inpoel )
123 const
124 // *****************************************************************************
125 // Write element block to ROOT file
126 //! \param[inout] elclass Count element class ids in file
127 //! \param[in] inpoel Element connectivity.
128 // *****************************************************************************
129 {
130  if (inpoel.empty()) return;
131 
132  // increase number of element classes in file
133  ++elclass;
134 
135  // Make sure element connectivity starts with zero
136  Assert( *std::minmax_element( begin(inpoel), end(inpoel) ).first == 0,
137  "node ids should start from zero" );
138 
139  // create a branch for storing the tetrahedrons
140  m_tree_connect->Branch( "tetconnect", &m_csobject->tet_connect );
141  for ( auto itr = inpoel.begin(); itr != inpoel.end(); itr++ )
142  m_csobject->tet_connect.push_back( *itr );
143 
144  m_tree_connect->Fill();
145  m_tree_connect->Write();
146 
147 }
