163 void
164 FileConvWriter::writeData()
165 //*****************************************************************************
166 // Write the Timestamp and Variables from Input to Output
167 //*****************************************************************************
168 {
169 
170  std::size_t timestep = 1;
171  while( true ) {
172  double dt = 0;
173 
174  std::string time_branch = "time_branch_" + std::to_string(timestep);
175  if( m_tree_local->GetBranch( time_branch.c_str() ) == nullptr )
176  break;
177 
178  for (std::size_t var_id=1; var_id<=m_nodal_size; ++var_id) {
179  std::string branch_var = "branch_" + std::to_string(timestep) + "_field_"
180  + std::to_string(var_id);
181 
182  if( m_tree_local->GetBranch( branch_var.c_str() ) == nullptr )
183  break;
184  else {
185 
186  std::vector< double> *var_fields = nullptr;
187 
188  m_tree_local->SetBranchAddress( branch_var.c_str(), &var_fields );
189  m_tree_local->SetBranchAddress(time_branch.c_str(), &dt );
190  m_tree_local->GetEntry(0);
191 
192  m_emw->writeNodeScalar( timestep,
193  static_cast<int>(var_id),
194  *var_fields );
195  }
196 
197  } // End the variables loop.
198 
199  // Write the timestamp variable, once per timestep
200  m_emw->writeTimeStamp( timestep, dt );
201  ++timestep;
202 
203  } // break after all the variables for all the timesteps are written
204  m_tree_local->ResetBranchAddresses();
205 
206 }
