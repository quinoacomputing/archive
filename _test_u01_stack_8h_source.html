48  //! \brief Add a statistical test to battery
49  //! \details Note that adding a test to the battery does not invoke the test
50  //! constructor, it only records the information on how to call the test
51  //! constructor in the future. That is it binds the constructor arguments
52  //! to the constructor call and records the the information so only a
53  //! function call "()" is necessary to instantiate it.
54  //! \param[in] proxy Charm++ host proxy to which the test calls back to
55  //! \param[in] tests Vector of test constructors to add tests to
56  //! \param[in] r RNG ID enum
57  //! \param[in] gen Raw function pointer to TestU01 statistical test
58  //! \param[in] names Vector of statisical test names (can be more than one
59  //! associated with a given test, since a test can contain more than one
60  //! statistical test evaluation, yielding multiple p-values)
61  //! \param[in] xargs Extra arguments to test-run
62  template< class TestType, class Proxy, typename... Ts >
63  void add( Proxy& proxy,
64  std::vector< std::function< StatTest() > >& tests,
65  tk::ctr::RNGType r,
66  unif01_Gen* const gen,
67  std::vector< std::string >&& names,
68  Ts&&... xargs ) const
69  {
70  using Model = TestType;
71  using Host = StatTest;
72  using Props = typename TestType::Props;
73  tests.emplace_back(
74  std::bind( boost::value_factory< Host >(),
75  std::function< Model() >(),
76  std::forward< Props >(
77  Props( proxy, r, std::move(names), gen,
78  std::forward<Ts>(xargs)... ) ) ) );
79  }
