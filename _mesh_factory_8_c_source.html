38 // Read unstructured mesh from file
39 //! \param[in] print Pretty printer
40 //! \param[in] filename Filename to read mesh from
41 //! \param[out] timestamp A time stamp consisting of a timer label (a string),
42 //! and a time state (a tk::real in seconds) measuring the mesh read time
43 //! \return Unstructured mesh object
44 // *****************************************************************************
45 {
46  print.diagstart( "Reading mesh from file ..." );
47 
48  // Read in mesh
49  tk::Timer t;
50 
51  //! Create unstructured mesh to store mesh
52  UnsMesh mesh;
53 
54  const auto meshtype = detectInput( filename );
55 
56  if (meshtype == MeshReaderType::GMSH)
57  GmshMeshReader( filename ).readMesh( mesh );
58  else if (meshtype == MeshReaderType::NETGEN)
59  NetgenMeshReader( filename ).readMesh( mesh );
60  else if (meshtype == MeshReaderType::EXODUSII)
61  ExodusIIMeshReader( filename ).readMesh( mesh );
62  else if (meshtype == MeshReaderType::ASC)
63  ASCMeshReader( filename ).readMesh( mesh );
64  else if (meshtype == MeshReaderType::HYPER)
65  HyperMeshReader( filename ).readMesh( mesh );
66 
67  timestamp = std::make_pair( "Read mesh from file", t.dsec() );
68 
69  print.diagend( "done" );
70 
71  // Return (move out) mesh object
72  return mesh;
73 }

76 writeUnsMesh( const tk::Print& print,
77  const std::string& filename,
78  UnsMesh& mesh,
79  bool reorder )
80 // *****************************************************************************
81 // Write unstructured mesh to file
82 //! \param[in] print Pretty printer
83 //! \param[in] filename Filename to write mesh to
84 //! \param[in] mesh Unstructured mesh object to write from
85 //! \param[in] reorder Whether to also reorder mesh nodes
86 //! \return Vector of time stamps consisting of a timer label (a string), and a
87 //! time state (a tk::real in seconds) measuring the renumber and the mesh
88 //! write time
89 // *****************************************************************************
90 {
91  std::vector< std::pair< std::string, tk::real > > times;
92 
93  tk::Timer t;
94 
95  // If mesh has tetrahedra but no triangles, generate triangle connectivity
96  if (!mesh.tetinpoel().empty() && mesh.triinpoel().empty()) {
97  print.diagstart( "Generating missing surface mesh ..." );
98 
99  const auto& inpoel = mesh.tetinpoel(); // get tet connectivity
100  auto esup = tk::genEsup( inpoel, 4 ); // elements surrounding points
101  auto esuel = tk::genEsuelTet( inpoel, esup ); // elems surrounding elements
102  auto& triinpoel = mesh.triinpoel();
103  // collect boundary faces
104  for (std::size_t e=0; e<esuel.size()/4; ++e) {
105  auto mark = e*4;
106  for (std::size_t f=0; f<4; ++f) {
107  if (esuel[mark+f] == -1) {
108  // extract triangle element connectivity from tetrahedron
109  triinpoel.push_back( inpoel[ mark+tk::lpofa[f][0] ] );
110  triinpoel.push_back( inpoel[ mark+tk::lpofa[f][1] ] );
111  triinpoel.push_back( inpoel[ mark+tk::lpofa[f][2] ] );
112  }
113  }
114  }
115 
116  print.diagend( "done" );
117  times.emplace_back( "Generate surface mesh", t.dsec() );
118  t.zero();
119  }
120 
121  if (reorder) {
122  print.diagstart( "Reordering mesh nodes ..." );
123 
124  // If mesh has tetrahedra elements, reorder based on those
125  if (!mesh.tetinpoel().empty()) {
126 
127  auto& inpoel = mesh.tetinpoel();
128  const auto psup = tk::genPsup( inpoel, 4, tk::genEsup( inpoel, 4 ) );
129  auto map = tk::renumber( psup );
130  tk::remap( inpoel, map );
131  tk::remap( mesh.triinpoel(), map );
132  tk::remap( mesh.x(), map );
133  tk::remap( mesh.y(), map );
134  tk::remap( mesh.z(), map );
135 
136  // If mesh has no tetrahedra elements, reorder based on triangle mesh if any
137  } else if (!mesh.triinpoel().empty()) {
138 
139  auto& inpoel = mesh.triinpoel();
140  const auto psup = tk::genPsup( inpoel, 3, tk::genEsup( inpoel, 3 ) );
141  auto map = tk::renumber( psup );
142  tk::remap( inpoel, map );
143  tk::remap( mesh.x(), map );
144  tk::remap( mesh.y(), map );
145  tk::remap( mesh.z(), map );
146  }
147 
148  print.diagend( "done" );
149  times.emplace_back( "Reorder mesh", t.dsec() );
150  t.zero();
151  }
152 
153  print.diagstart( "Writing mesh to file ..." );
154 
155  const auto meshtype = pickOutput( filename );
156 
157  if (meshtype == MeshWriterType::GMSH)
158  GmshMeshWriter( filename ).writeMesh( mesh );
159  else if (meshtype == MeshWriterType::NETGEN)
160  NetgenMeshWriter( filename ).writeMesh( mesh );
161  else if (meshtype== MeshWriterType::EXODUSII)
162  ExodusIIMeshWriter( filename, ExoWriter::CREATE ).writeMesh( mesh );
163 
164  print.diagend( "done" );
165  times.emplace_back( "Write mesh to file", t.dsec() );
166 
167  return times;
168 }
