75  //! Compute the left hand side sparse matrix
76  //! \param[in] coord Mesh node coordinates
77  //! \param[in] inpoel Mesh element connectivity
78  //! \param[in] psup Linked lists storing IDs of points surrounding points
79  //! \param[in,out] lhsd Diagonal of the sparse matrix storing nonzeros
80  //! \param[in,out] lhso Off-diagonal of the sparse matrix storing nonzeros
81  //! \details Sparse matrix storing the nonzero matrix values at rows and
82  //! columns given by psup. The format is similar to compressed row
83  //! storage, but the diagonal and off-diagonal data are stored in separate
84  //! vectors. For the off-diagonal data the local row and column indices,
85  //! at which values are nonzero, are stored by psup (psup1 and psup2,
86  //! where psup2 holds the indices at which psup1 holds the point ids
87  //! surrounding points, see also tk::genPsup()). Note that the number of
88  //! mesh points (our chunk) npoin = psup.second.size()-1.
89  void lhs( const std::array< std::vector< tk::real >, 3 >& coord,
90  const std::vector< std::size_t >& inpoel,
91  const std::pair< std::vector< std::size_t >,
92  std::vector< std::size_t > >& psup,
93  tk::Fields& lhsd,
94  tk::Fields& lhso ) const
95  {
96  Assert( psup.second.size()-1 == coord[0].size(),
97  "Number of mesh points and number of global IDs unequal" );
98  Assert( lhsd.nunk() == psup.second.size()-1, "Number of unknowns in "
99  "diagonal sparse matrix storage incorrect" );
100  Assert( lhso.nunk() == psup.first.size(), "Number of unknowns in "
101  "off-diagonal sparse matrix storage incorrect" );
102 
103  // Lambda to compute the sparse matrix vector index for row and column
104  // indices. Used only for off-diagonal entries.
105  auto spidx = [ &psup ]( std::size_t r, std::size_t c ) -> std::size_t {
106  Assert( r != c, "Only for computing the off-diagonal indices" );
107  for (auto i=psup.second[r]+1; i<=psup.second[r+1]; ++i)
108  if (c == psup.first[i]) return i;
109  Throw( "Cannot find row, column: " + std::to_string(r) + ',' +
110  std::to_string(c) + " in sparse matrix" );
111  };
112 
113  const auto& x = coord[0];
114  const auto& y = coord[1];
115  const auto& z = coord[2];
116 
117  // Zero matrix for all components
118  for (ncomp_t c=0; c<m_ncomp; ++c) {
119  lhsd.fill( c, m_offset, 0.0 );
120  lhso.fill( c, m_offset, 0.0 );
121  }
122 
123  for (std::size_t e=0; e<inpoel.size()/4; ++e) {
124  const auto A = inpoel[e*4+0];
125  const auto B = inpoel[e*4+1];
126  const auto C = inpoel[e*4+2];
127  const auto D = inpoel[e*4+3];
128  std::array< tk::real, 3 > ba{{ x[B]-x[A], y[B]-y[A], z[B]-z[A] }},
129  ca{{ x[C]-x[A], y[C]-y[A], z[C]-z[A] }},
130  da{{ x[D]-x[A], y[D]-y[A], z[D]-z[A] }};
131  const auto J = tk::triple( ba, ca, da ) / 120.0;
132  Assert( J > 0, "Element Jacobian non-positive" );
133 
134  for (ncomp_t c=0; c<m_ncomp; ++c) {
135  const auto r = lhsd.cptr( c, m_offset );
136  lhsd.var( r, A ) += 2.0 * J;
137  lhsd.var( r, B ) += 2.0 * J;
138  lhsd.var( r, C ) += 2.0 * J;
139  lhsd.var( r, D ) += 2.0 * J;
140 
141  const auto s = lhso.cptr( c, m_offset );
142  lhso.var( s, spidx(A,B) ) += J;
143  lhso.var( s, spidx(A,C) ) += J;
144  lhso.var( s, spidx(A,D) ) += J;
145 
146  lhso.var( s, spidx(B,A) ) += J;
147  lhso.var( s, spidx(B,C) ) += J;
148  lhso.var( s, spidx(B,D) ) += J;
149 
150  lhso.var( s, spidx(C,A) ) += J;
151  lhso.var( s, spidx(C,B) ) += J;
152  lhso.var( s, spidx(C,D) ) += J;
153 
154  lhso.var( s, spidx(D,A) ) += J;
155  lhso.var( s, spidx(D,B) ) += J;
156  lhso.var( s, spidx(D,C) ) += J;
157  }
158  }
159  }

161  //! Compute right hand side
162  //! \param[in] deltat Size of time step
163  //! \param[in] coord Mesh node coordinates
164  //! \param[in] inpoel Mesh element connectivity
165  //! \param[in] U Solution vector at recent time step
166  //! \param[in,out] R Right-hand side vector computed
167  void rhs( tk::real,
168  tk::real deltat,
169  const std::array< std::vector< tk::real >, 3 >& coord,
170  const std::vector< std::size_t >& inpoel,
171  const tk::Fields& U,
172  tk::Fields& Ue,
173  tk::Fields& R ) const
174  {
175  using tag::transport;
176  Assert( U.nunk() == coord[0].size(), "Number of unknowns in solution "
177  "vector at recent time step incorrect" );
178  Assert( R.nunk() == coord[0].size() && R.nprop() == m_ncomp,
179  "Number of unknowns and/or number of components in right-hand "
180  "side vector incorrect" );
181  Assert( U.nprop() == m_ncomp, "Number of components in solution vector "
182  "must equal " + std::to_string(m_ncomp) );
183  Assert( R.nprop() == m_ncomp, "Number of components in rhs must equal " +
184  std::to_string(m_ncomp) );
185 
186  const auto& x = coord[0];
187  const auto& y = coord[1];
188  const auto& z = coord[2];
189 
190  // 1st stage: update element values from node values (gather-add)
191  for (std::size_t e=0; e<inpoel.size()/4; ++e) {
192 
193  // access node IDs
194  const std::array< std::size_t, 4 > N{{ inpoel[e*4+0], inpoel[e*4+1],
195  inpoel[e*4+2], inpoel[e*4+3] }};
196  // compute element Jacobi determinant
197  const std::array< tk::real, 3 >
198  ba{{ x[N[1]]-x[N[0]], y[N[1]]-y[N[0]], z[N[1]]-z[N[0]] }},
199  ca{{ x[N[2]]-x[N[0]], y[N[2]]-y[N[0]], z[N[2]]-z[N[0]] }},
200  da{{ x[N[3]]-x[N[0]], y[N[3]]-y[N[0]], z[N[3]]-z[N[0]] }};
201  const auto J = tk::triple( ba, ca, da ); // J = 6V
202  Assert( J > 0, "Element Jacobian non-positive" );
203 
204  // shape function derivatives, nnode*ndim [4][3]
205  std::array< std::array< tk::real, 3 >, 4 > grad;
206  grad[1] = tk::crossdiv( ca, da, J );
207  grad[2] = tk::crossdiv( da, ba, J );
208  grad[3] = tk::crossdiv( ba, ca, J );
209  for (std::size_t i=0; i<3; ++i)
210  grad[0][i] = -grad[1][i]-grad[2][i]-grad[3][i];
211 
212  // access solution at element nodes
213  std::vector< std::array< tk::real, 4 > > u( m_ncomp );
214  for (ncomp_t c=0; c<m_ncomp; ++c) u[c] = U.extract( c, m_offset, N );
215  // access solution at elements
216  std::vector< const tk::real* > ue( m_ncomp );
217  for (ncomp_t c=0; c<m_ncomp; ++c) ue[c] = Ue.cptr( c, m_offset );
218 
219  // sum nodal averages to element
220  for (ncomp_t c=0; c<m_ncomp; ++c) {
221  Ue.var(ue[c],e) = 0.0;
222  for (std::size_t a=0; a<4; ++a)
223  Ue.var(ue[c],e) += u[c][a]/4.0;
224  }
225 
226  // get prescribed velocity
227  const std::array< std::vector<std::array<tk::real,3>>, 4 > vel{{
228  Problem::prescribedVelocity(x[N[0]], y[N[0]], z[N[0]], m_c, m_ncomp),
229  Problem::prescribedVelocity(x[N[0]], y[N[1]], z[N[1]], m_c, m_ncomp),
230  Problem::prescribedVelocity(x[N[0]], y[N[2]], z[N[2]], m_c, m_ncomp),
231  Problem::prescribedVelocity(x[N[0]], y[N[3]], z[N[3]], m_c, m_ncomp)}};
232 
233  // sum flux (advection) contributions to element
234  tk::real d = deltat/2.0;
235  for (std::size_t c=0; c<m_ncomp; ++c)
236  for (std::size_t j=0; j<3; ++j)
237  for (std::size_t a=0; a<4; ++a)
238  Ue.var(ue[c],e) -= d * grad[a][j] * vel[a][c][j]*u[c][a];
239 
240  }
241 
242 
243  // zero right hand side for all components
244  for (ncomp_t c=0; c<m_ncomp; ++c) R.fill( c, m_offset, 0.0 );
245 
246  // 2nd stage: form rhs from element values (scatter-add)
247  for (std::size_t e=0; e<inpoel.size()/4; ++e) {
248 
249  // access node IDs
250  const std::array< std::size_t, 4 > N{{ inpoel[e*4+0], inpoel[e*4+1],
251  inpoel[e*4+2], inpoel[e*4+3] }};
252  // compute element Jacobi determinant
253  const std::array< tk::real, 3 >
254  ba{{ x[N[1]]-x[N[0]], y[N[1]]-y[N[0]], z[N[1]]-z[N[0]] }},
255  ca{{ x[N[2]]-x[N[0]], y[N[2]]-y[N[0]], z[N[2]]-z[N[0]] }},
256  da{{ x[N[3]]-x[N[0]], y[N[3]]-y[N[0]], z[N[3]]-z[N[0]] }};
257  const auto J = tk::triple( ba, ca, da ); // J = 6V
258  Assert( J > 0, "Element Jacobian non-positive" );
259 
260  // shape function derivatives, nnode*ndim [4][3]
261  std::array< std::array< tk::real, 3 >, 4 > grad;
262  grad[1] = tk::crossdiv( ca, da, J );
263  grad[2] = tk::crossdiv( da, ba, J );
264  grad[3] = tk::crossdiv( ba, ca, J );
265  for (std::size_t i=0; i<3; ++i)
266  grad[0][i] = -grad[1][i]-grad[2][i]-grad[3][i];
267 
268  // access solution at elements
269  std::vector< tk::real > ue( m_ncomp );
270  for (ncomp_t c=0; c<m_ncomp; ++c) ue[c] = Ue( e, c, m_offset );
271  // access pointer to right hand side at component and offset
272  std::vector< const tk::real* > r( m_ncomp );
273  for (ncomp_t c=0; c<m_ncomp; ++c) r[c] = R.cptr( c, m_offset );
274  // access solution at nodes of element
275  std::vector< std::array< tk::real, 4 > > u( m_ncomp );
276  for (ncomp_t c=0; c<m_ncomp; ++c) u[c] = U.extract( c, m_offset, N );
277 
278  // get prescribed velocity
279  auto xc = (x[N[0]] + x[N[1]] + x[N[2]] + x[N[3]]) / 4.0;
280  auto yc = (y[N[0]] + y[N[1]] + y[N[2]] + y[N[3]]) / 4.0;
281  auto zc = (z[N[0]] + z[N[1]] + z[N[2]] + z[N[3]]) / 4.0;
282  const auto vel =
283  Problem::prescribedVelocity( xc, yc, zc, m_c, m_ncomp );
284 
285  // scatter-add flux contributions to rhs at nodes
286  tk::real d = deltat * J/6.0;
287  for (std::size_t c=0; c<m_ncomp; ++c)
288  for (std::size_t j=0; j<3; ++j)
289  for (std::size_t a=0; a<4; ++a)
290  R.var(r[c],N[a]) += d * grad[a][j] * vel[c][j]*ue[c];
291 
292  // add (optional) diffusion contribution to right hand side
293  Physics::diffusionRhs( m_c, m_ncomp, deltat, J, grad, N, u, r, R );
294 
295  }
296  }

298  //! Compute the minimum time step size
299  //! \param[in] U Solution vector at recent time step
300  //! \param[in] coord Mesh node coordinates
301  //! \param[in] inpoel Mesh element connectivity
302  //! \return Minimum time step size
303  tk::real dt( const std::array< std::vector< tk::real >, 3 >& coord,
304  const std::vector< std::size_t >& inpoel,
305  const tk::Fields& U ) const
306  {
307  using tag::transport;
308  Assert( U.nunk() == coord[0].size(), "Number of unknowns in solution "
309  "vector at recent time step incorrect" );
310  const auto& x = coord[0];
311  const auto& y = coord[1];
312  const auto& z = coord[2];
313  // compute the minimum dt across all elements we own
314  tk::real mindt = std::numeric_limits< tk::real >::max();
315  for (std::size_t e=0; e<inpoel.size()/4; ++e) {
316  const std::array< std::size_t, 4 > N{{ inpoel[e*4+0], inpoel[e*4+1],
317  inpoel[e*4+2], inpoel[e*4+3] }};
318  // compute cubic root of element volume as the characteristic length
319  const std::array< tk::real, 3 >
320  ba{{ x[N[1]]-x[N[0]], y[N[1]]-y[N[0]], z[N[1]]-z[N[0]] }},
321  ca{{ x[N[2]]-x[N[0]], y[N[2]]-y[N[0]], z[N[2]]-z[N[0]] }},
322  da{{ x[N[3]]-x[N[0]], y[N[3]]-y[N[0]], z[N[3]]-z[N[0]] }};
323  const auto L = std::cbrt( tk::triple( ba, ca, da ) / 6.0 );
324  // access solution at element nodes at recent time step
325  std::vector< std::array< tk::real, 4 > > u( m_ncomp );
326  for (ncomp_t c=0; c<m_ncomp; ++c) u[c] = U.extract( c, m_offset, N );
327  // get velocity for problem
328  const std::array< std::vector<std::array<tk::real,3>>, 4 > vel{{
329  Problem::prescribedVelocity(x[N[0]], y[N[0]], z[N[0]], m_c, m_ncomp),
330  Problem::prescribedVelocity(x[N[0]], y[N[1]], z[N[1]], m_c, m_ncomp),
331  Problem::prescribedVelocity(x[N[0]], y[N[2]], z[N[2]], m_c, m_ncomp),
332  Problem::prescribedVelocity(x[N[0]], y[N[3]], z[N[3]], m_c, m_ncomp)}};
333  // compute the maximum length of the characteristic velocity (advection
334  // velocity) across the four element nodes
335  tk::real maxvel = 0.0;
336  for (ncomp_t c=0; c<m_ncomp; ++c)
337  for (std::size_t i=0; i<4; ++i) {
338  auto v = std::sqrt( vel[i][c][0]*vel[i][c][0] +
339  vel[i][c][1]*vel[i][c][1] +
340  vel[i][c][2]*vel[i][c][2] );
341  if (v > maxvel) maxvel = v;
342  }
343  // compute element dt for the advection
344  auto advection_dt = L / maxvel;
345  // compute element dt based on diffusion
346  auto diffusion_dt = Physics::diffusion_dt( m_c, m_ncomp, L, u );
347  // compute minimum element dt
348  auto elemdt = std::min( advection_dt, diffusion_dt );
349  // find minimum dt across all elements
350  if (elemdt < mindt) mindt = elemdt;
351  }
352  return mindt;
353  }

361  //! \brief Query Dirichlet boundary condition value on a given side set for
362  //! all components in this PDE system
363  //! \param[in] t Physical time
364  //! \param[in] deltat Time step size
365  //! \param[in] ss Pair of side set ID and node IDs on the side set
366  //! \param[in] coord Mesh node coordinates
367  //! \return Vector of pairs of bool and boundary condition value associated
368  //! to mesh node IDs at which Dirichlet boundary conditions are set. Note
369  //! that instead of the actual boundary condition value, we return the
370  //! increment between t+dt and t, since that is what the solution requires
371  //! as we solve for the soution increments and not the solution itself.
372  std::unordered_map< std::size_t, std::vector< std::pair<bool,tk::real> > >
373  dirbc( tk::real t,
374  tk::real deltat,
375  const std::pair< const int, std::vector< std::size_t > >& ss,
376  const std::array< std::vector< tk::real >, 3 >& coord ) const
377  {
378  using tag::param; using tag::transport; using tag::bcdir;
379  using NodeBC = std::vector< std::pair< bool, tk::real > >;
380  std::unordered_map< std::size_t, NodeBC > bc;
381  const auto& ubc = g_inputdeck.get< param, transport, bcdir >();
382  if (!ubc.empty()) {
383  Assert( ubc.size() > m_c, "Indexing out of Dirichlet BC eq-vector" );
384  const auto& x = coord[0];
385  const auto& y = coord[1];
386  const auto& z = coord[2];
387  for (const auto& b : ubc[m_c])
388  if (std::stoi(b) == ss.first)
389  for (auto n : ss.second) {
390  Assert( x.size() > n, "Indexing out of coordinate array" );
391  auto s =
392  Problem::solinc( m_c, m_ncomp, x[n], y[n], z[n], t, deltat );
393  auto& nbc = bc[n] = NodeBC( m_ncomp );
394  for (ncomp_t c=0; c<m_ncomp; ++c)
395  nbc[c] = { true, s[c] };
396  }
397  }
398  return bc;
399  }

421  //! Return field output going to file
422  //! \param[in] t Physical time
423  //! \param[in] V Total mesh volume
424  //! \param[in] coord Mesh node coordinates
425  //! \param[in] v Nodal volumes
426  //! \param[in,out] U Solution vector at recent time step
427  //! \return Vector of vectors to be output to file
428  //! \details This functions should be written in conjunction with names(),
429  //! which provides the vector of field names
430  //! \note U is overwritten
431  std::vector< std::vector< tk::real > >
432  fieldOutput( tk::real t,
433  tk::real V,
434  const std::array< std::vector< tk::real >, 3 >& coord,
435  const std::vector< tk::real >& v,
436  tk::Fields& U ) const
437  {
438  std::vector< std::vector< tk::real > > out;
439  // will output numerical solution for all components
440  auto E = U;
441  for (ncomp_t c=0; c<m_ncomp; ++c)
442  out.push_back( U.extract( c, m_offset ) );
443  // evaluate analytic solution at time t
444  initialize( coord, U, t );
445  // will output analytic solution for all components
446  for (ncomp_t c=0; c<m_ncomp; ++c)
447  out.push_back( U.extract( c, m_offset ) );
448  // will output error for all components
449  for (ncomp_t c=0; c<m_ncomp; ++c) {
450  auto u = U.extract( c, m_offset );
451  auto e = E.extract( c, m_offset );
452  Assert( u.size() == e.size(), "Size mismatch" );
453  Assert( u.size() == v.size(), "Size mismatch" );
454  for (std::size_t i=0; i<u.size(); ++i)
455  e[i] = std::pow( e[i] - u[i], 2.0 ) * v[i] / V;
456  out.push_back( e );
457  }
458  return out;
459  }

461  //! Return names of integral variables to be output to diagnostics file
462  //! \return Vector of strings labelling integral variables output
463  std::vector< std::string > names() const {
