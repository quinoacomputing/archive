157  //! \brief Update coefficients using constant coefficients for b' and kappa'
158  //! \details This where the mix mass-fraction beta SDE is made consistent
159  //! with the no-mix and fully mixed limits by specifying the SDE
160  //! coefficients, b and kappa as functions of b' and kappa'. We leave S
161  //! unchanged.
162  void update(
163  char depvar,
164  ncomp_t ncomp,
165  const std::map< tk::ctr::Product, tk::real >& moments,
166  const std::vector< kw::sde_bprime::info::expect::type >& bprime,
167  const std::vector< kw::sde_kappaprime::info::expect::type >& kprime,
168  const std::vector< kw::sde_rho2::info::expect::type >&,
169  const std::vector< kw::sde_r::info::expect::type >&,
170  const std::vector< tk::Table >&,
171  const std::vector< tk::Table >&,
172  std::vector< kw::sde_b::info::expect::type >& b,
173  std::vector< kw::sde_kappa::info::expect::type >& k,
174  std::vector< kw::sde_S::info::expect::type >&,
175  tk::real ) const
176  {
177  for (ncomp_t c=0; c<ncomp; ++c) {
178  tk::real m = tk::ctr::lookup( tk::ctr::mean(depvar,c), moments );
179  tk::real v = tk::ctr::lookup( tk::ctr::variance(depvar,c), moments );
180 
181  if (m<1.0e-8 || m>1.0-1.0e-8) m = 0.5;
182  if (v<1.0e-8 && v>1.0-1.0e-8) v = 0.5;
183 
184  b[c] = bprime[c] * (1.0 - v / m / ( 1.0 - m ));
185  k[c] = kprime[c] * v;
186  }
187  }

240  //! \brief Update coefficients b', kappa', and S
241  //! \details This where the mix mass-fraction beta SDE is made consistent
242  //! with the no-mix and fully mixed limits by specifying the SDE
243  //! coefficients, b and kappa as functions of b' and kappa'. We also
244  //! specify S to force d<rho>/dt = 0, where <rho> = rho_2/(1+rY).
245  void update(
246  char depvar,
247  ncomp_t ncomp,
248  const std::map< tk::ctr::Product, tk::real >& moments,
249  const std::vector< kw::sde_bprime::info::expect::type >& bprime,
250  const std::vector< kw::sde_kappaprime::info::expect::type >& kprime,
251  const std::vector< kw::sde_rho2::info::expect::type >& rho2,
252  const std::vector< kw::sde_r::info::expect::type >& r,
253  const std::vector< tk::Table >&,
254  const std::vector< tk::Table >&,
255  std::vector< kw::sde_b::info::expect::type >& b,
256  std::vector< kw::sde_kappa::info::expect::type >& k,
257  std::vector< kw::sde_S::info::expect::type >& S,
258  tk::real ) const
259  {
260  using tk::ctr::lookup;
261  using tk::ctr::mean;
262  using tk::ctr::variance;
263  using tk::ctr::cen3;
264  // statistics nomenclature:
265  // Y = instantaneous mass fraction,
266  // R = instantaneous density,
267  // y = Y - <Y>, mass fraction fluctuation about its mean,
268  // r = R - <R>, density fluctuation about its mean,
269  // <Y> = mean mass fraction,
270  // <R> = mean density,
271  for (ncomp_t c=0; c<ncomp; ++c) {
272  tk::real m = lookup( mean(depvar,c), moments ); // <Y>
273  tk::real v = lookup( variance(depvar,c), moments ); // <y^2>
274  tk::real d = lookup( mean(depvar,c+ncomp), moments ); // <R>
275  tk::real d2 = lookup( variance(depvar,c+ncomp), moments ); // <r^2>
276  tk::real d3 = lookup( cen3(depvar,c+ncomp), moments ); // <r^3>
277 
278  if (m<1.0e-8 || m>1.0-1.0e-8) m = 0.5;
279  if (v<1.0e-8 && v>1.0-1.0e-8) v = 0.5;
280  b[c] = bprime[c] * (1.0 - v/m/(1.0-m));
281  //b[c] = bprime[c] * (1.0 - v/M[c]/(1.0-M[c]));
282  k[c] = kprime[c] * v;
283  //b[c] = 1.0;
284  //k[c] = 0.5*v/(m*(1.0-m));
285 
286  if (d < 1.0e-8) {
287  std::cout << "d:" << d << " ";
288  d = 0.5;
289  }
290  tk::real R = 1.0 + d2/d/d;
291  tk::real B = -1.0/r[c]/r[c];
292  tk::real C = (2.0+r[c])/r[c]/r[c];
293  tk::real D = -(1.0+r[c])/r[c]/r[c];
294  tk::real diff =
295  B*d/rho2[c] +
296  C*d*d*R/rho2[c]/rho2[c] +
297  D*d*d*d*(1.0 + 3.0*d2/d/d + d3/d/d/d)/rho2[c]/rho2[c]/rho2[c];
298  S[c] = (rho2[c]/d/R +
299  2.0*k[c]/b[c]*rho2[c]*rho2[c]/d/d*r[c]*r[c]/R*diff - 1.0) / r[c];
300  if (S[c] < 0.0 || S[c] > 1.0) {
301  std::cout << S[c] << " ";
302  S[c] = 0.5;
303  }
304  }
305  }

488  //! \brief Update coefficients b', kappa', and S
489  //! \details This where the mix mass-fraction beta SDE is made consistent
490  //! with the no-mix and fully mixed limits by specifying the SDE
491  //! coefficients, b and kappa as functions of b' and kappa'. Additionally,
492  //! we pull in a hydrodynamic timescale from an external function. We also
493  //! specify S to force d<rho>/dt = 0, where <rho> = rho_2/(1+rY).
494  void update(
495  char depvar,
496  ncomp_t ncomp,
497  const std::map< tk::ctr::Product, tk::real >& moments,
498  const std::vector< kw::sde_bprime::info::expect::type >& bprime,
499  const std::vector< kw::sde_kappaprime::info::expect::type >& kprime,
500  const std::vector< kw::sde_rho2::info::expect::type >& rho2,
501  const std::vector< kw::sde_r::info::expect::type >& r,
502  const std::vector< tk::Table >& hts,
503  const std::vector< tk::Table >& hp,
504  std::vector< kw::sde_b::info::expect::type >& b,
505  std::vector< kw::sde_kappa::info::expect::type >& k,
506  std::vector< kw::sde_S::info::expect::type >& S,
507  tk::real t ) const
508  {
509  using tk::ctr::lookup;
510  using tk::ctr::mean;
511  using tk::ctr::variance;
512  using tk::ctr::cen3;
513 
514  if (m_it == 0)
515  for (ncomp_t c=0; c<ncomp; ++c)
516  m_s.push_back( S[c] );
517 
518  // statistics nomenclature:
519  // Y = instantaneous mass fraction,
520  // R = instantaneous density,
521  // y = Y - <Y>, mass fraction fluctuation about its mean,
522  // r = R - <R>, density fluctuation about its mean,
523  // <Y> = mean mass fraction,
524  // <R> = mean density,
525  for (ncomp_t c=0; c<ncomp; ++c) {
526 
527  const tk::ctr::Term Y( static_cast<char>(std::toupper(depvar)),
528  c,
529  tk::ctr::Moment::ORDINARY );
530  const tk::ctr::Term dens( static_cast<char>(std::toupper(depvar)),
531  c+ncomp,
532  tk::ctr::Moment::ORDINARY );
533  const tk::ctr::Term s1( static_cast<char>(std::tolower(depvar)),
534  c+ncomp,
535  tk::ctr::Moment::CENTRAL );
536  const tk::ctr::Term s2( static_cast<char>(std::tolower(depvar)),
537  c+ncomp*2,
538  tk::ctr::Moment::CENTRAL );
539 
540  const auto RY = tk::ctr::Product( { dens, Y } );
541  tk::real ry = lookup( RY, moments ); // <RY>
542  const auto dscorr = tk::ctr::Product( { s1, s2 } );
543  tk::real ds = -lookup( dscorr, moments ); // b = -<rv>
544  tk::real d = lookup( mean(depvar,c+ncomp), moments ); // <R>
545  tk::real d2 = lookup( variance(depvar,c+ncomp), moments ); // <r^2>
546  tk::real d3 = lookup( cen3(depvar,c+ncomp), moments ); // <r^3>
547  tk::real yt = ry/d;
548 
549  // Sample hydrodynamics timescale and prod/diss at time t
550  auto ts = hydrotimescale( t, hts[c] ); // eps/k
551  auto pe = hydroproduction( t, hp[c] ); // P/eps = (dk/dt+eps)/eps
552 
553  tk::real a = r[c]/(1.0+r[c]*yt);
554  tk::real bnm = a*a*yt*(1.0-yt);
555  tk::real thetab = 1.0 - ds/bnm;
556  tk::real f2 =
557  1.0 / std::pow(1.0 + std::pow(pe-1.0,2.0)*std::pow(ds,0.25),0.5);
558  tk::real b1 = m_s[0];
559  tk::real b2 = m_s[1];
560  tk::real b3 = m_s[2];
561  tk::real eta = d2/d/d/ds;
562  tk::real beta2 = b2*(1.0+eta*ds);
563  tk::real Thetap = thetab*0.5*(1.0+eta/(1.0+eta*ds));
564  tk::real beta3 = b3*(1.0+eta*ds);
565  tk::real beta10 = b1 * (1.0+ds)/(1.0+eta*ds); //* 0.5*(1.0+d2/d/d+d2/d/d/ds);
566  tk::real beta1 = bprime[c] * 2.0/(1.0+eta+eta*ds) *
567  (beta10 + beta2*Thetap*f2 + beta3*Thetap*(1.0-Thetap)*f2);
568  b[c] = beta1 * ts;
569  k[c] = kprime[c] * beta1 * ts * ds * ds;
570 
571  tk::real R = 1.0 + d2/d/d;
572  tk::real B = -1.0/r[c]/r[c];
573  tk::real C = (2.0+r[c])/r[c]/r[c];
574  tk::real D = -(1.0+r[c])/r[c]/r[c];
575  tk::real diff =
576  B*d/rho2[c] +
577  C*d*d*R/rho2[c]/rho2[c] +
578  D*d*d*d*(1.0 + 3.0*d2/d/d + d3/d/d/d)/rho2[c]/rho2[c]/rho2[c];
579  S[c] = (rho2[c]/d/R +
580  2.0*k[c]/b[c]*rho2[c]*rho2[c]/d/d*r[c]*r[c]/R*diff - 1.0) / r[c];
581  }
582 
583  ++m_it;
584  }
585 
586  //! Sample the inverse hydrodynamics time scale at time t
587  //! \param[in] t Time at which to sample inverse hydrodynamics time scale
588  //! \param[in] ts Hydro time scale table to sample
589  //! \return Sampled value from discrete table of inverse hydro time scale
590  tk::real hydrotimescale( tk::real t, const tk::Table& ts ) const
591  { return tk::sample( t, ts ); }
592 
593  //! Sample the hydrodynamics production/dissipation rate (P/e) at time t
594  //! \param[in] t Time at which to sample hydrodynamics P/e
595  //! \param[in] p P/e table to sample
596  //! \return Sampled value from discrete table of P/e
597  tk::real hydroproduction( tk::real t, const tk::Table& p ) const
598  { return tk::sample( t, p ); }
599 
600  mutable std::size_t m_it = 0;
601  mutable std::vector< tk::real > m_s;
602 };
