88  //! \brief Advance particles according to the dissipation SDE
89  //! \param[in,out] particles Array of particle properties
90  //! \param[in] stream Thread (or more precisely stream) ID
91  //! \param[in] dt Time step size
92  //! \param[in] moments Map of statistical moments
93  void advance( tk::Particles& particles,
94  int stream,
95  tk::real dt,
96  tk::real,
97  const std::map< tk::ctr::Product, tk::real >& moments )
98  {
99  using tk::ctr::lookup;
100 
101  // Access mean turbulence frequency
102  tk::real O = lookup( m_O, moments );
103 
104  tk::ctr::Term u( static_cast<char>(std::toupper(m_velocity_depvar)), 0,
105  tk::ctr::Moment::ORDINARY );
106  tk::ctr::Term v( static_cast<char>(std::toupper(m_velocity_depvar)), 1,
107  tk::ctr::Moment::ORDINARY );
108  tk::ctr::Term w( static_cast<char>(std::toupper(m_velocity_depvar)), 2,
109  tk::ctr::Moment::ORDINARY );
110  auto r11 = tk::ctr::Product( { u, u } );
111  auto r22 = tk::ctr::Product( { v, v } );
112  auto r33 = tk::ctr::Product( { w, w } );
113  auto r12 = tk::ctr::Product( { u, v } );
114 
115  // Compute turbulent kinetic energy
116  tk::real k = ( lookup(r11,moments) +
117  lookup(r22,moments) +
118  lookup(r33,moments) ) / 2.0;
119 
120  // Production of turbulent kinetic energy
121  tk::real S = 1.0; // prescribed shear
122  tk::real P = -lookup(r12,moments)*S;
123 
124  // Source for turbulent frequency
125  tk::real Som = m_com2 - m_com1*P/(O*k);
126 
127  const auto npar = particles.nunk();
128  for (auto p=decltype(npar){0}; p<npar; ++p) {
129  // Generate a Gaussian random number with zero mean and unit variance
130  tk::real dW;
131  m_rng.gaussian( stream, m_ncomp, &dW );
132  // Advance particle frequency
133  tk::real& Op = particles( p, 0, m_offset );
134  tk::real d = 2.0*m_c3*m_c4*O*O*Op*dt;
135  d = (d > 0.0 ? std::sqrt(d) : 0.0);
136  Op += (-m_c3*(Op-O) - Som*Op)*O*dt + d*dW;
137  }
138  }

143  const ncomp_t m_ncomp; //!< Number of components
144  const ncomp_t m_offset; //!< Offset SDE operates from
