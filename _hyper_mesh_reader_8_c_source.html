40 HyperMeshReader::getFileNames() const
41 // *****************************************************************************
42 // Read Hyper mesh metadata and extract filenames we need to read
43 //! \return Vector of strings containing the filenames
44 // *****************************************************************************
45 {
46  // Read XML file for metadata
47  pugi::xml_document meta;
48  auto res = meta.load_file( m_filename.c_str() );
49  ErrChk( res == true, "Could not parse XML attempting to read HyperMesh "
50  "metadata file '" + m_filename + "', at character position " +
51  std::to_string(res.offset) + " : " + res.description() );
52 
53  // Extract path from metadata filename
54  std::string fn = m_filename;
55  std::string path( fn.substr(0,fn.find_last_of("/")+1) );
56  // Prepend path to filenames to be extracted
57  std::pair< std::string, std::string > filenames( path, path );
58 
59  for (const auto& child : meta.children("mesh"))
60  for (const auto& group : child.children()) {
61 
62  if (group.name() == std::string("coordinates")) {
63 
64  filenames.first += group.attribute("file").value();
65 
66  } else if (group.name() == std::string("element_set")) {
67 
68  filenames.second += group.attribute("file").value();
69  Assert( group.attribute("topology").value() ==
70  std::string("four_node_tet"), "Only pure tetrahedron-element "
71  "meshes are supported" );
72  }
73  }
74 
75  return filenames;
76 }

79 HyperMeshReader::readNodes( const std::string& filename, UnsMesh& mesh ) const
80 // *****************************************************************************
81 // Read nodes
82 //! \param[in] filename Filename to read nodes from
83 //! \param[in] mesh Unstructured mesh object to put nodes coordinates
84 //! \note We throw away the node ID, which means the nodes must be in order.
85 // *****************************************************************************
86 {
87  // Read in node coordinates: x-coord y-coord z-coord
88  for (auto& l : tk::Reader(filename).lines()) {
89  std::stringstream ss(l);
90  int id;
91  tk::real x, y, z;
92  ss >> id >> x >> y >> z;
93  mesh.x().push_back( x );
94  mesh.y().push_back( y );
95  mesh.z().push_back( z );
96  }
97 }
