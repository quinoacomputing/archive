26 /*! Transport PDE problem: diffusion of a shear layer
27  \details This class implements the analytical solutions for the test
28  problem, adopted from Okubo Akira Karweit Michael J. , (1969),
29  [DIFFUSION FROM A CONTINUOUS SOURCE IN A UNIFORM SHEAR
30  FLOW](http://onlinelibrary.wiley.com/doi/10.4319/lo.1969.14.4.0514/abstract),
31  Limnology and Oceanography, 14, doi: 10.4319/lo.1969.14.4.0514. In essence,
32  this is a test problem for the advection-diffusion equation in 3D where the
33  analytical solution is known in a closed form as the solution evolves in
34  time. The initial solution is a Gaussian that is advected and diffused in
35  time with an imposed constant-in-time velocity field that features
36  advection and shear. Also, the diffusion coefficients can be different in
37  the three coordinate directions. Note that t0 as well as all three
38  components of the diffusion must be larger than zero at t=t0 to have a
39  well-defined initial condition.
40 
41  In a nutshell, the equation solved is
42  \f[
43  \frac{\partial S}{\partial t} + \left(V_0 + \Omega_y y + \Omega_z z
44  \right) \frac{\partial S}{\partial x} =
45  A_x \frac{\partial^2S}{\partial x^2} +
46  A_y \frac{\partial^2S}{\partial y^2} +
47  A_z \frac{\partial^2S}{\partial z^2}
48  \f]
49  whose solution is given by
50  \f[
51  S(t,x,y,z,) = \frac{1}{8\pi^{3/2}(A_xA_yA_z)^{1/2}t^{3/2}
52  (1+\phi_3^2t^2)^{1/2}}
53  \exp\left[ -\frac{x-V_0t-(\Omega_yy+\Omega_zz)^2/2}
54  {4A_xt(1+\phi_3^2t^2}
55  -\frac{y^2}{4A_yt}
56  -\frac{z^2}{4A_zt} \right]
57  \f]
58  where \f$ \phi_3^2 = (\Omega_y^2A_y/A_x + \Omega_z^2A_z/A_x)/12\f$.
59  See also the paper.
60 */

64  //! Evaluate analytical solution at (x,y,z,t) for all components
65  //! \param[in] e Equation system index, i.e., which transport equation
66  //! system we operate on among the systems of PDEs
67  //! \param[in] ncomp Number of components in this transport equation system
68  //! \param[in] x X coordinate where to evaluate the solution
69  //! \param[in] y Y coordinate where to evaluate the solution
70  //! \param[in] z Z coordinate where to evaluate the solution
71  //! \param[in] t Time where to evaluate the solution
72  //! \return Values of all components evaluated at (x,y,z,t)
73  static std::vector< tk::real >
74  solution( ncomp_t e, ncomp_t ncomp, tk::real x, tk::real y, tk::real z,
75  tk::real t )
76  {
77  using tag::param; using tag::transport;
78  const auto& u0 = g_inputdeck.get< param, transport, tag::u0 >()[e];
79  const auto& d = g_inputdeck.get< param, transport, tag::diffusivity >()[e];
80  const auto& l = g_inputdeck.get< param, transport, tag::lambda >()[e];
81  std::vector< tk::real > r( ncomp );
82  for (ncomp_t c=0; c<ncomp; ++c) {
83  const auto li = 2*c;
84  const auto di = 3*c;
85  const auto phi3s = (l[li+0]*l[li+0]*d[di+1]/d[di+0] +
86  l[li+1]*l[li+1]*d[di+2]/d[di+0]) / 12.0;
87  r[c] =
88  1.0 / ( 8.0 * std::pow(M_PI,3.0/2.0) *
89  std::sqrt(d[di+0]*d[di+1]*d[di+2]) *
90  std::pow(t,3.0/2.0) * std::sqrt(1.0+phi3s*t*t) ) *
91  exp( -std::pow( x - u0[c]*t -
92  0.5*(l[li+0]*y + l[li+1]*z)*t, 2.0 ) /
93  ( 4.0 * d[di+0] * t * (1.0 + phi3s*t*t) )
94  -y*y / ( 4.0 * d[di+1] * t )
95  -z*z / ( 4.0 * d[di+2] * t ) );
96  }
97  return r;
98  }

100  //! \brief Evaluate the increment from t to t+dt of the analytical solution
101  //! at (x,y,z) for all components
102  //! \param[in] e Equation system index, i.e., which transporte equation
103  //! system we operate on among the systems of PDEs
104  //! \param[in] ncomp Number of components in this transport equation system
105  //! \param[in] x X coordinate where to evaluate the solution
106  //! \param[in] y Y coordinate where to evaluate the solution
107  //! \param[in] z Z coordinate where to evaluate the solution
108  //! \param[in] t Time where to evaluate the solution increment starting from
109  //! \param[in] dt Time increment at which evaluate the solution increment to
110  //! \return Increment in values of all components evaluated at (x,y,z,t+dt)
111  static std::vector< tk::real >
112  solinc( ncomp_t e, ncomp_t ncomp, tk::real x, tk::real y, tk::real z,
113  tk::real t, tk::real dt )
114  {
115  auto st1 = solution( e, ncomp, x, y, z, t );
116  auto st2 = solution( e, ncomp, x, y, z, t+dt );
117  std::transform( begin(st1), end(st1), begin(st2), begin(st2),
118  []( tk::real s, tk::real& d ){ return d -= s; } );
119  return st2;
120  }
