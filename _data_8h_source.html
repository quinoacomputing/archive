95  //! Const ptr to physical variable access dispatch
96  //! \details Public interface to the first half of a physical variable
97  //! access. cptr() and var() are two member functions intended to be used
98  //! together in case when component and offset would be expensive to
99  //! compute for data access via the function call operator, i.e., cptr()
100  //! can be used to pre-compute part of the address, which returns a
101  //! pointer and var() can be used to finish the data access using the
102  //! pointer returned by cptr(). In other words, cptr() returns part of the
103  //! address known based on component and offset and intended to be used in
104  //! a setup phase. Then var() takes this partial address and finishes the
105  //! address calculation given the unknown id. Thus the following two data
106  //! accesses are equivalent (modulo constness):
107  //! * real& value = operator()( unk, comp, offs ); and
108  //! * const real* p = cptr( comp, offs ); and
109  //! const real& value = var( p, unk ); or real& value = var( p, unk );
110  //! Requirement: offset + component < nprop, enforced with an assert in
111  //! DEBUG mode, see also the constructor.
112  //! \param[in] component Component index, i.e., position of a scalar within
113  //! a system
114  //! \param[in] offset System offset specifying the position of the system of
115  //! equations among other systems

371  //! Remove a number of unknowns
372  //! \param[in] unknown Set of indices of unknowns to remove
373  void rm( const std::set< ncomp_t >& unknown ) {
374  auto remove = [ &unknown ]( std::size_t i ) -> bool {
375  if (unknown.find(i) != end(unknown)) return true;
376  return false;
377  };
378  std::size_t last = 0;
379  for(std::size_t i=0; i<m_nunk; ++i, ++last) {
380  while( remove(i) ) ++i;
381  if (i >= m_nunk) break;
382  for (ncomp_t p = 0; p<m_nprop; ++p)
383  m_vec[ last*m_nprop+p ] = m_vec[ i*m_nprop+p ];
384  }
385  m_vec.resize( last*m_nprop );
386  m_nunk -= unknown.size();
387  }

625 //! Compute the maximum difference between the elements of two Data objects
626 //! \param[in] lhs 1st Data object
627 //! \param[in] rhs 2nd Data object
628 //! \return The index, i.e., the raw position, of and the largest absolute value
629 //! of the difference between all corresponding elements of _lhs_ and _rhs_.
630 //! \details The position returned is the position in the underlying raw data
631 //! structure, independent of components, offsets, etc. If lhs == rhs with
632 //! precision std::numeric_limits< tk::real >::epsilon(), a pair of (0,0.0)
633 //! is returned.
634 //! \note The Data objects _lhs_ and _rhs_ must have the same number of
635 //! unknowns and properties.
636 template< uint8_t Layout >
637 std::pair< std::size_t, tk::real >
638 maxdiff( const Data< Layout >& lhs, const Data< Layout >& rhs ) {
639  Assert( lhs.nunk() == rhs.nunk(), "Number of unknowns unequal" );
640  Assert( lhs.nprop() == rhs.nprop(), "Number of properties unequal" );
641  auto l = lhs.data().cbegin();
642  auto r = rhs.data().cbegin();
643  std::pair< std::size_t, tk::real > m( 0, std::abs(*l - *r) );
644  ++l; ++r;
645  while (l != lhs.data().cend()) {
646  const auto d = std::abs(*l - *r);
647  if (d > m.second) m = { std::distance(lhs.data().cbegin(),l), d };
648  ++l; ++r;
649  }
650  return m;
651 }
