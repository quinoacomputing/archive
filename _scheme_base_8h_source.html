35  //! Constructor
36  //! \param[in] scheme Discretization scheme
37  //! \details Based on the enum we create two empty chare arrays: (1)
38  //! discproxy which contains common functionality and data for all
39  //! discretizations, and (2) proxy, which have functionality and data
40  //! specific to a given discretization. Note that proxy is bound (in
41  //! migration behavior and properties) to discproxy.
42  explicit SchemeBase( ctr::SchemeType scheme ) :
43  discproxy( CProxy_Discretization::ckNew() )
44  {
45  CkArrayOptions bound;
46  bound.bindTo( discproxy );
47  if (scheme == ctr::SchemeType::CG) {
48  proxy = static_cast< CProxy_CG >( CProxy_CG::ckNew(bound) );
49  } else if (scheme == ctr::SchemeType::DG) {

96  //! Generic base for all call_* classes
97  //! \details This class stores the entry method arguments and contains a
98  //! helper function and classes that facilitate unpacking the tuple into a
99  //! variadic list of arguments passed to the called entry method. The Spec
100  //! template argument is the class inheriting from this class (see CRTP at
101  //! https://en.wikipedia.org/wiki/Curiously_recurring_template_pattern),
102  //! specializing functionality of this class. CRTP enables hiding all
103  //! generic functionality here while exposing only the specialized one
104  //! (the actual function call) to the base, which minimizes client code
105  //! (in class Scheme). Unpacking the tuple to a variadic argument list is
106  //! loosely inspired by https://stackoverflow.com/a/16868151.
107  template< class Spec, typename... Args >
108  struct Call : boost::static_visitor<> {
109  //! Constructor storing called member function arguments in tuple
110  Call( Args&&... args ) : arg( std::forward_as_tuple(args...) ) {}
111  //! Helper class for unpacking tuple into argument list
112  template< typename P, typename Tuple, bool Done, int Total, int... N >
113  struct invoke_impl {
114  static void invoke( P& p, Tuple&& t ) {
115  invoke_impl< P, Tuple, Total == 1 + sizeof...(N), Total, N...,
116  sizeof...(N) >::invoke( p, std::forward<Tuple>(t) );
117  }
118  };
119  //! Helper class for unpacking tuple into argument list (end of list)
120  template< typename P, typename Tuple, int Total, int... N >
121  struct invoke_impl< P, Tuple, true, Total, N... > {
122  static void invoke( P& p, Tuple&& t ) {
123  Spec::invoke( p, std::get< N >( std::forward<Tuple>(t) )... );
124  }
125  };
126  //! Invoke member function with arguments from tuple
127  //! \param[in,out] p Proxy behind which the entry method is called
128  //! \param[in] t Optional tuple holding argments to be passed to the call
129  template< typename P, typename Tuple = std::tuple<int> >
130  static void invoke( P& p, Tuple&& t = {} ) {
131  typedef typename std::decay<Tuple>::type ttype;
132  invoke_impl< P, Tuple, std::tuple_size<ttype>::value == 0,
133  std::tuple_size<ttype>::value >::invoke( p, std::forward<Tuple>(t) );
134  }
135  //! Function call operator overloading all types used with variant visitor
136  //! \param[in,out] p Proxy behind which the entry method is called
137  //! \details Classes inheriting from this class should also inherit from
138  //! boost::static_visitor which requires overloading operator(),
139  //! unambiguously accepting any value of types Ts over which classes
140  //! inheriting from this class may be used as functors for static
141  //! visitor with a variant of types Ts.
142  template< typename P > void operator()(P& p) const { invoke(p,arg); }
143  //! Tuple storing arguments of the entry method to be called
144  std::tuple< Args... > arg;
145  };
