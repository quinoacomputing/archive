60 //! \param[in] mesh Unstructured mesh object
61 // *****************************************************************************
62 {
63  if (mesh.tetinpoel().empty()) return;
64 
65  // Make sure tetrahedron element connectivity starts with zero
66  Assert( *std::minmax_element( begin(mesh.tetinpoel()),
67  end(mesh.tetinpoel()) ).first == 0,
68  "tetrahedron node ids should start from zero" );
69 
70  // Get number of tetrahedra in mesh
71  auto n = mesh.tetinpoel().size()/4;
72 
73  // Write out number of tetrahedra
74  m_outFile << n << std::endl;
75 
76  // Create empty tag vector
77  std::vector< std::vector< int > > tg;
78  tg.resize( n );
79  for (auto& t : tg) t.push_back( 1 );
80 
81  // Write out tetrehadra element tags and connectivity
82  for (std::size_t i=0; i<n; ++i) {
83  // tag n[1-4]
84  m_outFile << '\t' << tg[i][0]
85  << '\t' << mesh.tetinpoel()[i*4+3]+1
86  << '\t' << mesh.tetinpoel()[i*4+0]+1
87  << '\t' << mesh.tetinpoel()[i*4+1]+1
88  << '\t' << mesh.tetinpoel()[i*4+2]+1 << std::endl;
89  }
90 
91  if (mesh.triinpoel().empty()) return;
92 
93  // Make sure triangle element connectivity starts with zero
94  Assert( *std::minmax_element( begin(mesh.triinpoel()),
95  end(mesh.triinpoel()) ).first == 0,
96  "triangle node ids should start from zero" );
97 
98  // Get number of triangles in mesh
99  n = mesh.triinpoel().size()/3;
100 
101  // Write out number of triangles
102  m_outFile << n << std::endl;
103 
104  // Create empty tag vector if there is no tag
105  tg.clear();
106  tg.resize( n );
107  for (auto& t : tg) t.push_back( 1 );
108 
109  // Write out triangle element tags and connectivity
110  for (std::size_t i=0; i<n; ++i) {
111  // tag n[1-4]
112  m_outFile << '\t' << tg[i][0]
113  << '\t' << mesh.triinpoel()[i*3+0]+1
114  << '\t' << mesh.triinpoel()[i*3+1]+1
