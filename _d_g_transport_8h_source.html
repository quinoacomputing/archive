131  //! \brief Query Dirichlet boundary condition value on a given side set for
132  //! all components in this PDE system
133 // //! \param[in] t Physical time
134 // //! \param[in] deltat Time step size
135 // //! \param[in] sides Pair of side set ID and face IDs on the side set
136 // //! \param[in] coord Mesh face coordinates
137  //! \return Vector of pairs of bool and boundary condition value associated
138  //! to mesh face IDs at which Dirichlet boundary conditions are set. Note
139  //! that instead of the actual boundary condition value, we return the
140  //! increment between t+dt and t, since that is what the solution requires
141  //! as we solve for the soution increments and not the solution itself.
142  std::unordered_map< std::size_t, std::vector< std::pair<bool,tk::real> > >
143  dirbc( tk::real /*t*/,

146  const std::array< std::vector< tk::real >, 3 >& /*coord*/ ) const
147  {
148  // Call Problem::solinc() within a search for all face centers of a side
149  // set given in sides (key=setid, value=faceids). See
150  // cg::Transport::dirbc() for an example search for all nodes of a side
151  // set given in sides (key=setid, value=nodeids). Instead of coord, we
152  // probably want to pass in a const-ref to DG::m_geoFace and work with the
153  // face centroid coordinates.

179  //! Return field output going to file
180  //! \param[in] t Physical time
181  //! \param[in] V Total mesh volume
182  //! \param[in] coord Mesh node coordinates
183  //! \param[in] v Nodal volumes
184  //! \param[in,out] U Solution vector at recent time step
185  //! \return Vector of vectors to be output to file
186  //! \details This functions should be written in conjunction with names(),
187  //! which provides the vector of field names
188  //! \note U is overwritten
189  std::vector< std::vector< tk::real > >
190  fieldOutput( tk::real t,
191  tk::real V,
192  const std::array< std::vector< tk::real >, 3 >& coord,
193  const std::vector< tk::real >& v,
194  tk::Fields& U ) const
195  {
196  std::vector< std::vector< tk::real > > out;
197  // will output numerical solution for all components
198  auto E = U;
199  for (ncomp_t c=0; c<m_ncomp; ++c)
200  out.push_back( U.extract( c, m_offset ) );
201  // evaluate analytic solution at time t
202  initialize( coord, U, t );
203  // will output analytic solution for all components
204  for (ncomp_t c=0; c<m_ncomp; ++c)
205  out.push_back( U.extract( c, m_offset ) );
206  // will output error for all components
207  for (ncomp_t c=0; c<m_ncomp; ++c) {
208  auto u = U.extract( c, m_offset );
209  auto e = E.extract( c, m_offset );
210  Assert( u.size() == e.size(), "Size mismatch" );
211  Assert( u.size() == v.size(), "Size mismatch" );
212  for (std::size_t i=0; i<u.size(); ++i)
213  e[i] = std::pow( e[i] - u[i], 2.0 ) * v[i] / V;
214  out.push_back( e );
215  }
216  return out;
217  }
