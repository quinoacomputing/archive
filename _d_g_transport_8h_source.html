79  //! \param[in] geoElem Element geometry array
80  //! \param[in,out] unk Array of unknowns

85  {

89  for (std::size_t e=0; e<nelem; ++e)

101  //! Compute the left hand side mass matrix
102  //! \param[in] geoElem Element geometry array
103  //! \param[in,out] l Block diagonal mass matrix
104  void lhs( const tk::Fields& geoElem, tk::Fields& l ) const
105  {
106  Assert( geoElem.nunk() == l.nunk(), "Size mismatch" );
107  std::size_t nelem = geoElem.nunk();
108 
109  for (std::size_t e=0; e<nelem; ++e)
110  {
111  for (ncomp_t c=0; c<m_ncomp; ++c)
112  l(e, c, m_offset) = geoElem(e,0,0);
113  }
114  }

116  //! Compute right hand side
117  //! \param[in] geoFace Face geometry array
118  //! \param[in] fd Face connectivity and boundary conditions object
119  //! \param[in] U Solution vector at recent time step
120  //! \param[in,out] R Right-hand side vector computed
121  void rhs( tk::real,
122  const tk::Fields& geoFace,
123  const tk::Fields&,
124  const inciter::FaceData& fd,
125  const tk::Fields& U,
126  tk::Fields& R ) const
127  {
128  Assert( U.nunk() == R.nunk(), "Number of unknowns in solution "
129  "vector and right-hand side at recent time step incorrect" );
130  Assert( U.nprop() == m_ncomp && R.nprop() == m_ncomp,
131  "Number of components in solution and right-hand side vector "
132  "must equal "+ std::to_string(m_ncomp) );
133 
134  const auto& bface = fd.Bface();
135  const auto& esuf = fd.Esuf();
136 
137  // set rhs to zero
138  R.fill(0.0);
139 
140  // compute internal surface flux integrals
141  for (auto f=fd.Nbfac(); f<esuf.size()/2; ++f)
142  {
143  std::size_t el = static_cast< std::size_t >(esuf[2*f]);
144  std::size_t er = static_cast< std::size_t >(esuf[2*f+1]);
145  auto farea = geoFace(f,0,0);
146 
147  //--- upwind fluxes
148  auto flux = upwindFlux( f, geoFace, {{U.extract(el), U.extract(er)}} );
149 
150  for (ncomp_t c=0; c<m_ncomp; ++c) {
151  R(el, c, m_offset) -= farea * flux[c];
152  R(er, c, m_offset) += farea * flux[c];
153  }
154  }
155 
156  // compute boundary surface flux integrals
157  bndIntegral< Extrapolate >( m_bcextrapolate, bface, esuf, geoFace, U, R );
158  bndIntegral< Inlet >( m_bcinlet, bface, esuf, geoFace, U, R );
159  bndIntegral< Outlet >( m_bcoutlet, bface, esuf, geoFace, U, R );
160  }

201  //! Return field output going to file
202  //! \param[in] t Physical time
203  //! \param[in] geoElem Element geometry array
204  //! \param[in,out] U Solution vector at recent time step
205  //! \return Vector of vectors to be output to file
206  //! \details This functions should be written in conjunction with names(),
207  //! which provides the vector of field names
208  //! \note U is overwritten
209  std::vector< std::vector< tk::real > >
210  fieldOutput( tk::real t,
211  tk::real /*V*/,
212  const tk::Fields& geoElem,
213  tk::Fields& U ) const
214  {
215  Assert( geoElem.nunk() == U.nunk(), "Size mismatch" );
216  std::vector< std::vector< tk::real > > out;
217  // will output numerical solution for all components
218  for (ncomp_t c=0; c<m_ncomp; ++c)
219  out.push_back( U.extract( c, m_offset ) );
220  // evaluate analytic solution at time t
221  auto E = U;
222  initialize( geoElem, E, t );
223  // will output analytic solution for all components
224  for (ncomp_t c=0; c<m_ncomp; ++c)
225  out.push_back( E.extract( c, m_offset ) );
226  // will output error for all components
227  for (ncomp_t c=0; c<m_ncomp; ++c) {
228  auto u = U.extract( c, m_offset );
229  auto e = E.extract( c, m_offset );
230  for (std::size_t i=0; i<u.size(); ++i)
231  e[i] = std::pow( e[i] - u[i], 2.0 ) * geoElem(i,0,0);
232  out.push_back( e );
233  }
234  return out;
235  }

290  //! Compute boundary surface integral for a number of faces
291  //! \param[in] faces Face IDs at which to compute surface integral
292  //! \param[in] esuf Elements surrounding face, see tk::genEsuf()
293  //! \param[in] geoFace Face geometry array
294  //! \param[in] U Solution vector at recent time step
295  //! \param[in,out] R Right-hand side vector computed
296  //! \tparam State Policy class providing the left and right state at
297  //! boundaries by its member function State::LR()
298  template< class State >
299  void surfInt( const std::vector< std::size_t >& faces,
300  const std::vector< int >& esuf,
301  const tk::Fields& geoFace,
302  const tk::Fields& U,
303  tk::Fields& R ) const
304  {
305  for (const auto& f : faces) {
306  std::size_t el = static_cast< std::size_t >(esuf[2*f]);
307  Assert( esuf[2*f+1] == -1, "outside boundary element not -1" );
308  auto farea = geoFace(f,0,0);
309 
310  //--- upwind fluxes
311  auto flux = upwindFlux( f, geoFace, State::LR(U,el) );
312 
313  for (ncomp_t c=0; c<m_ncomp; ++c)
314  R(el, c, m_offset) -= farea * flux[c];
315  }
316  }

318  //! Compute boundary surface flux integrals for a given boundary type
319  //! \tparam BCType Specifies the type of boundary condition to apply
320  //! \param bcconfig BC configuration vector for multiple side sets
321  //! \param[in] bface Boundary faces side-set information
322  //! \param[in] esuf Elements surrounding faces
323  //! \param[in] geoFace Face geometry array
324  //! \param[in] U Solution vector at recent time step
325  //! \param[in,out] R Right-hand side vector computed
326  template< class BCType >
327  void
328  bndIntegral( const std::vector< bcconf_t >& bcconfig,
329  const std::map< int, std::vector< std::size_t > >& bface,
330  const std::vector< int >& esuf,
331  const tk::Fields& geoFace,
332  const tk::Fields& U,
333  tk::Fields& R ) const
334  {
335  for (const auto& s : bcconfig) { // for all bc sidesets
336  auto bc = bface.find( std::stoi(s) );// faces for side set
337  if (bc != end(bface))
338  surfInt< BCType >( bc->second, esuf, geoFace, U, R );
339  }
340  }

342  //! Riemann solver using upwind method
343  //! \param[in] f Face ID
344  //! \param[in] geoFace Face geometry array
345  //! \param[in] u Left and right unknown/state vector
346  //! \return Riemann solution using upwind method
347  std::vector< tk::real >
348  upwindFlux( std::size_t f,
349  const tk::Fields& geoFace,
350  const std::array< std::vector< tk::real >, 2 >& u ) const
351  {
352  std::vector< tk::real > flux( u[0].size(), 0 );
353 
354  auto xc = geoFace(f,4,0);
355  auto yc = geoFace(f,5,0);
356  auto zc = geoFace(f,6,0);
357 
358  std::array< tk::real, 3 > fn {{ geoFace(f,1,0),
359  geoFace(f,2,0),
360  geoFace(f,3,0) }};
361 
362  const auto vel = Problem::prescribedVelocity( xc, yc, zc, m_c, m_ncomp );
363 
364  for(ncomp_t c=0; c<m_ncomp; ++c)
365  {
366  auto ax = vel[c][0];
367  auto ay = vel[c][1];
368  auto az = vel[c][2];
369 
370  // wave speed
371  tk::real swave = ax*fn[0] + ay*fn[1] + az*fn[2];
372 
373  // upwinding
374  tk::real splus = 0.5 * (swave + fabs(swave));
375  tk::real sminus = 0.5 * (swave - fabs(swave));
376 
377  flux[c] = splus * u[0][c] + sminus * u[1][c];
378  }
379 
380  return flux;
381  }
