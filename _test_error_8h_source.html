114  //! Test jump error indicator (mostly bounds) for tetrahedron mesh
115  void TestErrorIndicator( inciter::ctr::AMRErrorType errtype ) {
116  // Shift node IDs to start from zero
117  tk::shiftToZero( inpoel );
118 
119  // find out number of points in mesh connectivity
120  auto minmax = std::minmax_element( begin(inpoel), end(inpoel) );
121  Assert( *minmax.first == 0, "node ids should start from zero" );
122  auto npoin = *minmax.second + 1;
123 
124  // Generate elements surrounding points
125  auto esup = tk::genEsup( inpoel, 4 );
126 
127  // Generate edge connectivity
128  auto inpoed = tk::genInpoed( inpoel, 4, esup );
129 
130  // create error indicator object to get access to error indicators
131  AMR::Error err;
132 
133  // generate a constant scalar field
134  tk::Fields uc( npoin, 1 );
135  uc.fill( 1.2 );
136  // test jump error indicator on all edges
137  for (std::size_t e=0; e<inpoed.size()/2; ++e) {
138  std::pair< std::size_t, std::size_t > edge{ inpoed[e*2],inpoed[e*2+1] };
139  auto r = err.scalar( uc, edge, 0, coord, inpoel, esup, errtype );
140  ensure_equals( "edge error of constant field incorrect", r, 0.0, pr );
141  }
142 
143  // generate a linear scalar field with a slope in only x direction
144  tk::Fields ux( npoin, 1 );
145  for (std::size_t p=0; p<npoin; ++p) ux(p,0,0) = coord[0][p];
146  // test jump error indicator on all edges
147  for (std::size_t e=0; e<inpoed.size()/2; ++e) {
148  std::pair< std::size_t, std::size_t > edge{ inpoed[e*2],inpoed[e*2+1] };
149  auto r = err.scalar( ux, edge, 0, coord, inpoel, esup, errtype );
150  ensure( "edge error < 0.0", r > -pr );
151  ensure( "edge error > 1.0", r < 1.0+pr );
152  }
153 
154  // generate a linear scalar field with a slope in only y direction
155  tk::Fields uy( npoin, 1 );
156  for (std::size_t p=0; p<npoin; ++p) uy(p,0,0) = coord[1][p];
157  // test jump error indicator on all edges
158  for (std::size_t e=0; e<inpoed.size()/2; ++e) {
159  std::pair< std::size_t, std::size_t > edge{ inpoed[e*2],inpoed[e*2+1] };
160  auto r = err.scalar( uy, edge, 0, coord, inpoel, esup, errtype );
161  ensure( "edge error < 0.0", r > -pr );
162  ensure( "edge error > 1.0", r < 1.0+pr );
163  }
164 
165  // generate a linear scalar field with a slope in only z direction
166  tk::Fields uz( npoin, 1 );
167  for (std::size_t p=0; p<npoin; ++p) uz(p,0,0) = coord[2][p];
168  // test jump error indicator on all edges
169  for (std::size_t e=0; e<inpoed.size()/2; ++e) {
170  std::pair< std::size_t, std::size_t > edge{ inpoed[e*2],inpoed[e*2+1] };
171  auto r = err.scalar( uz, edge, 0, coord, inpoel, esup, errtype );
172  ensure( "edge error < 0.0", r > -pr );
173  ensure( "edge error > 1.0", r < 1.0+pr );
174  }
175 
176  // generate linear vector field with different slopes for different components
177  tk::Fields u3( npoin, 3 );
178  for (std::size_t p=0; p<npoin; ++p) {
179  u3(p,0,0) = 2.0*coord[0][p];
180  u3(p,1,0) = 1.5*coord[1][p];
181  u3(p,2,0) = -0.5*coord[2][p];
182  }
183  // test jump error indicator on all edges
184  for (std::size_t e=0; e<inpoed.size()/2; ++e) {
185  std::pair< std::size_t, std::size_t > edge{ inpoed[e*2],inpoed[e*2+1] };
186  auto rx = err.scalar( u3, edge, 0, coord, inpoel, esup, errtype );
187  ensure( "edge error < 0.0", rx > -pr );
188  ensure( "edge error > 1.0", rx < 1.0+pr );
189  auto ry = err.scalar( u3, edge, 1, coord, inpoel, esup, errtype );
190  ensure( "edge error < 0.0", ry > -pr );
191  ensure( "edge error > 1.0", ry < 1.0+pr );
192  auto rz = err.scalar( u3, edge, 2, coord, inpoel, esup, errtype );
193  ensure( "edge error < 0.0", rz > -pr );

195  }
