102 //! \brief Products are arbitrary number of Terms to be multiplied and ensemble
103 //! averaged.
104 //! \details An example is the scalar flux in x direction which needs two terms
105 //! for ensemble averaging: (Y-<Y>) and (U-<U>), then the central moment is
106 //! <yu> = <(Y-<Y>)(U-<U>)>, another example is the third mixed central
107 //! moment of three scalars which needs three terms for ensemble averaging:
108 //! (Y1-<Y1>), (Y2-<Y2>), and (Y3-<Y3>), then the central moment is
109 //! <y1y2y3> = <(Y1-<Y1>)(Y2-<Y2>)(Y3-<Y3>)>.
110 //! \author J. Bakosi
111 using Product = std::vector< Term >;

277 //! \brief Find PDF information, see tk::ctr::PDFInfo
278 //! \note Size of binsizes, names, pdfs, and exts must all be equal
279 //! \note idx must be less than the length of binsizes, names, and pdfs
280 //! \param[in] binsizes Vector of sample space bin sizes for multiple PDFs
281 //! \param[in] names Vector of PDF names
282 //! \param[in] exts Vector of sample space extents. Note: if the user did not
283 //! specify extents for a PDF, the corresponding extents vector still exists
284 //! but it is empty.
285 //! \param[in] pdfs Vector of PDFs
286 //! \param[in] m ORDINARY or CENTRAL PDF we are looking for
287 //! \param[in] idx Index of the PDF within the list of matching (based on D and
288 //! m) PDFs requested
289 //! \return The PDF metadata requested
290 //! \details Find PDF information given the sample space dimension (template
291 //! argument D), ordinary or central PDF (m), and the index within the list of
292 //! matching (based on D and m) PDFs requested (idx). This function must find
293 //! the PDF, if it does not, it throws an exception.
294 //! \see walker::Distributor
295 //! \author J. Bakosi
296 template< std::size_t D >
297 PDFInfo pdfInfo( const std::vector< std::vector< tk::real > >& binsizes,
298  const std::vector< std::string >& names,
299  const std::vector< std::vector< tk::real > >& exts,
300  const std::vector< Probability >& pdfs,
301  tk::ctr::Moment m,
302  std::size_t idx )
303 {
304  Assert( binsizes.size() == names.size(),
305  "Length of binsizes vector and that of PDF names must equal" );
306  Assert( binsizes.size() == pdfs.size(),
307  "Length of binsizes vector and that of PDFs must equal" );
308  Assert( binsizes.size() == exts.size(),
309  "Length of binsizes vector and that of PDF extents must equal" );
310  Assert( binsizes.size() > idx, "Indexing out of bounds" );
311 
312  std::size_t i = 0; // will count all PDFs queried
313  std::size_t n = 0; // will count PDFs with sample space dimensions and type
314  // (orindary or central) we are looking for
315  for (const auto& bs : binsizes) {
316  if ( bs.size() == D &&
317  ((m == Moment::ORDINARY && ordinary(pdfs[i])) ||
318  (m == Moment::CENTRAL && central(pdfs[i]))) ) ++n;
319  if (n == idx+1) {
320  std::vector< std::string > vars;
321  for (const auto& term : pdfs[i])
322  vars.push_back( term.var + std::to_string(term.field+1) );
323  return { names[i], exts[i], std::move(vars) };
324  }
325  ++i;
326  }
327  Throw( "Cannot find PDF." );
328 }

330 //! Extract number of PDFs given sample space dimension
331 //! \details Count number of PDFs given the sample space dimension (template
332 //! argument D) and whether the PDF is ordinary or central (m)
333 //! \note Size of binsizes, names, pdfs, and exts must all be equal
334 //! \param[in] pdfs Vector of PDFs
335 //! \param[in] m ORDINARY or CENTRAL PDF we are looking for
336 //! \return The number of PDFs matchin the criteria discussed above
337 //! \author J. Bakosi
338 template< std::size_t D >
339 std::size_t numPDF( const std::vector< std::vector< tk::real > >& binsizes,
340  const std::vector< Probability >& pdfs,
341  ctr::Moment m )
342 {
343  Assert( binsizes.size() == pdfs.size(),
344  "Length of binsizes vector and that of PDFs must equal" );
345  auto& kind = (m == Moment::ORDINARY ? ordinary : central);
346  std::size_t i=0, n=0;
347  for (const auto& p : pdfs) {
348  const auto& bs = binsizes[i++];
349  if (kind(p) && bs.size() == D) ++n;
350  }
351  return n;
352 }
