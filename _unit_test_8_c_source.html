94 //! Pack/Unpack test runner. This Pack/Unpack method (re-)creates the
95 //! test runner singleton on all processing elements. Therefore we circumvent
96 //! Charm's usual pack/unpack for this type, and thus sizing does not make
97 //! sense: sizing is a no-op. We could initialize the stack in UnitTestDriver's
98 //! constructor and let this function re-create the runner only when unpacking,
99 //! but that leads to repeating the same code twice: once in UnitTestDriver's
100 //! constructor, once here. Another option is to use this pack/unpack routine to
101 //! both initially create (when packing) and to re-create (when unpacking) the
102 //! runner, which eliminates the need for pre-creating the object in
103 //! UnitTestDriver's constructor and therefore eliminates the repeated code.
104 //! This explains the guard for sizing: the code below is called for packing
105 //! only (in serial) and packing and unpacking (in parallel).
106 inline void operator|( PUP::er& p, tut::test_runner_singleton& runner )
107 { if (!p.isSizing()) runner = tut::test_runner_singleton(); }

111 //! \brief Charm++ main chare for the unit test suite executable, unittest.
112 //! \details Note that this object should not be in a namespace.
113 class Main : public CBase_Main {
114 
115  public:
116  //! \brief Constructor
117  //! \details The main chare constructor is the main entry point of the
118  //! program, called by the Charm++ runtime system. The constructor does
119  //! basic initialization steps, e.g., parser the command-line, prints out
120  //! some useful information to screen (in verbose mode), and instantiates
121  //! a driver. Since Charm++ is fully asynchronous, the constructor
122  //! usually spawns asynchronous objects and immediately exits. Thus in the
123  //! body of the main chare constructor we fire up an 'execute' chare,
124  //! which then calls back to Main::execute(). Finishing the main chare
125  //! constructor the Charm++ runtime system then starts the
126  //! network-migration of all global-scope data (if any). The execute chare
127  //! calling back to Main::execute() signals the end of the migration of
128  //! the global-scope data. Then we are ready to execute the driver. Since
129  //! the unit test suite is parallel and asynchronous, its driver fires up
130  //! additional Charm++ chare objects which then call back to
131  //! Main::finalize() at some point in the future when all work has been
132  //! finished. finalize() then exits by calling Charm++'s CkExit(),
133  //! shutting down the runtime system.
134  //! \see http://charm.cs.illinois.edu/manuals/html/charm++/manual.html
135  Main( CkArgMsg* msg )
136  try :
137  m_signal( tk::setSignalHandlers() ),
138  m_helped( false ),
139  m_cmdline(),
140  // Parse command line into m_cmdline using default simple pretty printer
141  m_cmdParser( msg->argc, msg->argv, tk::Print(), m_cmdline, m_helped ),
142  // Create pretty printer initializing output streams based on command line
143  m_print( m_cmdline.get< tag::verbose >() ? std::cout : std::clog ),
144  // Create UnitTest driver
145  m_driver( tk::Main< unittest::UnitTestDriver >
146  ( msg->argc, msg->argv,
147  m_cmdline,
148  tk::HeaderType::UNITTEST,
149  tk::meshconv_executable(),
150  m_print ) ),
151  m_timer(1), // Start new timer measuring the serial+Charm++ runtime
152  m_timestamp()
153  {
154  // Immediately exit if any help was requested; help is printed in main()
155  if (m_helped) CkExit();
156  // Save executable name to global-scope string so FileParser can access it
157  unittest::g_executable = msg->argv[0];
158  delete msg;
159  mainProxy = thisProxy;
160  // Optionally enable quiscence detection
161  if (m_cmdline.get< tag::quiescence >())
162  CkStartQD( CkCallback( CkIndex_Main::quiescence(), thisProxy ) );
163  // Fire up an asynchronous execute object, which when created at some
164  // future point in time will call back to this->execute(). This is
165  // necessary so that this->execute() can access already migrated
166  // global-scope data.
167  CProxy_execute::ckNew();
168  // Start new timer measuring the migration of global-scope data
169  m_timer.emplace_back();
170  } catch (...) { tk::processExceptionCharm(); }

172  void execute() {
173  try {
174  m_timestamp.emplace_back("Migrate global-scope data", m_timer[1].hms());

226 //! \brief UnitTest main()
227 //! \details UnitTest does have a main() function so that we can have tests
228 //! calling MPI functions. Thus we are using Charm++'s MPI-interoperation
229 //! capability as would have to be done with interoperation with an MPI
230 //! library. This is necessary, since MPI_Init() is a bit adamant about
231 //! capturing resources it wants and hence it has to be called before Charm is
232 //! initialized.
233 int main( int argc, char **argv ) {
234 
235  int peid, numpes;
236 
237  #if defined(__clang__)
238  #pragma clang diagnostic push
239  #pragma clang diagnostic ignored "-Wold-style-cast"
240  #endif
241 
242  // Initialize MPI
243  MPI_Init( &argc, &argv );
244  MPI_Comm_rank( MPI_COMM_WORLD, &peid );
245  MPI_Comm_size( MPI_COMM_WORLD, &numpes );
246 
247  // Run serial and Charm++ unit test suite
248  CharmLibInit( MPI_COMM_WORLD, argc, argv );
249  CharmLibExit();
250 
251  bool mpipass = true;
252 
253  // Lambda to compute exit code based on test failures and exit. This is the
254  // single exit point and we must exit from the program.
255  auto stop = [numpes](int pass) {
256  // Combine pass-status from Charm++/serial and MPI suites
257  int mypass = unittest::g_charmpass && pass ? 1 : 0;
258  // Add up every PE's pass status
259  int g = 0;
260  MPI_Allreduce( &mypass, &g, 1, MPI_INT, MPI_SUM, MPI_COMM_WORLD );
261  // Exit code: g==numpes: pass, g<numpes: fail
262  g /= numpes;
263  if (g == 1)
264  MPI_Finalize(); // will pass exit code 0 to shell
265  else
266  MPI_Abort( MPI_COMM_WORLD, tk::ErrCode::FAILURE ); // nonzero exit code
267  // Since this is an MPI program, the exit code passed to exit() does not
268  // matter, however, calling exit() here is important, because we must exit.
269  exit( tk::ErrCode::SUCCESS );
270  };
