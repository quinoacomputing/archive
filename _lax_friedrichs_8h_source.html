20 //! Lax-Friedrichs approximate Riemann solver
21 //! \details This class is used polymorphically with inciter::RiemannSolver
22 struct LaxFriedrichs {
23  //! Lax-Friedrichs approximate Riemann solver flux function
24  //! \param[in] f Face ID
25  //! \param[in] geoFace Face geometry array
26  //! \param[in] u Left and right unknown/state vector
27  //! \return Riemann solution using central difference method
28  std::vector< tk::real >
29  flux( std::size_t f,
30  const tk::Fields& geoFace,
31  const std::array< std::vector< tk::real >, 2 >& u ) const
32  {
33  std::vector< tk::real > flx( u[0].size(), 0.0 ),
34  fluxl( u[0].size(), 0.0 ),
35  fluxr( u[0].size(), 0.0 );
36 
37  std::array< tk::real, 3 > fn {{ geoFace(f,1,0),
38  geoFace(f,2,0),
39  geoFace(f,3,0) }};
40 
41  // ratio of specific heats
42  auto g = g_inputdeck.get< tag::param, tag::compflow, tag::gamma >()[0];
43 
44  // Primitive variables
45  auto rhol = u[0][0];
46  auto rhor = u[1][0];
47 
48  auto pl = (g-1.0)*(u[0][4] - (u[0][1]*u[0][1] +
49  u[0][2]*u[0][2] +
50  u[0][3]*u[0][3]) / (2.0*rhol));
51 
52  auto pr = (g-1.0)*(u[1][4] - (u[1][1]*u[1][1] +
53  u[1][2]*u[1][2] +
54  u[1][3]*u[1][3]) / (2.0*rhor));
55 
56  auto al = sqrt(g * pl / rhol);
57  auto ar = sqrt(g * pr / rhor);
58 
59  // Face-normal velocities
60  auto ul = u[0][1]/rhol;
61  auto vl = u[0][2]/rhol;
62  auto wl = u[0][3]/rhol;
63 
64  tk::real vnl = ul*fn[0] + vl*fn[1] + wl*fn[2];
65 
66  auto ur = u[1][1]/rhor;
67  auto vr = u[1][2]/rhor;
68  auto wr = u[1][3]/rhor;
69 
70  tk::real vnr = ur*fn[0] + vr*fn[1] + wr*fn[2];
71 
72  // Flux functions
73  fluxl[0] = u[0][0] * vnl;
74  fluxl[1] = u[0][1] * vnl + pl*fn[0];
75  fluxl[2] = u[0][2] * vnl + pl*fn[1];
76  fluxl[3] = u[0][3] * vnl + pl*fn[2];
77  fluxl[4] = ( u[0][4] + pl ) * vnl;
78 
79  fluxr[0] = u[1][0] * vnr;
80  fluxr[1] = u[1][1] * vnr + pr*fn[0];
81  fluxr[2] = u[1][2] * vnr + pr*fn[1];
82  fluxr[3] = u[1][3] * vnr + pr*fn[2];
83  fluxr[4] = ( u[1][4] + pr ) * vnr;
84 
85  auto lambda = fmax(al,ar) + fmax(fabs(vnl),fabs(vnr));
86 
87  // Numerical flux function
88  for(ncomp_t c=0; c<5; ++c)
89  {
90  flx[c] = 0.5 * ( fluxl[c] + fluxr[c]
91  - lambda * (u[1][c] - u[0][c]) );
92  }
93 
94  return flx;
95  }
96 
97  //! Flux type accessor
98  //! \return Flux type
99  static ctr::FluxType type() noexcept { return ctr::FluxType::LaxFriedrichs; }
100 };
