148 void
149 Carrier::vol()
150 // *****************************************************************************
151 // Read mesh node coordinates, sum mesh volumes to nodes, and start
152 // communicating them on chare-boundaries
153 //! \author J. Bakosi
154 // *****************************************************************************
155 {
156  // Read coordinates of nodes of the mesh chunk we operate on
157  readCoords();
158 
159  const auto& x = m_coord[0];
160  const auto& y = m_coord[1];
161  const auto& z = m_coord[2];
162 
163  // Compute nodal volumes on our chunk of the mesh
164  for (std::size_t e=0; e<m_inpoel.size()/4; ++e) {
165  const std::array< std::size_t, 4 > N{{ m_inpoel[e*4+0], m_inpoel[e*4+1],
166  m_inpoel[e*4+2], m_inpoel[e*4+3] }};
167  // compute element Jacobi determinant * 5/120 = element volume / 4
168  const std::array< tk::real, 3 >
169  ba{{ x[N[1]]-x[N[0]], y[N[1]]-y[N[0]], z[N[1]]-z[N[0]] }},
170  ca{{ x[N[2]]-x[N[0]], y[N[2]]-y[N[0]], z[N[2]]-z[N[0]] }},
171  da{{ x[N[3]]-x[N[0]], y[N[3]]-y[N[0]], z[N[3]]-z[N[0]] }};
172  const auto J = tk::triple( ba, ca, da ) * 5.0 / 120.0;
173  // scatter add V/4 to nodes
174  for (std::size_t j=0; j<4; ++j) m_vol[N[j]] += J;
175  }
176 
177  // Send our nodal volume contributions to neighbor chares
178  if (m_msum.empty())
179  contribute(
180  CkCallback(CkReductionTarget(Transporter,volcomplete), m_transporter) );
181  else
182  for (const auto& n : m_msum) {
183  std::vector< tk::real > v;
184  for (auto i : n.second) v.push_back( m_vol[ tk::cref_find(m_lid,i) ] );
185  thisProxy[ n.first ].comvol( n.second, v );
186  }
187 }

234 void
235 Carrier::bc()
236 // *****************************************************************************
237 // Extract node IDs from side set node lists and match to user-specified
238 // boundary conditions
239 //! \details Boundary conditions (BC), mathematically speaking, are applied on
240 //! finite surfaces. These finite surfaces are given by element sets. This
241 //! function queries the node lists associated to side set IDs as read in from
242 //! file (old mesh node IDs as in file associated to all side sets in file).
243 //! Then the user-specified boundary conditions, their values and which side
244 //! set they are assigned to, are interrogated and only those nodes and their
245 //! BC values are extracted that we operate on. This BC data structure is then
246 //! sent to the linear system merger which needs to know about this to apply
247 //! BCs before a linear solve. Note that the BC mesh nodes that this function
248 //! results in, stored in dirbc and sent to the linear system merger, only
249 //! contains those nodes that this chare contributes to, i.e., it does not
250 //! contain those BC nodes at which other chares enforce Dirichlet BCs. The
251 //! linear system merger then collects these and communicates to other PEs so
252 //! that BC data held in LinSysMerger::m_bc are the same on all PEs. That is
253 //! the authoritative BC data, which can be queried by LinSysMerger::bc().
254 //! \author J. Bakosi
255 // *****************************************************************************
256 {
257  // Access all side sets from LinSysMerger
258  auto& side = m_linsysmerger.ckLocalBranch()->side();
259 
260  // Invert m_cid, a map associating old node IDs (as in file) to new node IDs
261  // (as in producing contiguous-row-id linear system contributions), so we can
262  // search more efficiently for old node IDs.
263  decltype(m_cid) rcid;
264  for (const auto& i : m_cid) rcid[ i.second ] = i.first;
265 
266  // lambda to find out if we own the old (as in file) global node id
267  auto own = [ &rcid ]( std::size_t id ) -> std::pair< bool, std::size_t > {
268  auto it = rcid.find( id );
269  if (it != end(rcid)) return { true, it->second }; else return { false, 0 };
270  };
271 
272  // lambda to query the Dirichlet BCs on a side set for all components of all
273  // the PDEs integrated
274  auto bcval = []( int sideset ) {
275  // storage for whether the a BC is set and its value for all components of
276  // all PDEs configured to be solved
277  std::vector< std::pair< bool, tk::real > > b;
278  for (const auto& eq : g_pdes) {
279  auto e = eq.dirbc( sideset );
280  b.insert( end(b), begin(e), end(e) );
281  }
282  Assert( b.size() == g_inputdeck.get< tag::component >().nprop(),
283  "The total number of scalar components of all configured PDEs (" +
284  std::to_string( g_inputdeck.get< tag::component >().nprop() ) + ") "
285  "and the sum of the lengths of the Dirichlet BC vectors returned "
286  "from all configured PDE::dirbc() calls (" +
287  std::to_string( b.size() ) + ") does not match" );
288  return b;
289  };
290 
291  // lambda to find out whether 'new' node id is in the list of 'old' node ids
292  // given in s (which contains the old node ids of a side set). Here 'old'
293  // means as in file, while 'new' means as in producing contiguous-row-id
294  // linear system contributions. See also Partitioner.h.
295  auto inset = [ this ]( std::size_t id, const std::vector< std::size_t >& s )
296  -> bool {
297  for (auto n : s) if (tk::cref_find(this->m_cid,id) == n) return true;
298  return false;
299  };
300 
301  // Dirichlet boundary conditions storage: Vector of pairs of bool and boundary
302  // condition value associated to mesh node IDs at which to set Dirichlet
303  // boundary conditions
304  std::unordered_map< std::size_t,
305  std::vector< std::pair< bool, tk::real > > > dirbc;
306 
307  // Collect mesh node IDs we contribute to at which a Dirichlet BC is set. The
308  // data structure in which this information is stored associates a vector of
309  // pairs of bool and BC value to new global mesh node IDs. 'New' as in
310  // producing contiguous-row-id linear system contributions, see also
311  // Partitioner.h. The bool indicates whether the BC value is set at the given
312  // node by the user. The size of the vectors is the number of PDEs integrated
313  // times the number of scalar components in all PDEs. The vector is associated
314  // to global node IDs at which the boundary condition will be set. If a node
315  // gets boundary conditions from multiple side sets, true values (the fact
316  // that a BC is set) overwrite existing false ones, i.e., the union of the
317  // boundary conditions will be applied at the node.
318  for (const auto& s : side) {
319  // get BC values for side set
320  auto b = bcval( s.first );
321  // query if any BC is to be set on the side set for any component of any of
322  // the PDEs integrated
323  bool ubc = std::any_of( b.cbegin(), b.cend(),
324  [](const std::pair< bool, tk::real >& p)
325  { return p.first; } );
326  // for all node IDs we contribute to in the side set associate BC values and
327  // whether they are set for a component (for all PDE components)
328  for (auto o : s.second) {
329  auto n = own(o);
330  if (n.first && inset(n.second,s.second) && ubc) {
331  auto& v = dirbc[ n.second ];
332  v.resize( b.size() );
333  for (std::size_t i=0; i<b.size(); ++i) if (b[i].first) v[i] = b[i];
334  }
335  }
336  }
337 
338  // send progress report to host
339  if ( g_inputdeck.get< tag::cmd, tag::feedback >() )
340  m_transporter.chbcmatched();
341 
342  // Send off list of owned node IDs mapped to side sets to LinSysMerger
343  m_linsysmerger.ckLocalBranch()->charebc( dirbc );
344 }

378 void
379 Carrier::dt()
380 // *****************************************************************************
381 // Start computing minimum time step size
382 //! \author J. Bakosi
383 // *****************************************************************************
384 {
385  tk::real mindt = std::numeric_limits< tk::real >::max();
386 
387  auto const_dt = g_inputdeck.get< tag::discr, tag::dt >();
388  auto def_const_dt = g_inputdeck_defaults.get< tag::discr, tag::dt >();
389  auto eps = std::numeric_limits< tk::real >::epsilon();
390 
391  // use constant dt if configured
392  if (std::abs(const_dt - def_const_dt) > eps) {
393 
394  mindt = const_dt;
395 
396  } else { // compute dt based on CFL
397 
398  // find the minimum dt across all PDEs integrated
399  for (const auto& eq : g_pdes) {
400  auto eqdt = eq.dt( m_coord, m_inpoel, m_u );
401  if (eqdt < mindt) mindt = eqdt;
402  }
403 
404  // Scale smallest dt with CFL coefficient
405  mindt *= g_inputdeck.get< tag::discr, tag::cfl >();
406 
407  }
408 
409  // Contribute to mindt across all Carrier chares
410  contribute( sizeof(tk::real), &mindt, CkReduction::min_double,
411  CkCallback(CkReductionTarget(Transporter,dt), m_transporter) );
412 }

414 void
415 Carrier::lhs()
416 // *****************************************************************************
417 // Compute left-hand side of transport equations
418 //! \author J. Bakosi
419 // *****************************************************************************
420 {
421  // Compute left-hand side matrix for all equations
422  for (const auto& eq : g_pdes)
423  eq.lhs( m_coord, m_inpoel, m_psup, m_lhsd, m_lhso );
424 
425  // Send off left hand side for assembly
426  m_linsysmerger.ckLocalBranch()->
427  charelhs( thisIndex, m_gid, m_psup, m_lhsd, m_lhso );
428 
429  // Compute lumped mass lhs required for the low order solution
430  auto lump = m_fluxcorrector.lump( m_coord, m_inpoel );
431  // Send off lumped mass lhs for assembly
432  m_linsysmerger.ckLocalBranch()->chareauxlhs( thisIndex, m_gid, lump );
433 
434  // send progress report to host
435  if ( g_inputdeck.get< tag::cmd, tag::feedback >() )
436  m_transporter.chlhs();
437 
438  // Call back to Transporter::initcomplete(), signaling that the initialization
439  // is complete and we are now starting time stepping
440  contribute(
441  CkCallback(CkReductionTarget(Transporter,initcomplete), m_transporter));
442 }

444 void
445 Carrier::rhs( tk::real mult, const tk::Fields& sol )
446 // *****************************************************************************
447 // Compute right-hand side of transport equations
448 //! \param[in] mult Multiplier differentiating the different stages in
449 //! multi-stage time stepping
450 //! \param[in] sol Solution vector at current stage
451 //! \author J. Bakosi
452 // *****************************************************************************
453 {
454  // Initialize FCT data structures for new time step stage
455  m_p.fill( 0.0 );
456  m_a.fill( 0.0 );
457  for (std::size_t p=0; p<m_u.nunk(); ++p)
458  for (ncomp_t c=0; c<g_inputdeck.get<tag::component>().nprop(); ++c) {
459  m_q(p,c*2+0,0) = -std::numeric_limits< tk::real >::max();
460  m_q(p,c*2+1,0) = std::numeric_limits< tk::real >::max();
461  }
462 
463  for (auto& b : m_pc) std::fill( begin(b), end(b), 0.0 );
464  for (auto& b : m_ac) std::fill( begin(b), end(b), 0.0 );
465  for (auto& b : m_qc)
466  for (ncomp_t c=0; c<m_u.nprop(); ++c) {
467  b[c*2+0] = -std::numeric_limits< tk::real >::max();
468  b[c*2+1] = std::numeric_limits< tk::real >::max();
469  }
470 
471  // Compute right-hand side vector for all equations
472  tk::Fields r( m_gid.size(), g_inputdeck.get< tag::component >().nprop() );
473  for (const auto& eq : g_pdes) eq.rhs( mult, m_dt, m_coord, m_inpoel, sol, r );
474  // Send off right-hand sides for assembly
475  m_linsysmerger.ckLocalBranch()->charerhs( thisIndex, m_gid, r );
476 
477  // Compute mass diffusion rhs contribution required for the low order solution
478  auto diff = m_fluxcorrector.diff( m_coord, m_inpoel, sol );
479  // Send off mass diffusion rhs contribution for assembly
480  m_linsysmerger.ckLocalBranch()->chareauxrhs( thisIndex, m_gid, diff );
481 
482  // send progress report to host
483  if ( g_inputdeck.get< tag::cmd, tag::feedback >() )
484  m_transporter.chrhs();
485 }
