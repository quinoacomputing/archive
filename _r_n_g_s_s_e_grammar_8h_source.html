27  //! \brief Convert and insert RNGSSE sequence option value to map at position
28  //! given by tags
29  template< template < class > class use, class Option,
30  typename field, typename sel, typename vec, typename tag,
31  typename... tags >
32  struct action< insert_seq< use, Option, field, sel, vec, tag, tags... > > {
33  template< typename Input, typename Stack >
34  static void apply( const Input& in, Stack& stack ) {
35  ctr::RNGSSESeqLen opt;
36  using EnumType = ctr::RNGSSESeqLen::EnumType;
37  // get recently inserted key from <sel,vec>
38  using key_type =
39  typename Stack::template nT< sel >::template nT< vec >::value_type;
40  const key_type& key = stack.template get< sel, vec >().back();
41  // Error out if RNG does not support option specified
42  if ( !ctr::RNG().supportsOpt( key, opt.value(in.string()) ) ) {
43  Message< Stack, ERROR, MsgKey::UNSUPPORTED >( stack, in );
44  }
45  stack.template insert_opt< key_type, field, EnumType, tag, tags... >
46  ( key, opt.value(in.string()) );
47  // trigger error at compile-time if any of the expected option values
48  // is not in the keywords pool of the grammar
49  boost::mpl::for_each< typename Option::keywords >( is_keyword< use >() );
50  }
51  };
