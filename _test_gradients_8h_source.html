121 //! Test nodal gradients for tetrahedron-only mesh
122 template<> template<>
123 void Gradients_object::test< 1 >() {
124  set_test_name( "node gradients of tetrahedra mesh" );
125 
126  // Shift node IDs to start from zero
127  tk::shiftToZero( inpoel );
128 
129  // find out number of points in mesh connectivity
130  auto minmax = std::minmax_element( begin(inpoel), end(inpoel) );
131  Assert( *minmax.first == 0, "node ids should start from zero" );
132  auto npoin = *minmax.second + 1;
133 
134  // Generate elements surrounding points
135  auto esup = tk::genEsup( inpoel, 4 );
136 
137  // generate a constant scalar field
138  tk::Fields uc( npoin, 1 );
139  uc.fill( 1.2 );
140  // test gradients
141  for (std::size_t p=0; p<npoin; ++p) {
142  auto g = nodegrad( p, coord, inpoel, esup, uc, 0 );
143  ensure_equals( "x-gradient of constant field incorrect", g[0], 0, pr );
144  ensure_equals( "y-gradient of constant field incorrect", g[1], 0, pr );
145  ensure_equals( "z-gradient of constant field incorrect", g[2], 0, pr );
146  }
147 
148  // generate a linear scalar field with a slope in only x direction
149  tk::Fields ux( npoin, 1 );
150  for (std::size_t p=0; p<npoin; ++p) ux(p,0,0) = coord[0][p];
151  // test gradients
152  for (std::size_t p=0; p<npoin; ++p) {
153  auto g = nodegrad( p, coord, inpoel, esup, ux, 0 );
154  ensure_equals( "x-gradient of x-linear field incorrect", g[0], 1, pr );
155  ensure_equals( "y-gradient of x-linear field incorrect", g[1], 0, pr );
156  ensure_equals( "z-gradient of x-linear field incorrect", g[2], 0, pr );
157  }
158 
159  // generate a linear scalar field with a slope in only y direction
160  tk::Fields uy( npoin, 1 );
161  for (std::size_t p=0; p<npoin; ++p) uy(p,0,0) = coord[1][p];
162  // test gradients
163  for (std::size_t p=0; p<npoin; ++p) {
164  auto g = nodegrad( p, coord, inpoel, esup, uy, 0 );
165  ensure_equals( "x-gradient of y-linear field incorrect", g[0], 0, pr );
166  ensure_equals( "y-gradient of y-linear field incorrect", g[1], 1, pr );
167  ensure_equals( "z-gradient of y-linear field incorrect", g[2], 0, pr );
168  }
169 
170  // generate a linear scalar field with a slope in only z direction
171  tk::Fields uz( npoin, 1 );
172  for (std::size_t p=0; p<npoin; ++p) uz(p,0,0) = coord[2][p];
173  // test gradients
174  for (std::size_t p=0; p<npoin; ++p) {
175  auto g = nodegrad( p, coord, inpoel, esup, uz, 0 );
176  ensure_equals( "x-gradient of z-linear field incorrect", g[0], 0, pr );
177  ensure_equals( "y-gradient of z-linear field incorrect", g[1], 0, pr );
178  ensure_equals( "z-gradient of z-linear field incorrect", g[2], 1, pr );
179  }
180 
181  // generate linear vector field with different slopes for different components
182  tk::Fields u3( npoin, 3 );
183  for (std::size_t p=0; p<npoin; ++p) {
184  u3(p,0,0) = 2.0*coord[0][p];
185  u3(p,1,0) = 1.5*coord[1][p];
186  u3(p,2,0) = -0.5*coord[2][p];
187  }
188  // test gradients
189  for (std::size_t p=0; p<npoin; ++p) {
190  auto gx = nodegrad( p, coord, inpoel, esup, u3, 0 );
191  ensure_equals( "x-gradient of x-linear vector incorrect", gx[0], 2.0, pr );
192  ensure_equals( "y-gradient of x-linear vector incorrect", gx[1], 0, pr );
193  ensure_equals( "z-gradient of x-linear vector incorrect", gx[2], 0, pr );
194  auto gy = nodegrad( p, coord, inpoel, esup, u3, 1 );
195  ensure_equals( "x-gradient of y-linear vector incorrect", gy[0], 0, pr );
196  ensure_equals( "y-gradient of y-linear vector incorrect", gy[1], 1.5, pr );
197  ensure_equals( "z-gradient of y-linear vector incorrect", gy[2], 0, pr );
198  auto gz = nodegrad( p, coord, inpoel, esup, u3, 2 );

200  ensure_equals( "y-gradient of z-linear vector incorrect", gz[1], 0, pr );

202  }
