61  //! \brief Put option in state at position given by tags
62  //! \details This is simply a wrapper around tk::grm::store_option passing the
63  //! stack defaults.
64  //! \author J. Bakosi
65  template< class Option, typename... tags >
66  struct store_option : pegtl::action_base< store_option< Option, tags... > > {
67  static void apply(const std::string& value, Stack& stack) {
68  tk::grm::store_option< Stack, use, Option, ctr::InputDeck, tags... >
69  ( stack, value, g_inputdeck_defaults );
70  }
71  };
