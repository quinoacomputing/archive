51 // *****************************************************************************
52 {
53  using std::chrono::duration_cast;
54 
55  Dsec elapsed, estimated;
56 
57  if (it == 0) {
58 
59  // First iteration, just return zero
60  elapsed = estimated = clock::duration::zero();
61 
62  } else {
63 
64  // Compute time difference between start and now in seconds
65  elapsed = clock::now() - m_start;
66 
67  // Estimate time until term in seconds
68  Dsec est_term = elapsed * (term-time) / time;
69  // Estimate time until nstep in seconds
70  Dsec est_nstep = elapsed * static_cast<tk::real>(nstep-it) / it;
71 
72  // Time stepping will stop at term or nstep, whichever is sooner
73  estimated = min(est_term, est_nstep);
74 
75  }
