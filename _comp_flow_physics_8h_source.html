41 //! CompFlow system of PDEs problem: Navier-Stokes (viscous flow)
42 //! \details This class adds the viscous force contributions to the momentum and
43 //! energy conservation equations governing compressible flow.
44 class CompFlowPhysicsNavierStokes {
45 
46  public:
47  //! Add viscous stress contribution to momentum and energy rhs
48  //! \param[in] mult Multiplier differentiating the different stages in
49  //! multi-stage time stepping
50  //! \param[in] dt Size of time step
51  //! \param[in] J Element Jacobi determinant
52  //! \param[in] N Element node indices
53  //! \param[in] grad Shape function derivatives, nnode*ndim [4][3]
54  //! \param[in] u Solution at element nodes at recent time step stage
55  //! \param[in] r Pointers to right hand side at component and offset
56  //! \param[in,out] R Right-hand side vector contributing to
57  static void
58  viscousRhs( tk::real mult,
59  tk::real dt,
60  tk::real J,
61  const std::array< std::size_t, 4 >& N,
62  const std::array< std::array< tk::real, 3 >, 4 >& grad,
63  const std::array< std::array< tk::real, 4 >, 5 >& u,
64  const std::array< const tk::real*, 5 >& r,
65  tk::Fields& R )
66  {
67  // dynamic viscosity
68  auto mu = g_inputdeck.get< tag::param, tag::compflow, tag::mu >()[0];
69  // add deviatoric viscous stress contribution to momentum rhs
70  auto c = mult * dt * J/6.0 * mu;
71  for (std::size_t i=0; i<3; ++i)
72  for (std::size_t j=0; j<3; ++j)
73  for (std::size_t k=0; k<4; ++k) {
74  R.var(r[i+1],N[0]) -= c * grad[0][j]*(grad[k][j]*u[i+1][k] +
75  grad[k][i]*u[j+1][k])/u[0][k];
76  R.var(r[i+1],N[1]) -= c * grad[1][j]*(grad[k][j]*u[i+1][k] +
77  grad[k][i]*u[j+1][k])/u[0][k];
78  R.var(r[i+1],N[2]) -= c * grad[2][j]*(grad[k][j]*u[i+1][k] +
79  grad[k][i]*u[j+1][k])/u[0][k];
80  R.var(r[i+1],N[3]) -= c * grad[3][j]*(grad[k][j]*u[i+1][k] +
81  grad[k][i]*u[j+1][k])/u[0][k];
82  }
83  // add isotropic viscous stress contribution to momentum rhs
84  c = mult * dt * J/6.0 * mu * 2.0/3.0;
85  for (std::size_t i=0; i<3; ++i)
86  for (std::size_t j=0; j<3; ++j)
87  for (std::size_t k=0; k<4; ++k) {
88  R.var(r[i+1],N[0]) += c * grad[0][i]*grad[k][j]*u[j+1][k]/u[0][k];
89  R.var(r[i+1],N[1]) += c * grad[1][i]*grad[k][j]*u[j+1][k]/u[0][k];
90  R.var(r[i+1],N[2]) += c * grad[2][i]*grad[k][j]*u[j+1][k]/u[0][k];
91  R.var(r[i+1],N[3]) += c * grad[3][i]*grad[k][j]*u[j+1][k]/u[0][k];
92  }
93  // add deviatoric viscous stress contribution to energy rhs
94  c = mult * dt * J/24.0 * mu;
95  for (std::size_t i=0; i<3; ++i)
96  for (std::size_t j=0; j<3; ++j)
97  for (std::size_t k=0; k<4; ++k) {
98  R.var(r[4],N[0]) -= c * u[i+1][k]/u[0][k] *
99  grad[0][j]*(grad[k][j]*u[i+1][k] +
100  grad[k][i]*u[j+1][k])/u[0][k];
101  R.var(r[4],N[1]) -= c * u[i+1][k]/u[0][k] *
102  grad[1][j]*(grad[k][j]*u[i+1][k] +
103  grad[k][i]*u[j+1][k])/u[0][k];
104  R.var(r[4],N[2]) -= c * u[i+1][k]/u[0][k] *
105  grad[2][j]*(grad[k][j]*u[i+1][k] +
106  grad[k][i]*u[j+1][k])/u[0][k];
107  R.var(r[4],N[3]) -= c * u[i+1][k]/u[0][k] *
108  grad[3][j]*(grad[k][j]*u[i+1][k] +
109  grad[k][i]*u[j+1][k])/u[0][k];
110  }
111  // add isotropic viscous stress contribution to energy rhs
112  c = mult * dt * J/24.0 * mu * 2.0/3.0;
113  for (std::size_t i=0; i<3; ++i)
114  for (std::size_t j=0; j<3; ++j)
115  for (std::size_t k=0; k<4; ++k) {
116  R.var(r[4],N[0]) += c * u[i+1][k]/u[0][k] *
117  grad[0][i]*grad[k][j]*u[j+1][k]/u[0][k];
118  R.var(r[4],N[1]) += c * u[i+1][k]/u[0][k] *
119  grad[1][i]*grad[k][j]*u[j+1][k]/u[0][k];
120  R.var(r[4],N[2]) += c * u[i+1][k]/u[0][k] *
121  grad[2][i]*grad[k][j]*u[j+1][k]/u[0][k];
122  R.var(r[4],N[3]) += c * u[i+1][k]/u[0][k] *
123  grad[3][i]*grad[k][j]*u[j+1][k]/u[0][k];
124  }
125  }

127  //! Compute the minimum time step size based on the viscous force
128  //! \param[in] L Characteristic length scale
129  //! \param[in] u Solution at element nodes at recent time step stage
130  //! \return Minimum time step size based on viscous force
131  static tk::real
132  viscous_dt( tk::real L,
133  const std::array< std::array< tk::real, 4 >, 5 >& u )
134  {
135  // dynamic viscosity
136  auto mu = g_inputdeck.get< tag::param, tag::compflow, tag::mu >()[0];
137  // compute the minimum viscous time step size across the four nodes
138  tk::real mindt = std::numeric_limits< tk::real >::max();
139  for (std::size_t j=0; j<4; ++j) {
140  auto& r = u[0][j]; // rho
141  auto dt = L * L * r / mu; // dt ~ dx^2/nu = dx^2*rho/mu
142  if (dt < mindt) mindt = dt;
143  }
144  return mindt;
145  }

147  //! Add heat conduction contribution to energy rhs
148  //! \param[in] mult Multiplier differentiating the different stages in
149  //! multi-stage time stepping
150  //! \param[in] dt Size of time step
151  //! \param[in] J Element Jacobi determinant
152  //! \param[in] N Element node indices
153  //! \param[in] grad Shape function derivatives, nnode*ndim [4][3]
154  //! \param[in] u Solution at element nodes at recent time step stage
155  //! \param[in] r Pointers to right hand side at component and offset
156  //! \param[in,out] R Right-hand side vector contributing to
157  static void
158  conductRhs( tk::real mult,
159  tk::real dt,
160  tk::real J,
161  const std::array< std::size_t, 4 >& N,
162  const std::array< std::array< tk::real, 3 >, 4 >& grad,
163  const std::array< std::array< tk::real, 4 >, 5 >& u,
164  const std::array< const tk::real*, 5 >& r,
165  tk::Fields& R )
166  {
167  // specific heat at constant volume
168  auto cv = g_inputdeck.get< tag::param, tag::compflow, tag::cv >()[0];
169  // thermal conductivity
170  auto kc = g_inputdeck.get< tag::param, tag::compflow, tag::k >()[0];
171  // compute temperature
172  std::array< tk::real, 4 > T;
173  for (std::size_t i=0; i<4; ++i)
174  T[i] = cv*(u[4][i] - (u[1][i]*u[1][i] +
175  u[2][i]*u[2][i] +
176  u[3][i]*u[3][i])/2.0/u[0][i]) / u[0][i];
177  // add heat conduction contribution to energy rhs
178  auto c = mult * dt * J/24.0 * kc;
179  for (std::size_t i=0; i<3; ++i)
180  for (std::size_t k=0; k<4; ++k) {
181  R.var(r[4],N[0]) += c * grad[k][i] * T[k];
182  R.var(r[4],N[1]) += c * grad[k][i] * T[k];
183  R.var(r[4],N[2]) += c * grad[k][i] * T[k];
184  R.var(r[4],N[3]) += c * grad[k][i] * T[k];
185  }
186  }
187 
188  //! Compute the minimum time step size based on thermal diffusion
189  //! \param[in] L Characteristic length scale
190  //! \param[in] u Solution at element nodes at recent time step stage
191  //! \return Minimum time step size based on thermal diffusion
192  static tk::real
193  heat_diffusion_dt( tk::real L,
194  const std::array< std::array< tk::real, 4 >, 5 >& u )
195  {
196  // ratio of specific heats
197  auto g = g_inputdeck.get< tag::param, tag::compflow, tag::gamma >()[0];
198  // specific heat at constant volume
199  auto cv = g_inputdeck.get< tag::param, tag::compflow, tag::cv >()[0];
200  // thermal conductivity
201  auto k = g_inputdeck.get< tag::param, tag::compflow, tag::k >()[0];
202  // specific heat at constant pressure
203  auto cp = g * cv;
204  // compute the minimum conduction time step size across the four nodes
205  tk::real mindt = std::numeric_limits< tk::real >::max();
206  for (std::size_t j=0; j<4; ++j) {
207  auto& r = u[0][j]; // rho
208  auto dt = L * L * r * cp / k; // dt ~ dx^2/alpha = dx^2*rho*C_p/k
209  if (dt < mindt) mindt = dt;
210  }
211  return mindt;
212  }
