56  //! Compute the left hand side sparse matrix
57  //! \param[in] coord Mesh node coordinates
58  //! \param[in] inpoel Mesh element connectivity
59  //! \param[in] psup Linked lists storing IDs of points surrounding points
60  //! \param[in,out] lhsd Diagonal of the sparse matrix storing nonzeros
61  //! \param[in,out] lhso Off-diagonal of the sparse matrix storing nonzeros
62  //! \details Sparse matrix storing the nonzero matrix values at rows and
63  //! columns given by psup. The format is similar to compressed row
64  //! storage, but the diagonal and off-diagonal data are stored in separate
65  //! vectors. For the off-diagonal data the local row and column indices,
66  //! at which values are nonzero, are stored by psup (psup1 and psup2,
67  //! where psup2 holds the indices at which psup1 holds the point ids
68  //! surrounding points, see also tk::genPsup()). Note that the number of
69  //! mesh points (our chunk) npoin = psup.second.size()-1.
70  void lhs( const std::array< std::vector< tk::real >, 3 >& coord,
71  const std::vector< std::size_t >& inpoel,
72  const std::pair< std::vector< std::size_t >,
73  std::vector< std::size_t > >& psup,
74  tk::Fields& lhsd,
75  tk::Fields& lhso ) const
76  {
77  Assert( psup.second.size()-1 == coord[0].size(),
78  "Number of mesh points and number of global IDs unequal" );
79  Assert( lhsd.nunk() == psup.second.size()-1, "Number of unknowns in "
80  "diagonal sparse matrix storage incorrect" );
81  Assert( lhso.nunk() == psup.first.size(), "Number of unknowns in "
82  "off-diagonal sparse matrix storage incorrect" );
83 
84  // Lambda to compute the sparse matrix vector index for row and column
85  // indices. Used only for off-diagonal entries.
86  auto spidx = [ &psup ]( std::size_t r, std::size_t c ) -> std::size_t {
87  Assert( r != c, "Only for computing the off-diagonal indices" );
88  for (auto i=psup.second[r]+1; i<=psup.second[r+1]; ++i)
89  if (c == psup.first[i]) return i;
90  Throw( "Cannot find row, column: " + std::to_string(r) + ',' +
91  std::to_string(c) + " in sparse matrix" );
92  };

94  const auto& x = coord[0];
95  const auto& y = coord[1];
96  const auto& z = coord[2];

98  // Zero matrix for all components
99  for (ncomp_t c=0; c<5; ++c) {
100  lhsd.fill( c, m_offset, 0.0 );
101  lhso.fill( c, m_offset, 0.0 );
102  }

104  for (std::size_t e=0; e<inpoel.size()/4; ++e) {
105  const auto A = inpoel[e*4+0];
106  const auto B = inpoel[e*4+1];
107  const auto C = inpoel[e*4+2];
108  const auto D = inpoel[e*4+3];

110  ca{{ x[C]-x[A], y[C]-y[A], z[C]-z[A] }},
111  da{{ x[D]-x[A], y[D]-y[A], z[D]-z[A] }};
112  const auto J = tk::triple( ba, ca, da ) / 120.0;
113  Assert( J > 0, "Element Jacobian non-positive" );
114 
115  for (ncomp_t c=0; c<5; ++c) {
116  const auto r = lhsd.cptr( c, m_offset );
117  lhsd.var( r, A ) += 2.0 * J;
118  lhsd.var( r, B ) += 2.0 * J;
119  lhsd.var( r, C ) += 2.0 * J;
120  lhsd.var( r, D ) += 2.0 * J;
121 
122  const auto s = lhso.cptr( c, m_offset );
123  lhso.var( s, spidx(A,B) ) += J;
124  lhso.var( s, spidx(A,C) ) += J;
125  lhso.var( s, spidx(A,D) ) += J;
126 
127  lhso.var( s, spidx(B,A) ) += J;
128  lhso.var( s, spidx(B,C) ) += J;
129  lhso.var( s, spidx(B,D) ) += J;
130 
131  lhso.var( s, spidx(C,A) ) += J;
132  lhso.var( s, spidx(C,B) ) += J;
133  lhso.var( s, spidx(C,D) ) += J;
134 
135  lhso.var( s, spidx(D,A) ) += J;
136  lhso.var( s, spidx(D,B) ) += J;
137  lhso.var( s, spidx(D,C) ) += J;
138  }
139  }

142  //! Compute right hand side
143  //! \param[in] t Physical time
144  //! \param[in] deltat Size of time step
145  //! \param[in] coord Mesh node coordinates
146  //! \param[in] inpoel Mesh element connectivity
147  //! \param[in] U Solution vector at recent time step
148  //! \param[in,out] R Right-hand side vector computed
149  void rhs( tk::real t,
150  tk::real deltat,
151  const std::array< std::vector< tk::real >, 3 >& coord,
152  const std::vector< std::size_t >& inpoel,
153  const tk::Fields& U,
154  tk::Fields& Ue,
155  tk::Fields& R ) const
156  {
157  Assert( U.nunk() == coord[0].size(), "Number of unknowns in solution "
158  "vector at recent time step incorrect" );
159  Assert( R.nunk() == coord[0].size() && R.nprop() == 5,
160  "Number of unknowns and/or number of components in right-hand "
161  "side vector incorrect" );
162  Assert( U.nprop() == 5,
163  "Number of components in solution vector must be 5" );
164  Assert( R.nprop() == 5, "Number of components in rhs must be 5" );
165 
166  const auto& x = coord[0];
167  const auto& y = coord[1];
168  const auto& z = coord[2];
169 
170  // ratio of specific heats
171  auto g = g_inputdeck.get< tag::param, tag::compflow, tag::gamma >()[0];
172 
173  // 1st stage: update element values from node values (gather-add)
174  for (std::size_t e=0; e<inpoel.size()/4; ++e) {
175 
176  // access node IDs
177  const std::array< std::size_t, 4 > N{{ inpoel[e*4+0], inpoel[e*4+1],
178  inpoel[e*4+2], inpoel[e*4+3] }};
179  // compute element Jacobi determinant
180  const std::array< tk::real, 3 >
181  ba{{ x[N[1]]-x[N[0]], y[N[1]]-y[N[0]], z[N[1]]-z[N[0]] }},
182  ca{{ x[N[2]]-x[N[0]], y[N[2]]-y[N[0]], z[N[2]]-z[N[0]] }},
183  da{{ x[N[3]]-x[N[0]], y[N[3]]-y[N[0]], z[N[3]]-z[N[0]] }};
184  const auto J = tk::triple( ba, ca, da ); // J = 6V
185  Assert( J > 0, "Element Jacobian non-positive" );
186 
187  // shape function derivatives, nnode*ndim [4][3]
188  std::array< std::array< tk::real, 3 >, 4 > grad;
189  grad[1] = tk::crossdiv( ca, da, J );
190  grad[2] = tk::crossdiv( da, ba, J );
191  grad[3] = tk::crossdiv( ba, ca, J );
192  for (std::size_t i=0; i<3; ++i)
193  grad[0][i] = -grad[1][i]-grad[2][i]-grad[3][i];
194 
195  // access solution at element nodes
196  std::array< std::array< tk::real, 4 >, 5 > u;
197  for (ncomp_t c=0; c<5; ++c) u[c] = U.extract( c, m_offset, N );
198  // access solution at elements
199  std::array< const tk::real*, 5 > ue;
200  for (ncomp_t c=0; c<5; ++c) ue[c] = Ue.cptr( c, m_offset );
201 
202  // pressure
203  std::array< tk::real, 4 > p;
204  for (std::size_t a=0; a<4; ++a)
205  p[a] = (g-1.0)*(u[4][a] - (u[1][a]*u[1][a] +
206  u[2][a]*u[2][a] +
207  u[3][a]*u[3][a])/2.0/u[0][a]);
208 
209  // sum nodal averages to element
210  for (ncomp_t c=0; c<5; ++c) {
211  Ue.var(ue[c],e) = 0.0;
212  for (std::size_t a=0; a<4; ++a)
213  Ue.var(ue[c],e) += u[c][a]/4.0;
214  }
215 
216  // sum flux contributions to element
217  tk::real d = deltat/2.0;
218  for (std::size_t j=0; j<3; ++j)
219  for (std::size_t a=0; a<4; ++a) {
220  // mass: advection
221  Ue.var(ue[0],e) -= d * grad[a][j] * u[j+1][a];
222  // momentum: advection
223  for (std::size_t i=0; i<3; ++i)
224  Ue.var(ue[i+1],e) -= d * grad[a][j] * u[j+1][a]*u[i+1][a]/u[0][a];
225  // momentum: pressure
226  Ue.var(ue[j+1],e) -= d * grad[a][j] * p[a];
227  // energy: advection and pressure
228  Ue.var(ue[4],e) -= d * grad[a][j] *
229  (u[4][a] + p[a]) * u[j+1][a]/u[0][a];
230  }
231 
232  // add (optional) source to all equations
233  std::array< std::array< tk::real, 5 >, 4 > s{{
234  Problem::src( 0, x[N[0]], y[N[0]], z[N[0]], t ),
235  Problem::src( 0, x[N[1]], y[N[1]], z[N[1]], t ),
236  Problem::src( 0, x[N[2]], y[N[2]], z[N[2]], t ),
237  Problem::src( 0, x[N[3]], y[N[3]], z[N[3]], t ) }};
238  for (std::size_t c=0; c<5; ++c)
239  for (std::size_t a=0; a<4; ++a)
240  Ue.var(ue[c],e) += d/4.0 * s[a][c];
241 
242  }
243 
244 
245  // zero right hand side for all components
246  for (ncomp_t c=0; c<5; ++c) R.fill( c, m_offset, 0.0 );
247 
248  // 2nd stage: form rhs from element values (scatter-add)
249  for (std::size_t e=0; e<inpoel.size()/4; ++e) {
250 
251  // access node IDs
252  const std::array< std::size_t, 4 > N{{ inpoel[e*4+0], inpoel[e*4+1],
253  inpoel[e*4+2], inpoel[e*4+3] }};
254  // compute element Jacobi determinant
255  const std::array< tk::real, 3 >
256  ba{{ x[N[1]]-x[N[0]], y[N[1]]-y[N[0]], z[N[1]]-z[N[0]] }},
257  ca{{ x[N[2]]-x[N[0]], y[N[2]]-y[N[0]], z[N[2]]-z[N[0]] }},
258  da{{ x[N[3]]-x[N[0]], y[N[3]]-y[N[0]], z[N[3]]-z[N[0]] }};
259  const auto J = tk::triple( ba, ca, da ); // J = 6V
260  Assert( J > 0, "Element Jacobian non-positive" );
261 
262  // shape function derivatives, nnode*ndim [4][3]
263  std::array< std::array< tk::real, 3 >, 4 > grad;
264  grad[1] = tk::crossdiv( ca, da, J );
265  grad[2] = tk::crossdiv( da, ba, J );
266  grad[3] = tk::crossdiv( ba, ca, J );
267  for (std::size_t i=0; i<3; ++i)
268  grad[0][i] = -grad[1][i]-grad[2][i]-grad[3][i];
269 
270  // access solution at elements
271  std::array< tk::real, 5 > ue;
272  for (ncomp_t c=0; c<5; ++c) ue[c] = Ue( e, c, m_offset );
273  // access pointer to right hand side at component and offset
274  std::array< const tk::real*, 5 > r;
275  for (ncomp_t c=0; c<5; ++c) r[c] = R.cptr( c, m_offset );
276 
277  // pressure
278  auto p = (g-1.0)*(ue[4] -
279  (ue[1]*ue[1] + ue[2]*ue[2] + ue[3]*ue[3])/2.0/ue[0]);
280 
281  // scatter-add flux contributions to rhs at nodes
282  tk::real d = deltat * J/6.0;
283  for (std::size_t j=0; j<3; ++j)
284  for (std::size_t a=0; a<4; ++a) {
285  // mass: advection
286  R.var(r[0],N[a]) += d * grad[a][j] * ue[j+1];
287  // momentum: advection
288  for (std::size_t i=0; i<3; ++i)
289  R.var(r[i+1],N[a]) += d * grad[a][j] * ue[j+1]*ue[i+1]/ue[0];
290  // momentum: pressure
291  R.var(r[j+1],N[a]) += d * grad[a][j] * p;
292  // energy: advection and pressure
293  R.var(r[4],N[a]) += d * grad[a][j] * (ue[4] + p) * ue[j+1]/ue[0];
294  }
295 
296  // add (optional) source to all equations
297  auto xc = (x[N[0]] + x[N[1]] + x[N[2]] + x[N[3]]) / 4.0;
298  auto yc = (y[N[0]] + y[N[1]] + y[N[2]] + y[N[3]]) / 4.0;
299  auto zc = (z[N[0]] + z[N[1]] + z[N[2]] + z[N[3]]) / 4.0;
300  auto s = Problem::src( 0, xc, yc, zc, t+deltat/2 );
301  for (std::size_t c=0; c<5; ++c)
302  for (std::size_t a=0; a<4; ++a)
303  R.var(r[c],N[a]) += d/4.0 * s[c];
304 
305  }

308 // // add heat conduction contribution to energy rhs
309 // Physics::conductRhs( deltat, J, N, grad, u, r, R );
310  }

312  //! Compute the minimum time step size
313  //! \param[in] U Solution vector at recent time step
314  //! \param[in] coord Mesh node coordinates
315  //! \param[in] inpoel Mesh element connectivity
316  //! \return Minimum time step size
317  tk::real dt( const std::array< std::vector< tk::real >, 3 >& coord,
318  const std::vector< std::size_t >& inpoel,
319  const tk::Fields& U ) const
320  {
321  Assert( U.nunk() == coord[0].size(), "Number of unknowns in solution "
322  "vector at recent time step incorrect" );
323  const auto& x = coord[0];
324  const auto& y = coord[1];
325  const auto& z = coord[2];
326  // ratio of specific heats
327  auto g = g_inputdeck.get< tag::param, tag::compflow, tag::gamma >()[0];
328  // compute the minimum dt across all elements we own
329  tk::real mindt = std::numeric_limits< tk::real >::max();
330  for (std::size_t e=0; e<inpoel.size()/4; ++e) {
331  const std::array< std::size_t, 4 > N{{ inpoel[e*4+0], inpoel[e*4+1],
332  inpoel[e*4+2], inpoel[e*4+3] }};
333  // compute cubic root of element volume as the characteristic length
334  const std::array< tk::real, 3 >
335  ba{{ x[N[1]]-x[N[0]], y[N[1]]-y[N[0]], z[N[1]]-z[N[0]] }},
336  ca{{ x[N[2]]-x[N[0]], y[N[2]]-y[N[0]], z[N[2]]-z[N[0]] }},
337  da{{ x[N[3]]-x[N[0]], y[N[3]]-y[N[0]], z[N[3]]-z[N[0]] }};
338  const auto L = std::cbrt( tk::triple( ba, ca, da ) / 6.0 );
339  // access solution at element nodes at recent time step
340  std::array< std::array< tk::real, 4 >, 5 > u;
341  for (ncomp_t c=0; c<5; ++c) u[c] = U.extract( c, m_offset, N );
342  // compute the maximum length of the characteristic velocity (fluid
343  // velocity + sound velocity) across the four element nodes
344  tk::real maxvel = 0.0;
345  for (std::size_t j=0; j<4; ++j) {
346  auto& r = u[0][j]; // rho
347  auto& ru = u[1][j]; // rho * u
348  auto& rv = u[2][j]; // rho * v
349  auto& rw = u[3][j]; // rho * w
350  auto& re = u[4][j]; // rho * e
351  auto p = (g-1.0)*(re - (ru*ru + rv*rv + rw*rw)/2.0/r); // pressure
352  if (p < 0) p = 0.0;
353  auto c = std::sqrt(g*p/r); // sound speed
354  auto v = std::sqrt((ru*ru + rv*rv + rw*rw)/r/r) + c; // char. velocity
355  if (v > maxvel) maxvel = v;
356  }
357  // compute element dt for the Euler equations
358  auto euler_dt = L / maxvel;
359  // compute element dt based on the viscous force
360  auto viscous_dt = Physics::viscous_dt( L, u );
361  // compute element dt based on thermal diffusion
362  auto conduct_dt = Physics::conduct_dt( L, u );
363  // compute minimum element dt
364  auto elemdt = std::min( euler_dt, std::min( viscous_dt, conduct_dt ) );
365  // find minimum dt across all elements
366  if (elemdt < mindt) mindt = elemdt;
367  }
368  return mindt;
369  }

371  //! Extract the velocity field at cell nodes
372  //! \param[in] U Solution vector at recent time step
373  //! \param[in] N Element node indices
374  //! \return Array of the four values of the velocity field
375  std::array< std::array< tk::real, 4 >, 3 >

377  const std::array< std::vector< tk::real >, 3 >&,
378  const std::array< std::size_t, 4 >& N ) const
379  {

386  []( tk::real s, tk::real& d ){ return d /= s; } );
387  std::transform( r.begin(), r.end(), v[1].begin(), v[1].begin(),
388  []( tk::real s, tk::real& d ){ return d /= s; } );
389  std::transform( r.begin(), r.end(), v[2].begin(), v[2].begin(),
390  []( tk::real s, tk::real& d ){ return d /= s; } );
