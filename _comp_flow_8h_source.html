64  //! Compute the left hand side sparse matrix
65  //! \param[in] coord Mesh node coordinates
66  //! \param[in] inpoel Mesh element connectivity
67  //! \param[in] psup Linked lists storing IDs of points surrounding points
68  //! \param[in,out] lhsd Diagonal of the sparse matrix storing nonzeros
69  //! \param[in,out] lhso Off-diagonal of the sparse matrix storing nonzeros
70  //! \details Sparse matrix storing the nonzero matrix values at rows and
71  //! columns given by psup. The format is similar to compressed row
72  //! storage, but the diagonal and off-diagonal data are stored in separate
73  //! vectors. For the off-diagonal data the local row and column indices,
74  //! at which values are nonzero, are stored by psup (psup1 and psup2,
75  //! where psup2 holds the indices at which psup1 holds the point ids
76  //! surrounding points, see also tk::genPsup()). Note that the number of
77  //! mesh points (our chunk) npoin = psup.second.size()-1.
78  //! \author J. Bakosi
79  void lhs( const std::array< std::vector< tk::real >, 3 >& coord,
80  const std::vector< std::size_t >& inpoel,
81  const std::pair< std::vector< std::size_t >,
82  std::vector< std::size_t > >& psup,
83  tk::Fields& lhsd,
84  tk::Fields& lhso ) const
85  {
86  Assert( psup.second.size()-1 == coord[0].size(),
87  "Number of mesh points and number of global IDs unequal" );
88  Assert( lhsd.nunk() == psup.second.size()-1, "Number of unknowns in "
89  "diagonal sparse matrix storage incorrect" );
90  Assert( lhso.nunk() == psup.first.size(), "Number of unknowns in "
91  "off-diagonal sparse matrix storage incorrect" );
92 
93  // Lambda to compute the sparse matrix vector index for row and column
94  // indices. Used only for off-diagonal entries.
95  auto spidx = [ &psup ]( std::size_t r, std::size_t c ) -> std::size_t {
96  Assert( r != c, "Only for computing the off-diagonal indices" );
97  for (auto i=psup.second[r]+1; i<=psup.second[r+1]; ++i)
98  if (c == psup.first[i]) return i;
99  Throw( "Cannot find row, column: " + std::to_string(r) + ',' +
100  std::to_string(c) + " in sparse matrix" );
101  };
102 
103  const auto& x = coord[0];
104  const auto& y = coord[1];
105  const auto& z = coord[2];
106 
107  // Zero matrix for all components
108  for (ncomp_t c=0; c<5; ++c) {
109  lhsd.fill( c, m_offset, 0.0 );
110  lhso.fill( c, m_offset, 0.0 );
111  }
112 
113  for (std::size_t e=0; e<inpoel.size()/4; ++e) {
114  const auto A = inpoel[e*4+0];
115  const auto B = inpoel[e*4+1];
116  const auto C = inpoel[e*4+2];
117  const auto D = inpoel[e*4+3];
118  std::array< tk::real, 3 > ba{{ x[B]-x[A], y[B]-y[A], z[B]-z[A] }},
119  ca{{ x[C]-x[A], y[C]-y[A], z[C]-z[A] }},
120  da{{ x[D]-x[A], y[D]-y[A], z[D]-z[A] }};
121  const auto J = tk::triple( ba, ca, da ) / 120.0;
122  Assert( J > 0, "Element Jacobian non-positive" );
123 
124  for (ncomp_t c=0; c<5; ++c) {
125  const auto r = lhsd.cptr( c, m_offset );
126  lhsd.var( r, A ) += 2.0 * J;
127  lhsd.var( r, B ) += 2.0 * J;
128  lhsd.var( r, C ) += 2.0 * J;
129  lhsd.var( r, D ) += 2.0 * J;
130 
131  const auto s = lhso.cptr( c, m_offset );
132  lhso.var( s, spidx(A,B) ) += J;
133  lhso.var( s, spidx(A,C) ) += J;
134  lhso.var( s, spidx(A,D) ) += J;
135 
136  lhso.var( s, spidx(B,A) ) += J;
137  lhso.var( s, spidx(B,C) ) += J;
138  lhso.var( s, spidx(B,D) ) += J;
139 
140  lhso.var( s, spidx(C,A) ) += J;
141  lhso.var( s, spidx(C,B) ) += J;
142  lhso.var( s, spidx(C,D) ) += J;
143 
144  lhso.var( s, spidx(D,A) ) += J;
145  lhso.var( s, spidx(D,B) ) += J;
146  lhso.var( s, spidx(D,C) ) += J;
147  }
148  }

151  //! Compute right hand side
152  //! \param[in] t Physical time
153  //! \param[in] deltat Size of time step
154  //! \param[in] coord Mesh node coordinates
155  //! \param[in] inpoel Mesh element connectivity
156  //! \param[in] U Solution vector at recent time step stage
157  //! \param[in,out] R Right-hand side vector computed
158  //! \author J. Bakosi
159  void rhs( tk::real t,
160  tk::real deltat,
161  const std::array< std::vector< tk::real >, 3 >& coord,
162  const std::vector< std::size_t >& inpoel,
163  const tk::Fields& U,
164  tk::Fields& R ) const
165  {
166  Assert( U.nunk() == coord[0].size(), "Number of unknowns in solution "
167  "vector at recent time step incorrect" );
168  Assert( R.nunk() == coord[0].size() && R.nprop() == 5,
169  "Number of unknowns and/or number of components in right-hand "
170  "side vector incorrect" );
171 
172  const auto& x = coord[0];
173  const auto& y = coord[1];
174  const auto& z = coord[2];
175 
176  // zero right hand side for all components
177  for (ncomp_t c=0; c<5; ++c) R.fill( c, m_offset, 0.0 );
178 
179  // ratio of specific heats
180  auto g = g_inputdeck.get< tag::param, tag::compflow, tag::gamma >()[0];
181 
182  for (std::size_t e=0; e<inpoel.size()/4; ++e) {
183  const std::array< std::size_t, 4 > N{{ inpoel[e*4+0], inpoel[e*4+1],
184  inpoel[e*4+2], inpoel[e*4+3] }};
185  // compute element Jacobi determinant
186  const std::array< tk::real, 3 >
187  ba{{ x[N[1]]-x[N[0]], y[N[1]]-y[N[0]], z[N[1]]-z[N[0]] }},
188  ca{{ x[N[2]]-x[N[0]], y[N[2]]-y[N[0]], z[N[2]]-z[N[0]] }},
189  da{{ x[N[3]]-x[N[0]], y[N[3]]-y[N[0]], z[N[3]]-z[N[0]] }};
190  const auto J = tk::triple( ba, ca, da ); // J = 6V
191  Assert( J > 0, "Element Jacobian non-positive" );
192 
193  // consistent mass, nnode*nnode [4][4]
194  std::array< std::array< tk::real, 4 >, 4 > mass;
195  mass[0][0] = mass[1][1] = mass[2][2] = mass[3][3] = J/60.0; // diagonal
196  mass[0][1] = mass[0][2] = mass[0][3] = // off-diagonal
197  mass[1][0] = mass[1][2] = mass[1][3] =
198  mass[2][0] = mass[2][1] = mass[2][3] =
199  mass[3][0] = mass[3][1] = mass[3][2] = J/120.0;
200 
201  // shape function derivatives, nnode*ndim [4][3]
202  std::array< std::array< tk::real, 3 >, 4 > grad;
203  grad[1] = tk::crossdiv( ca, da, J );
204  grad[2] = tk::crossdiv( da, ba, J );
205  grad[3] = tk::crossdiv( ba, ca, J );
206  for (std::size_t i=0; i<3; ++i)
207  grad[0][i] = -grad[1][i]-grad[2][i]-grad[3][i];
208 
209  // access solution at element nodes at recent time step stage
210  std::array< std::array< tk::real, 4 >, 5 > u;
211  for (ncomp_t c=0; c<5; ++c) u[c] = U.extract( c, m_offset, N );
212  // access pointer to right hand side at component and offset
213  std::array< const tk::real*, 5 > r;
214  for (ncomp_t c=0; c<5; ++c) r[c] = R.cptr( c, m_offset );
215 
216  // compute pressure
217  std::array< tk::real, 4 > p;
218  for (std::size_t i=0; i<4; ++i)
219  p[i] = (g-1.0)*(u[4][i] - (u[1][i]*u[1][i] +
220  u[2][i]*u[2][i] +
221  u[3][i]*u[3][i])/2.0/u[0][i]);
222 
223  // scatter-add mass, momentum, and energy contributions to rhs
224  tk::real c = deltat * J/24.0;
225  for (std::size_t j=0; j<3; ++j)
226  for (std::size_t alpha=0; alpha<4; ++alpha)
227  for (std::size_t beta=0; beta<4; ++beta) {
228  // advection contribution to mass rhs
229  R.var(r[0],N[alpha]) -= c * grad[beta][j] * u[j+1][beta];
230  // advection contribution to momentum rhs
231  for (std::size_t k=0; k<3; ++k)
232  R.var(r[k+1],N[alpha]) -=
233  c * grad[beta][j] * u[k+1][beta]*u[j+1][beta]/u[0][beta];
234  // pressure gradient contribution to momentum rhs
235  R.var(r[j+1],N[alpha]) -= c * grad[beta][j] * p[beta];
236  // advection and pressure gradient contribution to energy rhs
237  R.var(r[4],N[alpha]) -= c * grad[beta][j] *
238  (u[4][beta] + p[beta]) * u[j+1][beta]/u[0][beta];
239  }
240 
241  // add viscous stress contribution to momentum and energy rhs
242  Physics::viscousRhs( deltat, J, N, grad, u, r, R );
243  // add heat conduction contribution to energy rhs
244  Physics::conductRhs( deltat, J, N, grad, u, r, R );
245  // add source to rhs for all equations
246  Problem::sourceRhs( t, coord, 0, deltat, J, N, mass, grad, r, u, R );
247  }
248  }

250  //! Compute the minimum time step size
251  //! \param[in] U Solution vector at recent time step stage
252  //! \param[in] coord Mesh node coordinates
253  //! \param[in] inpoel Mesh element connectivity
254  //! \return Minimum time step size
255  tk::real dt( const std::array< std::vector< tk::real >, 3 >& coord,
256  const std::vector< std::size_t >& inpoel,
257  const tk::Fields& U ) const
258  {
259  Assert( U.nunk() == coord[0].size(), "Number of unknowns in solution "
260  "vector at recent time step incorrect" );
261  const auto& x = coord[0];
262  const auto& y = coord[1];
263  const auto& z = coord[2];
264  // ratio of specific heats
265  auto g = g_inputdeck.get< tag::param, tag::compflow, tag::gamma >()[0];
266  // compute the minimum dt across all elements we own
267  tk::real mindt = std::numeric_limits< tk::real >::max();
268  for (std::size_t e=0; e<inpoel.size()/4; ++e) {
269  const std::array< std::size_t, 4 > N{{ inpoel[e*4+0], inpoel[e*4+1],
270  inpoel[e*4+2], inpoel[e*4+3] }};
271  // compute cubic root of element volume as the characteristic length
272  const std::array< tk::real, 3 >
273  ba{{ x[N[1]]-x[N[0]], y[N[1]]-y[N[0]], z[N[1]]-z[N[0]] }},
274  ca{{ x[N[2]]-x[N[0]], y[N[2]]-y[N[0]], z[N[2]]-z[N[0]] }},
275  da{{ x[N[3]]-x[N[0]], y[N[3]]-y[N[0]], z[N[3]]-z[N[0]] }};
276  const auto L = std::cbrt( tk::triple( ba, ca, da ) / 6.0 );
277  // access solution at element nodes at recent time step stage
278  std::array< std::array< tk::real, 4 >, 5 > u;
279  for (ncomp_t c=0; c<5; ++c) u[c] = U.extract( c, m_offset, N );
280  // compute the maximum length of the characteristic velocity (fluid
281  // velocity + sound velocity) across the four element nodes
282  tk::real maxvel = 0.0;
283  for (std::size_t j=0; j<4; ++j) {
284  auto& r = u[0][j]; // rho
285  auto& ru = u[1][j]; // rho * u
286  auto& rv = u[2][j]; // rho * v
287  auto& rw = u[3][j]; // rho * w
288  auto& re = u[4][j]; // rho * e
289  auto p = (g-1.0)*(re - (ru*ru + rv*rv + rw*rw)/2.0/r); // pressure
290  auto c = std::sqrt(g*p/r); // sound speed
291  auto v = std::sqrt((ru*ru + rv*rv + rw*rw)/r) + c; // char. velocity
292  if (v > maxvel) maxvel = v;
293  }
294  // compute element dt for the Euler equations
295  auto euler_dt = L / maxvel;
296  // compute element dt based on the viscous force
297  auto viscous_dt = Physics::viscous_dt( L, u );
298  // compute element dt based on thermal diffusion
299  auto heat_diffusion_dt = Physics::heat_diffusion_dt( L, u );
300  // compute minimum element dt
301  auto elemdt = std::min( euler_dt,
302  std::min( viscous_dt, heat_diffusion_dt ) );
303  // find minimum dt across all elements
304  if (elemdt < mindt) mindt = elemdt;
305  }
306  return mindt;
307  }

309  //! Extract the velocity field at cell nodes
310  //! \param[in] U Solution vector at recent time step stage
311  //! \param[in] N Element node indices

313  std::array< std::array< tk::real, 4 >, 3 >

324  []( tk::real s, tk::real& d ){ return d /= s; } );

328  []( tk::real s, tk::real& d ){ return d /= s; } );
