100  //! Test uniform distribution of a random number generator
101  //! \param[in] r RNG to test
102  //! \details In real code the member function used to generate random numbers
103  //! are called by different threads, but here we pass a different thread ID
104  //! (first argument to the member function called) will exercise multiple
105  //! streams in serial, i.e., emulate multi-threaded RNG on a single thread.
106  //! This is done here this way because the unit test harness deals out each
107  //! test to a PE, thus calling the generators from real threads would create
108  //! contention.
109  template< class rng >
110  static void test_uniform( const rng& r ) {
111  std::size_t num = 100000;
112  std::vector< double > numbers( num );
113  auto n = r.nthreads();
114  for (std::size_t i=0; i<n; ++i)
115  r.uniform( static_cast<int>(i), num/n, &numbers[i*num/n] );
116  for (auto m : numbers) ensure( "sample space incorrect", 0.0<m && m<1.0 );
117  test_stats( numbers, 0.5, 1.0/12.0, 0.0, -6.0/5.0 );
118  }

139  //! Test beta distribution of a random number generator
140  //! \param[in] r RNG to test
141  //! \details In real code the member function used to generate random numbers
142  //! are called by different threads, but here we pass a different thread ID
143  //! (first argument to the member function called) will exercise multiple
144  //! streams in serial, i.e., emulate multi-threaded RNG on a single thread.
145  //! This is done here this way because the unit test harness deals out each
146  //! test to a PE, thus calling the generators from real threads would create
147  //! contention.
148  template< class rng >
149  static void test_beta( const rng& r ) {
150  std::size_t num = 100000;
151  double a = 0.2;
152  double b = 0.6;
153  std::vector< double > numbers( num );
154  auto n = r.nthreads();
155  for (std::size_t i=0; i<n; ++i)
156  r.beta( static_cast<int>(i), num/n, a, b, 0.0, 1.0, &numbers[i*num/n] );
157  // test sample space
158  for (auto m : numbers)
159  ensure( "sample space incorrect", -eps<m && m<1.0+eps );
160  // test first four moments
161  test_stats( numbers, a/(a+b), a*b/(a+b)/(a+b)/(a+b+1.0),
162  2.0*(b-a)*std::sqrt(a+b+1.0)/(a+b+2.0)/std::sqrt(a*b),
163  6.0*((a-b)*(a-b)*(a+b+1.0)-a*b*(a+b+2.0))/a/b/(a+b+2.0)/(a+b+3.0) );
164  // test scaled sample space
165  for (std::size_t i=0; i<n; ++i)
166  r.beta( static_cast<int>(i), num/n, a, b, 0.5, 2.5, &numbers[i*num/n] );
167  for (auto m : numbers)
168  ensure( "sample space incorrect", 0.5-eps<m && m<3.0+eps );
169  }

209  // Test the first four moments of random numbers passed in
210  static void test_stats( const std::vector< double >& numbers,
211  double correct_mean,
212  double correct_variance,
213  double correct_skewness,
214  double correct_excess_kurtosis )
215  {
216  const double precision = 0.05;
217  std::array< double, 4 > s{{ 0.0, 0.0, 0.0, 0.0 }};
218  for (auto n : numbers) s[0] += n;
219  s[0] /= static_cast< double >( numbers.size() );
220  ensure_equals( "mean inaccurate", s[0], correct_mean, precision );
221  for (auto n : numbers) {
222  s[1] += (n-s[0])*(n-s[0]);
223  s[2] += (n-s[0])*(n-s[0])*(n-s[0]);
224  s[3] += (n-s[0])*(n-s[0])*(n-s[0])*(n-s[0]);
225  }
226  s[1] /= static_cast< double >( numbers.size() );
227  s[2] /= static_cast< double >( numbers.size() );
228  s[3] /= static_cast< double >( numbers.size() );
229  ensure_equals( "variance inaccurate", s[1], correct_variance, precision );
230  ensure_equals( "skewness inaccurate",
231  s[2]/std::pow(s[1],1.5), correct_skewness, precision );
232  ensure_equals( "excess kurtosis inaccurate",
233  s[3]/s[1]/s[1]-3.0, correct_excess_kurtosis, precision );
234  }
