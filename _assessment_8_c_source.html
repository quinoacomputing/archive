60 //! \param[in] print Pretty printer
61 //! \param[in] suite Name of the test suite
62 //! \param[in] nfail Number of failed tests
63 //! \param[in] nwarn Number of tests with a warning
64 //! \param[in] nskip Number of skipped tests
65 //! \param[in] nexcp Number of tests with an exception

69 // *****************************************************************************
70 {
71  if (!nfail && !nwarn && !nskip && !nexcp) {
72  print.note< tk::QUIET >
73  ( "All " + std::to_string(ncomplete) + " " + suite + " tests passed" );
74  } else {
75  std::string skip, warn, fail, excp;
76  if (nwarn) warn = "finished with a warning: " + std::to_string(nwarn);
77  if (nskip) skip = std::string(nwarn ? ", " : "") +
78  "skipped: " + std::to_string(nskip);
79  if (nexcp) excp = std::string(nskip || nwarn ? ", " : "") +
80  "threw exception: " + std::to_string(nexcp);
81  if (nfail) fail = std::string(nexcp || nskip || nwarn ?
82  ", " : "") + "failed: " + std::to_string(nfail);
83  print.note< tk::QUIET >
84  ( "Of " + std::to_string(ncomplete) + " " + suite +
85  " tests total: " + warn + skip + excp + fail );
86  }
87 
88  return (nfail || nexcp) ? false : true;
