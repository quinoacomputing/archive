25 //! AuxSolverLumpMassDiff: Auxiliary linear solver: lumped mass + diffusion
26 //! \details This class defines member functions that implement
27 //! assembling/merging the left and right hand sides of a linear system
28 //! resulting from a high order discretization obtained by lumping the
29 //! left hand side consistent mass matrix and adding mass diffusion to the
30 //! right hand side vector. Used in flux-corrected transport for transport
31 //! equations as the low order system. This is only the functionality that is
32 //! needed for the parallel merging of the linear system. Used by and
33 //! abstracted away from tk::LinSysMerger as its auxiliary solver, injected in
34 //! via the template argument AuxSolver.
35 //! \see src/LinSys/LinSysMerger.h
36 //! \note All member functions are static as they are called without an object.
37 class AuxSolverLumpMassDiff {

40  //! Chares contribute to the lhs of the low order linear system
41  //! \param[in] hostproxy The Charm++ proxy of the class we are used as a
42  //! template argument of
43  //! \param[in] host Pointer to the object instance we interoperate with
44  //! \param[in] lower Lower index of the global linear system rows on my PE
45  //! \param[in] upper Upper index of the global linear system rows on my PE
46  //! \param[in] fromch Charm chare array index contribution coming from
47  //! \param[in] gid Global row indices of the vector contributed
48  //! \param[in] mass Portion of the lhs vector contributed
49  //! \param[in,out] auxlhsimport Import map associating a list of global row
50  //! ids to a worker chare id during the communication of the low order lhs
51  //! vector
52  //! \param[in] auxlhs Part of the low order (auxiliary) system lhs vector
53  //! owned by my PE. These are the vector of values (for each scalar
54  //! equation solved) associated to global mesh point row ids. This vector
55  //! collects the nonzero values of the low order (auxiliary) linear system
56  //! lhs "matrix" solution, in the form of a vector, since the matrix of
57  //! this particular low order system is lumped, i.e., stored as a single

63  Host* const host,
64  std::size_t lower,
65  std::size_t upper,
66  int fromch,
67  const std::vector< std::size_t >& gid,
68  const tk::Fields& mass,
69  std::map< int, std::vector< std::size_t > >& auxlhsimport,
70  std::map< std::size_t, std::vector< tk::real > >& auxlhs )
71  {
72  using tk::operator+=;
73  Assert( gid.size() == mass.nunk(),
74  "Size of mass diffusion lhs and row ID vectors must equal" );
75  // Store+add vector of nonzero values owned and pack those to be exported
76  std::map< int, std::map< std::size_t, std::vector< tk::real > > > exp;
77  for (std::size_t i=0; i<gid.size(); ++i)
78  if (gid[i] >= lower && gid[i] < upper) { // if own
79  auxlhsimport[ fromch ].push_back( gid[i] );
80  auxlhs[ gid[i] ] += mass[i];
81  } else
82  exp[ host->pe(gid[i]) ][ gid[i] ] = mass[i];
83  // Export non-owned vector values to fellow branches that own them
84  for (const auto& p : exp) {
85  auto tope = static_cast< int >( p.first );
86  hostproxy[ tope ].addauxlhs( fromch, p.second );
87  }
88  }

196  //! Solve low order system
197  //! \param[in] host Pointer to the object instance we interoperate with
198  //! \param[in] ncomp Number of scalar components per unknown
199  //! \param[in] rhs Part of right-hand side vector owned by my PE. Vector of
200  //! values (for each scalar equation solved) associated to global mesh
201  //! point row ids. In the flux-corrected transport algorithm, this is the
202  //! rhs vector of the high order scheme, used to add mass diffusion to
203  //! create the low order rhs.
204  //! \param[in] lump Part of auxiliary system left-hand side "matrix"
205  //! (vector) owned by my PE. Vector of values (for each scalar equation
206  //! solved) associated to global mesh point row ids. This vector collects
207  //! the nonzero values of the auxiliary system lhs "matrix" solution. In
208  //! flux-corrected transport this is the lhs forthe low order system
209  //! only storing the nonzero diagonal entries of the lhs matrix.
210  //! \param[in,out] diff Part of the mass diffusion right-hand side vector
211  //! owned by my PE. Vector of values (for each scalar equation solved)
212  //! associated to global mesh point row ids. This vector collects the mass
213  //! diffusion terms to be added to the right-hand side for the low order
214  //! solution (for flux-corrected transport).
215  //! \note Static member function as it is called without an object.
216  //! \author J. Bakosi
217  template< class Host > static
218  void solve( Host* const host,
219  std::size_t ncomp,
220  const std::map< std::size_t, std::vector< tk::real > >& rhs,
221  const std::map< std::size_t, std::vector< tk::real > >& lump,
222  std::map< std::size_t, std::vector< tk::real > >& diff )
223  {
224  IGNORE(host);
225  Assert( host->rhscomplete(),
226  "Values of distributed right-hand-side vector on PE " +
227  std::to_string( CkMyPe() ) + " is incomplete: cannot solve low "
228  "order system" );
229  Assert( host->auxrhscomplete(),
230  "Values of distributed mass diffusion rhs vector on PE " +
231  std::to_string( CkMyPe() ) + " is incomplete: cannot solve low "
232  "order system" );
233  Assert( host->auxlhscomplete(),
234  "Values of distributed lumped mass lhs vector on PE " +
235  std::to_string( CkMyPe() ) + " is incomplete: cannot solve low "
236  "order system" );
237  Assert( tk::keyEqual( rhs, diff ), "Row IDs of rhs and mass "
238  "diffusion rhs vector unequal on PE " + std::to_string( CkMyPe() )
239  + ": cannot solve low order system" );
240  Assert( tk::keyEqual( rhs, lump ), "Row IDs of rhs and lumped mass "
241  "lhs vector unequal on PE " + std::to_string( CkMyPe() ) + ": "
242  "cannot solve low order system" );
243  auto ir = rhs.cbegin();
244  auto id = diff.begin();
245  auto im = lump.cbegin();
246  while (ir != rhs.cend()) {
247  const auto& r = ir->second;
248  const auto& m = im->second;
249  auto& d = id->second;
250  Assert( r.size()==ncomp && m.size()==ncomp && d.size()==ncomp,
251  "Wrong number of components in solving the low order system" );
252  for (std::size_t i=0; i<ncomp; ++i) d[i] = (r[i]+d[i])/m[i];
253  ++ir; ++id; ++im;
254  }
255  }

257  //! \brief Group low order solution vector by workers and send each the parts
258  //! back to workers that own them
259  //! \param[in] worker Worker proxy to use send update to
260  //! \param[in] solimport Import map associating a list of global row ids to a
261  //! worker chare id during the communication of the solution/unknown vector.
262  //! This is used here to find out which solution values associated to which
263  //! global nodes IDs to send the solution updates back to the workers.
264  //! \param[in] sol Part of solution vector owned by my PE. Vector of values
265  //! (for each scalar equation solved) associated to global mesh point row
266  //! IDs.
267  //! \note Static member function as it is called without an object.
268  //! \author J. Bakosi
269  template< class WorkerProxy > static
270  void update( const WorkerProxy& worker,
271  const std::map< int, std::vector< std::size_t > >& solimport,
272  const std::map< std::size_t, std::vector< tk::real > >& sol )
273  {
274  for (const auto& w : solimport) {
275  std::vector< std::size_t > gid;
276  std::vector< tk::real > solution;
277  for (auto r : w.second) {
278  const auto it = sol.find( r );
279  if (it != end(sol)) {
280  gid.push_back( it->first );
281  solution.insert(end(solution), begin(it->second), end(it->second));
282  } else
283  Throw( "Can't find global row id " + std::to_string(r) +
284  " to export in low order solution vector" );
285  }
286  worker[ w.first ].updateLowSol( gid, solution );
287  }
288  }
