32 // *****************************************************************************
33 // Match user-specified boundary conditions at nodes for side sets
34 //! \param[in] ncomp Number of scalar components in PDE system
35 //! \param[in] t Physical time at which to query boundary conditions
36 //! \param[in] dt Time step size (for querying BC increments in time)
37 //! \param[in] coord Mesh node coordinates
38 //! \param[in] gid Global node IDs
39 //! \param[in] lid Local node IDs associated to global node IDs
40 //! \param[in] sidenodes Map storing global mesh node IDs mapped to side set ids
41 //! \return Vector of pairs of bool and boundary condition value associated to
42 //! mesh node IDs at which the user has set Dirichlet boundary conditions for
43 //! all systems of PDEs integrated. The bool indicates whether the BC is set
44 //! at the node for that component: if true, the real value is the increment
45 //! (from t to dt) in the BC specified for a component.
46 //! \details Boundary conditions (BC), mathematically speaking, are applied on
47 //! finite surfaces. These finite surfaces are given by element sets (i.e., a
48 //! list of elements). This function queries Dirichlet boundary condition
49 //! values from all PDEs in the system of systems of PDEs integrated at the
50 //! node lists associated to side set IDs, given by sidenodes. Each
51 //! PDE system returns a BC data structure which is then returned. Note that
52 //! the BC mesh nodes that this function results in, stored in dirbc, only
53 //! contains those nodes that are supplied via sidenodes, i.e., in parallel
54 //! only a part of the mesh is worked on.
55 // *****************************************************************************
56 {
57  using inciter::g_cgpde;
58 
59  // Vector of pairs of bool and boundary condition value associated to mesh
60  // node IDs at which the user has set Dirichlet BCs for all PDEs integrated.
61  std::unordered_map< std::size_t,
62  std::vector< std::pair< bool, tk::real > > > dirbc;
63 
64  // Details for the algorithm below: PDE::dirbc() returns a new map that
65  // associates a vector of pairs associated to local node IDs. (The pair is a
66  // pair of bool and real value, the former is the fact that the BC is to be
67  // set while the latter is the value if it is to be set). The length of this
68  // NodeBC vector, returning from each system of PDEs equals to the number of
69  // scalar components the given PDE integrates. Here then we contatenate this
70  // map for all PDEs integrated. If there are multiple BCs set at a mesh node
71  // (dirbc::key), either because (1) in the same PDE system the user prescribed
72  // BCs on side sets that share nodes or (2) because more than a single PDE
73  // system assigns BCs to a given node (on different variables), the NodeBC
74  // vector must be correctly stored. "Correctly" here means that the size of
75  // the NodeBC vectors must all be the same and equal to the sum of all scalar
76  // components integrated by all PDE systems integrated. Example: single-phase
77  // compressible flow (density, momentum, energy = 5) + transported scalars of
78  // 10 variables -> NodeBC vector length = 15. Note that in case (1) above a
79  // new node encountered must "overwrite" the already existing space for the
80  // NodeBC vector. "Overwrite" here means that it should keep the existing BCs
81  // and add the new ones yielding the union the two prescription for BCs but in
82  // the same space that already exist in the NodeBC vector. In case (2),
83  // however, the NodeBC pairs must go to the location in the vector assigned to
84  // the given PDE system, i.e., using the above example BCs for the 10 (or
85  // less) scalars should go in the positions starting at 5, leaving the first 5
86  // false, indicating no BCs for the flow variables.
87  //
88  // Note that the logic described above is only partially implemented at this
89  // point. What works is the correct insertion of multiple BCs for nodes shared
90  // among multiple side sets, e.g., corners, originating from the same PDE
91  // system. What is not yet implemented is the case when there are no BCs set
92  // for flow variables but there are BCs for transport, the else branch below
93  // will incorrectly NOT skip the space for the flow variables. In other words,
94  // this only works for a single PDE system and a sytem of systems. This
95  // machinery is only tested with a single system of PDEs at this point.
96  //
97  // When a particular node belongs to two or more side sets with different BCs,
98  // there is an ambiguity as to which of the multiple BCs should be applied to
99  // the node. This issue is described in case (1) above. In the current
100  // implementation, every side set applies the BC to the common node in
101  // question, successively overwriting the BC applied by the previous side set.
102  // Effectively, the BC corresponding to the last side set ID is applied to the
103  // common node. Since sidenodes is an ordered map, the side set with a larger
104  // id wins if a node belongs to multiple side sets.
105 
106  // Lambda to convert global to local node ids of a list of nodes
107  auto local = [ &lid ]( const std::vector< std::size_t >& gnodes ){
108  std::vector< std::size_t > lnodes( gnodes.size() );
109  for (std::size_t i=0; i<gnodes.size(); ++i)
110  lnodes[i] = tk::cref_find( lid, gnodes[i] );
111  return lnodes;
112  };
113 
114  // Query Dirichlet BCs for all PDEs integrated and assign to nodes
115  for (const auto& s : sidenodes) { // for all side sets passed in
116  std::size_t c = 0;
117  auto l = local(s.second); // generate local node ids on side set
118  for (std::size_t eq=0; eq<g_cgpde.size(); ++eq) {
119  // query Dirichlet BCs at nodes of this side set
120  auto eqbc = g_cgpde[eq].dirbc( t, dt, {s.first,l}, coord );
121  for (const auto& n : eqbc) {
122  auto id = n.first; // BC node ID
123  const auto& bcs = n.second; // BCs
124  auto& nodebc = dirbc[ gid[id] ]; // BCs to be set for node
125  if (nodebc.size() > c) { // node already has BCs from this PDE
126  Assert( nodebc.size() == c+bcs.size(), "Size mismatch" );
127  for (std::size_t i=0; i<bcs.size(); i++) {
128  if (bcs[i].first) nodebc[c+i] = bcs[i];
129  }
130  } else { // node does not yet have BCs from this PDE
131  // This branch needs to be completed for system of systems of PDEs.
132  // See note above.
133  nodebc.insert( end(nodebc), begin(bcs), end(bcs) );
134  }
135  }
136  if (!eqbc.empty()) c += eqbc.cbegin()->second.size();
137  }
138  }
139 
140  // Verify the size of each NodeBC vectors. They must have the same lengths and
141  // equal to the total number of scalar components for all systems of PDEs
142  // integrated. This is intentional, because this way the linear system solver
143  // does not have to (and does not) know about individual equation systems.
144  // This entire loop is optimized away in RELEASE mode, thus the IGNOREs to
145  // silence compiler warnings.
146  Assert( std::all_of( begin(dirbc), end(dirbc),
147  [ ncomp ]( const decltype(dirbc)::value_type& n ){
148  return n.second.size() == ncomp; } ),
149  "Size of NodeBC vector incorrect" );
150  IGNORE( ncomp );
151 
152  return dirbc;
153 }

155 bool
156 correctBC( const tk::Fields& a,
157  const tk::Fields& dul,
158  const std::map< int, std::vector< std::size_t > >& bnode,
159  const std::unordered_map< std::size_t,
160  std::vector< std::pair< bool, tk::real > > >& bc,
161  const std::unordered_map< std::size_t, std::size_t >& lid )
162 // *****************************************************************************
163 // Verify that the change in the solution at those nodes where Dirichlet
164 // boundary conditions are set is exactly the amount the BCs prescribe
165 //! \param[in] a Limited antidiffusive element contributions (from FCT)
166 //! \param[in] dul Low order solution increment
167 //! \param[in] bnode Map of boundary-node lists mapped to corresponding
168 //! side set ids for this mesh chunk
169 //! \param[in] lid Local node IDs associated to global node IDs
170 //! \return True if solution is correct at Dirichlet boundary condition nodes
171 //! \details We loop through the map that associates a vector of local node IDs
172 //! to side set IDs for all side sets given in bnode. Then for each side set
173 //! for all mesh nodes on a given side set we attempt to find the global node
174 //! ID in bc, which stores nodes (and BC settings) at which the user has
175 //! configured Dirichlet BCs to be set. Then for all scalar components of all
176 //! systems of systems of PDEs integrated if a BC is to be set for a given
177 //! component, we compute the low order solution increment + the
178 //! anti-diffusive element contributions (in FCT), which is the current
179 //! solution increment (to be used to update the solution at time n in FCT) at
180 //! that node. This solution increment must equal the BC prescribed at the
181 //! given node as we solve for solution increments. If not, the BCs are not
182 //! set correctly, which is an error.
183 // *****************************************************************************
184 {
185  if (bc.empty()) return true;
186 
187  for (const auto& s : bnode)
188  for (auto i : s.second) {
189  auto u = bc.find( i );
190  auto l = tk::cref_find( lid, i );
191  if (u != end(bc)) {
192  const auto& b = u->second;
193  Assert( b.size() == dul.nprop(), "Size mismatch" );
194  for (std::size_t c=0; c<b.size(); ++c)
195  if ( b[c].first &&
196  std::abs( dul(l,c,0) + a(l,c,0) - b[c].second ) >
197  std::numeric_limits< tk::real >::epsilon() ) {
198  return false;
199  }
200  }
201  }
202 
203  return true;
204 }
