65  //! \brief Instantiate a differential equation object
66  //! \see walker::DiffEq
67  //! \details Since multiple systems of equations can be configured
68  //! with the same type, the map (cnt) is used to assign a counter to each
69  //! type. The template argument, EqTag, is used to find the given system
70  //! of differential equations in the input deck, the hierarchical data
71  //! filled during control file parsing, containing user input.
72  //! \param[in] eq The unique differential equation system key whose object
73  //! to instantiate.
74  //! \param[in,out] cnt Counter, a std::map, that counts all instantiated
75  //! differential equations systems by type.
76  template< class EqTag >
77  DiffEq createDiffEq( ctr::DiffEqType eq,
78  std::map< ctr::DiffEqType, ncomp_t >& cnt ) const {
79  auto c = ++cnt[ eq ]; // count eqs
80  --c; // used to index vectors starting with 0
81  if ( g_inputdeck.get< tag::component, EqTag >()[c] ) {
82  // create key and search for it
83  ctr::DiffEqKey key{ eq,
84  g_inputdeck.get< tag::param, EqTag, tag::initpolicy >()[c],
85  g_inputdeck.get< tag::param, EqTag, tag::coeffpolicy >()[c] };
86  const auto it = m_factory.find( key );
87  Assert( it != end( m_factory ),
88  "Can't find eq '" + ctr::DiffEq().name( eq ) +
89  "' in DiffEq factory with initialization policy '" +
90  ctr::InitPolicy().name(
91  g_inputdeck.get< tag::param, EqTag, tag::initpolicy >()[c] ) +
92  "' and coefficient policy '" +
93  ctr::CoeffPolicy().name(
94  g_inputdeck.get< tag::param, EqTag, tag::coeffpolicy >()[c] )
95  + "'" );
96  // instantiate and return diff eq object
97  return it->second( c );
98  } else Throw ( "Can't create DiffEq with zero independent variables" );
99  }
100 
101  //! \brief Instantiate tables from a differential equation
102  //! \details This function is used to instantiate a vector of tk::Tables and
103  //! their associated names for a system of differential equation selected
104  //! by the user. Since multiple systems of equations can be configured
105  //! with the same type, the map (cnt) is used to assign a counter to each
106  //! type. We then find out if the coefficients policy, configured to the
107  //! equation system, uses tables. If so, we return all the tables and
108  //! their names associated to the different components in the system. The
109  //! template argument, EqTag, is used to find the given differential
110  //! equation system in the input deck, the hierarchical data filled
111  //! during control file parsing, containing user input.
112  //! \param[in] eq The unique differential equation system key whose tables
113  //! object to instantiate (if the configured coefficients policy uses
114  //! tables).
115  //! \param[in,out] cnt Counter, a std::map, that counts all instantiated
116  //! vector of tables associated to a differential equations systems by
117  //! type.
118  template< class EqTag >
119  std::pair< std::vector< std::string >, std::vector< tk::Table > >
120  createTables( ctr::DiffEqType eq,
121  std::map< ctr::DiffEqType, ncomp_t >& cnt ) const
122  {
123  auto c = ++cnt[ eq ]; // count eqs
124  --c; // used to index vectors starting with 0
125  std::vector< tk::Table > tab;
126  std::vector< std::string > nam;
127  const auto& ncompeq = g_inputdeck.get< tag::component, EqTag >();
128  if (!ncompeq.empty()) {
129  if ( g_inputdeck.get< tag::component, EqTag >()[c] ) {
130  // find out if coefficients policy uses tables and return them if so
131  if (g_inputdeck.get< tag::param, EqTag, tag::coeffpolicy >()[c] ==
132  ctr::CoeffPolicyType::HYDROTIMESCALE)
133  {
134  const auto& hts = g_inputdeck.get< tag::param,
135  EqTag,
136  tag::hydrotimescales >().at(c);
137  ctr::HydroTimeScales ot;
138  for (auto t : hts) tab.push_back( ot.table(t) );
139  for (auto t : hts) nam.push_back( ot.name(t) );
140 
141  const auto& hp = g_inputdeck.get< tag::param,
142  EqTag,
143  tag::hydroproductions >().at(c);
144  ctr::HydroProductions op;
145  for (auto t : hp) tab.push_back( op.table(t) );
146  for (auto t : hp) nam.push_back( op.name(t) );
147 
148  }
149  } else Throw ( "DiffEq with zero independent variables" );
150  }
151  return { nam, tab };
152  }

154  DiffEqFactory m_factory; //!< Differential equations factory

156 };
