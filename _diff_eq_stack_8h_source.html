79  //! \brief Function object for registering a differential equation into the
80  //! differential equation factory
81  //! \details This functor is repeatedly called by MPL's cartesian_product,
82  //! sweeping all combinations of the differential equation policies. The
83  //! purpose of template template is to simplify client code as that will
84  //! not have to specify the template arguments of the template argument
85  //! (the policies of Eq), since we can figure it out here. See also
86  //! http://stackoverflow.com/a/214900
87  template< template< class, class > class Eq >
88  struct registerDiffEq {
89  //! Need to store the reference to factory we are registering into
90  DiffEqFactory& factory;
91  //! Need to store which differential equation we are registering
92  const ctr::DiffEqType type;
93  //! Constructor, also count number of unique equation types registered
94  explicit registerDiffEq( DiffEqFactory& f,
95  ctr::DiffEqType t,
96  std::set< ctr::DiffEqType >& eqTypes ) :
97  factory( f ), type( t ) { eqTypes.insert( t ); }
98  //! \brief Function call operator called with tk::cartesian_product for
99  //! each unique sequence of policy combinations
100  template< typename U > void operator()( brigand::type_<U> ) {
101  // Get Initialization policy: first type of brigand::list U
102  using InitPolicy = typename brigand::front< U >;
103  // Get coefficients policy: last type of brigand::list U
104  using CoeffPolicy = typename brigand::back< U >;
105  // Build differential equation key
106  ctr::DiffEqKey key{ type, InitPolicy::type(), CoeffPolicy::type() };
107  // Register equation (with policies given by brigand::list U) into
108  // factory
109  tk::recordModelLate< DiffEq, Eq< InitPolicy, CoeffPolicy > >
110  ( factory, key, static_cast<ncomp_t>(0) );
111  }
112  };

114  //! \brief Instantiate a differential equation object
115  //! \see walker::DiffEq
116  //! \details Since multiple systems of equations can be configured
117  //! with the same type, the map (cnt) is used to assign a counter to each
118  //! type. The template argument, EqTag, is used to find the given system
119  //! of differential equations in the input deck, the hierarchical data
120  //! filled during control file parsing, containing user input.
121  //! \param[in] eq The unique differential equation system key whose object
122  //! to instantiate.
123  //! \param[in,out] cnt Counter, a std::map, that counts all instantiated
124  //! differential equations systems by type.
125  template< class EqTag >
126  DiffEq createDiffEq( ctr::DiffEqType eq,
127  std::map< ctr::DiffEqType, ncomp_t >& cnt ) const {
128  auto c = ++cnt[ eq ]; // count eqs
129  --c; // used to index vectors starting with 0
130  if ( g_inputdeck.get< tag::component, EqTag >()[c] ) {
131  // create key and search for it
132  ctr::DiffEqKey key{ eq,
133  g_inputdeck.get< tag::param, EqTag, tag::initpolicy >()[c],
134  g_inputdeck.get< tag::param, EqTag, tag::coeffpolicy >()[c] };
135  const auto it = m_factory.find( key );
136  Assert( it != end( m_factory ),
137  "Can't find eq '" + ctr::DiffEq().name( eq ) +
138  "' in DiffEq factory with initialization policy '" +
139  ctr::InitPolicy().name(
140  g_inputdeck.get< tag::param, EqTag, tag::initpolicy >()[c] ) +
141  "' and coefficient policy '" +
142  ctr::CoeffPolicy().name(
143  g_inputdeck.get< tag::param, EqTag, tag::coeffpolicy >()[c] )
144  + "'" );
145  // instantiate and return diff eq object
146  return it->second( c );
147  } else Throw ( "Can't create DiffEq with zero independent variables" );
148  }

150  //! \brief Instantiate tables from a differential equation
151  //! \details This function is used to instantiate a vector of tk::Tables and
152  //! their associated names for a system of differential equation selected
153  //! by the user. Since multiple systems of equations can be configured
154  //! with the same type, the map (cnt) is used to assign a counter to each
155  //! type. We then find out if the coefficients policy, configured to the
156  //! equation system, uses tables. If so, we return all the tables and
157  //! their names associated to the different components in the system. The
158  //! template argument, EqTag, is used to find the given differential
159  //! equation system in the input deck, the hierarchical data filled
160  //! during control file parsing, containing user input.
161  //! \param[in] eq The unique differential equation system key whose tables
162  //! object to instantiate (if the configured coefficients policy uses
163  //! tables).
164  //! \param[in,out] cnt Counter, a std::map, that counts all instantiated
165  //! vector of tables associated to a differential equations systems by
166  //! type.
167  template< class EqTag >
168  std::pair< std::vector< std::string >, std::vector< tk::Table > >
169  createTables( ctr::DiffEqType eq,
170  std::map< ctr::DiffEqType, ncomp_t >& cnt ) const
171  {
172  auto c = ++cnt[ eq ]; // count eqs
173  --c; // used to index vectors starting with 0
174  std::vector< tk::Table > tab;
175  std::vector< std::string > nam;
176  const auto& ncompeq = g_inputdeck.get< tag::component, EqTag >();
177  if (!ncompeq.empty()) {
178  if ( g_inputdeck.get< tag::component, EqTag >()[c] ) {
179  // find out if coefficients policy uses tables and return them if so
180  if (g_inputdeck.get< tag::param, EqTag, tag::coeffpolicy >()[c] ==
181  ctr::CoeffPolicyType::HYDROTIMESCALE)
182  {
183  const auto& hts = g_inputdeck.get< tag::param,
184  EqTag,
185  tag::hydrotimescales >().at(c);
186  ctr::HydroTimeScales ot;
187  for (auto t : hts) tab.push_back( ot.table(t) );
188  for (auto t : hts) nam.push_back( ot.name(t) );
189 
190  const auto& hp = g_inputdeck.get< tag::param,
191  EqTag,
192  tag::hydroproductions >().at(c);
193  ctr::HydroProductions op;
194  for (auto t : hp) tab.push_back( op.table(t) );
195  for (auto t : hp) nam.push_back( op.name(t) );
196 
197  }
198  } else Throw ( "DiffEq with zero independent variables" );
199  }
200  return { nam, tab };
201  }
