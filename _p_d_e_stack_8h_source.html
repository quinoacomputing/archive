69  //! \brief Function object for registering a partial differential equation
70  //! into the partial differential equation factory
71  //! \details This functor is repeatedly called by MPL's cartesian_product,
72  //! sweeping all combinations of the partial differential equation
73  //! policies. The purpose of template template is to simplify client code
74  //! as that will not have to specify the template arguments of the
75  //! template argument (the policies of Eq), since we can figure it out
76  //! here. See also http://stackoverflow.com/a/214900
77  template< template< class, class > class Eq >
78  struct registerPDE {
79  //! Need to store the reference to factory we are registering into
80  PDEFactory& factory;
81  //! Need to store which differential equation we are registering
82  const ctr::PDEType type;
83  //! Constructor, also count number of unique equation types registered
84  explicit registerPDE( PDEFactory& f,
85  ctr::PDEType t,
86  std::set< ctr::PDEType >& eqTypes ) :
87  factory( f ), type( t ) { eqTypes.insert( t ); }
88  //! \brief Function call operator called by mpl::cartesian_product for
89  //! each unique sequence of policy combinations
90  template< typename U > void operator()( U ) {
91  namespace mpl = boost::mpl;
92  // Get problem policy: 1st type of mpl::vector U
93  using Physics = typename mpl::at< U, mpl::int_<0> >::type;
94  // Get problem policy: 2nd type of mpl::vector U
95  using Problem = typename mpl::at< U, mpl::int_<1> >::type;
96  // Build differential equation key
97  ctr::PDEKey key{ type, Physics::type(), Problem::type() };
98  // Register equation (with policies given by mpl::vector U) into factory
99  tk::recordModelLate< PDE, Eq< Physics, Problem > >
100  ( factory, key, static_cast<ncomp_t>(0) );
101  }
102  };
103 
104  //! \brief Instantiate a partial differential equation
105  //! \details The template argument, EqTag, is used to find the given
106  //! partial differential equation in the input deck, the hierarchical data
107  //! filled during control file parsing, containing user input.
108  //! \param[in] eq The unique partial differential equation key whose object
109  //! to instantiate.
110  //! \param[in,out] cnt Counter, a std::map, that counts all instantiated
111  //! partial differential equations by type.
112  template< class EqTag >
113  PDE createPDE( ctr::PDEType eq, std::map< ctr::PDEType, ncomp_t >& cnt )
114  const {
115  auto c = ++cnt[ eq ]; // count eqs
116  --c; // used to index vectors starting with 0
117  Assert( c < (g_inputdeck.get< tag::component, EqTag >().size()),
118  "The number of scalar components is unspecified for the PDE to "
119  "be instantiated. This is most likely a grammar error in the "
120  "parser. The parser should not allow the user to select a PDE "
121  "without configuring the number of scalar components the "
122  "equation consists of. See inciter::deck::check_eq." );
123  if ( g_inputdeck.get< tag::component, EqTag >()[c] ) {
124  // re-create key and search for it
125  ctr::PDEKey key{ eq,
126  g_inputdeck.get< tag::param, EqTag, tag::physics >()[c],
127  g_inputdeck.get< tag::param, EqTag, tag::problem >()[c] };
128  const auto it = m_factory.find( key );
129  Assert( it != end( m_factory ), "Can't find PDE in factory" );
130  return it->second( c ); // instantiate and return PDE object
131  } else Throw ( "Can't create PDE with zero components" );
132  }
