67  //! \brief Instantiate a partial differential equation
68  //! \details The template argument, EqTag, is used to find the given
69  //! partial differential equation in the input deck, the hierarchical data
70  //! filled during control file parsing, containing user input. The
71  //! template argument Factory specifies which factory we search in. The
72  //! template argument PDE specifies the "base" PDE type that the
73  //! instantiated "child" PDE class object is used polymorphically with.
74  //! \param[in] f Factory in which to search PDE in
75  //! \param[in] eq The unique partial differential equation key whose object
76  //! to instantiate.
77  //! \param[in,out] cnt Counter, a std::map, that counts all instantiated
78  //! partial differential equations by type.
79  template< class EqTag, class Factory, class PDE >
80  PDE createPDE( const Factory& f,
81  ctr::PDEType eq,
82  std::map< ctr::PDEType, ncomp_t >& cnt ) const
83  {
84  auto c = ++cnt[ eq ]; // count eqs
85  --c; // used to index vectors starting with 0
86  Assert( c < (g_inputdeck.get< tag::component, EqTag >().size()),
87  "The number of scalar components is unspecified for the PDE to "
88  "be instantiated. This is most likely a grammar error in the "
89  "parser. The parser should not allow the user to select a PDE "
90  "without configuring the number of scalar components the "
91  "equation consists of. See inciter::deck::check_eq." );
92  if ( g_inputdeck.get< tag::component, EqTag >()[c] ) {
93  // re-create key and search for it
94  ctr::PDEKey key{ eq,
95  g_inputdeck.get< tag::param, EqTag, tag::physics >()[c],
96  g_inputdeck.get< tag::param, EqTag, tag::problem >()[c] };
97  const auto it = f.find( key );
98  Assert( it != end( f ),
99  "Can't find PDE with key('" +
100  ctr::PDE().name( key.get< tag::pde >() ) + "', '" +
101  ctr::Physics().name( key.get< tag::physics >() ) + "', '" +
102  ctr::Problem().name( key.get< tag::problem >() ) +
103  + "') in factory" );
104  return it->second( c ); // instantiate and return PDE object
105  } else Throw ( "Can't create PDE with zero components" );
106  }
107 
108  //! Wrapper of createPDE specialized for registering CG PDEs
109  //! \param[in] t Enum selecting PDE type, Control/Inciter/Options/PDE.h
110  //! \param[in,out] cnt Counter, a std::map, that counts all instantiated
111  //! partial differential equations by type.
112  //! \details The sole reason for this function is to simplify client-code
113  //! calling createPDE specialized to CG PDEs
