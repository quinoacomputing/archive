77  //! \brief Function object for registering a partial differential equation
78  //! into the partial differential equation factory
79  //! \details This functor is repeatedly called by MPL's cartesian_product,
80  //! sweeping all combinations of the partial differential equation
81  //! policies. The purpose of template template is to simplify client code
82  //! as that will not have to specify the template arguments of the
83  //! template argument (the policies of Eq), since we can figure it out
84  //! here. See also http://stackoverflow.com/a/214900. The template
85  //! argument Eq specifies a "child" class that is used polymorphically
86  //! with a "base" class modeling a concept defined in the base. The base
87  //! is given by the template argument PDE. The template argument Factory
88  //! specifies which factory to store the registered and configured child
89  //1 PDE.
90  template< template< class, class > class Eq, class Factory, class PDE >
91  struct registerPDE {
92  //! Need to store the reference to factory we are registering into
93  Factory& factory;
94  //! Need to store which differential equation we are registering
95  const ctr::PDEType type;
96  //! Constructor, also count number of unique equation types registered
97  //! \param[in,out] f Factory into which to register PDE
98  //! \param[in] t Enum selecting PDE type, Control/Inciter/Options/PDE.h
99  //! \param[in] eqTypes Equation type counters
100  explicit registerPDE( Factory& f,
101  ctr::PDEType t,
102  std::set< ctr::PDEType >& eqTypes ) :
103  factory( f ), type( t ) { eqTypes.insert( t ); }
104  //! \brief Function call operator called by mpl::cartesian_product for
105  //! each unique sequence of policy combinations
106  template< typename U > void operator()( U ) {
107  namespace mpl = boost::mpl;
108  // Get problem policy: 1st type of mpl::vector U
109  using Physics = typename mpl::at< U, mpl::int_<0> >::type;
110  // Get problem policy: 2nd type of mpl::vector U
111  using Problem = typename mpl::at< U, mpl::int_<1> >::type;
112  // Build differential equation key
113  ctr::PDEKey key{ type, Physics::type(), Problem::type() };
114  // Register equation (with policies given by mpl::vector U) into factory
115  tk::recordModelLate< PDE, Eq< Physics, Problem > >
116  ( factory, key, static_cast<ncomp_t>(0) );
117  }
118  };

146  //! \brief Instantiate a partial differential equation
147  //! \details The template argument, EqTag, is used to find the given
148  //! partial differential equation in the input deck, the hierarchical data
149  //! filled during control file parsing, containing user input. The
150  //! template argument Factory specifies which factory we search in. The
151  //! template argument PDE specifies the "base" PDE type that the
152  //! instantiated "child" PDE class object is used polymorphically with.
153  //! \param[in] f Factory in which to search PDE in
154  //! \param[in] eq The unique partial differential equation key whose object
155  //! to instantiate.
156  //! \param[in,out] cnt Counter, a std::map, that counts all instantiated
157  //! partial differential equations by type.
158  template< class EqTag, class Factory, class PDE >
159  PDE createPDE( const Factory& f,
160  ctr::PDEType eq,
161  std::map< ctr::PDEType, ncomp_t >& cnt ) const
162  {
163  auto c = ++cnt[ eq ]; // count eqs
164  --c; // used to index vectors starting with 0
165  Assert( c < (g_inputdeck.get< tag::component, EqTag >().size()),
166  "The number of scalar components is unspecified for the PDE to "
167  "be instantiated. This is most likely a grammar error in the "
168  "parser. The parser should not allow the user to select a PDE "
169  "without configuring the number of scalar components the "
170  "equation consists of. See inciter::deck::check_eq." );
171  if ( g_inputdeck.get< tag::component, EqTag >()[c] ) {
172  // re-create key and search for it
173  ctr::PDEKey key{ eq,
174  g_inputdeck.get< tag::param, EqTag, tag::physics >()[c],
175  g_inputdeck.get< tag::param, EqTag, tag::problem >()[c] };
176  const auto it = f.find( key );
177  Assert( it != end( f ),
178  "Can't find PDE with key('" +
179  ctr::PDE().name( key.get< tag::pde >() ) + "', '" +
180  ctr::Physics().name( key.get< tag::physics >() ) + "', '" +
181  ctr::Problem().name( key.get< tag::problem >() ) +
182  + "') in factory" );
183  return it->second( c ); // instantiate and return PDE object
184  } else Throw ( "Can't create PDE with zero components" );
185  }
