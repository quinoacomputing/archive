32 // *****************************************************************************
33 // Signal handler for multiple signals, SIGABRT, SIGSEGV, etc.
34 //! \param[in] signum Signal number
35 //! \see https://oroboro.com/stack-trace-on-crash
36 //! \details Signals caught:
37 //! SIGABRT is generated when the program calls the abort() function, such as
38 //! when an assert() triggers
39 //! SIGSEGV is generated when the program makes an illegal memory access, such
40 //! as reading unaligned memory, dereferencing a null pointer, reading
41 //! memory out of bounds etc.
42 //! SIGILL is generated when the program tries to execute a malformed
43 //! instruction. This happens when the execution pointer starts reading
44 //! non-program data, or when a pointer to a function is corrupted.
45 //! SIGFPE is generated when executing an illegal floating point instruction,
46 //! most commonly division by zero or floating point overflow.
47 // *****************************************************************************
48 {
49  // associate each signal with a signal name string.
50  const char* name = nullptr;
51  switch( signum ) {
52  case SIGABRT: name = "SIGABRT"; break;
53  case SIGSEGV: name = "SIGSEGV"; break;
54  case SIGILL: name = "SIGILL"; break;
55  case SIGFPE: name = "SIGFPE"; break;
56  }
57 
58  // Echo what signal is caught
59  if ( name )
60  fprintf( stderr, "Caught signal %d (%s)\n", signum, name );
61  else
62  fprintf( stderr, "Caught signal %d\n", signum );
63 
64  // Get and display backtrace
65  tk::Exception("signal").handleException();
66 
67  // Tell the runtime system to exit with a message
68  CkAbort( "Signal caught" );
69 }
