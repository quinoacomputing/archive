31  //! Compute internal energy parameter
32  //! \param[in] bx Parameter betax
33  //! \param[in] by Parameter betay
34  //! \param[in] bz Parameter betaz
35  //! \param[in] x X coordinate to evaluate at
36  //! \param[in] y Y coordinate to evaluate at
37  //! \param[in] z Z coordinate to evaluate at
38  //! \return Internal energy parameter
39  static tk::real hx( tk::real bx, tk::real by, tk::real bz,
40  tk::real x, tk::real y, tk::real z )
41  { return std::cos(bx*M_PI*x) * std::cos(by*M_PI*y) * std::cos(bz*M_PI*z); }
42 
43  //! Compute a power of the internal energy
44  //! \param[in] ce Internal energy parameter
45  //! \param[in] kappa Internal energy parameter
46  //! \param[in] t Physical time
47  //! \param[in] h Internal energy parameter
48  //! \param[in] p Power
49  //! \return Internal energy raised to power p
50  static tk::real ec( tk::real ce, tk::real kappa, tk::real t, tk::real h,
51  tk::real p )
52  { return std::pow( -3.0*(ce + kappa*h*h*t), p ); }

54  //! Evaluate analytical solution at (x,y,z,t) for all components
55  //! \param[in] e Equation system index, i.e., which compressible
56  //! flow equation system we operate on among the systems of PDEs
57  //! \param[in] x X coordinate where to evaluate the solution
58  //! \param[in] y Y coordinate where to evaluate the solution
59  //! \param[in] z Z coordinate where to evaluate the solution
60  //! \param[in] t Time where to evaluate the solution
61  //! \return Values of all components evaluated at (x,y,z,t)
62  static std::array< tk::real, 5 >
63  solution( tk::ctr::ncomp_type e,
64  tk::real x, tk::real y, tk::real z, tk::real t )
65  {
66  using tag::param; using tag::compflow;
67  // manufactured solution parameters
68  const auto ce = g_inputdeck.get< param, compflow, tag::ce >()[e];
69  const auto r0 = g_inputdeck.get< param, compflow, tag::r0 >()[e];
70  const auto a = g_inputdeck.get< param, compflow, tag::alpha >()[e];
71  const auto k = g_inputdeck.get< param, compflow, tag::kappa >()[e];
72  const auto bx = g_inputdeck.get< param, compflow, tag::betax >()[e];
73  const auto by = g_inputdeck.get< param, compflow, tag::betay >()[e];
74  const auto bz = g_inputdeck.get< param, compflow, tag::betaz >()[e];
75  // spatial component of density field
76  const tk::real gx = 1.0 - x*x - y*y - z*z;
77  // internal energy parameter
78  const auto h = hx( bx, by, bz, x, y, z );
79  // temporal component of the density field
80  tk::real ft = std::exp( -a*t );
81  // solution at t
82  auto r = r0 + ft*gx;
83  return {{ r, 0.0, 0.0, 0.0, r*ec(ce,k,t,h,-1.0/3.0) }};
84  }

139  //! Compute and return source term for NLEG manufactured solution
140  //! \param[in] e Equation system index, i.e., which compressible
141  //! flow equation system we operate on among the systems of PDEs
142  //! \param[in] x X coordinate where to evaluate the solution
143  //! \param[in] y Y coordinate where to evaluate the solution
144  //! \param[in] z Z coordinate where to evaluate the solution
145  //! \param[in] t Physical time at which to evaluate the source
146  //! \return Array of reals containing the source for all components
147  static std::array< tk::real, 5 >
148  src( tk::ctr::ncomp_type e, tk::real x, tk::real y, tk::real z, tk::real t )
149  {
150  using tag::param; using tag::compflow; using std::sin; using std::cos;
151  // manufactured solution parameters
152  const auto a = g_inputdeck.get< param, compflow, tag::alpha >()[e];
153  const auto bx = g_inputdeck.get< param, compflow, tag::betax >()[e];
154  const auto by = g_inputdeck.get< param, compflow, tag::betay >()[e];
155  const auto bz = g_inputdeck.get< param, compflow, tag::betaz >()[e];
156  const auto ce = g_inputdeck.get< param, compflow, tag::ce >()[e];
157  const auto kappa = g_inputdeck.get< param, compflow, tag::kappa >()[e];
158  const auto r0 = g_inputdeck.get< param, compflow, tag::r0 >()[e];
159  // ratio of specific heats
160  const auto g = g_inputdeck.get< param, compflow, tag::gamma >()[e];
161  // spatial component of density field
162  const auto gx = 1.0 - x*x - y*y - z*z;
163  // derivative of spatial component of density field
164  const std::array< tk::real, 3 > dg{{ -2.0*x, -2.0*y, -2.0*z }};
165  // spatial component of energy field
166  const auto h = hx( bx, by, bz, x, y, z );
167  // derivative of spatial component of energy field
168  std::array< tk::real, 3 >
169  dh{{ -bx*M_PI*sin(bx*M_PI*x)*cos(by*M_PI*y)*cos(bz*M_PI*z),
170  -by*M_PI*cos(bx*M_PI*x)*sin(by*M_PI*y)*cos(bz*M_PI*z),
171  -bz*M_PI*cos(bx*M_PI*x)*cos(by*M_PI*y)*sin(bz*M_PI*z) }};
172  // temporal function f and its derivative
173  const auto ft = std::exp(-a*t);
174  const auto dfdt = -a*ft;
175  // density and its derivatives
176  const auto rho = r0 + ft*gx;
177  const std::array< tk::real, 3 > drdx{{ ft*dg[0], ft*dg[1], ft*dg[2] }};
178  const auto drdt = gx*dfdt;
179  // internal energy and its derivatives
180  const auto ie = ec( ce, kappa, t, h, -1.0/3.0 );
181  const std::array< tk::real, 3 > dedx{{
182  2.0*std::pow(ie,4.0)*kappa*h*dh[0]*t,
183  2.0*std::pow(ie,4.0)*kappa*h*dh[1]*t,
184  2.0*std::pow(ie,4.0)*kappa*h*dh[2]*t }};
185  const auto dedt = kappa*h*h*std::pow(ie,4.0);
186  // sources
187  std::array< tk::real, 5 > r;
188  // density source
189  r[0] = drdt;
190  // momentum source
191  r[1] = (g-1.0)*(rho*dedx[0] + ie*drdx[0]);
192  r[2] = (g-1.0)*(rho*dedx[1] + ie*drdx[1]);
193  r[3] = (g-1.0)*(rho*dedx[2] + ie*drdx[2]);
194  // energy source
195  r[4] = rho*dedt + ie*drdt;
196  return r;
197  }

209  //! \brief Query Dirichlet boundary condition value on a given side set for
210  //! all components in this PDE system
211  //! \param[in] e Equation system index, i.e., which compressible
212  //! flow equation system we operate on among the systems of PDEs
213  //! \param[in] t Physical time
214  //! \param[in] deltat Time step size
215  //! \param[in] side Pair of side set ID and node IDs on the side set
216  //! \param[in] coord Mesh node coordinates
217  //! \return Vector of pairs of bool and boundary condition value associated
218  //! to mesh node IDs at which Dirichlet boundary conditions are set. Note
219  //! that instead of the actual boundary condition value, we return the
220  //! increment between t+dt and t, since that is what the solution requires
221  //! as we solve for the soution increments and not the solution itself.
222  static std::unordered_map< std::size_t,
223  std::vector< std::pair< bool, tk::real > > >
224  dirbc( tk::ctr::ncomp_type e,
225  tk::real t,
226  tk::real deltat,
227  const std::pair< const int, std::vector< std::size_t > >& side,
228  const std::array< std::vector< tk::real >, 3 >& coord )
229  {
230  using tag::param; using tag::compflow; using tag::bcdir;
231  using NodeBC = std::vector< std::pair< bool, tk::real > >;
232  std::unordered_map< std::size_t, NodeBC > bc;
233  const auto& ubc = g_inputdeck.get< param, compflow, bcdir >();
234  if (!ubc.empty()) {
235  Assert( ubc.size() > e, "Indexing out of Dirichlet BC eq-vector" );
236  const auto& x = coord[0];
237  const auto& y = coord[1];
238  const auto& z = coord[2];
239  for (const auto& b : ubc[e])
240  if (std::stoi(b) == side.first)
241  for (auto n : side.second) {
242  Assert( x.size() > n, "Indexing out of coordinate array" );
243  auto s = solinc( e, x[n], y[n], z[n], t, deltat );
244  bc[n] = {{ {true,s[0]}, {true,s[1]}, {true,s[2]}, {true,s[3]},
245  {true,s[4]} }};
246  }
247  }
248  return bc;
249  }

272  //! Return field output going to file
273  //! \param[in] e Equation system index, i.e., which compressible
274  //! flow equation system we operate on among the systems of PDEs
275  //! \param[in] offset System offset specifying the position of the system of
276  //! PDEs among other systems
277  //! \param[in] t Physical time
278  //! \param[in] coord Mesh node coordinates
279  //! \param[in] U Solution vector at recent time step
280  //! \return Vector of vectors to be output to file
281  static std::vector< std::vector< tk::real > >
282  fieldOutput( tk::ctr::ncomp_type e,
283  tk::ctr::ncomp_type offset,
284  tk::real t,
285  tk::real V,
286  const std::vector< tk::real >& vol,
287  const std::array< std::vector< tk::real >, 3 >& coord,
288  tk::Fields& U )
289  {
290  // ratio of specific heats
291  tk::real g =
292  g_inputdeck.get< tag::param, tag::compflow, tag::gamma >()[e];
293 
294  std::vector< std::vector< tk::real > > out;
295  auto r = U.extract( 0, offset );
296  auto u = U.extract( 1, offset );
297  auto v = U.extract( 2, offset );
298  auto w = U.extract( 3, offset );
299  auto E = U.extract( 4, offset );
300 
301  // mesh node coordinates
302  const auto& x = coord[0];
303  const auto& y = coord[1];
304  const auto& z = coord[2];
305 
306  out.push_back( r );
307  std::transform( r.begin(), r.end(), u.begin(), u.begin(),
308  []( tk::real s, tk::real& d ){ return d /= s; } );
309  out.push_back( u );
310  std::transform( r.begin(), r.end(), v.begin(), v.begin(),
311  []( tk::real s, tk::real& d ){ return d /= s; } );
312  out.push_back( v );
313  std::transform( r.begin(), r.end(), w.begin(), w.begin(),
314  []( tk::real s, tk::real& d ){ return d /= s; } );
315  out.push_back( w );
316  std::transform( r.begin(), r.end(), E.begin(), E.begin(),
317  []( tk::real s, tk::real& d ){ return d /= s; } );
318  out.push_back( E );
319 
320  auto p = r;
321  for (std::size_t i=0; i<r.size(); ++i)
322  p[i] = (g-1.0)*r[i]*(E[i] - (u[i]*u[i] + v[i]*v[i] + w[i]*w[i])/2.0);
323  out.push_back( p );
324 
325  auto er = r, ee = r;
326  for (std::size_t i=0; i<r.size(); ++i) {
327  auto s = solution( e, x[i], y[i], z[i], t );
328  er[i] = std::pow( r[i] - s[0], 2.0 ) * vol[i] / V;
329  ee[i] = std::pow( E[i] - s[4]/s[0], 2.0 ) * vol[i] / V;
330  r[i] = s[0];
331  u[i] = s[1]/s[0];
332  v[i] = s[2]/s[0];
333  w[i] = s[3]/s[0];
334  E[i] = s[4]/s[0];
335  p[i] = (g-1.0)*r[i]*(E[i] - (u[i]*u[i] + v[i]*v[i] + w[i]*w[i])/2.0);
336  }
337 
338  out.push_back( r );
339  out.push_back( u );
340  out.push_back( v );
341  out.push_back( w );
342  out.push_back( E );
343  out.push_back( p );
344 
345  out.push_back( er );
346  out.push_back( ee );
347 
348  return out;
349  }
