2 /*!
3  \file src/DiffEq/SkewNormal.h
4  \copyright 2012-2015, J. Bakosi, 2016-2018, Los Alamos National Security, LLC.
5  \brief System of skew-normal SDEs
6  \details This file implements the time integration of a system of stochastic
7  differential equations (SDEs), whose invariant is the joint [skew-normal
8  distribution](http://en.wikipedia.org/wiki/Skew_normal_distribution).
9 
10  In a nutshell, the equation integrated governs a set of scalars,
11  \f$x_\alpha\f$, \f$\alpha\!=\!1,\dots,N\f$, as
12  \f[
13  \mathrm{d}x_\alpha(t) = -\frac{1}{T_\alpha}\left[x_\alpha -
14  \lambda_\alpha\sigma^2_\alpha\sqrt{\frac{2}{\pi}} \cdot
15  \frac{\exp{\left(-\lambda_\alpha^2x^2_\alpha/2\right)}}{1+\mathrm{erf}
16  \left( \lambda_\alpha x_\alpha/\sqrt{2}\right)} \right] \mathrm{d}t +
17  \sqrt{\frac{2\sigma^2_\alpha}{T_\alpha}}\mathrm{d}W_\alpha(t).
18  \f]
19  The invariant distribution is the joint skew-normal distribution
20  \f[
21  p(x_\alpha) = \frac{1}{\sigma_\alpha\sqrt{2\pi}} \exp\left(
22  -\frac{x^2_\alpha}{2\sigma^2_\alpha} \right) \left[1 + \mathrm{erf}\left(
23  \frac{\lambda_\alpha x_\alpha}{\sqrt{2}}\right) \right].
24  \f]
25  Here \f$\mathrm{erf}(y) = 2/\sqrt{\pi} \int_0^y \exp(-u^2) \mathrm{d}u\f$,
26  \f$T_\alpha\f$ are time scales, \f$\sigma_\alpha\f$ are variance parameters,
27  \f$\mathrm{d}W_\alpha(t)\f$ is an isotropic [Wiener
28  process](http://en.wikipedia.org/wiki/Wiener_process) with independent
29  increments, and \f$\lambda_\alpha\f$ are the parameters that control the
30  asymmetry and skewness for variable \f$x_\alpha\f$: \f$\lambda_\alpha<0\f$
31  and \f$\lambda_\alpha>0\f$ give negative and positive skewness,
32  respectively, while \f$\lambda_\alpha=0\f$ reduces the system to a set of
33  independent Ornstein-Uhlenbeck processes. See also
34  DiffEq/DiagOrnsteinUhlenbeck.h. For more details on the skew-normal
35  distribution, see http://www.jstor.org/stable/2337278.
36 */

103  //! \brief Advance particles according to the system of skew-normal SDEs
104  //! \param[in,out] particles Array of particle properties
105  //! \param[in] stream Thread (or more precisely stream) ID
106  //! \param[in] dt Time step size
107  void advance( tk::Particles& particles,
108  int stream,
109  tk::real dt,
110  tk::real,
111  const std::map< tk::ctr::Product, tk::real >& )
112  {
113  const auto npar = particles.nunk();
114  for (auto p=decltype(npar){0}; p<npar; ++p) {
115  // Generate Gaussian random numbers with zero mean and unit variance
116  std::vector< tk::real > dW( m_ncomp );
117  m_rng.gaussian( stream, m_ncomp, dW.data() );
118 
119  // Advance all m_ncomp scalars
120  for (ncomp_t i=0; i<m_ncomp; ++i) {
121  tk::real& x = particles( p, i, m_offset );
122  tk::real d = 2.0 * m_sigmasq[i] / m_T[i] * dt;
123  d = (d > 0.0 ? std::sqrt(d) : 0.0);
124  x += - ( x - m_lambda[i] * m_sigmasq[i]
125  * std::sqrt( 2.0 / M_PI )
126  * std::exp( - m_lambda[i] * m_lambda[i] * x * x / 2.0 )
127  / ( 1.0 + std::erf( m_lambda[i] * x / std::sqrt(2.0) ) )
128  ) / m_T[i] * dt
129  + d*dW[i];
130  }
131  }
132  }
