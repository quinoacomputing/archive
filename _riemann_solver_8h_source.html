27 //! \brief Generic Riemann solver interface class for various Riemann solvers
28 //! \details This class uses runtime polymorphism without client-side
29 //! inheritance: inheritance is confined to the internals of this class,
30 //! invisble to client-code. The class exclusively deals with ownership
31 //! enabling client-side value semantics. Credit for this idea goes to Sean

34 //! Papers-and-Presentations.

37  public:
38  //! \brief Constructor taking an object modeling Concept
39  //! \details The object of class T comes pre-constructed.
40  //! \param[in] x Instantiated object of type T given by the template
41  //! argument.
42  template< typename T > explicit RiemannSolver( T x ) :
43  self( tk::make_unique< Model<T> >( std::move(x) ) ) {}
44 
45  //! \brief Constructor taking a function pointer (std::function) to a
46  //! constructor of an object modeling Concept.
47  //! \details Passing std::function allows late execution of the constructor,
48  //! i.e., as late as inside this class' constructor, and thus usage from
49  //! a factory. Note that there are at least two different ways of using
50  //! this constructor:
51  //! - Bind T's constructor arguments and place it in std::function<T()>
52  //! and passing no arguments as args.... This case then instantiates the
53  //! model via its constructor and stores it in here.
54  //! - Bind a single placeholder argument to T's constructor and pass it in
55  //! as host's args..., which then forwards it to model's constructor. This
56  //! allows late binding, i.e., binding the argument only here.
57  //! \see See also the wrapper tk::recordModel() which does the former for
58  //! arbitrary number of model constructor arguments and
59  //! tk::recordModelLate() for a single model constructor argument which
60  //! does the latter, both defined in src/Base/Factory.h.
61  //! \param[in] x Function pointer (std::function) to a constructor of an
62  //! object modeling Concept.
63  //! \param[in] args Zero or more constructor arguments
64  template< typename T, typename...Args >
65  explicit RiemannSolver( std::function<T(Args...)> x, Args&&... args ) :
66  self( tk::make_unique< Model<T> >(
67  std::move( x( std::forward<Args>(args)... ) ) ) ) {}
