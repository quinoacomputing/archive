130  //! \brief Update coefficients using constant coefficients for b' and kappa'
131  //! \details This where the mix number-fraction beta SDE is made consistent
132  //! with the no-mix and fully mixed limits by specifying the SDE
133  //! coefficients, b and kappa as functions of b' and kappa'.
134  void update(
135  char depvar,
136  ncomp_t ncomp,
137  const std::map< tk::ctr::Product, tk::real >& moments,
138  const std::vector< kw::sde_bprime::info::expect::type >& bprime,
139  const std::vector< kw::sde_kappaprime::info::expect::type >& kprime,
140  std::vector< kw::sde_b::info::expect::type >& b,
141  std::vector< kw::sde_kappa::info::expect::type >& k ) const
142  {
143  for (ncomp_t c=0; c<ncomp; ++c) {
144  tk::real m = tk::ctr::lookup( tk::ctr::mean(depvar,c), moments );
145  tk::real v = tk::ctr::lookup( tk::ctr::variance(depvar,c), moments );
146 
147  if (m<1.0e-8 || m>1.0-1.0e-8) m = 0.5;
148  if (v<1.0e-8 && v>1.0-1.0e-8) v = 0.5;
149 
150  b[c] = bprime[c] * (1.0 - v / m / ( 1.0 - m ));
151  k[c] = kprime[c] * v;
152  }
153  }
154 };
