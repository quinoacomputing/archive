53 // *****************************************************************************
54 // Read ASCII STL mesh
55 // \param[in] store Whether to store the facets or not (i.e., only count)
56 // \param[in] x Vertex x coordinates
57 // \param[in] y Vertex y coordinates
58 // \param[in] z Vertex z coordinates
59 // \return Number of vertices counted
60 //! \author J. Bakosi
61 // *****************************************************************************
62 {
63  #if defined(__GNUC__)
64  #pragma GCC diagnostic push
65  #pragma GCC diagnostic ignored "-Winline"
66  #endif
67  // Define possible keywords in ASCI STL file: objects and their correct
68  // string values (for error checking)
69  STLKeyword solid("solid"), facet("facet"), normal("normal"), outer("outer"),
70  loop("loop"), vertex("vertex"), endloop("endloop"),
71  endfacet("endfacet");
72  #if defined(__GNUC__)
73  #pragma GCC diagnostic pop
74  #endif
75 
76  // Read in solids with their facets until eof
77  std::size_t num = 0;
78  while ( !m_inFile.eof() ) {
79  // Start reading new solid
80  std::string solidname;
81  m_inFile >> solid >> solidname;
82 
83  // Read and store facets
84  bool newfacet = true;
85  while (newfacet) {
86  // Coordinates of normal, triangle vertex A, B, and C
87  tk::real nx, ny, nz, Ax, Ay, Az, Bx, By, Bz, Cx, Cy, Cz;
88 
89  // Read in normal (and throw it away as it is redundant)
90  m_inFile >> facet >> normal >> nx >> ny >> nz;
91 
92  // Read in and store off triangle
93  m_inFile >> outer >> loop;
94  m_inFile >> vertex >> Ax >> Ay >> Az;
95  m_inFile >> vertex >> Bx >> By >> Bz;
96  m_inFile >> vertex >> Cx >> Cy >> Cz;
97  m_inFile >> endloop >> endfacet;
98 
99  // Store coordinates of facet if requested
100  if (store) {
101  x[num] = Ax;
102  y[num] = Ay;
103  z[num] = Az;
104  ++num;
105  x[num] = Bx;
106  y[num] = By;
107  z[num] = Bz;
108  ++num;
109  x[num] = Cx;
110  y[num] = Cy;
111  z[num] = Cz;
112  ++num;
113  } else {
114  num += 3; // only increase number of vertices
115  }
116 
117  // Read in next keyword
118  std::streampos back = m_inFile.tellg(); // save file position
119  std::string kw; // not an STLKeyword: no error checking
120  m_inFile >> kw;
121  if (kw == "facet") {
122  m_inFile.seekg(back); // seek back
123  newfacet = true; // there is more to this solid
124  } else if (kw == "endsolid") {
125  m_inFile >> solidname; // read in solidname last time
126  newfacet = false; // solid finished, try to read next one
127  back = m_inFile.tellg();
128  m_inFile >> kw; // try to read on
129  if (!m_inFile.eof()) m_inFile.seekg(back);
130  } else {
131  Throw( "Corruption in ASCII STL file while parsing keyword '" + kw +
132  "': keyword 'endfacet' must be followed by either 'facet' or "
133  "'endsolid'" );
134  }
135  } // while (newfacet)
136  } // while (newsolid)
137 
138  // Seek to beginning of file
139  m_inFile.seekg(0, std::ios::beg);
140 
141  // Return number of vertices
142  return num;
143 }
