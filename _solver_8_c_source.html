3  \file src/LinSys/Solver.C
4  \copyright 2012-2015, J. Bakosi, 2016-2018, Los Alamos National Security, LLC.
5  \brief Charm++ chare linear system merger group to solve a linear system
6  \details Charm++ chare linear system merger group used to collect and
7  assemble the left hand side matrix (lhs), the right hand side (rhs) vector,
8  and the solution (unknown) vector from individual worker
9  chares. Beside collection and assembly, the system is also solved. The
10  solution is outsourced to hypre, an MPI-only library. Once the solution is
11  available, the individual worker chares are updated with the new solution.
12 
13  The implementation uses the Charm++ runtime system and is fully
14  asynchronous, overlapping computation and communication. The algorithm
15  utilizes the structured dagger (SDAG) Charm++ functionality. The high-level
16  overview of the algorithm structure and how it interfaces with Charm++ is
17  discussed in the Charm++ interface file src/LinSys/solver.ci. See also
18  src/LinSys/Solver.h for a discussion of the asynchronous call logic.
19 */

131 void
132 Solver::bounds( int p, std::size_t lower, std::size_t upper )
133 // *****************************************************************************
134 // Receive lower and upper global node IDs all PEs will operate on
135 //! \param[in] p PE whose bounds being received
136 //! \param[in] lower Lower index of the global rows on my PE
137 //! \param[in] upper Upper index of the global rows on my PE
138 // *****************************************************************************
139 {
140  Assert( lower < upper, "Lower bound must be lower than the upper bound: "
141  "(" + std::to_string(lower) + "..." + std::to_string(upper) +
142  ") sent by PE " + std::to_string(p) );
143 
144  // Store our bounds
145  if (p == CkMyPe()) {
146  m_lower = lower;
147  m_upper = upper;
148  }
149 
150  // Store inverse of PE-division map stored on all PE
151  m_div[ {lower,upper} ] = p;
152 
153  // If we have all PEs' bounds, signal the runtime system to continue
154  if (m_div.size() == static_cast<std::size_t>(CkNumPes())) {
155  // Create my PE's lhs matrix distributed across all PEs
156  m_A.create( m_lower*m_ncomp, m_upper*m_ncomp );
157  // Create my PE's rhs and unknown vectors distributed across all PEs
158  m_b.create( m_lower*m_ncomp, m_upper*m_ncomp );
159  m_x.create( m_lower*m_ncomp, m_upper*m_ncomp );
160  // Create linear solver
161  m_solver.create();
162  bounds_complete();
163  }

210 void
211 Solver::charecom( const inciter::CProxy_MatCG& worker,
212  int fromch,
213  const std::vector< std::size_t >& row )
214 // *****************************************************************************
215 // Chares contribute their global row ids for establishing communications
216 //! \param[in] worker Charm chare array proxy contribution coming from
217 //! \param[in] fromch Charm chare array index contribution coming from
218 //! \param[in] row Global mesh point (row) indices contributed
219 //! \note This function does not have to be declared as a Charm++ entry
220 //! method since it is always called by chares on the same PE.
221 // *****************************************************************************
222 {
223  // Store worker proxy for solution update later
224  m_worker = worker;
225 
226  // Collect chare ids of workers on my PE
227  m_myworker.push_back( fromch );
228 
229  // Store rows owned and pack those to be exported, also build import map
230  // used to test for completion
231  std::map< int, std::set< std::size_t > > exp;
232  for (auto gid : row) {
233  if (gid >= m_lower && gid < m_upper) { // if own
234  m_rowimport[ fromch ].push_back( gid );
235  m_row.insert( gid );
236  } else exp[ pe(gid) ].insert( gid );
237  }
238 
239  // Export non-owned parts to fellow branches that own them
240  m_nperow += exp.size();
241  for (const auto& p : exp) {
242  auto tope = static_cast< int >( p.first );
243  thisProxy[ tope ].addrow( fromch, CkMyPe(), p.second );
244  }
245 
246  if (comcomplete()) contribute( m_cb.get< tag::com >() );
247 }

289 // Chares contribute their solution nonzero values
290 //! \param[in] fromch Charm chare array index contribution coming from
291 //! \param[in] gid Global row indices of the vector contributed
292 //! \param[in] solution Portion of the unknown/solution vector contributed
293 //! \note This function does not have to be declared as a Charm++ entry

295 // *****************************************************************************

301  // exported, also build import map used to test for completion
302  std::map< int, std::map< std::size_t, std::vector< tk::real > > > exp;

307  m_sol[ gid[i] ] = solution[i];

340 void
341 Solver::charelhs( int fromch,
342  const std::vector< std::size_t >& gid,
343  const std::pair< std::vector< std::size_t >,
344  std::vector< std::size_t > >& psup,
345  const tk::Fields& lhsd,
346  const tk::Fields& lhso )
347 // *****************************************************************************
348 // Chares contribute their matrix nonzero values
349 //! \param[in] fromch Charm chare array index contribution coming from
350 //! \param[in] gid Global row indices of the matrix chunk contributed
351 //! \param[in] psup Points surrounding points using local indices. See also
352 //! tk::genPsup().
353 //! \param[in] lhsd Portion of the left-hand side matrix contributed,
354 //! containing non-zero values (for all scalar components of the equations
355 //! solved) as a sparse matrix diagonal
356 //! \param[in] lhso Portion of the left-hand side matrix contributed,
357 //! containing non-zero values (for all scalar components of the equations
358 //! solved) as a sparse matrix off-diagonal entries in compressed row
359 //! storage format
360 //! \note This function does not have to be declared as a Charm++ entry
361 //! method since it is always called by chares on the same PE.
362 // *****************************************************************************
363 {
364  Assert( psup.second.size()-1 == gid.size(),
365  "Number of mesh points and number of global IDs unequal" );
366  Assert( psup.second.size()-1 == lhsd.nunk(),
367  "Number of mesh points and number of diagonals unequal" );
368  Assert( psup.first.size() == lhso.nunk(),
369  "Number of off-diagonals and their number of indices unequal" );
370 
371  // Store matrix nonzero values owned and pack those to be exported, also
372  // build import map used to test for completion
373  std::map< int, std::map< std::size_t,
374  std::map< std::size_t,
375  std::vector< tk::real > > > > exp;
376 
377  for (std::size_t i=0; i<gid.size(); ++i)
378  if (gid[i] >= m_lower && gid[i] < m_upper) { // if own
379  m_lhsimport[ fromch ].push_back( gid[i] );
380  auto& row = m_lhs[ gid[i] ];
381  row[ gid[i] ] += lhsd[i];
382  for (auto j=psup.second[i]+1; j<=psup.second[i+1]; ++j)
383  row[ gid[ psup.first[j] ] ] += lhso[j];
384  } else {
385  auto& row = exp[ pe(gid[i]) ][ gid[i] ];
386  row[ gid[i] ] = lhsd[i];
387  for (auto j=psup.second[i]+1; j<=psup.second[i+1]; ++j)
388  row[ gid[ psup.first[j] ] ] = lhso[j];
389  }
390 
391  // Export non-owned matrix rows values to fellow branches that own them
392  for (const auto& p : exp)
393  thisProxy[ p.first ].addlhs( fromch, p.second );
394 
395  if (lhscomplete()) lhs_complete();
396 }

589 void
590 Solver::comfinal()
591 // *****************************************************************************
592 // All communications have been establised among PEs
593 //! \details At this point all solver objects on all PEs must have received
594 //! their global row ids which means that the communications have been
595 //! established among all PEs and this the communications (maps) are final on
596 //! all PEs.
597 // *****************************************************************************
598 {
599  // Assert that all global row indices have been received on my PE. The assert
600  // consists of three necessary conditions, which together comprise the
601  // sufficient condition that all global row indices have been received owned
602  // by this PE.
603  Assert( // 1. have heard from every chare on my PE
604  m_myworker.size() == m_nchare &&
605  // 2. number of rows equals that of the expected on my PE
606  m_row.size() == m_upper-m_lower &&
607  // 3. all fellow PEs have received my row ids contribution
608  m_nperow == 0,
609  // if any of the above is unsatisfied, the row ids are incomplete
610  "Row ids are incomplete on PE " + std::to_string(CkMyPe()) );
611 
612  // now that the global row ids are complete, build Hypre data from it
613  hyprerow();
614 }

1059 void
1060 Solver::lowsolve()
1061 // *****************************************************************************
1062 // Solve low-order linear system
1063 // *****************************************************************************
1064 {
1065  // Set boundary conditions on the low-order system
1066  for (const auto& n : m_bc)
1067  if (n.first >= m_lower && n.first < m_upper) {
1068  // lhs
1069  auto& l = tk::ref_find( m_lowlhs, n.first );
1070  for (std::size_t i=0; i<m_ncomp; ++i)
1071  if (n.second[i].first) l[i] = 1.0;
1072  // rhs (set to zero instead of the solution increment at Dirichlet
1073  // BCs, because for the low order solution the right hand side is the sum
1074  // of the high order right hand side and mass diffusion, so the low order
1075  // system is L = R + D, where L is the lumped mass matrix, R is the high
1076  // order RHS, and D is mass diffusion, and R already has the Dirichlet BC
1077  // set)
1078  auto& r = tk::ref_find( m_lowrhs, n.first );
1079  for (std::size_t i=0; i<m_ncomp; ++i)
1080  if (n.second[i].first) r[i] = 0.0;
1081  }
1082 
1083  // Solve low-order system
1084  Assert( rhscomplete(),
1085  "Values of distributed right-hand-side vector on PE " +
1086  std::to_string( CkMyPe() ) + " is incomplete: cannot solve low "
1087  "order system" );
1088  Assert( lowrhscomplete(),
1089  "Values of distributed mass diffusion rhs vector on PE " +
1090  std::to_string( CkMyPe() ) + " is incomplete: cannot solve low "
1091  "order system" );
1092  Assert( lowlhscomplete(),
1093  "Values of distributed lumped mass lhs vector on PE " +
1094  std::to_string( CkMyPe() ) + " is incomplete: cannot solve low "
1095  "order system" );
1096  Assert( tk::keyEqual( m_rhs, m_lowrhs ), "Row IDs of rhs and mass "
1097  "diffusion rhs vector unequal on PE " + std::to_string( CkMyPe() )
1098  + ": cannot solve low order system" );
1099  Assert( tk::keyEqual( m_rhs, m_lowlhs ), "Row IDs of rhs and lumped mass "
1100  "lhs vector unequal on PE " + std::to_string( CkMyPe() ) + ": "
1101  "cannot solve low order system" );
1102 
1103  auto ir = m_rhs.cbegin();
1104  auto id = m_lowrhs.begin();
1105  auto im = m_lowlhs.cbegin();
1106 
1107  while (ir != m_rhs.cend()) {
1108  const auto& r = ir->second;
1109  const auto& m = im->second;
1110  auto& d = id->second;
1111  Assert( r.size()==m_ncomp && m.size()==m_ncomp && d.size()==m_ncomp,
1112  "Wrong number of components in solving the low order system" );
1113  for (std::size_t i=0; i<m_ncomp; ++i) d[i] = (r[i]+d[i])/m[i];
1114  ++ir; ++id; ++im;
1115  }
1116 
1117  //lowsolve_complete();
1118  updateLowSol();
1119 }
