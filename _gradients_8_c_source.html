27 // Compute gradient at a mesh node
28 //! \param[in] node Node id at which to compute gradient
29 //! \param[in] coord Mesh node coordinates
30 //! \param[in] inpoel Mesh element connectivity
31 //! \param[in] esup Linked lists storing elements surrounding points, see
32 //! tk::genEsup()
33 //! \param[in] U Field vector whose component gradient to compute
34 //! \param[in] c Scalar component to compute gradient of
35 //! \return Gradient of U(c) at mesh node
36 // *****************************************************************************
37 {
38  Assert( c < U.nprop(), "Indexing out of field data" );
39 
40  const auto& x = coord[0];
41  const auto& y = coord[1];
42  const auto& z = coord[2];
43 
44  // storage for gradient and volume at the mesh node
45  std::array< tk::real, 3 > g{{ 0.0, 0.0, 0.0 }};
46  tk::real vol = 0.0;
47 
48  // loop over cells surrounding mesh node
49  for (auto e : tk::Around(esup,node)) {
50  // access node IDs
51  const std::array< std::size_t, 4 > N{{ inpoel[e*4+0], inpoel[e*4+1],
52  inpoel[e*4+2], inpoel[e*4+3] }};
53 
54  // compute element Jacobi determinant
55  const std::array< tk::real, 3 >
56  ba{{ x[N[1]]-x[N[0]], y[N[1]]-y[N[0]], z[N[1]]-z[N[0]] }},
57  ca{{ x[N[2]]-x[N[0]], y[N[2]]-y[N[0]], z[N[2]]-z[N[0]] }},
58  da{{ x[N[3]]-x[N[0]], y[N[3]]-y[N[0]], z[N[3]]-z[N[0]] }};
59  const auto J = tk::triple( ba, ca, da ); // J = 6V
60  Assert( J > 0, "Element Jacobian non-positive" );
61 
62  // shape function derivatives, nnode*ndim [4][3]
63  std::array< std::array< tk::real, 3 >, 4 > grad;
64  grad[1] = tk::crossdiv( ca, da, J );
65  grad[2] = tk::crossdiv( da, ba, J );
66  grad[3] = tk::crossdiv( ba, ca, J );
67  for (std::size_t i=0; i<3; ++i)
68  grad[0][i] = -grad[1][i]-grad[2][i]-grad[3][i];
69 
70  // access field data for scalar component c at nodes of element
71  auto u = U.extract( c, 0, N );
72 
73  // compute nodal volume: every element contributes their volume / 4
74  vol += 5.0*J/120.0;
75 
76  // compute gradient over element weighed by cell volume / 4 and sum to node
77  for (std::size_t j=0; j<3; ++j) {
78  for (std::size_t i=0; i<4; ++i) {
79  g[j] += grad[i][j] * u[i] * 5.0*J/120.0;
80  }
81  }
82  }
83 
84  // divide components of nodal gradient by nodal volume
85  for (std::size_t j=0; j<3; ++j) g[j] /= vol;
86 
87  return g;
88 }
