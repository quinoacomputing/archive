59  // Get group name string passed in by -g

61 
62  // If only select groups to be run, see if there is any that will run
63  bool work = false;
64  if (grp.empty())
65  work = true;
66  else
67  for (const auto& g : groups)
68  if ( g.find("MPI") == std::string::npos && // don't consider MPI groups
69  g.find(grp) != std::string::npos )
70  work = true;
71 
72  // Quit if there is no work to be done
73  if (!work) {

77  mainProxy.finalize( false, true );
78 
79  } else {
80 
81  m_print.endpart();
82  m_print.part( "Serial and Charm++ unit test suite" );
83  m_print.unithead( "Unit tests computed", grp );
84 
85  // Fire up all tests in all groups using the Charm++ runtime system
86  for (const auto& g : groups)
87  if (g.find("MPI") == std::string::npos) { // don't start MPI test groups
88  if (grp.empty()) { // consider all test groups
89  spawngrp( g );

99 void
100 TUTSuite::spawngrp( const std::string& g )
101 // *****************************************************************************
102 // Fire up all tests in a test group
103 //! \param[in] g Name of the test group
104 // *****************************************************************************
105 {
106  ++m_ngroup; // increase number of test groups to run
107 
108  // Add up number of Charm++ migration tests (this is so we know how many to
109  // expect results from)
110  const auto it = m_migrations.find( g );
111  if (it != m_migrations.end()) m_nmigr += it->second;
112 
113  // Asynchronously fire up all tests in test group
114  for (int t=1; t<=g_maxTestsInGroup; ++t) {
115  auto i = m_fromPE0.find( g );
116  if (i != end(m_fromPE0))
117  CProxy_TUTTest< CProxy_TUTSuite >::ckNew( thisProxy, g, t, 0 );
118  else
119  CProxy_TUTTest< CProxy_TUTSuite >::ckNew( thisProxy, g, t );
120  }
121 }

128 //! unittest::TUTTest constructor for the expected structure of status.
129 // *****************************************************************************
130 {

136 
137  // Echo one-liner info on result of test
138  m_print.test( m_ncomplete, m_nfail, status );
139 
140  // Wait for all tests to finish, then quit
141  if (m_nrun == m_ngroup*static_cast<std::size_t>(g_maxTestsInGroup) + m_nmigr)
142  {
143  auto pass =
144  assess( m_print, "serial and Charm++", m_nfail, m_nwarn, m_nskip, m_nexcp,
145  m_ncomplete );
146  // Quit
147  mainProxy.finalize( true, pass );
148  }
