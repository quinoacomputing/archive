79 SiloWriter::write()
80 // *****************************************************************************
81 // Write out Silo file
82 // *****************************************************************************
83 {
84  tk::real* coords[] = { m_mesh.getx(), m_mesh.gety(), m_mesh.getz() };
85  int nnodes = m_mesh.nnodes();
86  int nfaces = nnodes/3;
87  int zshapetype = DB_ZONETYPE_TRIANGLE;
88  int zshapesize = 3;
89  int zshapecnt = nfaces;
90 
91  // Write out STL face connectivity
92  DBPutZonelist2(m_dbfile, "zonelist", nfaces, 3, m_mesh.nodelist(), nnodes,
93  0, 0, 0, &zshapetype, &zshapesize, &zshapecnt, 1, NULL);
94 
95  // Write out STL mesh: no zones, only faces with a simple face connectivity
96  DBPutUcdmesh(m_dbfile, m_mesh.name().c_str(), 3, NULL, coords,
97  nnodes, nfaces, "zonelist", NULL, DB_DOUBLE, NULL);
98 }
