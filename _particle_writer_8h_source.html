68  signal2host_outcomplete( m_host );

77  m_z.insert( end(m_z), begin(z), end(z) );

79  if (++m_nchare == nchare) {

82  m_x.clear(); // prepare for next step

103  /** @name Host signal calls
104  * \brief These functions signal back to the host via a global reduction
105  * originating from each PE branch
106  * \details Singal calls contribute to a reduction on all branches (PEs)
107  * of ParticleWriter to the host, e.g., inciter::CProxy_Transporter,
108  * given by the template argument HostProxy. The signal functions are
109  * overloads on the specialization, e.g., inciter::CProxy_Transporter, of
110  * the ParticleWriter template. They create Charm++ reduction targets via
111  * creating a callback that invokes the typed reduction client, where
112  * host is the proxy on which the reduction target method, given by the
113  * string followed by "redn_wrapper_", e.g., parcomplete(), is called
114  * upon completion of the reduction.
115  *
116  * Note that we do not use Charm++'s CkReductionTarget macro here,
117  * but instead explicitly generate the code that that macro would
118  * generate. To explain why, here is Charm++'s CkReductionTarget macro's
119  * definition, given in ckreduction.h:
120  * \code{.cpp}
121  * #define CkReductionTarget(me, method) \
122  * CkIndex_##me::redn_wrapper_##method(NULL)
123  * \endcode
124  * This macro takes arguments 'me' (a class name) and 'method' a member
125  * function of class 'me' and generates the call
126  * 'CkIndex_<class>::redn_wrapper_<method>(NULL)'. With the overloads the
127  * signal2* functions generate, we do the above macro's job for
128  * ParticleWriter specialized by HostProxy, hard-coded here, as well its
129  * reduction target. This is required since
130  * * Charm++'s CkReductionTarget macro's preprocessing happens earlier
131  * than type resolution and the string of the template argument would
132  * be substituted instead of the type specialized (which is not what
133  * we want here), and
134  * * the template argument class, e.g, CProxy_Transporter, is in a
135  * namespace different than that of ParticleWriter. When a new class
136  * is used to specialize ParticleWriter, the compiler will alert that
137  * a new overload needs to be defined.
138  *
139  * \note This simplifies client-code, e.g., inciter::Transporter, which now
140  * requires no explicit book-keeping with counters, etc. Also a reduction
141  * (instead of a direct call to the host) better utilizes the
142  * communication network as computational nodes can send their aggregated
143  * contribution to other nodes on a network instead of all chares sending
144  * their (smaller) contributions to the same host, (hopefully)
145  * implemented using a tree among the PEs.
146  * \see http://charm.cs.illinois.edu/manuals/html/charm++/manual.html,
147  * Sections "Processor-Aware Chare Collections" and "Chare Arrays".
148  * */
149  ///@{
150  //! \brief Signal back to host that the output of particles is complete
151  void signal2host_outcomplete( const inciter::CProxy_Transporter& host )
152  {
153  using inciter::CkIndex_Transporter;
154  Group::contribute(
155  CkCallback(CkIndex_Transporter::redn_wrapper_outcomplete(NULL), host) );
156  }
157  ///@}
158  #if defined(__clang__)
159  #pragma clang diagnostic pop
160  #endif
