26 //! Register class into factory with given key. This is used to register a
27 //! derived-class object's constructor (deriving from some base class) to a
28 //! factory. The factory itself is a std::map< key, std::function< Child*() > >,
29 //! i.e., an associative container, associating some key to a std::function
30 //! object holding a pointer of Child's base class constructor. The constructor
31 //! and its bound arguments are stored via boost::factory, which, in this
32 //! use-case, yields the correct function object of type Base constructor pointer
33 //! and thus facilitates runtime polymorphism. This function works in conjunction
34 //! with boost::factory, i.e., uses reference semantics (works with storing
35 //! pointers of objects). For a simple example on how to use this function, see
36 //! UnitTest/tests/Base/Factory.h.
37 //! \param[in] f Factory to register to (std::map with value using reference
38 //! semantics)
39 //! \param[in] key Key used to identify the entry in the factory
40 //! \param[in] args Variable number of arguments to pass to the constructor
41 //! being registered. Note that the constructor arguments are only bound to
42 //! the constructor and stored in the factory (an std::map with value using
43 //! reference semantics). The object is not instantiated here, i.e., the
44 //! constructor is not called here. The object can be instantiated by function
45 //! instantiate. \see instantiate
46 template< class C, class Key, class Factory, typename... ConstructorArgs >
47 void record( Factory& f, const Key& key, ConstructorArgs&&... args ) {
48  f.emplace( key,
49  std::bind( boost::factory< C* >(),
50  std::forward< ConstructorArgs >( args )... ) );
51 }

53 //! Instantiate object from factory. Factory must have a mapped_value which must
54 //! have a result_type ptr, e.g., std::map< Key, std::function< Obj*() > >. This
55 //! wrapper function can be used to instantiate an derived-class object from a
56 //! factory, repeatedly filled with wrapper function 'record' above. The
57 //! factory, as described in the documentation of 'record', stores base class
58 //! pointers in an associative container, thereby facilitating runtime
59 //! polymorphism and a simple lookup-and-instantiate-style object creation. The
60 //! object instantiated is of type Child class. This function works in
61 //! conjunction with boost::factory, i.e., uses reference semantics (works with
62 //! storing pointers of objects). For a simple example on how to
63 //! use this function, see UnitTest/tests/Base/Factory.h.
64 //! \param[in] f Factory to instantiate object from (std::map with value using
65 //! reference semantics)
66 //! \param[in] key Key used to identify the object to instantiate from factory
67 //! \return std::unique_ptr pointing to the object instantiated from factory
68 //! \see record
69 template< class Factory, class Key,
70  class Obj = typename std::remove_pointer<
71  typename Factory::mapped_type::result_type >::type >
72 std::unique_ptr< Obj > instantiate( const Factory& f, const Key& key ) {
73  const auto it = f.find( key );
74  Assert( it != end( f ), "No such object registered in factory" );
75  return std::unique_ptr< Obj >( it->second() );
76 }

78 //! Register "model" class of "host" into factory with given key. This wrapper
79 //! can be used to in a similar manner to 'record', but uses
80 //! boost::value_factory to bind the model object constructor to its arguments
81 //! and place it in the associative container storing host class objects. The
82 //! container is thus of type std::map< key, std::function< T() > >, i.e.,
83 //! associating a key to a function holding a constructor (and not its
84 //! pointer). Runtime polymorphism here is realized entirely within the "base"
85 //! class. See walker::DiffEq in DiffEq/DiffEq.h for an example and more
86 //! information on runtime polymorphism without client-side inheritance. As a
87 //! result, this wrapper works with factories that use value semantics, as opposed
88 //! to 'record' and instantiate which work with reference semantics factories.
89 //! In order to differentiate between runtime polymorphic classes using
90 //! reference semantics, consistent with classes realizing runtime polymorphism
91 //! without client-side inheritance, we call Host as the "Base" class and Model
92 //! as the "derived" (or child) class. This wrapper function works in
93 //! conjunction with boost::value_factory, i.e., uses value semantics (works
94 //! with storing objects instead of object pointers). For a simple example on
95 //! how to use this function, see UnitTest/tests/Base/Factory.h.
96 //! \param[in] f Factory to register to (std::map with value using value
97 //! semantics)
98 //! \param[in] key Key used to identify the entry in the factory
99 //! \param[in] args Variable number of arguments to pass to the constructor
100 //! being registered. Note that the constructor arguments are only bound to
101 //! the constructor and stored in the factory (an std::map with value using
102 //! value semantics). The object is not instantiated here, i.e., the
103 //! constructor is not called here. The object can be instantiated by simply
104 //! calling the function call operator () on the mapped value. For an example,
105 //! RNGStack::selected() in RNG/RNGStack.C.
106 template< class Host, class ModelConstructor, class Factory, class Key,
107  typename... ModelConstrArgs >
108 void recordModel( Factory& f, const Key& key, ModelConstrArgs&&... args ) {
109  // Bind model constructor to its arguments
110  std::function< ModelConstructor() > c =
111  std::bind( boost::value_factory< ModelConstructor >(),
112  std::forward< ModelConstrArgs >( args )... );
113  // Bind host to std::function of model constructor and place in factory
114  f.emplace( key, std::bind( boost::value_factory< Host >(), std::move(c) ) );
115 }

117 //! Register model class of host into factory with given key using late binding.
118 //! This variant of 'record' is very similar to 'recordModel', but registers a
119 //! model class constructor to a factory with late binding of the constructor
120 //! argument. Late binding allows specifying the constructor argument at the
121 //! time when the object is instantiated instead of at the time when it is
122 //! registered. This has all the benefits of using a factory and allows passing
123 //! information into the model object only when it is available. The late bind
124 //! is facilitated via boost::bind instead of std::bind using a placeholder,
125 //! _1, which stands for the first argument (bound later, i.e., not here). The
126 //! value of the model constructor argument is then not used here, only its
127 //! type, used to perform the late binding. The binding happens to both the
128 //! model constructor via std::function (passed to the host constructor) as well
129 //! as explicitly to the host constructor. Prescribing late binding to the model
130 //! constructor ensures that the compiler requires the argument to the model
131 //! constructor, i.e., ensures that the host constructor is required to pass the
132 //! argument to the model constructor. Prescribing late binding to the host
133 //! constructor puts in the actual request that an argument (with the correct
134 //! type) must be passed to the host constructor at instantiate time, which then
135 //! will forward it to the model constructor. See also, for example,
136 //! walker::DiffEq's corresponding constructor. An example of client-side code
137 //! is in walker::DiffEqStack::registerDiffEq for registration into factory, and
138 //! DiffEqStack::createDiffEq for instantiation late-passing the argument.
139 //! \param[in] f Factory to register to (std::map with value using value
140 //! semantics)
141 //! \param[in] key Key used to identify the entry in the factory
142 //! \warning Only works with a single constructor argument
143 template< class Host, class ModelConstructor, class Factory, class Key,
144  typename ModelConstrArg >
145 void recordModelLate( Factory& f, const Key& key, ModelConstrArg ) {
146  // Prescribe late binding the model constructor to its single argument
147  std::function< ModelConstructor(const ModelConstrArg&) > c =
148  boost::bind( boost::value_factory< ModelConstructor >(), boost::arg<1>() );
149  // Bind host to std::function of model constructor and place in factory and
150  // also explicitly bind single model constructor argument to host constructor
151  f.emplace( key,
152  boost::bind( boost::value_factory< Host >(), std::move(c), boost::arg<1>() )
153  );
154 }

156 //! Register Charm++ model class of host into factory with given key. We bind a
157 //! host constructor to its arguments of which the first one is a std::function
158 //! holding a model constructor type (modeling, i.e., used polymorhically with
159 //! host), followed by an optional number of others (possibly zero) with
160 //! arbitrary types. Note that the model constructor is a nullptr (default-
161 //! constructed) and only used to forward its type to the call site inside
162 //! std::function. The host constructor function is then placed into the
163 //! factory. This is because Charm++ chares do not explicitly invoke
164 //! constructors, only call ckNew() on their proxy, which requires all
165 //! constructor arguments to be present and forwarded to the actual constructor
166 //! that is only called at a later point in time. This can then be used by those
167 //! constructors of hosts that invoke the model constructors' proxies' ckNew()
168 //! and ignore the std::function. See, e.g., rngtest::Battery() and the
169 //! associated unit tests in UnitTest/tests/Base/Factory.h.
170 //! \param[in] f Factory to register to (std::map with value using value
171 //! semantics)
172 //! \param[in] key Key used to identify the entry in the factory
173 //! \param[in] args Variable number of arguments to pass to the constructor
174 //! being registered.
175 template< class Host, class ModelConstructor, class Factory, class Key,
176  typename... ModelConstrArgs >
177 void recordCharmModel( Factory& f, const Key& key, ModelConstrArgs&&... args ) {
178  f.emplace( key, std::bind( boost::value_factory< Host >(),
179  std::function< ModelConstructor() >(), // nullptr
180  std::forward< ModelConstrArgs >( args )...) );
181 }
