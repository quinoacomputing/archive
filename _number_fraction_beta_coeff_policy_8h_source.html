14  \code{.cpp}
15  CoeffPolicyName(
16  tk::ctr::ncomp_type ncomp,
17  const std::vector< kw::sde_b::info::expect::type >& b_,
18  const std::vector< kw::sde_S::info::expect::type >& S_,
19  const std::vector< kw::sde_kappa::info::expect::type >& k_,
20  const std::vector< kw::sde_rho2::info::expect::type >& rho2_,
21  const std::vector< kw::sde_rcomma::info::expect::type >& rcomma_,
22  std::vector< kw::sde_b::info::expect::type >& b,
23  std::vector< kw::sde_S::info::expect::type >& S,
24  std::vector< kw::sde_kappa::info::expect::type >& k,
25  std::vector< kw::sde_rho2::info::expect::type >& rho2_,
26  std::vector< kw::sde_rcomma::info::expect::type >& rcomma_ );
27  \endcode
28  where
29  - ncomp denotes the number of scalar components of the system of
30  number-fraction beta SDEs.
31  - Constant references to b_, S_, k_, rho2_, and rcomma_, which denote five
32  vectors of real values used to initialize the parameter vectors of the
33  system of number-fraction beta SDEs. The length of the vectors must be
34  equal to the number of components given by ncomp.
35  - References to b, S, k, rho2_, and rcomma, which denote the parameter
36  vectors to be initialized based on b_, S_, k_, rho2_, and rcomma_.
37 
38  - Must define the static function _type()_, returning the enum value of the
39  policy option. Example:
40  \code{.cpp}
41  static ctr::CoeffPolicyType type() noexcept {
42  return ctr::CoeffPolicyType::CONSTANT;
43  }
44  \endcode
45  which returns the enum value of the option from the underlying option
46  class, collecting all possible options for coefficients policies.
47 */
