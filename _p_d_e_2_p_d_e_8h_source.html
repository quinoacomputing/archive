55  //! \brief Constructor taking a function pointer to a constructor of an
56  //! object modeling Concept.
57  //! \details Passing std::function allows late execution of the constructor,
58  //! i.e., as late as inside this class' constructor, and thus usage from
59  //! a factory. Note that there are at least two different ways of using
60  //! this constructor:
61  //! - Bind T's constructor arguments and place it in std::function<T()>
62  //! and passing no arguments as args.... This case then instantiates the
63  //! model via its constructor and stores it in here.
64  //! - Bind a single placeholder argument to T's constructor and pass it in
65  //! as host's args..., which then forwards it to model's constructor. This
66  //! allows late binding, i.e., binding the argument only here.
67  //! \see See also the wrapper tk::recordModel() which does the former and
68  //! tk::recordModelLate() which does the latter, both defined in
69  //! src/Base/Factory.h.
70  //! \param[in] x Function pointer to a constructor of an object modeling
71  //! Concept.
72  //! \param[in] args Zero or more constructor arguments
73  template< typename T, typename...Args >
74  explicit PDE( std::function<T(Args...)> x, Args&&... args ) :
75  self( tk::make_unique< Model<T> >(
