161 std::size_t
162 ExodusIIMeshReader::readElemBlockIDs()
163 // *****************************************************************************
164 // Read element block IDs from ExodusII file
165 //! \return Total number of nodes in mesh
166 // *****************************************************************************
167 {
168  // Read ExodusII file header
169  auto nnode = readHeader();
170 
171  std::vector< int > eid( m_neblk );
172 
173  // Read element block ids
174  ErrChk( ex_get_ids( m_inFile, EX_ELEM_BLOCK, eid.data()) == 0,
175  "Failed to read element block ids from ExodusII file: " +
176  m_filename );
177 
178  // Fill element block ID vector
179  for (auto id : eid) {
180  char eltype[MAX_STR_LENGTH+1];
181  int n, nnpe, nattr;
182 
183  // Read element block information
184  ErrChk( ex_get_block( m_inFile, EX_ELEM_BLOCK, id, eltype, &n, &nnpe,
185  &nattr, nullptr, nullptr ) == 0,
186  "Failed to read element block information from ExodusII file: " +
187  m_filename );
188 
189  // Store ExodusII element block ID
190  m_eid.push_back( id );
191 
192  // Store ExodusII element block IDs mapped to elem type, and add up the
193  // number of elements per for each elem type
194  if (nnpe == 4) { // tetrahedra
195  auto e = static_cast< std::size_t >( ExoElemType::TET );
196  m_eidt[ e ].push_back( id );
197  m_nel[ e ].push_back( static_cast< std::size_t >( n ) );
198  Assert( m_eidt[e].size() == m_nel[e].size(), "Size mismatch" );
199  } else if (nnpe == 3) { // triangles
200  auto e = static_cast< std::size_t >( ExoElemType::TRI );
201  m_eidt[ e ].push_back( id );
202  m_nel[ e ].push_back( static_cast< std::size_t >( n ) );
203  Assert( m_eidt[e].size() == m_nel[e].size(), "Size mismatch" );
204  }
205  }
206 
207  return nnode;
208 }

211 void
212 ExodusIIMeshReader::readAllElements( UnsMesh& mesh )
213 // *****************************************************************************
214 // Read all element blocks and mesh connectivity from ExodusII file
215 //! \param[inout] mesh Unstructured mesh object to store mesh in
216 // *****************************************************************************
217 {
218  // Read element block ids
219  readElemBlockIDs();
220 
221  for (auto id : m_eid) {
222  char eltype[MAX_STR_LENGTH+1];
223  int nel, nnpe, nattr;
224 
225  // Read element block information
226  ErrChk( ex_get_block( m_inFile, EX_ELEM_BLOCK, id, eltype, &nel, &nnpe,
227  &nattr, nullptr, nullptr ) == 0,
228  "Failed to read element block information from ExodusII file: " +
229  m_filename );
230 
231  // Read element connectivity
232  auto connectsize = static_cast< std::size_t >( nel*nnpe );
233  if (nnpe == 4) { // tetrahedra
234 
235  std::vector< int > inpoel( connectsize );
236  ErrChk( ex_get_conn( m_inFile, EX_ELEM_BLOCK, id, inpoel.data(),
237  nullptr, nullptr ) == 0,
238  "Failed to read " + std::string(eltype) + " element connectivity from "
239  "ExodusII file: " + m_filename );
240  for (auto n : inpoel)
241  mesh.tetinpoel().push_back( static_cast< std::size_t >( n ) );
242 
243  } else if (nnpe == 3) { // triangles
244 
245  std::vector< int > inpoel( connectsize );
246  ErrChk( ex_get_conn( m_inFile, EX_ELEM_BLOCK, id, inpoel.data(),
247  nullptr, nullptr ) == 0,
248  "Failed to read " + std::string(eltype) + " element connectivity from "
249  "ExodusII file: " + m_filename );
250  for (auto n : inpoel)
251  mesh.triinpoel().push_back( static_cast< std::size_t >( n ) );
252 
253  }
254  }
255 
256  // Shift node IDs to start from zero
257  shiftToZero( mesh.triinpoel() );
258  shiftToZero( mesh.tetinpoel() );
259 }

261 void
262 ExodusIIMeshReader::readElements( const std::array< std::size_t, 2 >& ext,
263  tk::ExoElemType elemtype,
264  std::vector< std::size_t >& conn ) const
265 // *****************************************************************************
266 // Read element connectivity of a number of mesh cells from ExodusII file
267 //! \param[in] ext Extents of element IDs whose connectivity to read:
268 //! [from...till), using zero-based element IDs, where 'from' >=0, inclusive
269 //! and 'till < 'maxelements', where 'maxelements' is the total number of
270 //! elements of all element blocks in the file of the requested cell type.
271 //! Note that 'maxelements' can be queried by nelem().
272 //! \param[in] elemtype Element type
273 //! \param[inout] conn Connectivity vector to push to
274 //! \note Must be preceded by a call to readElemBlockIDs()
275 //! \details This function takes the extents of element IDs in a zero-based
276 //! fashion. These input extents can be thought of "absolute" extents that
277 //! denote lowest and the largest-1 element IDs to be read from file.
278 // *****************************************************************************
279 {
280  Assert( tk::sumsize(m_eidt) > 0,
281  "A call to this function must be preceded by a call to "
282  "ExodusIIMeshReader::readElemBlockIDs()" );
283  Assert( ext[0] <= ext[1] &&
284  ext[0] < nelem(elemtype) &&
285  ext[1] < nelem(elemtype),
286  "Invalid element ID extents. Of the requested extents [from...till), "
287  "'from' must be lower than or equal to 'till', and they must be in "
288  "the range [0...maxelements), where 'maxelements' is the total "
289  "number of elements of all element blocks in the file of the "
290  "requested cell type. Requested element ID extents: ["
291  + std::to_string(ext[0]) + "..." + std::to_string(ext[1])
292  + "), 'maxelements' of cell type with "
293  + std::to_string( m_nnpe[ static_cast<std::size_t>(elemtype) ] )
294  + " nodes per cell in file '" + m_filename + "': "
295  + std::to_string( nelem( elemtype ) ) );
296 
297  auto e = static_cast< std::size_t >( elemtype );
298  // List of number of elements of all blocks of element type requested
299  const auto& nel = m_nel[e];
300  // List of element block IDs for element type requested
301  const auto& bid = m_eidt[e];
302 
303  // Compute lower and upper element block ids to read from based on extents
304  std::size_t lo_bid = 0, hi_bid = 0, offset = 0;
305  for (std::size_t b=0; b<nel.size(); ++b) {
306  std::size_t lo = offset; // lo (min) elem ID in block
307  std::size_t hi = offset + nel[b] - 1; // hi (max) elem ID in block
308  if (ext[0] >= lo && ext[0] <= hi) lo_bid = b;
309  if (ext[1] >= lo && ext[1] <= hi) hi_bid = b;
310  offset += nel[b];
311  }
312 
313  Assert( lo_bid < nel.size() && lo_bid < bid.size(),
314  "Invalid start block ID" );
315  Assert( hi_bid < nel.size() && hi_bid < bid.size(),
316  "Invalid end block ID" );
317 
318  // Compute relative extents based on absolute ones for each block to read from
319  std::vector< std::array< std::size_t, 2 > > rext;
320  offset = 0;
321  for (std::size_t b=0; b<lo_bid; ++b) offset += nel[b];
322  for (std::size_t b=lo_bid; b<=hi_bid; ++b) {
323  std::size_t lo = offset;
324  std::size_t hi = offset + nel[b] - 1;
325  std::size_t le = 1, he = nel[b];
326  if (ext[0] >= lo && ext[0] <= hi) le = ext[0] - lo + 1;
327  if (ext[1] >= lo && ext[1] <= hi) he = ext[1] - lo + 1;
328  Assert( le >= 1 && le <= nel[b] && he >= 1 && he <= nel[b],
329  "Relative index out of block" );
330  rext.push_back( {{ le, he }} );
331  offset += nel[b];
332  }
333 
334  Assert( std::accumulate(
335  std::next(rext.cbegin()), rext.cend(), rext[0][1]-rext[0][0]+1,
336  []( std::size_t n, const std::array< std::size_t, 2 >& r )
337  { return n + r[1] - r[0] + 1; }
338  ) == ext[1]-ext[0]+1,
339  "Total number of elements to read incorrect, requested extents: " +
340  std::to_string(ext[0]) + " ... " + std::to_string(ext[1]) );
341 
342  std::vector< int > inpoel;
343 
344  // Read element connectivity from file
345  std::size_t B = 0;
346  for (auto b=lo_bid; b<=hi_bid; ++b, ++B) {
347  const auto& r = rext[B];
348  std::vector< int > c( (r[1]-r[0]+1) * m_nnpe[e] );
349  ErrChk( ex_get_partial_conn( m_inFile,
350  EX_ELEM_BLOCK,
351  bid[b],
352  static_cast< int64_t >( r[0] ),
353  static_cast< int64_t >( r[1]-r[0]+1 ),
354  c.data(),
355  nullptr,
356  nullptr ) == 0,
357  "Failed to read element connectivity of elements [" +
358  std::to_string(r[0]) + "..." + std::to_string(r[1]) +
359  "] from element block " + std::to_string(bid[b]) + " in ExodusII "
360  "file: " + m_filename );
361  inpoel.reserve( inpoel.size() + c.size() );
362  std::move( begin(c), end(c), std::back_inserter(inpoel) );
363  }
364 
365  Assert( inpoel.size() == (ext[1]-ext[0]+1)*m_nnpe[e],
366  "Failed to read element connectivity of elements [" +
367  std::to_string(ext[0]) + "..." + std::to_string(ext[1]) + ") from "
368  "ExodusII file: " + m_filename );
369 
370  // Put in element connectivity using zero-based node indexing
371  for (auto& i : inpoel) --i;
372  conn.reserve( conn.size() + inpoel.size() );
373  std::move( begin(inpoel), end(inpoel), std::back_inserter(conn) );
374 }

378  std::vector< std::size_t >& conn )
379 // *****************************************************************************
380 // Read face connectivity of a number of boundary faces from ExodusII file
381 //! \param[in] nbfac Number of boundary faces
382 //! \param[inout] conn Connectivity vector to push to
383 //! \details This function reads-in the total number of boundary faces
384 //! also called triangle-elements in the EXO2 file, and their connectivity.
385 // *****************************************************************************
386 {
387  ErrChk( nbfac > 0, "Number of boundary faces must be larger than zero" );
388 
389  std::size_t nnpf(3);
390 
391  // Read triangle boundary-face connectivity
392  std::vector< std::size_t > l_triinpoel;
393  readElements( {{0,nbfac-1}}, tk::ExoElemType::TRI, l_triinpoel );
394 
395  std::size_t count(0);
396  // Use node_map to get the global-IDs of the face-node connectivity
397  for (std::size_t f=0; f<nbfac; ++f)
398  {
399  count = f*nnpf;
400  for (std::size_t i=0; i<nnpf; ++i)
401  {
402  conn.push_back( l_triinpoel[count+i] );
403  }
404  }

408 ExodusIIMeshReader::readNodemap()
409 // *****************************************************************************
410 // Read local to global node-ID map from ExodusII file
411 //! \return node_map Vector mapping the local Exodus node-IDs to global node-IDs
412 //! \details The node-map is required to get the "Exodus-global" node-IDs from
413 //! the "Exodus-internal" node-IDs, which are returned from the exodus APIs.
414 //! The node-IDs in the exodus file are referred to as the "Exodus-global"
415 //! node-IDs or "fileIDs" in Quinoa.
416 // *****************************************************************************
417 {
418  // Read triangle boundary-face connectivity
419  auto nnode = readElemBlockIDs();
420 
421  // Create array to store node-number map
422  std::vector< int > node_map( nnode );
423 
424  // Read in the node number map to map the above nodes to the global node-IDs
425  ErrChk( ex_get_id_map( m_inFile, EX_NODE_MAP, node_map.data() ) == 0,
426  "Failed to read node map length from ExodusII file: " );
427 
428  std::vector< std::size_t > node_map1( nnode );
429 
430  for (std::size_t i=0; i<nnode; ++i)
431  {
432  node_map1[i] = static_cast< std::size_t >(node_map[i]-1);
433  }
434 
435  return node_map1;

450

464  ErrChk( ex_get_side_set_node_list_len( m_inFile, i, &nnode ) == 0,
465  "Failed to read side set " + std::to_string(i) + " node list "

468  std::vector< int > df( static_cast< std::size_t >( nface ) );
469  std::vector< int > nodes( static_cast< std::size_t >( nnode ) );
470  // Read in node list for side set i
471  ErrChk( ex_get_side_set_node_list( m_inFile, i, df.data(), nodes.data() )
472  == 0, "Failed to read node list of side set " +
473  std::to_string(i) + " from ExodusII file: " +
474  m_filename );

476  tk::unique( nodes );

480  }

505  "Failed to read side set ids from ExodusII file: " + m_filename );

522  "Failed to read side set " + std::to_string(i) + " parameters "

526  nbfac += static_cast< std::size_t >(nface);
527 
528  Assert(nface > 0, "Number of faces = 0 in side set" + std::to_string(i));
529  std::vector< int > tbface( static_cast< std::size_t >( nface ) );

531  std::vector< int > gbelem( static_cast< std::size_t >( nface ) );
532
