32  //! Evaluate analytical solution at (x,y,z) for all components
33  //! \param[in] e Equation system index, i.e., which compressible
34  //! flow equation system we operate on among the systems of PDEs
35  //! \param[in] x X coordinate where to evaluate the solution
36  //! \param[in] y Y coordinate where to evaluate the solution
37  //! \param[in] z Z coordinate where to evaluate the solution
38  //! \return Values of all components evaluated at (x,y,z)
39  static std::array< tk::real, 5 >
40  solution( ncomp_t e, tk::real x, tk::real y, tk::real z, tk::real ) {
41  using tag::param; using tag::compflow;
42  // manufactured solution parameters
43  const auto& a = g_inputdeck.get< param, compflow, tag::alpha >()[ e ];
44  const auto& b = g_inputdeck.get< param, compflow, tag::beta >()[ e ];
45  const auto& p0 = g_inputdeck.get< param, compflow, tag::p0 >()[ e ];
46  // ratio of specific heats
47  tk::real g = g_inputdeck.get< param, compflow, tag::gamma >()[ e ];
48  // velocity
49  const tk::real ru = a*x - b*y;
50  const tk::real rv = b*x + a*y;
51  const tk::real rw = -2.0*a*z;
52  // total specific energy
53  const tk::real rE = (ru*ru+rv*rv+rw*rw)/2.0 + (p0-2.0*a*a*z*z)/(g-1.0);
54  return {{ 1.0, ru, rv, rw, rE }};
55  }

67  //! Compute and return source term for vortical flow manufactured solution
68  //! \param[in] e Equation system index, i.e., which compressible
69  //! flow equation system we operate on among the systems of PDEs
70  //! \param[in] x X coordinate where to evaluate the solution
71  //! \param[in] y Y coordinate where to evaluate the solution
72  //! \param[in] z Z coordinate where to evaluate the solution
73  //! \return Array of reals containing the source for all components
74  static std::array< tk::real, 5 >
75  src( ncomp_t e, tk::real x, tk::real y, tk::real z, tk::real ) {
76  using tag::param; using tag::compflow;
77  // manufactured solution parameters
78  const auto& a = g_inputdeck.get< param, compflow, tag::alpha >()[ e ];
79  const auto& b = g_inputdeck.get< param, compflow, tag::beta >()[ e ];
80  // ratio of specific heats
81  tk::real g = g_inputdeck.get< param, compflow, tag::gamma >()[ e ];
82  // evaluate solution at x,y,z
83  auto s = solution( e, x, y, z, 0.0 );
84  std::array< tk::real, 5 > r;
85  // density source
86  r[0] = 0.0;
87  // momentum source
88  r[1] = a*s[1]/s[0] - b*s[2]/s[0];
89  r[2] = b*s[1]/s[0] + a*s[2]/s[0];
90  r[3] = 0.0;
91  // energy source
92  r[4] = (r[1]*s[1] + r[2]*s[2])/s[0] + 8.0*a*a*a*z*z/(g-1.0);
93  return r;
94  }

125  //! Return field output going to file
126  //! \param[in] e Equation system index, i.e., which compressible
127  //! flow equation system we operate on among the systems of PDEs
128  //! \param[in] offset System offset specifying the position of the system of
129  //! PDEs among other systems
130  //! \param[in] coord Mesh node coordinates
131  //! \param[in] U Solution vector at recent time step
132  //! \return Vector of vectors to be output to file
133  static std::vector< std::vector< tk::real > >
134  fieldOutput( ncomp_t e,
135  ncomp_t offset,
136  tk::real,
137  tk::real,
138  const std::vector< tk::real >&,
139  const std::array< std::vector< tk::real >, 3 >& coord,
140  tk::Fields& U )
141  {
142  // manufactured solution parameters
143  const auto& a =
144  g_inputdeck.get< tag::param, tag::compflow, tag::alpha >()[e];
145  const auto& b =
146  g_inputdeck.get< tag::param, tag::compflow, tag::beta >()[e];
147  const auto& p0 =
148  g_inputdeck.get< tag::param, tag::compflow, tag::p0 >()[e];
149  // ratio of specific heats
150  tk::real g =
151  g_inputdeck.get< tag::param, tag::compflow, tag::gamma >()[e];
152 
153  std::vector< std::vector< tk::real > > out;
154  const auto r = U.extract( 0, offset );
155  const auto ru = U.extract( 1, offset );
156  const auto rv = U.extract( 2, offset );
157  const auto rw = U.extract( 3, offset );
158  const auto re = U.extract( 4, offset );
159 
160  // mesh node coordinates
161  const auto& x = coord[0];
162  const auto& y = coord[1];
163  const auto& z = coord[2];
164 
165  out.push_back( r );
166  out.push_back( std::vector< tk::real >( r.size(), 1.0 ) );
167 
168  std::vector< tk::real > u = ru;
169  std::transform( r.begin(), r.end(), u.begin(), u.begin(),
170  []( tk::real s, tk::real& d ){ return d /= s; } );
171  out.push_back( u );
172  for (std::size_t i=0; i<u.size(); ++i) u[i] = a*x[i] - b*y[i];
173  out.push_back( u );
174 
175  std::vector< tk::real > v = rv;
176  std::transform( r.begin(), r.end(), v.begin(), v.begin(),
177  []( tk::real s, tk::real& d ){ return d /= s; } );
178  out.push_back( v );
179  for (std::size_t i=0; i<v.size(); ++i) v[i] = b*x[i] + a*y[i];
180  out.push_back( v );
181 
182  std::vector< tk::real > w = rw;
183  std::transform( r.begin(), r.end(), w.begin(), w.begin(),
184  []( tk::real s, tk::real& d ){ return d /= s; } );
185  out.push_back( w );
186  for (std::size_t i=0; i<w.size(); ++i) w[i] = -2.0*a*z[i];
187  out.push_back( w );
188 
189  std::vector< tk::real > E = re;
190  std::transform( r.begin(), r.end(), E.begin(), E.begin(),
191  []( tk::real s, tk::real& d ){ return d /= s; } );
192  out.push_back( E );
193  for (std::size_t i=0; i<E.size(); ++i)
194  E[i] = 0.5*(u[i]*u[i] + v[i]*v[i] + w[i]*w[i]) +
195  (p0 - 2.0*a*a*z[i]*z[i])/(g-1.0);
196  out.push_back( E );
197 
198  std::vector< tk::real > P( r.size(), 0.0 );
199  for (std::size_t i=0; i<P.size(); ++i)
200  P[i] = (g-1.0)*(re[i] - r[i]*(u[i]*u[i] + v[i]*v[i] + w[i]*w[i])/2.0);
201  out.push_back( P );
202  for (std::size_t i=0; i<P.size(); ++i)
203  P[i] = p0 - 2.0*a*a*z[i]*z[i];
204  out.push_back( P );
205 
206  return out;
207  }
