28 // *****************************************************************************
29 // Read a part of the mesh (graph and coordinates) from Omega_h file
30 //! \param[in,out] ginpoel Container to store element connectivity of this PE's
31 //! chunk of the mesh (global ids)
32 //! \param[in,out] inpoel Container to store element connectivity with local
33 //! node IDs of this PE's mesh chunk
34 //! \param[in,out] gid Container to store global node IDs of elements of this
35 //! PE's mesh chunk
36 //! \param[in,out] lid Container to store global->local node IDs of elements of
37 //! this PE's mesh chunk
38 //! \param[in,out] coord Container to store coordinates of mesh nodes of this
39 //! PE's mesh chunk
40 //! \param[in] numpes Total number of PEs (default n = 1, for a single-CPU read)
41 //! \param[in] mype This PE (default m = 0, for a single-CPU read)
42 //! \note The last two integer arguments are unused. They are needed because
43 //! this function can be used via a polymorphic interface via a base class,
44 //! see tk::MeshReader, and other specialized mesh readers, e.g.,
45 //! tk::ExodusIIMeshReader, use these arguments. Here we require the Omega_h
46 //! input mesh to be pre-partitioned, with Omega_h's osh_part tool, into the
47 //! number of partitions that is equal to the number of PEs this fuinction is
48 //! called on in parallel.
49 // *****************************************************************************
50 {
51  IGNORE( mype ); // Avoid compiler warning in Release mode
52  Assert( mype < numpes, "Invalid input: PE id must be lower than NumPEs" );
53  Assert( ginpoel.empty() && inpoel.empty() && gid.empty() && lid.empty() &&
54  coord[0].empty() && coord[1].empty() && coord[2].empty(),
55  "Containers to store mesh must be empty" );
56 
57  // Create Omega_h library instance
58  auto lib = Omega_h::Library( nullptr, nullptr, MPI_COMM_WORLD );
59 
60  // Find out how many partitions the Omega_h mesh was saved with
61  auto nparts = Omega_h::binary::read_nparts( m_filename, lib.world() );
62 
63  if (numpes < nparts)
64  Throw( "The Omega_h mesh reader only supports NumPEs >= nparts, where "
65  "nparts is the number of partitions the mesh is partitioned into. "
66  "Also note that NumPEs must be a power of 2 if NumPEs > nparts." );
67 
68  // Read mesh
69  auto mesh = Omega_h::binary::read( m_filename, &lib );
70 
71  // Lambda to check if int is a power of two
72  auto isPowerOfTwo = []( int x ) { return (x != 0) && ((x & (x - 1)) == 0); };
73 
74  if (nparts != numpes) {
75  if (!isPowerOfTwo(numpes))
76  Throw( "The Omega_h mesh reader only supports NumPEs of power of 2" );
77  else
78  mesh.balance();
79  }
80 
81  // Extract connectivity from Omega_h's mesh object
82  auto ntets = mesh.nelems();
83  ginpoel.resize( static_cast< std::size_t >( ntets ) * 4 );
84  auto o_inpoel = mesh.ask_elem_verts();
85  auto o_gid = mesh.globals( Omega_h::VERT );
86  for (int i=0; i<ntets; ++i)
87  for (int j=0; j<4; ++j) {
88  auto I = static_cast< std::size_t >( i );
89  auto J = static_cast< std::size_t >( j );
90  ginpoel[ I*4+J ] = static_cast<std::size_t>( o_gid[ o_inpoel[i*4+j] ] );
91  }
92 
93  // Extract node coordinates from Omega_h's mesh object
94  auto o_coord = mesh.coords();
95 
96  // Extract number of vertices from Omega_h's mesh object
97  auto nnode = static_cast< std::size_t >( mesh.nverts() );
98 
99  // Extract node coordinates from Omega_h's mesh object
100  auto& x = coord[0];
101  auto& y = coord[1];
102  auto& z = coord[2];
103  x.resize( nnode );
104  y.resize( nnode );
105  z.resize( nnode );
106 
107  for (std::size_t I=0; I<nnode; ++I) {
108  auto i = static_cast< int >( I );
109  x[I] = o_coord[ i*3+0 ];
110  y[I] = o_coord[ i*3+1 ];
111  z[I] = o_coord[ i*3+2 ];
112  }
113 
114  // Compute local data from global mesh connectivity
115  std::tie( inpoel, gid, lid ) = tk::global2local( ginpoel );
116 }
