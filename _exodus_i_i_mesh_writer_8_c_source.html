160 void
161 ExodusIIMeshWriter::writeElements( const UnsMesh& mesh ) const
162 // *****************************************************************************
163 // Write element connectivity to ExodusII file
164 //! \param[in] mesh Unstructured mesh object
165 // *****************************************************************************
166 {
167  int elclass = 0;
168 
169  // For meshes that have no triangle element block (only tetrahedra), keeping
170  // the order as tets first followed by triangles allows keeping the tet ids
171  // associated to side sets the same when adding the missing triangle elements
172  // by meshconv. Hence this order should be kept as tets first triangles next.
173 
174  writeElemBlock( elclass, 4, "TETRAHEDRA", mesh.tetinpoel() );
175  writeElemBlock( elclass, 3, "TRIANGLES", mesh.triinpoel() );
176 }

178 void
179 ExodusIIMeshWriter::writeElemBlock( int& elclass,
180  int64_t nnpe,
181  const std::string& eltype,
182  const std::vector< std::size_t >& inpoel )
183 const
184 // *****************************************************************************
185 // Write element block to ExodusII file
186 //! \param[inout] elclass Count element class ids in file
187 //! \param[in] nnpe Number of nodes per element for block
188 //! \param[in] eltype String describing element type
189 //! \param[in] inpoel Element connectivity
190 // *****************************************************************************
191 {
192  if (inpoel.empty()) return;
193 
194  // increase number of element classes in file
195  ++elclass;
196 
197  // Write element block information
198  ErrChk(
199  ex_put_block( m_outFile, // exo file handle
200  EX_ELEM_BLOCK, // block type: elem block
201  elclass, // element class id
202  eltype.c_str(), // element block description
203  static_cast< int64_t >( inpoel.size() ) / nnpe, // nof cells
204  nnpe, // number of nodes per element
205  6, // number of edges per element
206  4, // number of faces per element
207  0 // number of attributes per element
208  ) == 0,
209  "Failed to write " + eltype + " element block to ExodusII file: " +
210  m_filename );
211 
212  // Write element connectivity with 1-based node ids
213  std::vector< int > inp;
214  for (auto p : inpoel) inp.push_back( static_cast< int >( p+1 ) );
215  ErrChk( ex_put_conn( m_outFile, EX_ELEM_BLOCK, elclass, inp.data(),
216  nullptr, nullptr ) == 0,
217  "Failed to write " + eltype + " element connectivity to ExodusII "
218  "file: " + m_filename );
219 }

221 void
222 ExodusIIMeshWriter::writeSidesets( const UnsMesh& mesh ) const
223 // *****************************************************************************
224 // Write side sets and their face connectivity to ExodusII file
225 //! \param[in] mesh Unstructured mesh object
226 // *****************************************************************************
227 {
228  // Write all side sets in mesh
229  for (const auto& s : mesh.sidetet()) {
230  // Write side set parameters
231  ErrChk( ex_put_set_param( m_outFile, EX_SIDE_SET, s.first,
232  static_cast<int64_t>(s.second.size()), 0 ) == 0,
233  "Failed to write side set parameters to ExodusII file: " + m_filename );
234 
235  // ExodusII wants 32-bit integers as IDs of tets adjacent to side set faces
236  std::vector< int > sidetet;
237  for (auto f : s.second) sidetet.push_back( static_cast<int>(f)+1 );
238 
239  // Write side set data: tet ids adjacent to side set and face id relative to
240  // tet (1...4) indicating which face is aligned with the side set.
241  ErrChk( ex_put_set( m_outFile, EX_SIDE_SET, s.first, sidetet.data(),
242  tk::cref_find( mesh.faceid(), s.first ).data() ) == 0,
243  "Failed to write side set to ExodusII file: " + m_filename );
244  }
245 }
