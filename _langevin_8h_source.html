3  \file src/DiffEq/Langevin.h
4  \copyright 2016-2018, Los Alamos National Security, LLC.
5  \brief A Langevin model for velocity in variable-density turbulence
6  \details This file implements the time integration of a system of stochastic
7  differential equations (SDEs) to model the fluctuating velocity components
8  in homogeneous variable-density turbulence. This model is an extension of
9  the generalized Langevin (GLM) model for constant-density flows by Haworth &
10  Pope (https://doi.org/10.1063/1.865723). The extension is roughly along the
11  lines of https://doi.org/10.1080/14685248.2011.554419.
12 */

32 //! \brief Langevin SDE used polymorphically with DiffEq
33 //! \details The template arguments specify policies and are used to configure
34 //! the behavior of the class. The policies are:
35 //! - Init - initialization policy, see DiffEq/InitPolicy.h
36 //! - Coefficients - coefficients policy, see DiffEq/LangevinCoeffPolicy.h
37 template< class Init, class Coefficients >
38 class Langevin {

40  private:
41  using ncomp_t = tk::ctr::ncomp_type;
42 
43  public:
44  //! \brief Constructor
45  //! \param[in] c Index specifying which system of beta SDEs to construct.
46  //! There can be multiple beta ... end blocks in a control file. This
47  //! index specifies which beta SDE system to instantiate. The index
48  //! corresponds to the order in which the beta ... end blocks are given
49  //! the control file.
50  explicit Langevin( ncomp_t c ) :
51  m_c( c ),
52  m_depvar(
53  g_inputdeck.get< tag::param, tag::langevin, tag::depvar >().at(c) ),
54  m_ncomp(
55  g_inputdeck.get< tag::component >().get< tag::langevin >().at(c) ),
56  m_offset(
57  g_inputdeck.get< tag::component >().offset< tag::langevin >(c) ),
58  m_rng( g_rng.at( tk::ctr::raw(
59  g_inputdeck.get< tag::param, tag::langevin, tag::rng >().at(c) ) ) ),
60  m_c0(),
61  m_g(),
62  m_U( {{ tk::ctr::Product( {
63  tk::ctr::Term( static_cast<char>(std::toupper(m_depvar)),
64  0, tk::ctr::Moment::ORDINARY ) } ),
65  tk::ctr::Product( {
66  tk::ctr::Term( static_cast<char>(std::toupper(m_depvar)),
67  1, tk::ctr::Moment::ORDINARY ) } ),
68  tk::ctr::Product( {
69  tk::ctr::Term( static_cast<char>(std::toupper(m_depvar)),
70  1, tk::ctr::Moment::ORDINARY ) } ) }} ),
71  coeff( g_inputdeck.get< tag::param, tag::langevin, tag::c0 >().at(c),
72  m_c0 )
73  {
74  Assert( m_ncomp == 3, "Langevin eq number of components must be 3" );
75  // Populate inverse hydrodynamics time scales extracted from DNS
76  if ( Coefficients::type() == ctr::CoeffPolicyType::HYDROTIMESCALE ) {
77  // Configure inverse hydrodyanmics time scale from DNS
78  const auto& hts = g_inputdeck.get< tag::param,
79  tag::langevin,
80  tag::hydrotimescales >().at(c);
81  Assert( hts.size() == 1,
82  "Langevin eq Hydrotimescales vector size must be 1" );
83  m_hts = ctr::HydroTimeScales().table( hts[0] );
84  }
85  }

87  //! Initalize SDE, prepare for time integration
88  //! \param[in] stream Thread (or more precisely stream) ID
89  //! \param[in,out] particles Array of particle properties
90  void initialize( int stream, tk::Particles& particles ) {
91  //! Set initial conditions using initialization policy
92  Init::template
93  init< tag::langevin >
94  ( g_inputdeck, m_rng, stream, particles, m_c, m_ncomp, m_offset );
95  }

97  //! \brief Advance particles according to the system of beta SDEs
98  //! \param[in,out] particles Array of particle properties
99  //! \param[in] stream Thread (or more precisely stream) ID
100  //! \param[in] dt Time step size
101  void advance( tk::Particles& particles,
102  int stream,
103  tk::real dt,
104  tk::real t,
105  const std::map< tk::ctr::Product, tk::real >& moments )
106  {
107  using tk::ctr::lookup;
108 
109  // Update SDE coefficients
110  tk::real eps = 0.0;
111  coeff.update( m_depvar, moments, m_hts, m_c0, t, eps, m_g );
112 
113  // Access mean velocity
114  std::array< tk::real, 3 > U{{ lookup(m_U[0],moments),
115  lookup(m_U[1],moments),
116  lookup(m_U[2],moments) }};
117 
118  const auto npar = particles.nunk();
119  for (auto p=decltype(npar){0}; p<npar; ++p) {
120  // Generate Gaussian random numbers with zero mean and unit variance
121  std::vector< tk::real > dW( m_ncomp );
122  m_rng.gaussian( stream, m_ncomp, dW.data() );
123 
124  // Advance all m_ncomp scalars
125  tk::real& Up = particles( p, 0, m_offset );
126  tk::real& Vp = particles( p, 1, m_offset );
127  tk::real& Wp = particles( p, 2, m_offset );
128  tk::real d = m_c0 * eps * dt;
129  d = (d > 0.0 ? std::sqrt(d) : 0.0);
130  Up += (m_g[0]*(Up-U[0]) + m_g[1]*(Vp-U[1]) + m_g[2]*(Wp-U[2]))*dt +
131  d*dW[0];
132  Vp += (m_g[3]*(Up-U[0]) + m_g[4]*(Vp-U[1]) + m_g[5]*(Wp-U[2]))*dt +
133  d*dW[1];
134  Wp += (m_g[6]*(Up-U[0]) + m_g[7]*(Vp-U[1]) + m_g[8]*(Wp-U[2]))*dt +
135  d*dW[2];
136  }
137  }
138

140  const ncomp_t m_c; //!< Equation system index
141  const char m_depvar; //!< Dependent variable
142  const ncomp_t m_ncomp; //!< Number of components
143  const ncomp_t m_offset; //!< Offset SDE operates from
144  const tk::RNG& m_rng; //!< Random number generator
145 
146  //! Selected inverse hydrodynamics time scale (if used)
147  //! \details This is only used if the coefficients policy is
148  //! LangevinCoeffHydroTimeScale. See constructor.

154
