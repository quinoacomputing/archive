79  //! \brief Advance particles according to the system of beta SDEs
80  //! \param[in,out] particles Array of particle properties
81  //! \param[in] dt Time step size
82  void advance( tk::Particles& particles,
83  int,
84  tk::real dt,
85  tk::real,
86  const std::map< tk::ctr::Product, tk::real >& )
87  {
88  const auto npar = particles.nunk();
89  for (auto p=decltype(npar){0}; p<npar; ++p) {
90  // Access particle velocity
91  tk::real u = particles( p, 0, m_velocity_offset );
92  tk::real v = particles( p, 1, m_velocity_offset );
93  tk::real w = particles( p, 2, m_velocity_offset );
94  // Advance all particle positions
95  tk::real& Xp = particles( p, 0, m_offset );
96  tk::real& Yp = particles( p, 1, m_offset );
97  tk::real& Zp = particles( p, 2, m_offset );
98  // Advance particle position
99  Xp += (m_dU[0]*Xp + m_dU[1]*Yp + m_dU[2]*Zp + u)*dt;
100  Yp += (m_dU[3]*Xp + m_dU[4]*Yp + m_dU[5]*Zp + v)*dt;
101  Zp += (m_dU[6]*Xp + m_dU[7]*Yp + m_dU[8]*Zp + w)*dt;
102  }
103  }
