125  //! \brief Print one-liner info for test
126  //! \details Columns: [done/total/failed]
127  //! - done: number of tests completed so far (note that a test may produce
128  //! more than one statistics and thus p-values)
129  //! - total: total number of tests: number of tests in the suite times the
130  //! number of RNGs tested (note that a test may produce more than one
131  //! statistics and thus p-values)
132  //! - failed: number of failed tests by a given RNG so far
133  //! name of the statistical test
134  //! name of RNG
135  //! result of test: "pass" or "fail, p-value = ..."
136  //! \param[in] ncomplete Number of completed tests
137  //! \param[in] ntest Total number of tests
138  //! \param[in] nfail Number of failed tests for RNG
139  //! \param[in] status Vector of vector of string with the following assumed
140  //! structure:
141  //! - status[0]: vector of name(s) of the test(s),
142  //! length: number of p-values
143  //! - status[1]: vector of p-value strings: "pass" or "fail, p-value = ...",
144  //! length: number of p-values
145  //! - status[2]: vector of length 1: RNG name used to run the test
146  //! \author J. Bakosi
147  void test( std::size_t ncomplete,
148  std::size_t ntest,
149  std::map< std::string, std::size_t >& nfail,
150  const std::vector< std::vector< std::string > >& status ) const
151  {
152  const auto& numfail = nfail.find( status[2][0] );
153  Assert( numfail != nfail.end(), "Cannot find RNG" );
154 
155  // Lambda to count number of failed tests
156  auto nfailed = [ &numfail, &status ]() {
157  for (const auto& p : status[1]) if (p.size() > 4) ++numfail->second;
158  return numfail->second;
159  };
160 
161  // Construct and echo info-line for all statistics resulted from test
162  for (std::size_t t=0; t<status[0].size(); ++t) {
163  std::stringstream ss;
164  ss << "[" << ncomplete << "/" << ntest << "/" << nfailed()
165  << "] " << status[0][t];

167  (status[1][t] == "pass" ? m_stream : m_qstream) <<
168  m_item_widename_value_fmt % m_item_indent % ss.str() % status[1][t];
169  }
