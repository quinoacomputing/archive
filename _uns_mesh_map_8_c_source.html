5  \brief Advanced Charm++ array creation with a map using an unstructured
6  grid
7  \details Advanced Charm++ array creation refers to various ways arrays can
8  be created with the Charm++ runtime system. See
9  http://charm.cs.illinois.edu/manuals/html/charm++/manual.html, Sec.
10  Advanced Array Creation. This class does a distribution that is based on
11  which portion of a distributed sparse matrix resulting from discretization
12  on an unstructured grid residing on a PE should hold a given chare array
13  element. (The one that owns most on-PE rows to minimize off-PE
14  communication.)
15 
16  Note that to help with performance, it is not advised to do heavy
17  computations in the overridden member functions, procNum() and
18  populateInitial(), since they can be potentially called many times. See
19  also this note on the inner workings of the Charm++ runtime system
20  regarding map objects and array element creation from long-time Charm++
21  developer, Eric Bohm, also at:
22  http://lists.cs.uiuc.edu/pipermail/charm/2015-May/002047.html
23 
24  _Procnum will be called to construct your chare, the first time a message
25  is sent to it from a node, and each time subsequent sends do not find a
26  precached location record for it. The latter event can occur when many
27  sends have pushed it out of cache, or after migration._
28 
29  _The potential global memory footprint of location management caching is
30  proportional to the total number of objects multiplied by the number of
31  nodes. Therefore, the runtime system keeps a finite number on each node.
32  At the limit, procnum could be called for nearly every message send,
33  therefore procnum should be designed to be inexpensive._
34 
35  The heavy portion of array element placement should therefore be done in
36  the constructor.
37 */

48 UnsMeshMap::UnsMeshMap( std::size_t npoin,
49  const std::vector< std::vector< std::size_t > >& point )
50  : m_pe()
51 // *****************************************************************************
52 // Constructor
53 //! \param[in] npoin Total number of points in mesh
54 //! \param[in] point Global mesh point ids owned by each array element
55 // *****************************************************************************
56 {
57  Assert( npoin > 0, "Need at least a single mesh point" );
58  Assert( point.size() >= static_cast< std::size_t >( CkNumPes() ),
59  "UnsMeshMap only works with nchare >= numPEs" );
60 
61  // Vector of maps to associate the number of mesh points an array element
62  // contributes to a pe to the array element
63  std::vector< std::map< std::size_t, std::size_t > > owner( point.size() );
64 
65  // Compute number of mesh points per PE (chunksize)
66  const auto numpes = static_cast< std::size_t >( CkNumPes() );
67  const auto chunksize = npoin>numpes ? npoin/numpes : 1;
68 
69  // Count up number of mesh points contributing to a PE for each array element
70  for (std::size_t e=0; e<point.size(); ++e) // for all array elements
71  for (auto p : point[e]) { // for all points array element e owns
72  auto pe = p / chunksize; // PE array element e contributes to
73  if (pe == static_cast<std::size_t>(CkNumPes())) --pe;
74  ++owner[e][pe]; // count up number of points element e contributing to pe
75  }
76 
77  // Find and store the PE with the largest number of points owned by an array
78  // element. This is the key (PE) of the largest mapped_type of the element's
79  // map, owner[e].
80  using pr = decltype(owner)::value_type::value_type;
81  m_pe.resize( owner.size() );
82  for (std::size_t e=0; e<owner.size(); ++e) {
83  const auto most = std::max_element( begin(owner[e]), end(owner[e]),
84  []( const pr& p1, const pr& p2 )
85  { return p1.second < p2.second; } );
86  m_pe[e] = most->first;
87  }
88 
89  // Check if all PEs have at least one array element to create, fix if not
90  fixPEs();
91 }

93 void
94 UnsMeshMap::fixPEs()
95 // *****************************************************************************
96 // Check that all PEs create at least a single array element, fix if not
97 //! \details This is required because if the array elements, placed using this
98 //! map object, send messages to some Charm++ chare group branches and the
99 //! group happens to use Charm++'s structured dagger, such as LinSysMerger,
100 //! memory problems will occur.
101 // *****************************************************************************
102 {
103  // Build unique set of PEs the array elements are assigned to
104  std::set< std::size_t > nkind;
105  for (auto p : m_pe) nkind.insert( p );
106 
107  // If not all PEs have at least one array element to create, go through all
108  // PEs that have no array elements assigned, and construct a map that
109  // associates the number of PEs to array elements. Then as a replacement PE,
110  // pick the PE that has the most array elements assigned. Then assign the
111  // first of the elements assigned to the PE with the most work to the
112  // replacement PE.
113  if (nkind.size() != static_cast<std::size_t>(CkNumPes()))
114  for (std::size_t p=0; p<static_cast<std::size_t>(CkNumPes()); ++p)
115  if (!nkind.count(p)) { // if PE p has no array elements to create
116  // Count up number elements for each PE
117  std::map< std::size_t, std::size_t > npe;
118  using pr = decltype(npe)::value_type;
119  for (auto q : m_pe) ++npe[q];
120  // Find the PE with the most array elements assigned
121  const auto most = std::max_element( begin(npe), end(npe),
122  []( const pr& p1, const pr& p2 )
123  { return p1.second < p2.second; } );
124  for (std::size_t q=0; q<m_pe.size(); ++q)
125  if (m_pe[q] == most->first) { // first element of the most loaded PE
126  //std::cout << CkMyPe() << ": replace " << m_pe[q] << " by " << p
127  // << " for e " << q << std::endl;
128  m_pe[q] = p; // replace PE of m_pe[q] of element q by p
129  q = m_pe.size(); // quit (enough to replace one)
130  }
131  }
132 
133  nkind.clear();
134  for (auto p : m_pe) nkind.insert( p );
135  Assert( nkind.size() == static_cast<std::size_t>(CkNumPes()),
136  "There are still PE(s) with no array elements assigned" );
137 }
