20 MeshReaderType
21 detectInput( const std::string& filename )
22 // *****************************************************************************
23 // Detect input mesh file type
24 //! \param[in] filename File to open and detect its type
25 //! \return enum specifying the mesh reader type
26 // *****************************************************************************
27 {
28  std::ifstream inFile;
29 
30  // Check if file exists, throw exception if it does not
31  inFile.open( filename, std::ifstream::in );
32  ErrChk( inFile.good(), "Failed to open file: " + filename );
33 
34  // Attempt to read a character, throw if it fails
35  // It is curious that on some systems opening a directory instead of a file
36  // with the above ifstream::open() call does not set the failbit. Thus we get
37  // here fine, so we try to read a character from it. If the read fails, we
38  // assume it is a directory and assume that it is in Omega_h osh file format.
39  // Read more at: http://stackoverflow.com/questions/9591036/
40  // ifstream-open-doesnt-set-error-bits-when-argument-is-a-directory.
41  inFile.get();
42  if (!inFile.good()) return MeshReaderType::OMEGA_H;
43 
44  // Close it
45  inFile.close();
46  ErrChk( !inFile.fail(), "Failed to close file: " + filename );
47 
48  // Get first three letters from input file
49  std::string s( Reader( filename ).firstline().substr(0,4) );
50 
51  if ( s.find("$Me") != std::string::npos ) {
52  return MeshReaderType::GMSH;
53  } else if ( s.find("CDF") != std::string::npos ||
54  s.find("HDF") != std::string::npos ) {
55  return MeshReaderType::EXODUSII;
56  } else if ( s.find("<?x") != std::string::npos ) {
57  return MeshReaderType::HYPER;
58  } else if ( s.find("*nd") != std::string::npos ) {
59  return MeshReaderType::ASC;
60  } else {
61  try {
62  std::stoi(s); // try to convert to an integer
63  } catch ( std::invalid_argument ) {
64  Throw( "Input mesh file type could not be determined from header: " +
65  filename );
66  }
67  // could also catch std::out_of_range, the other exception potentially
68  // thrown by std::stoi(), but a three-digit integer will always fit into int
69 
70  // if we got here, the above string-to-integer conversion succeeded
71  return MeshReaderType::NETGEN;
72  }
73 }

75 MeshWriterType
76 pickOutput( const std::string& filename )
77 // *****************************************************************************
78 // Determine output mesh file type
79 //! \param[in] filename Filename to pick its type based on extension given
80 //! \return enum specifying the mesh writer type
81 // *****************************************************************************
82 {
83  // Get extension of input file name
84  std::string fn = filename;
85  std::string ext( fn.substr(fn.find_last_of(".") + 1) );
86 
87  if ( ext == "msh" ) {
88  return MeshWriterType::GMSH;
89  } else if ( ext == "exo" || ext == "h5" ) {
90  return MeshWriterType::EXODUSII;
91  } else if ( ext == "mesh" ) {
92  return MeshWriterType::NETGEN;
93  } else {
94  Throw( "Output mesh file type could not be determined from extension of "
95  "filename '" + filename + "'; valid extensions are: "
96  "'msh' for Gmsh, 'exo' or 'h5' for ExodusII, 'mesh' for Netgen's "
97  "neutral" );
98  }
99 }
