63  //! Set initial conditions for dispersion in simple shear flow
64  //! \param[in] coord Mesh node coordinates
65  //! \param[in,out] unk Array of unknowns
66  //! \param[in] e Equation system index, i.e., which transport equation
67  //! system we operate on among the systems of PDEs
68  //! \param[in] ncomp Number of components in this transport equation
69  //! \param[in] offset System offset specifying the position of the system of
70  //! PDEs among other systems
71  //! \param[in] t Physical time
72  template< class eq >
73  static void init( const std::array< std::vector< tk::real >, 3 >& coord,
74  tk::Fields& unk,
75  tk::ctr::ncomp_type e,
76  tk::ctr::ncomp_type ncomp,
77  tk::ctr::ncomp_type offset,
78  tk::real t )
79  {
80  const auto& u0 = g_inputdeck.get< tag::param, eq, tag::u0 >()[e];
81  const auto& lambda = g_inputdeck.get< tag::param, eq, tag::lambda >()[e];
82  const auto& diff =
83  g_inputdeck.get< tag::param, eq, tag::diffusivity >()[e];
84  const tk::real X0 = 7200.0; // x position of source
85  const tk::real t0 = g_inputdeck.get< tag::discr, tag::t0 >(); // initial t
86  const auto& x = coord[0];
87  const auto& y = coord[1];
88  for (ncomp_t c=0; c<ncomp; ++c) {
89  const auto b = 1.0 + lambda[c]*lambda[c]*t*t/12.0;
90  const auto M =
91  4.0*M_PI*t0*std::sqrt( 1.0 + lambda[c]*lambda[c]*t0*t0/12.0 );
92  for (ncomp_t i=0; i<x.size(); ++i) {
93  tk::real a = x[i] - X0 - u0[c]*t - 0.5*lambda[c]*y[i]*t;
94  unk( i, c, offset ) =
95  M * exp( -a*a/(4.0*M_PI*diff[c]*t*b) - y[i]*y[i]/(4.0*diff[c]*t) )
96  / ( 4.0*M_PI*t*std::sqrt(b) );
97  }
98  }
99  }

101  //! Assign prescribed shear velocity to nodes of tetrahedron element
102  //! \param[in] N Element node indices
103  //! \param[in] coord Mesh node coordinates
104  //! \param[in] e Equation system index, i.e., which transport equation
105  //! system we operate on among the systems of PDEs
106  //! \param[in] ncomp Number of components in this transport equation
107  //! \return Velocity assigned to all vertices of a tetrehedron, size:
108  //! ncomp * ndim * nnode = [ncomp][3][4]
109  template< class eq >
110  static std::vector< std::array< std::array< tk::real, 4 >, 3 > >
111  prescribedVelocity( const std::array< std::size_t, 4 >& N,
112  const std::array< std::vector< tk::real >, 3 >& coord,
113  tk::ctr::ncomp_type e,
114  tk::ctr::ncomp_type ncomp )
115  {
116  const auto& u0 = g_inputdeck.get< tag::param, eq, tag::u0 >()[e];
117  const auto& lambda = g_inputdeck.get< tag::param, eq, tag::lambda >()[e];
118  const auto& y = coord[1];
119  std::vector< std::array< std::array< tk::real, 4 >, 3 > > vel( ncomp );
120  for (ncomp_t c=0; c<ncomp; ++c) {
121  std::array< std::array< tk::real, 4 >, 3 > v;
122  v[0][0] = u0[c] + lambda[c]*y[N[0]]; v[1][0] = 0.0; v[2][0] = 0.0;
123  v[0][1] = u0[c] + lambda[c]*y[N[1]]; v[1][1] = 0.0; v[2][1] = 0.0;
124  v[0][2] = u0[c] + lambda[c]*y[N[2]]; v[1][2] = 0.0; v[2][2] = 0.0;
125  v[0][3] = u0[c] + lambda[c]*y[N[3]]; v[1][3] = 0.0; v[2][3] = 0.0;
126  vel[c] = std::move(v);
127  }
128  return vel;
129  }

147 //! Transport PDE problem: rotation of Zalesak's slotted cylinder
148 //! \see Zalesak, S. (1978) Fully Multidimensional Flux-Corrected Transport
149 //! Algorithms for Fluids. Journal of Computational Physics 31, 335-362.
150 //! \see Leveque R.J. (1996) High-Resolution Conservative Algorithms for
151 //! Advection in Incompressible Flow. SIAM Journal on Numerical Analyis 33,
152 //! 627-665.
153 class TransportProblemSlotCyl {
154 
155  public:
156  //! Do error checking on PDE parameters
157  template< class eq >
158  static void errchk( tk::ctr::ncomp_type, tk::ctr::ncomp_type ) {}
159 
160  //! Set initial conditions for Zalesak's slotted cylinder case
161  //! \param[in] coord Mesh node coordinates
162  //! \param[in,out] unk Array of unknowns
163  //! \param[in] ncomp Number of components in this transport equation
164  //! \param[in] offset System offset specifying the position of the system of
165  //! PDEs among other systems
166  //! \param[in] time Physical time
167  template< class eq >
168  static void init( const std::array< std::vector< tk::real >, 3 >& coord,
169  tk::Fields& unk,
170  tk::ctr::ncomp_type,
171  tk::ctr::ncomp_type ncomp,
172  tk::ctr::ncomp_type offset,
173  tk::real time )
174  {
175  for (ncomp_t c=0; c<ncomp; ++c) {
176  auto t = time + 2.0*M_PI/ncomp * c;
177  const tk::real R0 = 0.15;
178  // center of the cone
179  tk::real x0 = 0.5;
180  tk::real y0 = 0.25;
181  tk::real r = std::sqrt((x0-0.5)*(x0-0.5) + (y0-0.5)*(y0-0.5));
182  tk::real kx = 0.5 + r*std::sin( t );
183  tk::real ky = 0.5 - r*std::cos( t );
184  // center of the hump
185  x0 = 0.25;
186  y0 = 0.5;
187  r = std::sqrt((x0-0.5)*(x0-0.5) + (y0-0.5)*(y0-0.5));
188  tk::real hx = 0.5 + r*std::sin( t-M_PI/2.0 ),
189  hy = 0.5 - r*std::cos( t-M_PI/2.0 );
190  // center of the slotted cylinder
191  x0 = 0.5;
192  y0 = 0.75;
193  r = std::sqrt((x0-0.5)*(x0-0.5) + (y0-0.5)*(y0-0.5));
194  tk::real cx = 0.5 + r*std::sin( t+M_PI ),
195  cy = 0.5 - r*std::cos( t+M_PI );
196  // end points of the cylinder slot
197  tk::real i1x = 0.525, i1y = cy - r*std::cos( std::asin(0.025/r) ),
198  i2x = 0.525, i2y = 0.8,
199  i3x = 0.475, i3y = 0.8;
200  // rotate end points of cylinder slot
201  tk::real ri1x = 0.5 + std::cos(t)*(i1x-0.5) - std::sin(t)*(i1y-0.5),
202  ri1y = 0.5 + std::sin(t)*(i1x-0.5) + std::cos(t)*(i1y-0.5),
203  ri2x = 0.5 + std::cos(t)*(i2x-0.5) - std::sin(t)*(i2y-0.5),
204  ri2y = 0.5 + std::sin(t)*(i2x-0.5) + std::cos(t)*(i2y-0.5),
205  ri3x = 0.5 + std::cos(t)*(i3x-0.5) - std::sin(t)*(i3y-0.5),
206  ri3y = 0.5 + std::sin(t)*(i3x-0.5) + std::cos(t)*(i3y-0.5);
207  // direction of slot sides
208  tk::real v1x = ri2x-ri1x, v1y = ri2y-ri1y,
209  v2x = ri3x-ri2x, v2y = ri3y-ri2y;
210  // lengths of direction of slot sides vectors
211  tk::real v1 = std::sqrt(v1x*v1x + v1y*v1y),
212  v2 = std::sqrt(v2x*v2x + v2y*v2y);
213  const auto& x = coord[0];
214  const auto& y = coord[1];
215  unk.fill( c, offset, 0.0 );
216  for (ncomp_t i=0; i<x.size(); ++i) {
217  // cone
218  r = std::sqrt((x[i]-kx)*(x[i]-kx) + (y[i]-ky)*(y[i]-ky)) / R0;
219  if (r<1.0) unk( i, c, offset ) = 0.6*(1.0-r);
220  // hump
221  r = std::sqrt((x[i]-hx)*(x[i]-hx) + (y[i]-hy)*(y[i]-hy)) / R0;
222  if (r<1.0)
223  unk( i, c, offset ) = 0.2*(1.0+std::cos(M_PI*std::min(r,1.0)));
224  // cylinder
225  r = std::sqrt((x[i]-cx)*(x[i]-cx) + (y[i]-cy)*(y[i]-cy)) / R0;
226  const std::array< tk::real, 2 > r1{{ v1x, v1y }},
227  r2{{ x[i]-ri1x, y[i]-ri1y }};
228  const auto d1 = (r1[0]*r2[1] - r2[0]*r1[1]) / v1;
229  const std::array< tk::real, 2 > r3{{ v2x, v2y }},
230  r4{{ x[i]-ri2x, y[i]-ri2y }};
231  const auto d2 = (r3[0]*r4[1] - r4[0]*r3[1]) / v2;
232  if (r<1.0 && (d1>0.05 || d1<0.0 || d2<0.0)) unk( i, c, offset ) = 0.6;
233  }
234  }
235  }

237  //! Assign prescribed velocity to nodes of tetrahedron element
238  //! \param[in] N Element node indices
239  //! \param[in] coord Mesh node coordinates
240  //! \param[in] ncomp Number of components in this transport equation
241  //! \return Velocity assigned to all vertices of a tetrehedron, size:
242  //! ncomp * ndim * nnode = [ncomp][3][4]
243  template< class eq >
244  static std::vector< std::array< std::array< tk::real, 4 >, 3 > >
245  prescribedVelocity( const std::array< std::size_t, 4 >& N,
246  const std::array< std::vector< tk::real >, 3 >& coord,
247  tk::ctr::ncomp_type,
248  tk::ctr::ncomp_type ncomp )
249  {
250  const auto& x = coord[0];
251  const auto& y = coord[1];
252  std::vector< std::array< std::array< tk::real, 4 >, 3 > > vel( ncomp );
253  for (ncomp_t c=0; c<ncomp; ++c) {
254  std::array< std::array< tk::real, 4 >, 3 > v;
255  v[0][0] = 0.5-y[N[0]]; v[1][0] = x[N[0]]-0.5; v[2][0] = 0.0;
256  v[0][1] = 0.5-y[N[1]]; v[1][1] = x[N[1]]-0.5; v[2][1] = 0.0;
257  v[0][2] = 0.5-y[N[2]]; v[1][2] = x[N[2]]-0.5; v[2][2] = 0.0;
258  v[0][3] = 0.5-y[N[3]]; v[1][3] = x[N[3]]-0.5; v[2][3] = 0.0;
259  vel[c] = std::move(v);
260  }
261  return vel;
262  }

265  //! \return Array of the four values of the three velocity coordinates
266  static std::array< std::array< tk::real, 4 >, 3 >
267  velocity( const tk::Fields&,
268  const std::array< std::vector< tk::real >, 3 >& coord,
269  const std::array< std::size_t, 4 >& N )

275  {{ 0.0, 0.0, 0.0, 0.0 }} }};
276  }
