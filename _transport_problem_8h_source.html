39 /*! Transport PDE problem: diffusion of a shear layer
40  \details This class implements the analytical solutions for the test
41  problem, adopted from Okubo Akira Karweit Michael J. , (1969),
42  [DIFFUSION FROM A CONTINUOUS SOURCE IN A UNIFORM SHEAR
43  FLOW](http://onlinelibrary.wiley.com/doi/10.4319/lo.1969.14.4.0514/abstract),
44  Limnology and Oceanography, 14, doi: 10.4319/lo.1969.14.4.0514. In essence,
45  this is a test problem for the advection-diffusion equation in 3D where the
46  analytical solution is known in a closed form as the solution evolves in
47  time. The initial solution is a Gaussian that is advected and diffused in
48  time with an imposed constant-in-time velocity field that features
49  advection and shear. Also, the diffusion coefficients can be different in
50  the three coordinate directions. Note that t0 as well as all three
51  components of the diffusion must be larger than zero at t=t0 to have a
52  well-defined initial condition.
53 
54  In a nutshell, the equation solved is
55  \f[
56  \frac{\partial S}{\partial t} + \left(V_0 + \Omega_y y + \Omega_z z
57  \right) \frac{\partial S}{\partial x} =
58  A_x \frac{\partial^2S}{\partial x^2} +
59  A_y \frac{\partial^2S}{\partial y^2} +
60  A_z \frac{\partial^2S}{\partial z^2}
61  \f]
62  whose solution is given by
63  \f[
64  S(t,x,y,z,) = \frac{1}{8\pi^{3/2}(A_xA_yA_z)^{1/2}t^{3/2}
65  (1+\phi_3^2t^2)^{1/2}}
66  \exp\left[ -\frac{x-V_0t-(\Omega_yy+\Omega_zz)^2/2}
67  {4A_xt(1+\phi_3^2t^2}
68  -\frac{y^2}{4A_yt}
69  -\frac{z^2}{4A_zt} \right]
70  \f]
71  where \f$ \phi_3^2 = (\Omega_y^2A_y/A_x + \Omega_z^2A_z/A_x)/12\f$.
72  See also the paper.
73 */

95  //! Set initial conditions for dispersion in simple shear flow
96  //! \param[in] coord Mesh node coordinates
97  //! \param[in,out] unk Array of unknowns
98  //! \param[in] e Equation system index, i.e., which transport equation
99  //! system we operate on among the systems of PDEs
100  //! \param[in] ncomp Number of components in this transport equation
101  //! \param[in] offset System offset specifying the position of the system of
102  //! PDEs among other systems
103  //! \param[in] t Physical time
104  template< class eq >
105  static void init( const std::array< std::vector< tk::real >, 3 >& coord,
106  tk::Fields& unk,
107  tk::ctr::ncomp_type e,
108  tk::ctr::ncomp_type ncomp,
109  tk::ctr::ncomp_type offset,
110  tk::real t )
111  {
112  const auto& u0 = g_inputdeck.get< tag::param, eq, tag::u0 >()[e];
113  const auto& d = g_inputdeck.get< tag::param, eq, tag::diffusivity >()[e];
114  const auto& l = g_inputdeck.get< tag::param, eq, tag::lambda >()[e];
115  const auto& x = coord[0];
116  const auto& y = coord[1];
117  const auto& z = coord[2];
118  for (ncomp_t c=0; c<ncomp; ++c) {
119  const auto li = 2*c;
120  const auto di = 3*c;
121  const auto phi3s = (l[li+0]*l[li+0]*d[di+1]/d[di+0] +
122  l[li+1]*l[li+1]*d[di+2]/d[di+0]) / 12.0;
123  for (ncomp_t i=0; i<x.size(); ++i)
124  unk( i, c, offset ) =
125  1.0 / ( 8.0 * std::pow(M_PI,3.0/2.0) *
126  std::sqrt(d[di+0]*d[di+1]*d[di+2]) *
127  std::pow(t,3.0/2.0) * std::sqrt(1.0+phi3s*t*t) ) *
128  exp( -std::pow( x[i] - u0[c]*t -
129  0.5*(l[li+0]*y[i] -l[li+1]*z[i])*t, 2.0 ) /
130  ( 4.0 * d[di+0] * t * (1.0 + phi3s*t*t) )
131  -y[i]*y[i] / ( 4.0 * d[di+1] * t )
132  -z[i]*z[i] / ( 4.0 * d[di+2] * t ) );
133  }
134  }

136  //! Assign prescribed shear velocity to nodes of tetrahedron element
137  //! \param[in] N Element node indices
138  //! \param[in] coord Mesh node coordinates
139  //! \param[in] e Equation system index, i.e., which transport equation
140  //! system we operate on among the systems of PDEs

143  //! ncomp * ndim * nnode = [ncomp][3][4]

157  std::array< std::array< tk::real, 4 >, 3 > v;
158  const auto li = 2*c;

160  v[1][0] = 0.0;

163  v[1][1] = 0.0;
164  v[2][1] = 0.0;
165  v[0][2] = u0[c] + l[li+0]*y[N[2]] + l[li+1]*z[N[2]];
166  v[1][2] = 0.0;
167  v[2][2] = 0.0;
168  v[0][3] = u0[c] + l[li+0]*y[N[3]] + l[li+1]*z[N[3]];
169  v[1][3] = 0.0;
170  v[2][3] = 0.0;
171  vel[c] = std::move(v);
172  }

192 //! Transport PDE problem: rotation of Zalesak's slotted cylinder
193 //! \see Zalesak, S. (1978) Fully Multidimensional Flux-Corrected Transport
194 //! Algorithms for Fluids. Journal of Computational Physics 31, 335-362.
195 //! \see Leveque R.J. (1996) High-Resolution Conservative Algorithms for
196 //! Advection in Incompressible Flow. SIAM Journal on Numerical Analyis 33,
197 //! 627-665.
198 class TransportProblemSlotCyl {
199 
200  public:

205  //! Set initial conditions for Zalesak's slotted cylinder case
206  //! \param[in] coord Mesh node coordinates
207  //! \param[in,out] unk Array of unknowns
208  //! \param[in] ncomp Number of components in this transport equation
209  //! \param[in] offset System offset specifying the position of the system of
210  //! PDEs among other systems
211  //! \param[in] time Physical time
212  template< class eq >
213  static void init( const std::array< std::vector< tk::real >, 3 >& coord,
214  tk::Fields& unk,
215  tk::ctr::ncomp_type,
216  tk::ctr::ncomp_type ncomp,
217  tk::ctr::ncomp_type offset,
218  tk::real time )
219  {
220  for (ncomp_t c=0; c<ncomp; ++c) {
221  auto t = time + 2.0*M_PI/ncomp * c;
222  const tk::real R0 = 0.15;
223  // center of the cone
224  tk::real x0 = 0.5;
225  tk::real y0 = 0.25;
226  tk::real r = std::sqrt((x0-0.5)*(x0-0.5) + (y0-0.5)*(y0-0.5));
227  tk::real kx = 0.5 + r*std::sin( t );
228  tk::real ky = 0.5 - r*std::cos( t );
229  // center of the hump
230  x0 = 0.25;
231  y0 = 0.5;
232  r = std::sqrt((x0-0.5)*(x0-0.5) + (y0-0.5)*(y0-0.5));
233  tk::real hx = 0.5 + r*std::sin( t-M_PI/2.0 ),
234  hy = 0.5 - r*std::cos( t-M_PI/2.0 );
235  // center of the slotted cylinder
236  x0 = 0.5;
237  y0 = 0.75;
238  r = std::sqrt((x0-0.5)*(x0-0.5) + (y0-0.5)*(y0-0.5));
239  tk::real cx = 0.5 + r*std::sin( t+M_PI ),
240  cy = 0.5 - r*std::cos( t+M_PI );
241  // end points of the cylinder slot
242  tk::real i1x = 0.525, i1y = cy - r*std::cos( std::asin(0.025/r) ),
243  i2x = 0.525, i2y = 0.8,
244  i3x = 0.475, i3y = 0.8;
245  // rotate end points of cylinder slot
246  tk::real ri1x = 0.5 + std::cos(t)*(i1x-0.5) - std::sin(t)*(i1y-0.5),
247  ri1y = 0.5 + std::sin(t)*(i1x-0.5) + std::cos(t)*(i1y-0.5),
248  ri2x = 0.5 + std::cos(t)*(i2x-0.5) - std::sin(t)*(i2y-0.5),
249  ri2y = 0.5 + std::sin(t)*(i2x-0.5) + std::cos(t)*(i2y-0.5),
250  ri3x = 0.5 + std::cos(t)*(i3x-0.5) - std::sin(t)*(i3y-0.5),
251  ri3y = 0.5 + std::sin(t)*(i3x-0.5) + std::cos(t)*(i3y-0.5);
252  // direction of slot sides
253  tk::real v1x = ri2x-ri1x, v1y = ri2y-ri1y,
254  v2x = ri3x-ri2x, v2y = ri3y-ri2y;
255  // lengths of direction of slot sides vectors
256  tk::real v1 = std::sqrt(v1x*v1x + v1y*v1y),
257  v2 = std::sqrt(v2x*v2x + v2y*v2y);
258  const auto& x = coord[0];
259  const auto& y = coord[1];
260  unk.fill( c, offset, 0.0 );
261  for (ncomp_t i=0; i<x.size(); ++i) {
262  // cone
263  r = std::sqrt((x[i]-kx)*(x[i]-kx) + (y[i]-ky)*(y[i]-ky)) / R0;
264  if (r<1.0) unk( i, c, offset ) = 0.6*(1.0-r);
265  // hump
266  r = std::sqrt((x[i]-hx)*(x[i]-hx) + (y[i]-hy)*(y[i]-hy)) / R0;
267  if (r<1.0)
268  unk( i, c, offset ) = 0.2*(1.0+std::cos(M_PI*std::min(r,1.0)));
269  // cylinder
270  r = std::sqrt((x[i]-cx)*(x[i]-cx) + (y[i]-cy)*(y[i]-cy)) / R0;
271  const std::array< tk::real, 2 > r1{{ v1x, v1y }},
272  r2{{ x[i]-ri1x, y[i]-ri1y }};
273  const auto d1 = (r1[0]*r2[1] - r2[0]*r1[1]) / v1;
274  const std::array< tk::real, 2 > r3{{ v2x, v2y }},
275  r4{{ x[i]-ri2x, y[i]-ri2y }};
276  const auto d2 = (r3[0]*r4[1] - r4[0]*r3[1]) / v2;
277  if (r<1.0 && (d1>0.05 || d1<0.0 || d2<0.0)) unk( i, c, offset ) = 0.6;
278  }
279  }
280  }

285  //! \param[in] ncomp Number of components in this transport equation

305  }
