69  m_volc(),

72 // *****************************************************************************
73 // Constructor
74 //! \param[in] transporter Host (Transporter) proxy
75 //! \param[in] fctproxy Distributed FCT proxy
76 //! \param[in] solver Linear system solver (Solver) proxy
77 //! \param[in] conn Vector of mesh element connectivity owned (global IDs)
78 //! \param[in] msum Global mesh node IDs associated to chare IDs bordering the
79 //! mesh chunk we operate on
80 //! \param[in] filenodes Map associating old node IDs (as in file) to new node
81 //! IDs (as in producing contiguous-row-id linear system contributions)
82 //! \param[in] edgenodes Map associating new node IDs ('new' as in producing
83 //! contiguous-row-id linear system contributions) to edges (a pair of old
84 //! node IDs ('old' as in file). These 'new' node IDs are the ones newly
85 //! added during inital uniform mesh refinement.
86 //! \param[in] nchare Total number of Discretization chares
87 //! \details "Contiguous-row-id" here means that the numbering of the mesh nodes
88 //! (which corresponds to rows in the linear system) are (approximately)
89 //! contiguous (as much as this can be done with an unstructured mesh) as the
90 //! problem is distirbuted across PEs, held by Solver objects. This ordering
91 //! is in start contrast with "as-in-file" ordering, which is the ordering of
92 //! the mesh nodes as it is stored in the file from which the mesh is read in.
93 //! The as-in-file ordering is highly non-contiguous across the distributed
94 //! problem.
95 // *****************************************************************************
96 {
97  Assert( m_psup.second.size()-1 == m_gid.size(),
98  "Number of mesh points and number of global IDs unequal" );
99 
100  // Convert neighbor nodes to vectors from sets
101  for (const auto& n : msum) {
102  auto& v = m_msum[ n.first ];
103  v.insert( end(v), begin(n.second), end(n.second) );
104  }
105 
106  // Count the number of mesh nodes at which we receive data from other chares
107  // and compute map associating boundary-chare node ID associated to global ID
108  std::vector< std::size_t > c;
109  for (const auto& n : m_msum) for (auto i : n.second) c.push_back( i );
110  tk::unique( c );
111  m_bid = tk::assignLid( c );
112 
113  // Allocate receive buffer for nodal volumes
114  m_volc.resize( m_bid.size(), 0.0 );
115 
116  // Insert DistFCT chare array element if FCT is needed. Note that even if FCT
117  // is configured false in the input deck, at this point, we still need the FCT
118  // object as FCT is still being performed, only its results are ignored. See
119  // also, e.g., MatCG::next().
120  const auto sch = g_inputdeck.get< tag::selected, tag::scheme >();
121  const auto nprop = g_inputdeck.get< tag::component >().nprop();
122  if ((sch == ctr::SchemeType::MatCG || sch == ctr::SchemeType::DiagCG))
123  m_fct[ thisIndex ].insert( m_transporter, nchare, m_gid.size(), nprop,
124  m_msum, m_bid, m_lid, m_inpoel, CkMyPe() );
125 }

158 void
159 Discretization::vol()
160 // *****************************************************************************
161 // Sum mesh volumes to nodes, start communicating them on chare-boundaries
162 // *****************************************************************************
163 {
164  const auto& x = m_coord[0];
165  const auto& y = m_coord[1];
166  const auto& z = m_coord[2];
167 
168  // Compute nodal volumes on our chunk of the mesh
169  for (std::size_t e=0; e<m_inpoel.size()/4; ++e) {
170  const std::array< std::size_t, 4 > N{{ m_inpoel[e*4+0], m_inpoel[e*4+1],
171  m_inpoel[e*4+2], m_inpoel[e*4+3] }};
172  // compute element Jacobi determinant * 5/120 = element volume / 4
173  const std::array< tk::real, 3 >
174  ba{{ x[N[1]]-x[N[0]], y[N[1]]-y[N[0]], z[N[1]]-z[N[0]] }},
175  ca{{ x[N[2]]-x[N[0]], y[N[2]]-y[N[0]], z[N[2]]-z[N[0]] }},
176  da{{ x[N[3]]-x[N[0]], y[N[3]]-y[N[0]], z[N[3]]-z[N[0]] }};
177  const auto J = tk::triple( ba, ca, da ) * 5.0 / 120.0;
178  ErrChk( J > 0, "Element Jacobian non-positive: PE:" +
179  std::to_string(CkMyPe()) + ", node IDs: " +
180  std::to_string(m_gid[N[0]]) + ',' +
181  std::to_string(m_gid[N[1]]) + ',' +
182  std::to_string(m_gid[N[2]]) + ',' +
183  std::to_string(m_gid[N[3]]) + ", coords: (" +
184  std::to_string(x[N[0]]) + ", " +
185  std::to_string(y[N[0]]) + ", " +
186  std::to_string(z[N[0]]) + "), (" +
187  std::to_string(x[N[1]]) + ", " +
188  std::to_string(y[N[1]]) + ", " +
189  std::to_string(z[N[1]]) + "), (" +
190  std::to_string(x[N[2]]) + ", " +
191  std::to_string(y[N[2]]) + ", " +
192  std::to_string(z[N[2]]) + "), (" +
193  std::to_string(x[N[3]]) + ", " +
194  std::to_string(y[N[3]]) + ", " +
195  std::to_string(z[N[3]]) + ')' );
196  // scatter add V/4 to nodes
197  for (std::size_t j=0; j<4; ++j) m_vol[N[j]] += J;
198  }
199 
200  // Store nodal volumes without contributions from other chares on
201  // chare-boundaries
202  m_v = m_vol;
203 
204  // Send our nodal volume contributions to neighbor chares
205  if (m_msum.empty())
206  contribute( CkCallback(CkReductionTarget(Transporter,vol), m_transporter) );
207  else
208  for (const auto& n : m_msum) {
209  std::vector< tk::real > v;
210  for (auto i : n.second) v.push_back( m_vol[ tk::cref_find(m_lid,i) ] );
211  thisProxy[ n.first ].comvol( n.second, v );
212  }
213 }

215 void
216 Discretization::comvol( const std::vector< std::size_t >& gid,
217  const std::vector< tk::real >& nodevol )
218 // *****************************************************************************
219 // Receive nodal volumes on chare-boundaries
220 //! \param[in] gid Global mesh node IDs at which we receive volume contributions
221 //! \param[in] nodevol Partial sums of nodal volume contributions to
222 //! chare-boundary nodes
223 //! \details This function receives contributions to m_vol, which stores the
224 //! nodal volumes. While m_vol stores own contributions, m_volc collects the
225 //! neighbor chare contributions during communication. This way work on m_vol
226 //! and m_volc is overlapped. The two are combined in totalvol().
227 // *****************************************************************************
228 {
229  Assert( nodevol.size() == gid.size(), "Size mismatch" );
230 
231  for (std::size_t i=0; i<gid.size(); ++i) {
232  auto bid = tk::cref_find( m_bid, gid[i] );
233  Assert( bid < m_volc.size(), "Indexing out of bounds" );
234  m_volc[ bid ] += nodevol[i];
235  }
236 
237  if (++m_nvol == m_msum.size()) {
238  m_nvol = 0;
239  contribute( CkCallback(CkReductionTarget(Transporter,vol), m_transporter) );
240  }
241 }

243 void
244 Discretization::totalvol()
245 // *****************************************************************************
246 // Sum mesh volumes and contribute own mesh volume to total volume
247 // *****************************************************************************
248 {
249  // Combine own and communicated contributions of nodal volumes
250  for (const auto& b : m_bid) {
251  auto lid = tk::cref_find( m_lid, b.first );
252  m_vol[ lid ] += m_volc[ b.second ];
253  }
254 
255  // Sum mesh volume to host
256  tk::real tvol = 0.0;
257  for (auto v : m_v) tvol += v;
258  contribute( sizeof(tk::real), &tvol, CkReduction::sum_double,
259  CkCallback(CkReductionTarget(Transporter,totalvol), m_transporter) );
260 }

262 void
263 Discretization::stat()
264 // *****************************************************************************
265 // Compute mesh cell statistics
266 // *****************************************************************************
267 {
268  const auto& x = m_coord[0];
269  const auto& y = m_coord[1];
270  const auto& z = m_coord[2];
271 
272  auto MIN = -std::numeric_limits< tk::real >::max();
273  auto MAX = std::numeric_limits< tk::real >::max();
274  std::array< tk::real, 2 > min = {{ MAX, MAX }};
275  std::array< tk::real, 2 > max = {{ MIN, MIN }};
276  std::array< tk::real, 4 > sum{{ 0.0, 0.0, 0.0, 0.0 }};
277  tk::UniPDF edgePDF( 1e-4 );
278  tk::UniPDF volPDF( 1e-4 );
279 
280  // Compute edge length statistics
281  // Note that while the min and max edge lengths are independent of the number
282  // of CPUs (by the time they are aggregated across all chares), the sum of
283  // the edge lengths and the edge length PDF are not. This is because the
284  // edges on the chare-boundary are counted multiple times and we
285  // conscientiously do not make an effort to precisely compute this, because
286  // that would require communication and more complex logic. Since these
287  // statistics are intended as simple average diagnostics, we ignore these
288  // small differences. For reproducible average edge lengths and edge length
289  // PDFs, run the mesh in serial.
290  for (std::size_t p=0; p<m_gid.size(); ++p)
291  for (auto i=m_psup.second[p]+1; i<=m_psup.second[p+1]; ++i) {
292  const auto dx = x[ m_psup.first[i] ] - x[ p ];
293  const auto dy = y[ m_psup.first[i] ] - y[ p ];
294  const auto dz = z[ m_psup.first[i] ] - z[ p ];
295  const auto length = std::sqrt( dx*dx + dy*dy + dz*dz );
296  if (length < min[0]) min[0] = length;
297  if (length > max[0]) max[0] = length;
298  sum[0] += 1.0;
299  sum[1] += length;
300  edgePDF.add( length );
301  }
302 
303  // Compute mesh cell volume statistics
304  for (std::size_t e=0; e<m_inpoel.size()/4; ++e) {
305  const std::array< std::size_t, 4 > N{{ m_inpoel[e*4+0], m_inpoel[e*4+1],
306  m_inpoel[e*4+2], m_inpoel[e*4+3] }};
307  const std::array< tk::real, 3 >
308  ba{{ x[N[1]]-x[N[0]], y[N[1]]-y[N[0]], z[N[1]]-z[N[0]] }},
309  ca{{ x[N[2]]-x[N[0]], y[N[2]]-y[N[0]], z[N[2]]-z[N[0]] }},
310  da{{ x[N[3]]-x[N[0]], y[N[3]]-y[N[0]], z[N[3]]-z[N[0]] }};
311  const auto L = std::cbrt( tk::triple( ba, ca, da ) / 6.0 );
312  if (L < min[1]) min[1] = L;
313  if (L > max[1]) max[1] = L;
314  sum[2] += 1.0;
315  sum[3] += L;
316  volPDF.add( L );
317  }
318 
319  // Contribute to mesh statistics across all Discretization chares
320  contribute( min.size()*sizeof(tk::real), min.data(), CkReduction::min_double,
321  CkCallback(CkReductionTarget(Transporter,minstat), m_transporter) );
322  contribute( max.size()*sizeof(tk::real), max.data(), CkReduction::max_double,
323  CkCallback(CkReductionTarget(Transporter,maxstat), m_transporter) );
324  contribute( sum.size()*sizeof(tk::real), sum.data(), CkReduction::sum_double,
325  CkCallback(CkReductionTarget(Transporter,sumstat), m_transporter) );
326 
327  // Serialize PDFs to raw stream
328  auto stream = tk::serialize( { edgePDF, volPDF } );
329  // Create Charm++ callback function for reduction of PDFs with
330  // Transporter::pdfstat() as the final target where the results will appear.
331  CkCallback cb( CkIndex_Transporter::pdfstat(nullptr), m_transporter );
332  // Contribute serialized PDF of partial sums to host via Charm++ reduction
333  contribute( stream.first, stream.second.get(), PDFMerger, cb );
334 }

336 void
337 Discretization::readCoords()
338 // *****************************************************************************
339 // Read coordinates of mesh nodes from file
340 // *****************************************************************************
341 {
342  tk::ExodusIIMeshReader
343  er( g_inputdeck.get< tag::cmd, tag::io, tag::input >() );
344 
345  auto nnode = er.readHeader();
346 
347  auto& x = m_coord[0];
348  auto& y = m_coord[1];
349  auto& z = m_coord[2];
350 
351  auto nn = m_lid.size();
352  x.resize( nn );
353  y.resize( nn );
354  z.resize( nn );
355 
356  for (auto p : m_gid) {
357  auto n = m_filenodes.find(p);
358  if (n != end(m_filenodes) && n->second < nnode)
359  er.readNode( n->second, tk::cref_find(m_lid,n->first), x, y, z );
360  }

363 void
364 Discretization::addEdgeNodeCoords()
365 // *****************************************************************************
366 // Add coordinates of mesh nodes newly generated to edge-mid points during
367 // initial refinement
368 // *****************************************************************************
369 {
370  if (m_edgenodes.empty()) return;
371 
372  auto& x = m_coord[0];
373  auto& y = m_coord[1];
374  auto& z = m_coord[2];
375  Assert( x.size() == y.size() && x.size() == z.size(), "Size mismatch" );
376 
377  // Lambda to add coordinates for a single new node on an edge
378  auto addnode = [ this, &x, &y, &z ]
379  ( const decltype(m_edgenodes)::value_type& e )
380  {
381  auto p = tk::cref_find( m_lid, e.first[0] );
382  auto q = tk::cref_find( m_lid, e.first[1] );
383  auto i = tk::cref_find( m_lid, e.second );
384  Assert( p < x.size(), "Discretization chare " + std::to_string(thisIndex) +
385  " indexing out of bounds: " + std::to_string(p)
386  + " must be lower than " + std::to_string(x.size()) );
387  Assert( q < x.size(), "Discretization chare " + std::to_string(thisIndex) +
388  " indexing out of bounds: " + std::to_string(q)
389  + " must be lower than " + std::to_string(x.size()) );
390  Assert( i < x.size(), "Discretization chare " + std::to_string(thisIndex) +
391  " indexing out of bounds: " + std::to_string(i)
392  + " must be lower than " + std::to_string(x.size()) );
393  x[i] = (x[p]+x[q])/2.0;
394  y[i] = (y[p]+y[q])/2.0;
395  z[i] = (z[p]+z[q])/2.0;
396  };
397 
398  // add new nodes
399  for (const auto& e : m_edgenodes) addnode( e );
400 }
