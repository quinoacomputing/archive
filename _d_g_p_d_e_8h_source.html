52  //! \brief Constructor taking a function pointer to a constructor of an
53  //! object modeling Concept.
54  //! \details Passing std::function allows late execution of the constructor,
55  //! i.e., as late as inside this class' constructor, and thus usage from
56  //! a factory. Note that there are at least two different ways of using
57  //! this constructor:
58  //! - Bind T's constructor arguments and place it in std::function<T()>
59  //! and passing no arguments as args.... This case then instantiates the
60  //! model via its constructor and stores it in here.
61  //! - Bind a single placeholder argument to T's constructor and pass it in
62  //! as host's args..., which then forwards it to model's constructor. This
63  //! allows late binding, i.e., binding the argument only here.
64  //! \see See also the wrapper tk::recordModel() which does the former and
65  //! tk::recordModelLate() which does the latter, both defined in
66  //! src/Base/Factory.h.
67  //! \param[in] x Function pointer to a constructor of an object modeling
68  //! Concept.
69  //! \param[in] args Zero or more constructor arguments
70  template< typename T, typename...Args >
71  explicit DGPDE( std::function<T(Args...)> x, Args&&... args ) :
72  self( tk::make_unique< Model<T> >(
73  std::move( x( std::forward<Args>(args)... ) ) ) ) {}
