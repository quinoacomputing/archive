51 // *****************************************************************************
52 //! \brief Wrapper for the standard C library's gettimeofday() from
53 //! \return A stirng containing the current date and time
54 // *****************************************************************************
55 {
56  time_t current_time;
57  char* c_time_string;
58 
59  // Obtain current time as seconds elapsed since the Epoch
60  current_time = time( nullptr );
61 
62  if (current_time == static_cast<time_t>(-1))
63  Throw( "Failure to compute the current time" );
64 
65  // Convert to local time format

161 //! \brief Generic Main() used for all executables for code-reuse and a uniform
162 //! output
163 //! \details The template arguments configure this Main class that is
164 //! practically used instead of the usual main(). This allows code-reuse and a
165 //! unfirom screen-output. The template arguments are:
166 //! - Driver, specializaing the driver type to be created, see tk::Driver
167 //! - Printer, specializaing the pretty printer type to use, see tk::Print
168 //! - CmdLine, specializing the command line object storing data parsed from
169 //! the command line
170 //! \param[in] argc Number of command-line arguments to executable
171 //! \param[in] argv C-style string array to command-line arguments to executable
172 //! \param[in] cmdline Command line object storing data parsed from the command
173 //! line arguments
174 //! \param[in] header Header type enum indicating which executable header to
175 //! print
176 //! \param[in] executable Name of the executable
177 //! \param[in] print Pretty printer to use
178 //! \return Instantiated driver object which can then be used to execute()
179 //! whatever it is intended to drive
180 template< class Driver, class Printer, class CmdLine >
181 Driver Main( int argc, char* argv[],
182  const CmdLine& cmdline,
183  HeaderType header,
184  const std::string& executable,
185  const Printer& print )
186 {
187  // Echo program header
188  echoHeader( print, header );
189 
190  // Echo environment
191  print.part( "Environment" );
192  // Build environment
193  echoBuildEnv( print, executable );
194  // Runtime environment
195  echoRunEnv( print, argc, argv, cmdline.template get< tag::verbose >(),
196  cmdline.template get< tag::quiescence >() );
197 
198  // Create and return driver
199  return Driver( print, cmdline );
200 }
