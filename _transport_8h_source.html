70  //! Compute the left hand side sparse matrix
71  //! \param[in] coord Mesh node coordinates
72  //! \param[in] inpoel Mesh element connectivity
73  //! \param[in] psup Linked lists storing IDs of points surrounding points
74  //! \param[in,out] lhsd Diagonal of the sparse matrix storing nonzeros
75  //! \param[in,out] lhso Off-diagonal of the sparse matrix storing nonzeros
76  //! \details Sparse matrix storing the nonzero matrix values at rows and
77  //! columns given by psup. The format is similar to compressed row
78  //! storage, but the diagonal and off-diagonal data are stored in separate
79  //! vectors. For the off-diagonal data the local row and column indices,
80  //! at which values are nonzero, are stored by psup (psup1 and psup2,
81  //! where psup2 holds the indices at which psup1 holds the point ids
82  //! surrounding points, see also tk::genPsup()). Note that the number of
83  //! mesh points (our chunk) npoin = psup.second.size()-1.
84  void lhs( const std::array< std::vector< tk::real >, 3 >& coord,
85  const std::vector< std::size_t >& inpoel,
86  const std::pair< std::vector< std::size_t >,
87  std::vector< std::size_t > >& psup,
88  tk::Fields& lhsd,
89  tk::Fields& lhso ) const
90  {
91  Assert( psup.second.size()-1 == coord[0].size(),
92  "Number of mesh points and number of global IDs unequal" );
93  Assert( lhsd.nunk() == psup.second.size()-1, "Number of unknowns in "
94  "diagonal sparse matrix storage incorrect" );
95  Assert( lhso.nunk() == psup.first.size(), "Number of unknowns in "
96  "off-diagonal sparse matrix storage incorrect" );
97 
98  // Lambda to compute the sparse matrix vector index for row and column
99  // indices. Used only for off-diagonal entries.
100  auto spidx = [ &psup ]( std::size_t r, std::size_t c ) -> std::size_t {
101  Assert( r != c, "Only for computing the off-diagonal indices" );
102  for (auto i=psup.second[r]+1; i<=psup.second[r+1]; ++i)
103  if (c == psup.first[i]) return i;
104  Throw( "Cannot find row, column: " + std::to_string(r) + ',' +
105  std::to_string(c) + " in sparse matrix" );
106  };
107 
108  const auto& x = coord[0];
109  const auto& y = coord[1];
110  const auto& z = coord[2];
111 
112  // Zero matrix for all components
113  for (ncomp_t c=0; c<m_ncomp; ++c) {
114  lhsd.fill( c, m_offset, 0.0 );
115  lhso.fill( c, m_offset, 0.0 );
116  }
117 
118  for (std::size_t e=0; e<inpoel.size()/4; ++e) {
119  const auto A = inpoel[e*4+0];
120  const auto B = inpoel[e*4+1];
121  const auto C = inpoel[e*4+2];
122  const auto D = inpoel[e*4+3];
123  std::array< tk::real, 3 > ba{{ x[B]-x[A], y[B]-y[A], z[B]-z[A] }},
124  ca{{ x[C]-x[A], y[C]-y[A], z[C]-z[A] }},
125  da{{ x[D]-x[A], y[D]-y[A], z[D]-z[A] }};
126  const auto J = tk::triple( ba, ca, da ) / 120.0;
127  Assert( J > 0, "Element Jacobian non-positive" );
128 
129  for (ncomp_t c=0; c<m_ncomp; ++c) {
130  const auto r = lhsd.cptr( c, m_offset );
131  lhsd.var( r, A ) += 2.0 * J;
132  lhsd.var( r, B ) += 2.0 * J;
133  lhsd.var( r, C ) += 2.0 * J;
134  lhsd.var( r, D ) += 2.0 * J;
135 
136  const auto s = lhso.cptr( c, m_offset );
137  lhso.var( s, spidx(A,B) ) += J;
138  lhso.var( s, spidx(A,C) ) += J;
139  lhso.var( s, spidx(A,D) ) += J;
140 
141  lhso.var( s, spidx(B,A) ) += J;
142  lhso.var( s, spidx(B,C) ) += J;
143  lhso.var( s, spidx(B,D) ) += J;
144 
145  lhso.var( s, spidx(C,A) ) += J;
146  lhso.var( s, spidx(C,B) ) += J;
147  lhso.var( s, spidx(C,D) ) += J;
148 
149  lhso.var( s, spidx(D,A) ) += J;
150  lhso.var( s, spidx(D,B) ) += J;
151  lhso.var( s, spidx(D,C) ) += J;
152  }
153  }

156  //! Compute right hand side
157  //! \param[in] deltat Size of time step
158  //! \param[in] coord Mesh node coordinates
159  //! \param[in] inpoel Mesh element connectivity
160  //! \param[in] U Solution vector at recent time step stage
161  //! \param[in,out] R Right-hand side vector computed
162  void rhs( tk::real,
163  tk::real deltat,
164  const std::array< std::vector< tk::real >, 3 >& coord,
165  const std::vector< std::size_t >& inpoel,
166  const tk::Fields& U,
167  tk::Fields& R ) const
168  {
169  Assert( U.nunk() == coord[0].size(), "Number of unknowns in solution "
170  "vector at recent time step incorrect" );
171  Assert( R.nunk() == coord[0].size() && R.nprop() == m_ncomp,
172  "Number of unknowns and/or number of components in right-hand "
173  "side vector incorrect" );
174 
175  const auto& x = coord[0];
176  const auto& y = coord[1];
177  const auto& z = coord[2];
178 
179  // zero right hand side for all components
180  for (ncomp_t c=0; c<m_ncomp; ++c) R.fill( c, m_offset, 0.0 );
181 
182  for (std::size_t e=0; e<inpoel.size()/4; ++e) {
183  const std::array< std::size_t, 4 > N{{ inpoel[e*4+0], inpoel[e*4+1],
184  inpoel[e*4+2], inpoel[e*4+3] }};
185  // compute element Jacobi determinant
186  const std::array< tk::real, 3 >
187  ba{{ x[N[1]]-x[N[0]], y[N[1]]-y[N[0]], z[N[1]]-z[N[0]] }},
188  ca{{ x[N[2]]-x[N[0]], y[N[2]]-y[N[0]], z[N[2]]-z[N[0]] }},
189  da{{ x[N[3]]-x[N[0]], y[N[3]]-y[N[0]], z[N[3]]-z[N[0]] }};
190  const auto J = tk::triple( ba, ca, da );
191  Assert( J > 0, "Element Jacobian non-positive" );
192 
193  // construct tetrahedron element-level matrices
194 
195  // consistent mass
196  std::array< std::array< tk::real, 4 >, 4 > mass; // nnode*nnode [4][4]
197  mass[0][0] = mass[1][1] = mass[2][2] = mass[3][3] = J/60.0; // diagonal
198  mass[0][1] = mass[0][2] = mass[0][3] = // off-diagonal
199  mass[1][0] = mass[1][2] = mass[1][3] =
200  mass[2][0] = mass[2][1] = mass[2][3] =
201  mass[3][0] = mass[3][1] = mass[3][2] = J/120.0;
202 
203  // shape function derivatives
204  std::array< std::array< tk::real, 3 >, 4 > grad; // nnode*ndim [4][3]
205  grad[1] = tk::crossdiv( ca, da, J );
206  grad[2] = tk::crossdiv( da, ba, J );
207  grad[3] = tk::crossdiv( ba, ca, J );
208  for (std::size_t i=0; i<3; ++i)
209  grad[0][i] = -grad[1][i]-grad[2][i]-grad[3][i];
210 
211  // access solution at element nodes at recent time step stage
212  std::vector< std::array< tk::real, 4 > > u( m_ncomp );
213  for (ncomp_t c=0; c<m_ncomp; ++c) u[c] = U.extract( c, m_offset, N );
214  // access pointer to right hand side at component and offset
215  std::vector< const tk::real* > r( m_ncomp );
216  for (ncomp_t c=0; c<m_ncomp; ++c) r[c] = R.cptr( c, m_offset );
217 
218  // get velocity for problem
219  const auto vel =
220  Problem::template
221  prescribedVelocity< tag::transport >( N, coord, m_c, m_ncomp );
222 
223  // add advection contribution to right hand side
224  for (ncomp_t c=0; c<m_ncomp; ++c)
225  for (std::size_t i=0; i<4; ++i)
226  for (std::size_t j=0; j<4; ++j)
227  for (std::size_t k=0; k<3; ++k)
228  for (std::size_t l=0; l<4; ++l)
229  R.var(r[c],N[j]) -= deltat * mass[j][i] * vel[c][k][i]
230  * grad[l][k] * u[c][l];
231 
232  // add diffusion contribution to right hand side
233  Physics::diffusionRhs( m_c, m_ncomp, deltat, J, N, grad, u, r, R );
234  }
235  }

237  //! Compute the minimum time step size
238  //! \param[in] U Solution vector at recent time step stage
239  //! \param[in] coord Mesh node coordinates
240  //! \param[in] inpoel Mesh element connectivity
241  //! \return Minimum time step size
242  tk::real dt( const std::array< std::vector< tk::real >, 3 >& coord,
243  const std::vector< std::size_t >& inpoel,
244  const tk::Fields& U ) const
245  {
246  Assert( U.nunk() == coord[0].size(), "Number of unknowns in solution "
247  "vector at recent time step incorrect" );
248  const auto& x = coord[0];
249  const auto& y = coord[1];
250  const auto& z = coord[2];
251  // compute the minimum dt across all elements we own
252  tk::real mindt = std::numeric_limits< tk::real >::max();
253  for (std::size_t e=0; e<inpoel.size()/4; ++e) {
254  const std::array< std::size_t, 4 > N{{ inpoel[e*4+0], inpoel[e*4+1],
255  inpoel[e*4+2], inpoel[e*4+3] }};
256  // compute cubic root of element volume as the characteristic length
257  const std::array< tk::real, 3 >
258  ba{{ x[N[1]]-x[N[0]], y[N[1]]-y[N[0]], z[N[1]]-z[N[0]] }},
259  ca{{ x[N[2]]-x[N[0]], y[N[2]]-y[N[0]], z[N[2]]-z[N[0]] }},
260  da{{ x[N[3]]-x[N[0]], y[N[3]]-y[N[0]], z[N[3]]-z[N[0]] }};
261  const auto L = std::cbrt( tk::triple( ba, ca, da ) / 6.0 );
262  // access solution at element nodes at recent time step stage
263  std::vector< std::array< tk::real, 4 > > u( m_ncomp );
264  for (ncomp_t c=0; c<m_ncomp; ++c) u[c] = U.extract( c, m_offset, N );
265  // get velocity for problem
266  const auto vel =
267  Problem::template
268  prescribedVelocity< tag::transport >( N, coord, m_c, m_ncomp );
269  // compute the maximum length of the characteristic velocity (advection
270  // velocity) across the four element nodes
271  tk::real maxvel = 0.0;
272  for (ncomp_t c=0; c<m_ncomp; ++c)
273  for (std::size_t i=0; i<4; ++i) {
274  auto v = std::sqrt( vel[c][0][i]*vel[c][0][i] +
275  vel[c][1][i]*vel[c][1][i] +
276  vel[c][2][i]*vel[c][2][i] );
277  if (v > maxvel) maxvel = v;
278  }
279  // compute element dt for the advection
280  auto advection_dt = L / maxvel;
281  // compute element dt based on diffusion
282  auto diffusion_dt = Physics::diffusion_dt( m_c, m_ncomp, L, u );
283  // compute minimum element dt
284  auto elemdt = std::min( advection_dt, diffusion_dt );
285  // find minimum dt across all elements
286  if (elemdt < mindt) mindt = elemdt;
287  }
288  return mindt;
289  }

347  //! \param[in] t Physical time
348  //! \param[in] V Total mesh volume
349  //! \param[in] coord Mesh node coordinates

360  const std::vector< tk::real >& v,
361  tk::Fields& U ) const
362  {

378  Assert( u.size() == v.size(), "Size mismatch" );
379  for (std::size_t i=0; i<u.size(); ++i)
380  e[i] = std::pow( e[i] - u[i], 2.0 ) * v[i] / V;
381  out.push_back( e );
