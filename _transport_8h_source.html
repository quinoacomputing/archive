75  //! Compute the left hand side sparse matrix
76  //! \param[in] coord Mesh node coordinates
77  //! \param[in] inpoel Mesh element connectivity
78  //! \param[in] psup Linked lists storing IDs of points surrounding points
79  //! \param[in,out] lhsd Diagonal of the sparse matrix storing nonzeros
80  //! \param[in,out] lhso Off-diagonal of the sparse matrix storing nonzeros
81  //! \details Sparse matrix storing the nonzero matrix values at rows and
82  //! columns given by psup. The format is similar to compressed row
83  //! storage, but the diagonal and off-diagonal data are stored in separate
84  //! vectors. For the off-diagonal data the local row and column indices,
85  //! at which values are nonzero, are stored by psup (psup1 and psup2,
86  //! where psup2 holds the indices at which psup1 holds the point ids
87  //! surrounding points, see also tk::genPsup()). Note that the number of
88  //! mesh points (our chunk) npoin = psup.second.size()-1.
89  //! \author J. Bakosi
90  void lhs( const std::array< std::vector< tk::real >, 3 >& coord,
91  const std::vector< std::size_t >& inpoel,
92  const std::pair< std::vector< std::size_t >,
93  std::vector< std::size_t > >& psup,
94  tk::Fields& lhsd,
95  tk::Fields& lhso ) const
96  {
97  Assert( psup.second.size()-1 == coord[0].size(),
98  "Number of mesh points and number of global IDs unequal" );
99  Assert( lhsd.nunk() == psup.second.size()-1, "Number of unknowns in "
100  "diagonal sparse matrix storage incorrect" );
101  Assert( lhso.nunk() == psup.first.size(), "Number of unknowns in "
102  "off-diagonal sparse matrix storage incorrect" );
103 
104  // Lambda to compute the sparse matrix vector index for row and column
105  // indices. Used only for off-diagonal entries.
106  auto spidx = [ &psup ]( std::size_t r, std::size_t c ) -> std::size_t {
107  Assert( r != c, "Only for computing the off-diagonal indices" );
108  for (auto i=psup.second[r]+1; i<=psup.second[r+1]; ++i)
109  if (c == psup.first[i]) return i;
110  Throw( "Cannot find row, column: " + std::to_string(r) + ',' +
111  std::to_string(c) + " in sparse matrix" );
112  };
113 
114  const auto& x = coord[0];
115  const auto& y = coord[1];
116  const auto& z = coord[2];
117 
118  // Zero matrix for all components
119  for (ncomp_t c=0; c<m_ncomp; ++c) {
120  lhsd.fill( c, m_offset, 0.0 );
121  lhso.fill( c, m_offset, 0.0 );
122  }
123 
124  for (std::size_t e=0; e<inpoel.size()/4; ++e) {
125  const auto A = inpoel[e*4+0];
126  const auto B = inpoel[e*4+1];
127  const auto C = inpoel[e*4+2];
128  const auto D = inpoel[e*4+3];
129  std::array< tk::real, 3 > ba{{ x[B]-x[A], y[B]-y[A], z[B]-z[A] }},
130  ca{{ x[C]-x[A], y[C]-y[A], z[C]-z[A] }},
131  da{{ x[D]-x[A], y[D]-y[A], z[D]-z[A] }};
132  const auto J = tk::triple( ba, ca, da ) / 120.0;
133 
134  for (ncomp_t c=0; c<m_ncomp; ++c) {
135  const auto r = lhsd.cptr( c, m_offset );
136  lhsd.var( r, A ) += 2.0 * J;
137  lhsd.var( r, B ) += 2.0 * J;
138  lhsd.var( r, C ) += 2.0 * J;
139  lhsd.var( r, D ) += 2.0 * J;
140 
141  const auto s = lhso.cptr( c, m_offset );
142  lhso.var( s, spidx(A,B) ) += J;
143  lhso.var( s, spidx(A,C) ) += J;
144  lhso.var( s, spidx(A,D) ) += J;
145 
146  lhso.var( s, spidx(B,A) ) += J;
147  lhso.var( s, spidx(B,C) ) += J;
148  lhso.var( s, spidx(B,D) ) += J;
149 
150  lhso.var( s, spidx(C,A) ) += J;
151  lhso.var( s, spidx(C,B) ) += J;
152  lhso.var( s, spidx(C,D) ) += J;
153 
154  lhso.var( s, spidx(D,A) ) += J;
155  lhso.var( s, spidx(D,B) ) += J;
156  lhso.var( s, spidx(D,C) ) += J;
157  }
158  }
159  }

161  //! Compute right hand side
162  //! \param[in] mult Multiplier differentiating the different stages in
163  //! multi-stage time stepping
164  //! \param[in] deltat Size of time step
165  //! \param[in] coord Mesh node coordinates
166  //! \param[in] inpoel Mesh element connectivity
167  //! \param[in] U Solution vector at recent time step stage
168  //! \param[in,out] R Right-hand side vector computed
169  //! \author J. Bakosi
170  void rhs( tk::real mult,
171  tk::real deltat,
172  const std::array< std::vector< tk::real >, 3 >& coord,
173  const std::vector< std::size_t >& inpoel,
174  const tk::Fields& U,
175  tk::Fields& R ) const
176  {
177  Assert( U.nunk() == coord[0].size(), "Number of unknowns in solution "
178  "vector at recent time step incorrect" );
179  Assert( R.nunk() == coord[0].size() && R.nprop() == m_ncomp,
180  "Number of unknowns and/or number of components in right-hand "
181  "side vector incorrect" );
182 
183  const auto& x = coord[0];
184  const auto& y = coord[1];
185  const auto& z = coord[2];
186 
187  // zero right hand side for all components
188  for (ncomp_t c=0; c<m_ncomp; ++c) R.fill( c, m_offset, 0.0 );
189 
190  for (std::size_t e=0; e<inpoel.size()/4; ++e) {
191  const std::array< std::size_t, 4 > N{{ inpoel[e*4+0], inpoel[e*4+1],
192  inpoel[e*4+2], inpoel[e*4+3] }};
193  // compute element Jacobi determinant
194  const std::array< tk::real, 3 >
195  ba{{ x[N[1]]-x[N[0]], y[N[1]]-y[N[0]], z[N[1]]-z[N[0]] }},
196  ca{{ x[N[2]]-x[N[0]], y[N[2]]-y[N[0]], z[N[2]]-z[N[0]] }},
197  da{{ x[N[3]]-x[N[0]], y[N[3]]-y[N[0]], z[N[3]]-z[N[0]] }};
198  const auto J = tk::triple( ba, ca, da );
199 
200  // construct tetrahedron element-level matrices
201 
202  // consistent mass
203  std::array< std::array< tk::real, 4 >, 4 > mass; // nnode*nnode [4][4]
204  mass[0][0] = mass[1][1] = mass[2][2] = mass[3][3] = J/60.0; // diagonal
205  mass[0][1] = mass[0][2] = mass[0][3] = // off-diagonal
206  mass[1][0] = mass[1][2] = mass[1][3] =
207  mass[2][0] = mass[2][1] = mass[2][3] =
208  mass[3][0] = mass[3][1] = mass[3][2] = J/120.0;
209 
210  // shape function derivatives
211  std::array< std::array< tk::real, 3 >, 4 > grad; // nnode*ndim [4][3]
212  grad[1] = tk::crossdiv( ca, da, J );
213  grad[2] = tk::crossdiv( da, ba, J );
214  grad[3] = tk::crossdiv( ba, ca, J );
215  for (std::size_t i=0; i<3; ++i)
216  grad[0][i] = -grad[1][i]-grad[2][i]-grad[3][i];
217 
218  // access solution at element nodes at recent time step stage
219  std::vector< std::array< tk::real, 4 > > u( m_ncomp );
220  for (ncomp_t c=0; c<m_ncomp; ++c) u[c] = U.extract( c, m_offset, N );
221  // access pointer to right hand side at component and offset
222  std::vector< const tk::real* > r( m_ncomp );
223  for (ncomp_t c=0; c<m_ncomp; ++c) r[c] = R.cptr( c, m_offset );
224 
225  // get velocity for problem
226  const auto vel =
227  Problem::template
228  prescribedVelocity< tag::transport >( N, coord, m_c, m_ncomp );
229 
230  // add advection contribution to right hand side
231  tk::real a = mult * deltat;
232  for (ncomp_t c=0; c<m_ncomp; ++c)
233  for (std::size_t i=0; i<4; ++i)
234  for (std::size_t j=0; j<4; ++j)
235  for (std::size_t k=0; k<3; ++k)
236  for (std::size_t l=0; l<4; ++l)
237  R.var(r[c],N[j]) -= a * mass[j][i] * vel[c][k][i]
238  * grad[l][k] * u[c][l];
239 
240  // add diffusion contribution to right hand side
241  Physics::diffusionRhs( m_c, m_ncomp, mult, deltat, J, N, grad, u, r, R );
242  }
243  }

245  //! Compute the minimum time step size
246  //! \param[in] U Solution vector at recent time step stage
247  //! \param[in] coord Mesh node coordinates
248  //! \param[in] inpoel Mesh element connectivity
249  //! \return Minimum time step size
250  tk::real dt( const std::array< std::vector< tk::real >, 3 >& coord,
251  const std::vector< std::size_t >& inpoel,
252  const tk::Fields& U ) const
253  {
254  Assert( U.nunk() == coord[0].size(), "Number of unknowns in solution "
255  "vector at recent time step incorrect" );
256  const auto& x = coord[0];
257  const auto& y = coord[1];
258  const auto& z = coord[2];
259  // compute the minimum dt across all elements we own
260  tk::real mindt = std::numeric_limits< tk::real >::max();
261  for (std::size_t e=0; e<inpoel.size()/4; ++e) {
262  const std::array< std::size_t, 4 > N{{ inpoel[e*4+0], inpoel[e*4+1],
263  inpoel[e*4+2], inpoel[e*4+3] }};
264  // compute cubic root of element volume as the characteristic length
265  const std::array< tk::real, 3 >
266  ba{{ x[N[1]]-x[N[0]], y[N[1]]-y[N[0]], z[N[1]]-z[N[0]] }},
267  ca{{ x[N[2]]-x[N[0]], y[N[2]]-y[N[0]], z[N[2]]-z[N[0]] }},
268  da{{ x[N[3]]-x[N[0]], y[N[3]]-y[N[0]], z[N[3]]-z[N[0]] }};
269  const auto L = std::cbrt( tk::triple( ba, ca, da ) / 6.0 );
270  // access solution at element nodes at recent time step stage
271  std::vector< std::array< tk::real, 4 > > u( m_ncomp );
272  for (ncomp_t c=0; c<m_ncomp; ++c) u[c] = U.extract( c, m_offset, N );
273  // get velocity for problem
274  const auto vel =
275  Problem::template
276  prescribedVelocity< tag::transport >( N, coord, m_c, m_ncomp );
277  // compute the maximum length of the characteristic velocity (advection
278  // velocity) across the four element nodes
279  tk::real maxvel = 0.0;
280  for (ncomp_t c=0; c<m_ncomp; ++c)
281  for (std::size_t i=0; i<4; ++i) {
282  auto v = std::sqrt( vel[c][0][i]*vel[c][0][i] +
283  vel[c][1][i]*vel[c][1][i] +
284  vel[c][2][i]*vel[c][2][i] );
285  if (v > maxvel) maxvel = v;
286  }
287  // compute element dt for the advection
288  auto advection_dt = L / maxvel;
289  // compute element dt based on diffusion
290  auto diffusion_dt = Physics::diffusion_dt( m_c, m_ncomp, L, u );
291  // compute minimum element dt
292  auto elemdt = std::min( advection_dt, diffusion_dt );
293  // find minimum dt across all elements
294  if (elemdt < mindt) mindt = elemdt;
295  }
296  return mindt;
297  }
