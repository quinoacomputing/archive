85 // *****************************************************************************
86 //! \brief Find and return minimum and maximum values in associative container
87 //! \param[in] map Map whose extents of values to find
88 //! \return Array of two values with the minimum and maximum values in the map
89 //! \Note This function should not be called with heavy Value types, as the a

100 template< class T, class Allocator >
101 std::vector< T, Allocator >&
102 operator+=( std::vector< T, Allocator >& dst,
103  const std::vector< T, Allocator >& src )
104 // *****************************************************************************
105 //! \brief Add all elements of a vector to another one
106 //! \param[inout] dst Destination vector, i.e., left-hand side of v1 += v2
107 //! \param[in] src Source vector, i.e., righ-hand side of v1 += v2
108 //! \return Destination containing v1[0] += v2[0], v1[1] += v2[1], ...
109 //! \details If src.size() > dst.size() will grow dst to that of src.size()
110 //! padding with zeros.
111 //! \note Will throw exception in DEBUG if src is empty (to warn on no-op), and
112 //! if src.size() < dst.size() (to warn on loosing data).
113 // *****************************************************************************
114 {
115  Assert( !src.empty(), "src empty in std::vector<T,Allocator>::operator+=()" );
116  Assert( src.size() >= dst.size(), "src.size() < dst.size() would loose data "
117  "in std::vector<T,Allocator>::operator+=()" );
118  dst.resize( src.size() );
119  std::transform( src.begin(), src.end(), dst.begin(), dst.begin(),
120  []( const T& s, T& d ){ return d += s; } );
121  return dst;
122 }

124 // *****************************************************************************
125 //! Test if all keys of two associative containers are equal
126 //! \param[in] a 1st container to compare
127 //! \param[in] b 2nd container to compare
128 //! \return True if the containers have the same size and all keys (and only the
129 //! keys) of the two containers are equal
130 //! \note It is an error to call this function with unequal-size containers,
131 //! triggering an exception in DEBUG mode.
132 //! \note Operator != is used to compare the container keys.
133 // *****************************************************************************
134 template< class Container >
135 bool keyEqual( const Container& a, const Container& b ) {
136  Assert( a.size() == b.size(), "Size mismatch comparing containers" );
137  auto ia = a.cbegin();
138  auto ib = b.cbegin();
139  while (ia != a.cend()) {
140  if (ia->first != ib->first) return false;
141  ++ia;
142  ++ib;
143  }
144  return true;
145 }
