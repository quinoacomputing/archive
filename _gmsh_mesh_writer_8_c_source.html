140 GmshMeshWriter::writeElemBlock( std::size_t nnpe,
141  GmshElemType type,
142  const std::vector< std::size_t >& inpoel )
143 // *****************************************************************************
144 // Write element block: element ids and connectivity (node list)
145 //! \param[in] nnpe Number of nodes per element
146 //! \param[in] type Element type
147 //! \param[in] inpoel Element connectivity (must be zero-based)
148 // *****************************************************************************
149 {
150  // Return if connectivity is empty, there is no such element block in mesh
151  if (inpoel.empty()) return;
152 
153  // Make sure element connectivity starts with zero
154  Assert( *std::minmax_element( begin(inpoel), end(inpoel) ).first == 0,
155  "node ids should start from zero" );
156 
157  // Get number of elements in mesh
158  auto n = inpoel.size()/nnpe;
159 
160  // Ignore element tags
161  std::vector< std::vector< int > > tg;
162  tg.resize( n );
163  for (auto& t : tg) t.push_back( 0 );
164 
165  if (isASCII()) {
166 
167  for (std::size_t i=0; i<n; i++) {
168  // elm-number elm-type number-of-tags < tag > ... node-number-list
169  m_outFile << i+1 << " " << type << " " << tg[i].size() << " ";
170  copy( tg[i].begin(), tg[i].end()-1,
171  std::ostream_iterator< int >( m_outFile, " " ) );
172  m_outFile << tg[i].back() << " ";
173 
174  // gmsh likes one-based node ids
175  for (std::size_t k=0; k<nnpe; k++) m_outFile << inpoel[i*nnpe+k]+1 << " ";
176  m_outFile << std::endl;
177  }
178 
179  } else {
180 
181  int ntags = static_cast< int >( tg[0].size() );
182  int nel = static_cast< int >( n );
183  // elm-type num-of-elm-follow number-of-tags
184  m_outFile.write( reinterpret_cast<char*>(&type), sizeof(int) );
185  m_outFile.write( reinterpret_cast<char*>(&nel), sizeof(int) );
186  m_outFile.write( reinterpret_cast<char*>(&ntags), sizeof(int) );
187  for (std::size_t i=0; i<n; i++) {
188  int I = static_cast< int >( i );
189  // gmsh likes one-based node ids
190  std::vector< int > Inpoel;
191  for (std::size_t k=0; k<nnpe; ++k)
192  Inpoel.push_back( static_cast< int >( inpoel[i*nnpe+k]+1 ) );
193  // element id
194  m_outFile.write( reinterpret_cast<const char*>(&I), sizeof(int) );
195  // element tags
196  m_outFile.write( reinterpret_cast<const char*>(tg[i].data()),
197  static_cast<std::streamsize>(tg[i].size()*sizeof(int)) );
198  // element node list (i.e. connectivity)
199  m_outFile.write( reinterpret_cast<const char*>(Inpoel.data()),
200  static_cast<std::streamsize>(nnpe*sizeof(int)) );
201  }
