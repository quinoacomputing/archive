77 GmshMeshWriter::writeNodes( const UnsMesh& mesh )
78 // *****************************************************************************
79 // Write "$Nodes--$EndNodes" section
80 //! \param[in] mesh Unstructured mesh object
81 //! \author J. Bakosi
82 // *****************************************************************************
83 {
84  m_outFile << "$Nodes" << std::endl;
85 
86  // Write out number of nodes
87  m_outFile << mesh.nnode() << std::endl;
88 
89  // Write node ids and coordinates: node-number x-coord y-coord z-coord
90  if (isASCII()) {
91  for (std::size_t i=0; i<mesh.nnode(); ++i) {
92  m_outFile << i+1 << " " << std::setprecision(16)
93  << mesh.x()[i] << " "
94  << mesh.y()[i] << " "
95  << mesh.z()[i] << std::endl;
96  }
97  } else {
98  for (std::size_t i=0; i<mesh.nnode(); ++i) {
99  // gmsh likes one-based node ids
100  int I = static_cast< int >( i+1 );
101  m_outFile.write(
102  reinterpret_cast<const char*>(&I), sizeof(int) );
103  m_outFile.write(
104  reinterpret_cast<const char*>(&mesh.x()[i]), sizeof(double) );
105  m_outFile.write(
106  reinterpret_cast<const char*>(&mesh.y()[i]), sizeof(double) );
107  m_outFile.write(
108  reinterpret_cast<const char*>(&mesh.z()[i]), sizeof(double) );
109  }
110  m_outFile << std::endl;
111  }
112 
113  m_outFile << "$EndNodes" << std::endl;
114 }

146 GmshMeshWriter::writeElemBlock( std::size_t nnpe,
147  GmshElemType type,
148  const std::vector< std::size_t >& inpoel )
149 // *****************************************************************************
150 // Write element block: element ids and connectivity (node list)
151 //! \param[in] nnpe Number of nodes per element
152 //! \param[in] type Element type
153 //! \param[in] inpoel Element connectivity (must be zero-based)
154 //! \author J. Bakosi
155 // *****************************************************************************
156 {
157  // Return if connectivity is empty, there is no such element block in mesh
158  if (inpoel.empty()) return;
159 
160  // Make sure element connectivity starts with zero
161  Assert( *std::minmax_element( begin(inpoel), end(inpoel) ).first == 0,
162  "node ids should start from zero" );
163 
164  // Get number of elements in mesh
165  auto n = inpoel.size()/nnpe;
166 
167  // Ignore element tags
168  std::vector< std::vector< int > > tg;
169  tg.resize( n );
170  for (auto& t : tg) t.push_back( 0 );
171 
172  if (isASCII()) {
173 
174  for (std::size_t i=0; i<n; i++) {
175  // elm-number elm-type number-of-tags < tag > ... node-number-list
176  m_outFile << i+1 << " " << type << " " << tg[i].size() << " ";
177  copy( tg[i].begin(), tg[i].end()-1,
178  std::ostream_iterator< int >( m_outFile, " " ) );
179  m_outFile << tg[i].back() << " ";
180 
181  // gmsh likes one-based node ids
182  for (std::size_t k=0; k<nnpe; k++) m_outFile << inpoel[i*nnpe+k]+1 << " ";
183  m_outFile << std::endl;
184  }
185 
186  } else {
187 
188  int ntags = static_cast< int >( tg[0].size() );
189  // elm-type num-of-elm-follow number-of-tags
190  m_outFile.write( reinterpret_cast<char*>(&type), sizeof(int) );
191  m_outFile.write( reinterpret_cast<char*>(&n), sizeof(int) );
192  m_outFile.write( reinterpret_cast<char*>(&ntags), sizeof(int) );
193  for (std::size_t i=0; i<n; i++) {
194  int I = static_cast< int >( i );
195  // gmsh likes one-based node ids
196  std::vector< int > Inpoel;
197  for (std::size_t k=0; k<nnpe; ++k)
198  Inpoel.push_back( static_cast< int >( inpoel[i*nnpe+k]+1 ) );
199  // element id
200  m_outFile.write( reinterpret_cast<const char*>(&I), sizeof(int) );
201  // element tags
202  m_outFile.write( reinterpret_cast<const char*>(tg[i].data()),
203  static_cast<std::streamsize>(tg[i].size()*sizeof(int)) );
204  // element node list (i.e. connectivity)
205  m_outFile.write( reinterpret_cast<const char*>(Inpoel.data()),
206  static_cast<std::streamsize>(nnpe*sizeof(int)) );
207  }
208 
209  }
