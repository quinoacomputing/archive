30 GmshMeshReader::readMesh( UnsMesh& mesh )
31 // *****************************************************************************
32 // Public interface for read a Gmsh mesh from file
33 //! \param[in] mesh Unstructured mesh object
34 //! \author J. Bakosi
35 // *****************************************************************************
36 {
37  // Read in mandatory "$MeshFormat" section
38  readMeshFormat();
39 
40  // Keep reading in sections until end of file. These sections can be in
41  // arbitrary order, hence a while loop.
42  while ( !m_inFile.eof() ) {
43  std::string s;
44  getline( m_inFile, s );
45  if ( s == "$Nodes" )
46  readNodes( mesh );
47  else if ( s == "$Elements" )
48  readElements( mesh );
49  else if ( s == "$PhysicalNames" )
50  readPhysicalNames();
51  }
52 }

54 void
55 GmshMeshReader::readMeshFormat()
56 // *****************************************************************************
57 // Read mandatory "$MeshFormat--$EndMeshFormat" section
58 //! \author J. Bakosi
59 // *****************************************************************************
60 {
61  using tk::operator<<;
62  std::string s;
63 
64  // Read in beginning of header: $MeshFormat
65  getline( m_inFile, s );
66  ErrChk( s == "$MeshFormat",
67  std::string("Unsupported mesh format '") + s + "' in file " +
68  m_filename );
69 
70  // Read in "version-number file-type data-size"
71  int type;
72  m_inFile >> m_version >> type >> m_datasize;
73  if (type == 0 )
74  m_type = GmshFileType::ASCII;
75  else if (type == 1 )
76  m_type = GmshFileType::BINARY;
77 
78  ErrChk( ( fabs(m_version-2.2) < std::numeric_limits<tk::real>::epsilon() ||
79  fabs(m_version-2.0) < std::numeric_limits<tk::real>::epsilon() ),
80  std::string("Unsupported mesh version '") << m_version <<
81  "' in file " << m_filename );
82 
83  ErrChk( ( m_type == GmshFileType::ASCII || m_type == GmshFileType::BINARY ),
84  std::string("Unsupported mesh type '") << m_type << "' in file " <<
85  m_filename );
86 
87  ErrChk( m_datasize == sizeof(tk::real),
88  std::string("Unsupported mesh datasize '") << m_datasize <<
89  "' in file " << m_filename );
90 
91  getline( m_inFile, s ); // finish reading the line
92 
93  // if file is binary, binary-read in binary "one"
94  if ( isBinary() ) {
95  int one;
96  m_inFile.read( reinterpret_cast<char*>(&one), sizeof(int) );
97  ErrChk( one == 1, "Endianness does not match in file " + m_filename );
98  getline( m_inFile, s ); // finish reading the line
99  }
100 
101  // Read in end of header: $EndMeshFormat
102  getline( m_inFile, s );
103  ErrChk( s == "$EndMeshFormat",
104  "'$EndMeshFormat' keyword is missing in file " + m_filename );
105 }

107 void
108 GmshMeshReader::readNodes( UnsMesh& mesh )
109 // *****************************************************************************
110 // Read "$Nodes--$EndNodes" section
111 //! \param[in] mesh Unstructured mesh object
112 //! \author J. Bakosi
113 // *****************************************************************************
114 {
115  // Read in number of nodes in this node set
116  std::size_t nnode;
117  m_inFile >> nnode;
118  ErrChk( nnode > 0,
119  "Number of nodes must be greater than zero in file " + m_filename );
120  std::string s;
121  if (isBinary()) getline( m_inFile, s ); // finish reading the line
122 
123  // Read in node ids and coordinates: node-number x-coord y-coord z-coord
124  for ( std::size_t i=0; i<nnode; ++i ) {
125  int id;
126  std::array< tk::real, 3 > coord;
127 
128  if (isASCII()) {
129  m_inFile >> id >> coord[0] >> coord[1] >> coord[2];
130  } else {
131  m_inFile.read( reinterpret_cast<char*>(&id), sizeof(int) );
132  m_inFile.read( reinterpret_cast<char*>(coord.data()), 3*sizeof(double) );
133  }
134 
135  mesh.x().push_back( coord[0] );
136  mesh.y().push_back( coord[1] );
137  mesh.z().push_back( coord[2] );
138  }
139  getline( m_inFile, s ); // finish reading the last line
140 
141  // Read in end of header: $EndNodes
142  getline( m_inFile, s );
143  ErrChk( s == "$EndNodes",
144  "'$EndNodes' keyword is missing in file" + m_filename );
145 }

147 void
148 GmshMeshReader::readElements( UnsMesh& mesh )
149 // *****************************************************************************
150 // Read "$Elements--$EndElements" section
151 //! \param[in] mesh Unstructured mesh object
152 //! \author J. Bakosi
153 // *****************************************************************************
154 {
155  using tk::operator<<;
156 
157  std::string s;
158 
159  // Read in number of elements in this element set
160  int nel;
161  m_inFile >> nel;
162  ErrChk( nel > 0, "Number of elements must be greater than zero in file " +
163  m_filename );
164  getline( m_inFile, s ); // finish reading the last line
165 
166  // Read in element ids, tags, and element connectivity (node list)
167  int n=1;
168  for (int i=0; i<nel; i+=n) {
169  int id, elmtype, ntags;
170 
171  if (isASCII()) {
172  // elm-number elm-type number-of-tags < tag > ... node-number-list
173  m_inFile >> id >> elmtype >> ntags;
174  } else {
175  // elm-type num-of-elm-follow number-of-tags
176  m_inFile.read( reinterpret_cast<char*>(&elmtype), sizeof(int) );
177  m_inFile.read( reinterpret_cast<char*>(&n), sizeof(int) );
178  m_inFile.read( reinterpret_cast<char*>(&ntags), sizeof(int) );
179  }
180 
181  // Find element type, throw exception if not supported
182  const auto it = m_elemNodes.find( elmtype );
183  ErrChk( it != m_elemNodes.end(),
184  std::string("Unsupported element type ") << elmtype <<
185  " in mesh file: " << m_filename );
186 
187  for (int e=0; e<n; ++e) {
188  // Read element id if binary
189  if (isBinary()) {
190  m_inFile.read( reinterpret_cast<char*>(&id), sizeof(int) );
191  }
192 
193  // Read and ignore element tags
194  std::vector< int > tags( static_cast<std::size_t>(ntags), 0 );
195  if (isASCII()) {
196  for (std::size_t j=0; j<static_cast<std::size_t>(ntags); j++)
197  m_inFile >> tags[j];
198  } else {
199  m_inFile.read(
200  reinterpret_cast<char*>(tags.data()),
201  static_cast<std::streamsize>(
202  static_cast<std::size_t>(ntags) * sizeof(int) ) );
203  }
204 
205  // Read and add element node list (i.e. connectivity)
206  std::size_t nnode = static_cast< std::size_t >( it->second );
207  std::vector< std::size_t > nodes( nnode, 0 );
208  if (isASCII()) {
209  for (std::size_t j=0; j<nnode; j++)
210  m_inFile >> nodes[j];
211  } else {
212  std::vector< int > nds( nnode, 0 );
213  m_inFile.read(
214  reinterpret_cast< char* >( nds.data() ),
215  static_cast< std::streamsize >( nnode * sizeof(int) ) );
216  for (std::size_t j=0; j<nnode; j++)
217  nodes[j] = static_cast< std::size_t >( nds[j] );
218  }
219  // Put in element connectivity for different types of elements
220  switch ( elmtype ) {
221  case GmshElemType::LIN:
222  for (const auto& j : nodes) mesh.lininpoel().push_back( j );
223  break;
224  case GmshElemType::TRI:
225  for (const auto& j : nodes) mesh.triinpoel().push_back( j );
226  break;
227  case GmshElemType::TET:
228  for (const auto& j : nodes) mesh.tetinpoel().push_back( j );
229  break;
230  case GmshElemType::PNT:
231  break; // ignore 1-node 'point element' type
232  default: Throw( std::string("Unsupported element type ") << elmtype <<
233  " in mesh file: " << m_filename );
234  }
235  }
236  }
237  getline( m_inFile, s ); // finish reading the last line
238 
239  // Shift node IDs to start from zero (gmsh likes one-based node ids)
240  shiftToZero( mesh.lininpoel() );
241  shiftToZero( mesh.triinpoel() );
242  shiftToZero( mesh.tetinpoel() );
243 
244  // Read in end of header: $EndNodes
245  getline( m_inFile, s );
246  ErrChk( s == "$EndElements",
247  "'$EndElements' keyword is missing in file" + m_filename );
248 }
