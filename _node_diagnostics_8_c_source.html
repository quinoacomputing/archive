59 bool
60 NodeDiagnostics::compute( Discretization& d, const tk::Fields& u )
61 // *****************************************************************************
62 // Compute diagnostics, e.g., residuals, norms of errors, etc.
63 //! \param[in] d Discretization proxy to read from
64 //! \param[in] u Current solution vector
65 //! \return True if diagnostics have been computed
66 // *****************************************************************************
67 {
68  // Optionally collect diagnostics and send for aggregation across all workers
69 
70  // Query after how many time steps user wants to dump diagnostics
71  auto diagfreq = g_inputdeck.get< tag::interval, tag::diag >();
72 
73  if ( !((d.It()+1) % diagfreq) ) { // if remainder, don't dump
74 
75  // Collect analytical solutions (if available) from all PDEs. Note that
76  // calling the polymorphic PDE::initialize() is assumed to evaluate the
77  // analytical solution for a PDE. For those PDE problems that have
78  // analytical solutions, this is the same as used for setting the initial
79  // conditions, since if the analytical solution is available for a problem,
80  // it is (so far anyway) always initialized to its analytical solution and
81  // that is done by calling PDE::initialize(). If the analytical solution is
82  // a function of time, that is already incorporated in setting initial
83  // conditions. For those PDEs where the analytical solution is not
84  // available, initialize() returns the initial conditions (obviously), and
85  // thus the "error", defined between this "analytical" and numerical
86  // solution will be a measure of the "distance" between the initial
87  // condition and the current numerical solution. This is not necessarily
88  // useful, but simplies the logic because all PDEs can be treated as being
89  // able to compute an error based on some "analytical" solution, which is
90  // really the initial condition.
91  auto a = u;
92  for (const auto& eq : g_cgpde)
93  eq.initialize( d.Coord(), a, d.T()+d.Dt() );
94 
95  // Prepare for computing diagnostics. Diagnostics are defined as some norm,
96  // .e.g., L2 norm, of a quantity, computed in mesh nodes, A, as ||A||_2 =
97  // sqrt[ sum_i(A_i)^2 V_i ], where the sum is taken over all mesh nodes and
98  // V_i is the nodal volume. We send multiple sets of quantities to the host
99  // for aggregation across the whole mesh: (1) the numerical solutions of all
100  // components of all PDEs, and their error, defined as A_i = (a_i - n_i),
101  // where a_i and n_i are the analytical and numerical solutions at node i,
102  // respectively. The final aggregated solution will end up in
103  // Transporter::diagnostics().
104 
105  // Diagnostics vector (of vectors) during aggregation:
106  // 0: L2-norm of all scalar components of the numerical solution
107  // 1: L2-norm of all scalar components of the numerical-analytic solution
108  // 2: Linf-norm of all scalar components of the numerical-analytic solution
109  std::vector< std::vector< tk::real > >
110  diag( NUMDIAG, std::vector< tk::real >( u.nprop(), 0.0 ) );
111 
112  // Put in norms sweeping our mesh chunk
113  for (std::size_t i=0; i<u.nunk(); ++i)
114  if (m_slave.find(i) == end(m_slave)) { // ignore non-owned nodes
115  // Compute sum for L2 norm of the numerical solution
116  for (std::size_t c=0; c<u.nprop(); ++c)
117  diag[L2SOL][c] += u(i,c,0) * u(i,c,0) * d.Vol()[i];
118  // Compute sum for L2 norm of the numerical-analytic solution
119  for (std::size_t c=0; c<u.nprop(); ++c)
120  diag[L2ERR][c] +=
121  (u(i,c,0)-a(i,c,0)) * (u(i,c,0)-a(i,c,0)) * d.Vol()[i];
122  // Compute max for Linf norm of the numerical-analytic solution
123  for (std::size_t c=0; c<u.nprop(); ++c) {
124  auto err = std::abs( u(i,c,0) - a(i,c,0) );
125  if (err > diag[LINFERR][c]) diag[LINFERR][c] = err;
126  }
127  }
128 
129  // Append diagnostics vector with metadata on the current time step
130  // 3: Current iteration count (only the first entry is used)
131  // 4: Current physical time (only the first entry is used)
132  // 5: Current physical time step size (only the first entry is used)
133  diag[ITER][0] = static_cast< tk::real >( d.It()+1 );
134  diag[TIME][0] = d.T() + d.Dt();
135  diag[DT][0] = d.Dt();
136 
137  // Contribute to diagnostics
138  auto stream = serialize( diag );
139  d.contribute( stream.first, stream.second.get(), DiagMerger,
140  CkCallback(CkIndex_Transporter::diagnostics(nullptr), d.Tr()) );
141 
142  return true;
143  }
144 
145  return false;
146 }
