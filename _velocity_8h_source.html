3  \file src/DiffEq/Velocity.h
4  \copyright 2016-2018, Los Alamos National Security, LLC.
5  \brief A model for velocity in variable-density turbulence
6  \details This file implements the time integration of a system of stochastic
7  differential equations (SDEs) to model the fluctuating velocity components
8  in homogeneous variable-density turbulence. This model is an extension of
9  the generalized Langevin (GLM) model for constant-density flows by Haworth &
10  Pope (https://doi.org/10.1063/1.865723). The extension is roughly along the
11  lines of https://doi.org/10.1080/14685248.2011.554419.
12 */

31 //! \brief Velocity SDE used polymorphically with DiffEq
32 //! \details The template arguments specify policies and are used to configure
33 //! the behavior of the class. The policies are:
34 //! - Init - initialization policy, see DiffEq/InitPolicy.h
35 //! - Coefficients - coefficients policy, see DiffEq/VelocityCoeffPolicy.h
36 template< class Init, class Coefficients >
37 class Velocity {

39  private:
40  using ncomp_t = tk::ctr::ncomp_type;

42  public:
43  //! \brief Constructor
44  //! \param[in] c Index specifying which system of beta SDEs to construct.
45  //! There can be multiple beta ... end blocks in a control file. This
46  //! index specifies which beta SDE system to instantiate. The index
47  //! corresponds to the order in which the beta ... end blocks are given
48  //! the control file.
49  explicit Velocity( ncomp_t c ) :
50  m_c( c ),
51  m_depvar(
52  g_inputdeck.get< tag::param, tag::velocity, tag::depvar >().at(c) ),
53  m_ncomp(
54  g_inputdeck.get< tag::component >().get< tag::velocity >().at(c) ),
55  m_offset(
56  g_inputdeck.get< tag::component >().offset< tag::velocity >(c) ),
57  m_dissipation_depvar(
58  g_inputdeck.get< tag::param, tag::velocity, tag::dissipation >().at(c) ),
59  m_rng( g_rng.at( tk::ctr::raw(
60  g_inputdeck.get< tag::param, tag::velocity, tag::rng >().at(c) ) ) ),
61  m_solve(g_inputdeck.get< tag::param, tag::velocity, tag::solve >().at(c)),
62  m_U( {{ tk::ctr::mean( m_depvar, 0 ),
63  tk::ctr::mean( m_depvar, 1 ),
64  tk::ctr::mean( m_depvar, 2 ) }} ),
65  m_variant(
66  g_inputdeck.get< tag::param, tag::velocity, tag::variant >().at(c) ),
67  m_c0(),
68  m_G(),
69  m_coeff( g_inputdeck.get< tag::param, tag::velocity, tag::c0 >().at(c),
70  m_c0,
71  m_dU )
72  {
73  Assert( m_ncomp == 3, "Velocity eq number of components must be 3" );
74  // Zero prescribed mean velocity gradient if full variable is solved for
75  if (m_solve == ctr::DepvarType::FULLVAR) m_dU.fill( 0.0 );
76  // Populate inverse hydrodynamics time scales extracted from DNS
77  if ( Coefficients::type() == ctr::CoeffPolicyType::HYDROTIMESCALE ) {
78  // Configure inverse hydrodyanmics time scale from DNS
79  const auto& hts = g_inputdeck.get< tag::param,
80  tag::velocity,
81  tag::hydrotimescales >().at(c);
82  Assert( hts.size() == 1,
83  "Velocity eq Hydrotimescales vector size must be 1" );
84  m_hts = ctr::HydroTimeScales().table( hts[0] );
85  }
86  }

88  //! Initalize SDE, prepare for time integration
89  //! \param[in] stream Thread (or more precisely stream) ID
90  //! \param[in,out] particles Array of particle properties
91  void initialize( int stream, tk::Particles& particles ) {
92  // Set initial conditions using initialization policy

96  }

98  //! \brief Advance particles according to the system of beta SDEs
99  //! \param[in,out] particles Array of particle properties
100  //! \param[in] stream Thread (or more precisely stream) ID
101  //! \param[in] dt Time step size
102  void advance( tk::Particles& particles,
103  int stream,
104  tk::real dt,
105  tk::real t,
106  const std::map< tk::ctr::Product, tk::real >& moments )
107  {
108  // Update coefficients
109  tk::real eps = 0.0;
110  m_coeff.update( m_depvar, m_dissipation_depvar, moments, m_hts, m_solve,
111  m_variant, m_c0, t, eps, m_G );
112 
113  // Access mean velocity (if needed)
114  std::array< tk::real, 3 > U{{ 0.0, 0.0, 0.0 }};
115  if (m_solve == ctr::DepvarType::FULLVAR) {
116  using tk::ctr::lookup;
117  U[0] = lookup( m_U[0], moments );
118  U[1] = lookup( m_U[1], moments );
119  U[2] = lookup( m_U[2], moments );
120  }
121 
122  // Modify G with the mean velocity gradient
123  for (std::size_t i=0; i<9; ++i) m_G[i] -= m_dU[i];
124 
125  const auto npar = particles.nunk();
126  for (auto p=decltype(npar){0}; p<npar; ++p) {
127  // Generate Gaussian random numbers with zero mean and unit variance
128  std::vector< tk::real > dW( m_ncomp );
129  m_rng.gaussian( stream, m_ncomp, dW.data() );
130  // Acces particle velocity
131  tk::real& Up = particles( p, 0, m_offset );
132  tk::real& Vp = particles( p, 1, m_offset );
133  tk::real& Wp = particles( p, 2, m_offset );
134  // Compute diffusion
135  tk::real d = m_c0 * eps * dt;
136  d = (d > 0.0 ? std::sqrt(d) : 0.0);
137  // Compute velocity fluctuation
138  tk::real u = Up - U[0];
139  tk::real v = Vp - U[1];
140  tk::real w = Wp - U[2];
141  // Update particle velocity
142  Up += (m_G[0]*u + m_G[1]*v + m_G[2]*w)*dt + d*dW[0];
143  Vp += (m_G[3]*u + m_G[4]*v + m_G[5]*w)*dt + d*dW[1];
144  Wp += (m_G[6]*u + m_G[7]*v + m_G[8]*w)*dt + d*dW[2];
145  }
146  }
