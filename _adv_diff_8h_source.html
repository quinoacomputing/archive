27  //! \param[in] e Equation system index, i.e., which transport equation

38  diffusionRhs( tk::ctr::ncomp_type e,
39  tk::ctr::ncomp_type ncomp,
40  tk::real deltat,
41  tk::real J,
42  const std::array< std::array< tk::real, 3 >, 4 >& grad,
43  const std::array< std::size_t, 4 >& N,
44  const std::vector< std::array< tk::real, 4 > >& u,
45  const std::vector< const tk::real* >& r,
46  tk::Fields& R )
47  {
48  // diffusivities for all components
49  const auto& diff =
50  g_inputdeck.get< tag::param, tag::transport, tag::diffusivity >().at(e);
51  // add diffusion contribution to right hand side
52  const auto d = deltat * J/6.0;
53  for (ncomp_t c=0; c<ncomp; ++c)
54  for (std::size_t k=0; k<3; ++k) {
55  const auto D = diff[ 3*c+k ];
56  for (std::size_t a=0; a<4; ++a)
57  for (std::size_t b=0; b<4; ++b)
58  R.var(r[c],N[a]) -= d * D * grad[a][k] * grad[b][k] * u[c][b];
59  }
60  }

62  //! Compute the minimum time step size based on the diffusion
63  //! \param[in] e Equation system index, i.e., which transport equation
64  //! system we operate on among the systems of PDEs
65  //! \param[in] ncomp Number of components in this PDE
66  //! \param[in] L Characteristic length scale
67  //! \return Minimum time step size based on diffusion
68  static tk::real
69  diffusion_dt( tk::ctr::ncomp_type e,
70  tk::ctr::ncomp_type ncomp,
71  tk::real L,
72  const std::vector< std::array< tk::real, 4 > >& )
73  {
74  // diffusivities for all components
75  const auto& df =
76  g_inputdeck.get< tag::param, tag::transport, tag::diffusivity >().at(e);
77  // compute the minimum diffusion time step size across the four nodes
78  tk::real mindt = std::numeric_limits< tk::real >::max();
79  for (ncomp_t c=0; c<ncomp; ++c) {
80  const auto di = 3*c;
81  const auto d = std::max( df[di+2], std::max( df[di+0], df[di+1] ) );
82  const auto dt = L * L / (2.0*d); // dt ~ dx^2/(2D)
83  if (dt < mindt) mindt = dt;
84  }
85  return mindt;
86  }
