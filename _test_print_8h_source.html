87 //! Test saving pointer to quiet stream
88 template<> template<>
89 void Print_object::test< 5 >() {
90  set_test_name( "save/reset pointer to quiet stream" );
91 
92  prq % "blah"; // write to quiet stream
93  auto p = prq.save< tk::QUIET >(); // save quiet stream pointer
94  // Instantiate new printer using two new stringstreams. Note, if we use the
95  // default constructor arguments here, the quiet stream will be std::cout,
96  // which then will be overwritten by the stream saved out of prq above,
97  // containing "blah" so far. Overwriting std::cout results in hijacking
98  // std::cout and thus we would see no screen output after the reset call
99  // below.
100  std::stringstream v, q;
101  tk::Print pr( v, q );
102  // Reset quiet stream in new printer to saved one, has "blah"
103  pr.reset< tk::QUIET >( p );
104  // Write to new quiet stream, now should have "blah blah"
105  pr % " blah";
106  ensure_equals( "initialize new object with previously saved quiet stream "
107  "of another object", quiet.str(), "blah blah" );
108 }
