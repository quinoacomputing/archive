68  //! Rule used to trigger action
69  template< class eq > struct check_transport : pegtl::success {};
70  //! \brief Set defaults and do error checking on the transport equation block
71  //! \details This is error checking that only the transport equation block
72  //! must satisfy. Besides error checking we also set defaults here as
73  //! this block is called when parsing of a transport...end block has
74  //! just finished.
75  template< class eq >
76  struct action< check_transport< eq > > {
77  template< typename Input, typename Stack >
78  static void apply( const Input& in, Stack& stack ) {
79  using inciter::deck::neq;
80  // Error out if no dependent variable has been selected
81  auto& depvar = stack.template get< tag::param, eq, tag::depvar >();
82  if (depvar.empty() || depvar.size() != neq.get< eq >())
83  Message< Stack, ERROR, MsgKey::NODEPVAR >( stack, in );
84  // If no number of components has been selected, default to 1
85  auto& ncomp = stack.template get< tag::component, eq >();
86  if (ncomp.empty() || ncomp.size() != neq.get< eq >())
87  ncomp.push_back( 1 );
88  // If physics type is not given, default to 'advection'
89  auto& physics = stack.template get< tag::param, eq, tag::physics >();
90  if (physics.empty() || physics.size() != neq.get< eq >())
91  physics.push_back( inciter::ctr::PhysicsType::ADVECTION );
92  // If physics type is advection-diffusion, check for correct number of
93  // advection velocity, shear, and diffusion coefficients
94  if (physics.back() == inciter::ctr::PhysicsType::ADVDIFF) {
95  auto& u0 = stack.template get< tag::param, eq, tag::u0 >();
96  if (u0.back().size() != ncomp.back()) // must define 1 component
97  Message< Stack, ERROR, MsgKey::WRONGSIZE >( stack, in );
98  auto& diff = stack.template get< tag::param, eq, tag::diffusivity >();
99  if (diff.back().size() != 3*ncomp.back()) // must define 3 components
100  Message< Stack, ERROR, MsgKey::WRONGSIZE >( stack, in );
101  auto& lambda = stack.template get< tag::param, eq, tag::lambda >();
102  if (lambda.back().size() != 2*ncomp.back()) // must define 2 shear comps
103  Message< Stack, ERROR, MsgKey::WRONGSIZE >( stack, in );
104  }
105  // If problem type is not given, error out
106  auto& problem = stack.template get< tag::param, eq, tag::problem >();
107  if (problem.empty() || problem.size() != neq.get< eq >())
108  Message< Stack, ERROR, MsgKey::NOPROBLEM >( stack, in );
109  // Error check Dirichlet boundary condition block for all transport eq
110  // configurations
111  for (const auto& s : stack.template get< tag::param, eq, tag::bcdir >())
112  if (s.empty())
113  Message< Stack, ERROR, MsgKey::BC_EMPTY >( stack, in );
114  }
115  };

117  //! Rule used to trigger action
118  template< class eq > struct check_compflow : pegtl::success {};
119  //! \brief Set defaults and do error checking on the compressible flow
120  //! equation block
121  //! \details This is error checking that only the compressible flow equation
122  //! block must satisfy. Besides error checking we also set defaults here as
123  //! this block is called when parsing of a compflow...end block has
124  //! just finished.
125  template< class eq >
126  struct action< check_compflow< eq > > {
127  template< typename Input, typename Stack >
128  static void apply( const Input& in, Stack& stack ) {
129  using inciter::deck::neq;
130  // Error out if no dependent variable has been selected
131  auto& depvar = stack.template get< tag::param, eq, tag::depvar >();
132  if (depvar.empty() || depvar.size() != neq.get< eq >())
133  Message< Stack, ERROR, MsgKey::NODEPVAR >( stack, in );
134  // Set number of components to 5 (mass, 3 x mom, energy)
135  stack.template get< tag::component, eq >().push_back( 5 );
136  // If physics type is not given, default to 'euler'
137  auto& physics = stack.template get< tag::param, eq, tag::physics >();
138  if (physics.empty() || physics.size() != neq.get< eq >())
139  physics.push_back( inciter::ctr::PhysicsType::EULER );
140  // If problem type is not given, default to 'user_defined'
141  auto& problem = stack.template get< tag::param, eq, tag::problem >();
142  if (problem.empty() || problem.size() != neq.get< eq >())
143  problem.push_back( inciter::ctr::ProblemType::USER_DEFINED );
144  else if (problem.back() == inciter::ctr::ProblemType::VORTICAL_FLOW) {
145  const auto& alpha = stack.template get< tag::param, eq, tag::alpha >();
146  const auto& beta = stack.template get< tag::param, eq, tag::beta >();
147  const auto& p0 = stack.template get< tag::param, eq, tag::p0 >();
148  if ( alpha.size() != problem.size() ||
149  beta.size() != problem.size() ||
150  p0.size() != problem.size() )
151  Message< Stack, ERROR, MsgKey::VORTICAL_UNFINISHED >( stack, in );
152  }
153  else if (problem.back() == inciter::ctr::ProblemType::NL_ENERGY_GROWTH) {
154  const auto& alpha = stack.template get< tag::param, eq, tag::alpha >();
155  const auto& betax = stack.template get< tag::param, eq, tag::betax >();
156  const auto& betay = stack.template get< tag::param, eq, tag::betay >();
157  const auto& betaz = stack.template get< tag::param, eq, tag::betaz >();
158  const auto& kappa = stack.template get< tag::param, eq, tag::kappa >();
159  const auto& r0 = stack.template get< tag::param, eq, tag::r0 >();
160  const auto& ce = stack.template get< tag::param, eq, tag::ce >();
161  if ( alpha.size() != problem.size() ||
162  betax.size() != problem.size() ||
163  betay.size() != problem.size() ||
164  betaz.size() != problem.size() ||
165  kappa.size() != problem.size() ||
166  r0.size() != problem.size() ||
167  ce.size() != problem.size() )
168  Message< Stack, ERROR, MsgKey::ENERGY_UNFINISHED >( stack, in);
169  }
170  else if (problem.back() == inciter::ctr::ProblemType::RAYLEIGH_TAYLOR) {
171  const auto& alpha = stack.template get< tag::param, eq, tag::alpha >();
172  const auto& betax = stack.template get< tag::param, eq, tag::betax >();
173  const auto& betay = stack.template get< tag::param, eq, tag::betay >();
174  const auto& betaz = stack.template get< tag::param, eq, tag::betaz >();
175  const auto& kappa = stack.template get< tag::param, eq, tag::kappa >();
176  const auto& p0 = stack.template get< tag::param, eq, tag::p0 >();
177  const auto& r0 = stack.template get< tag::param, eq, tag::r0 >();
178  if ( alpha.size() != problem.size() ||
179  betax.size() != problem.size() ||
180  betay.size() != problem.size() ||
181  betaz.size() != problem.size() ||
182  kappa.size() != problem.size() ||
183  p0.size() != problem.size() ||
184  r0.size() != problem.size() )
185  Message< Stack, ERROR, MsgKey::RT_UNFINISHED >( stack, in);
186  }
187  // Error check Dirichlet boundary condition block for all compflow
188  // configurations
189  for (const auto& s : stack.template get< tag::param, eq, tag::bcdir >())
190  if (s.empty())
191  Message< Stack, ERROR, MsgKey::BC_EMPTY >( stack, in );
192  }
193  };

211  //! Rule used to trigger action
212  struct check_inciter : pegtl::success {};
213  //! \brief Do error checking on the inciter block
214  template<> struct action< check_inciter > {
215  template< typename Input, typename Stack >
216  static void apply( const Input& in, Stack& stack ) {
217  using inciter::deck::neq;
218  using inciter::g_inputdeck_defaults;
219  // Error out if no dt policy has been selected
220  const auto& dt = stack.template get< tag::discr, tag::dt >();
221  const auto& cfl = stack.template get< tag::discr, tag::cfl >();
222  if ( std::abs(dt - g_inputdeck_defaults.get< tag::discr, tag::dt >()) <
223  std::numeric_limits< tk::real >::epsilon() &&
224  std::abs(cfl - g_inputdeck_defaults.get< tag::discr, tag::cfl >()) <
225  std::numeric_limits< tk::real >::epsilon() )
226  Message< Stack, ERROR, MsgKey::NODT >( stack, in );
227  // If both dt and cfl are given, warn that dt wins over cfl
228  if ( std::abs(dt - g_inputdeck_defaults.get< tag::discr, tag::dt >()) >
229  std::numeric_limits< tk::real >::epsilon() &&
230  std::abs(cfl - g_inputdeck_defaults.get< tag::discr, tag::cfl >()) >
231  std::numeric_limits< tk::real >::epsilon() )
232  Message< Stack, WARNING, MsgKey::MULDT >( stack, in );
233  // if MatCG is configured, turn on reordering
234  if (stack.template get< tag::discr, tag::scheme >() ==
235  inciter::ctr::SchemeType::MatCG)
236  stack.template get< tag::discr, tag::reorder >() = true;
237  }

251  //! Rule used to trigger action
252  struct compute_refvar_idx : pegtl::success {};
253  //! Compute indices of refinement variables
254  //! \details This functor computes the indices in the unknown vector for all
255  //! refinement variables in the system of systems of dependent variables
256  //! after the refvar...end block has been parsed in the amr...end block.
257  //! After basic error checking, the vector at stack.get<tag::amr,tag::id>()
258  //! is filled.
259  template<>
260  struct action< compute_refvar_idx > {
261  template< typename Input, typename Stack >
262  static void apply( const Input& in, Stack& stack ) {
263  // reference variables just parsed by refvar...end block
264  const auto& refvar = stack.template get< tag::amr, tag::refvar >();
265  // get ncomponents object from this input deck
266  const auto& ncomps = stack.template get< tag::component >();
267  // compute offset map associating offsets to dependent variables
268  auto offsetmap = ncomps.offsetmap( stack );
269  // compute number of components associated to dependent variabels
270  auto ncompmap = ncomps.ncompmap( stack );
271  // reference variable index vector to fill
272  auto& refidx = stack.template get< tag::amr, tag::id >();
273  // Compute indices for all refvars
274  for (const auto& v : refvar) { // for all reference variables parsed
275  // depvar is the first char of a refvar
276  auto depvar = v[0];
277  // the field ID is optional and is the rest of the depvar string
278  std::size_t f = (v.size()>1 ? std::stoul(v.substr(1)) : 1) - 1;
279  // field ID must be less than or equal to the number of scalar
280  // components configured for the eq system for this dependent variable
281  if (f >= tk::cref_find( ncompmap, depvar ))
282  Message< Stack, ERROR, MsgKey::NOSUCHCOMPONENT >( stack, in );
283  // get offset for depvar
284  auto eqsys_offset = tk::cref_find( offsetmap, depvar );
285  // the index is the eq offset + field ID
286  auto idx = eqsys_offset + f;
287  // save refvar index in system of all systems
288  refidx.push_back( idx );
289  }
290  }
291  };

293  //! Rule used to trigger action
294  struct check_amr_errors : pegtl::success {};
295  //! Do error checking for the amr...end block
296  //! \details This is error checking that only the amr...end block
297  //! must satisfy. Besides error checking this can also set defaults
298  //! as this block is called when parsing of a amr...end block has
299  //! just finished.
300  template<>
301  struct action< check_amr_errors > {
302  template< typename Input, typename Stack >
303  static void apply( const Input& in, Stack& stack ) {
304  // Error out if refvar size does not equal refidx size
305  Assert( (stack.template get< tag::amr, tag::refvar >().size() ==
306  stack.template get< tag::amr, tag::id >().size()),
307  "The size of refvar and refidx vectors must equal" );
308  // Error out if initref list is not divisible by 2
309  const auto& edgeref = stack.template get< tag::amr, tag::edge >();
310  if (edgeref.size() % 2 == 1)
311  Message< Stack, ERROR, MsgKey::INITREFODD >( stack, in );
312  // Set initial AMR (on/off) if either the initial refinement type list or
313  // an edge list is configured
314  Assert( (stack.template get< tag::amr, tag::initamr >() == false),
315  "InitAMR switch must be false by default" );
316  const auto& initref = stack.template get< tag::amr, tag::init >();
317  if (!initref.empty() || !edgeref.empty())
318  stack.template get< tag::amr, tag::initamr >() = true;
319  }
320  };
