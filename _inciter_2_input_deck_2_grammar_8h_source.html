72  //! Rule used to trigger action
73  template< class eq > struct check_inciter_eq : pegtl::success {};
74  //! \brief Do general error checking on the differential equation block
75  //! \details This is error checking that all equation types must satisfy. For
76  //! more specific equations, such as compressible flow, a more specialized
77  //! equation checker does and can do better error checking.
78  template< class eq >
79  struct action< check_inciter_eq< eq > > {
80  template< typename Input, typename Stack >
81  static void apply( const Input& in, Stack& stack ) {
82  using inciter::deck::neq;
83  // Error out if no dependent variable has been selected
84  const auto& depvar = stack.template get< tag::param, eq, tag::depvar >();
85  if (depvar.empty() || depvar.size() != neq.get< eq >())
86  Message< Stack, ERROR, MsgKey::NODEPVAR >( stack, in );
87  // Error out if no number of components has been selected
88  const auto& ncomp = stack.template get< tag::component, eq >();
89  if (ncomp.empty() || ncomp.size() != neq.get< eq >())
90  Message< Stack, ERROR, MsgKey::NONCOMP >( stack, in );
91  // Error out if no test problem has been selected
92  const auto& problem = stack.template get< tag::param, eq, tag::problem >();
93  if (problem.empty() || problem.size() != neq.get< eq >())
94  Message< Stack, ERROR, MsgKey::NOINIT >( stack, in );
95  }
96  };

98  //! Rule used to trigger action
99  template< class eq > struct check_transport : pegtl::success {};
100  //! \brief Set defaults and do error checking on the transport equation block
101  //! \details This is error checking that only the transport equation block
102  //! must satisfy. Besides error checking we also set defaults here as
103  //! this block is called when parsing of a transport...end block has
104  //! just finished.
105  template< class eq >
106  struct action< check_transport< eq > > {
107  template< typename Input, typename Stack >
108  static void apply( const Input& in, Stack& stack ) {
109  using inciter::deck::neq;
110  // Error out if no dependent variable has been selected
111  auto& depvar = stack.template get< tag::param, eq, tag::depvar >();
112  if (depvar.empty() || depvar.size() != neq.get< eq >())
113  depvar.push_back( 'c' );
114  // If no number of components has been selected, default to 1
115  auto& ncomp = stack.template get< tag::component, eq >();
116  if (ncomp.empty() || ncomp.size() != neq.get< eq >())
117  ncomp.push_back( 1 );
118  // If physics type is not given, default to 'advection'
119  auto& physics = stack.template get< tag::param, eq, tag::physics >();
120  if (physics.empty() || physics.size() != neq.get< eq >())
121  physics.push_back( inciter::ctr::PhysicsType::ADVECTION );
122  // If physics type is advection-diffusion, check for correct number of
123  // advection velocity, shear, and diffusion coefficients
124  if (physics.back() == inciter::ctr::PhysicsType::ADVDIFF) {
125  auto& u0 = stack.template get< tag::param, eq, tag::u0 >();
126  if (u0.back().size() != ncomp.back()) // must define 1 component
127  Message< Stack, ERROR, MsgKey::WRONGSIZE >( stack, in );
128  auto& diff = stack.template get< tag::param, eq, tag::diffusivity >();
129  if (diff.back().size() != 3*ncomp.back()) // must define 3 components
130  Message< Stack, ERROR, MsgKey::WRONGSIZE >( stack, in );
131  auto& lambda = stack.template get< tag::param, eq, tag::lambda >();
132  if (lambda.back().size() != 2*ncomp.back()) // must define 2 shear comps
133  Message< Stack, ERROR, MsgKey::WRONGSIZE >( stack, in );
134  }
135  // If problem type is not given, error out
136  auto& problem = stack.template get< tag::param, eq, tag::problem >();
137  if (problem.empty() || problem.size() != neq.get< eq >())
138  Message< Stack, ERROR, MsgKey::NOPROBLEM >( stack, in );
139  // Error check Dirichlet boundary condition block for all transport eq
140  // configurations
141  for (const auto& s : stack.template get< tag::param, eq, tag::bcdir >())
142  if (s.empty())
143  Message< Stack, ERROR, MsgKey::BC_EMPTY >( stack, in );
144  }
145  };

147  //! Rule used to trigger action
148  template< class eq > struct check_compflow : pegtl::success {};
149  //! \brief Set defaults and do error checking on the compressible flow
150  //! equation block
151  //! \details This is error checking that only the compressible flow equation
152  //! block must satisfy. Besides error checking we also set defaults here as
153  //! this block is called when parsing of a compflow...end block has
154  //! just finished.
155  template< class eq >
156  struct action< check_compflow< eq > > {
157  template< typename Input, typename Stack >
158  static void apply( const Input& in, Stack& stack ) {
159  using inciter::deck::neq;
160  // Set number of components to 5 (mass, 3 x mom, energy)
161  stack.template get< tag::component, eq >().push_back( 5 );
162  // If physics type is not given, default to 'euler'
163  auto& physics = stack.template get< tag::param, eq, tag::physics >();
164  if (physics.empty() || physics.size() != neq.get< eq >())
165  physics.push_back( inciter::ctr::PhysicsType::EULER );
166  // If problem type is not given, default to 'user_defined'
167  auto& problem = stack.template get< tag::param, eq, tag::problem >();
168  if (problem.empty() || problem.size() != neq.get< eq >())
169  problem.push_back( inciter::ctr::ProblemType::USER_DEFINED );
170  else if (problem.back() == inciter::ctr::ProblemType::VORTICAL_FLOW) {
171  const auto& alpha = stack.template get< tag::param, eq, tag::alpha >();
172  const auto& beta = stack.template get< tag::param, eq, tag::beta >();
173  const auto& p0 = stack.template get< tag::param, eq, tag::p0 >();
174  if ( alpha.size() != problem.size() ||
175  beta.size() != problem.size() ||
176  p0.size() != problem.size() )
177  Message< Stack, ERROR, MsgKey::VORTICAL_UNFINISHED >( stack, in );
178  }
179  else if (problem.back() == inciter::ctr::ProblemType::NL_ENERGY_GROWTH) {
180  const auto& alpha = stack.template get< tag::param, eq, tag::alpha >();
181  const auto& betax = stack.template get< tag::param, eq, tag::betax >();
182  const auto& betay = stack.template get< tag::param, eq, tag::betay >();
183  const auto& betaz = stack.template get< tag::param, eq, tag::betaz >();
184  const auto& kappa = stack.template get< tag::param, eq, tag::kappa >();
185  const auto& r0 = stack.template get< tag::param, eq, tag::r0 >();
186  const auto& ce = stack.template get< tag::param, eq, tag::ce >();
187  if ( alpha.size() != problem.size() ||
188  betax.size() != problem.size() ||
189  betay.size() != problem.size() ||
190  betaz.size() != problem.size() ||
191  kappa.size() != problem.size() ||
192  r0.size() != problem.size() ||
193  ce.size() != problem.size() )
194  Message< Stack, ERROR, MsgKey::ENERGY_UNFINISHED >( stack, in);
195  }
196  else if (problem.back() == inciter::ctr::ProblemType::RAYLEIGH_TAYLOR) {
197  const auto& alpha = stack.template get< tag::param, eq, tag::alpha >();
198  const auto& betax = stack.template get< tag::param, eq, tag::betax >();
199  const auto& betay = stack.template get< tag::param, eq, tag::betay >();
200  const auto& betaz = stack.template get< tag::param, eq, tag::betaz >();
201  const auto& kappa = stack.template get< tag::param, eq, tag::kappa >();
202  const auto& p0 = stack.template get< tag::param, eq, tag::p0 >();
203  const auto& r0 = stack.template get< tag::param, eq, tag::r0 >();
204  if ( alpha.size() != problem.size() ||
205  betax.size() != problem.size() ||
206  betay.size() != problem.size() ||
207  betaz.size() != problem.size() ||
208  kappa.size() != problem.size() ||
209  p0.size() != problem.size() ||
210  r0.size() != problem.size() )
211  Message< Stack, ERROR, MsgKey::RT_UNFINISHED >( stack, in);
212  }
213  // Error check Dirichlet boundary condition block for all compflow
214  // configurations
215  for (const auto& s : stack.template get< tag::param, eq, tag::bcdir >())
216  if (s.empty())
217  Message< Stack, ERROR, MsgKey::BC_EMPTY >( stack, in );
218  }
219  };

237  //! Rule used to trigger action
238  struct check_dt : pegtl::success {};
239  //! \brief Do error checking on setting the time step calculation policy
240  template<> struct action< check_dt > {
241  template< typename Input, typename Stack >
242  static void apply( const Input& in, Stack& stack ) {
243  using inciter::deck::neq;
244  using inciter::g_inputdeck_defaults;
245  // Error out if no dt policy has been selected
246  const auto& dt = stack.template get< tag::discr, tag::dt >();
247  const auto& cfl = stack.template get< tag::discr, tag::cfl >();
248  if ( std::abs(dt - g_inputdeck_defaults.get< tag::discr, tag::dt >()) <
249  std::numeric_limits< tk::real >::epsilon() &&
250  std::abs(cfl - g_inputdeck_defaults.get< tag::discr, tag::cfl >()) <
251  std::numeric_limits< tk::real >::epsilon() )
252  Message< Stack, ERROR, MsgKey::NODT >( stack, in );
253  // If both dt and cfl are given, warn that dt wins over cfl
254  if ( std::abs(dt - g_inputdeck_defaults.get< tag::discr, tag::dt >()) >
255  std::numeric_limits< tk::real >::epsilon() &&
256  std::abs(cfl - g_inputdeck_defaults.get< tag::discr, tag::cfl >()) >
257  std::numeric_limits< tk::real >::epsilon() )
258  Message< Stack, WARNING, MsgKey::MULDT >( stack, in );
259  }
