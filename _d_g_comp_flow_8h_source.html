48  //! Factory for Riemann solvers
49  //! \details This factory is used to store the constructors as a
50  //! std::function of specific Riemann solvers that can be invoked at a
51  //! later time compared to the point where the map is populated. The key
52  //! is an enum, uniquely idenfitying a specific Riemann solver. The value
53  //! is std::function storing a constructor to be invoked. The type of
54  //! object stored in std::function is a generic (base) class constructor,
55  //! which provides a polymorphyic interface (overridable functions) that
56  //! specific (child) Riemann solvers override.
57  using RiemannFactory =
58  std::map< ctr::FluxType, std::function< RiemannSolver() > >;

60  //! Register a Riemann solver into the Riemann solver factory
61  struct registerRiemannSolver {
62  //! Factory to which to register the Riemann solver
63  RiemannFactory& factory;
64  //! Constructor
65  //! \param[in] f Factory
66  registerRiemannSolver( RiemannFactory& f ) : factory( f ) {}
67  //! \brief Function call operator templated on the type that implements
68  //! a specific Riemann solver
69  template< typename U > void operator()( U ) {
70  // Function object holding the (default) constructor to be called later
71  // without bound arguments, since all specific Riemann solvers'
72  // constructors are compiler-generated (default) constructors, and thus
73  // taking no arguments.
74  std::function< U() > c = boost::value_factory< U >();
75  // Associate constructor function object to flux type in factory
76  factory.emplace( U::type(),
77  boost::bind(boost::value_factory< RiemannSolver >(), std::move(c)) );
78  }
79  };

122  //! \param[in] geoElem Element geometry array

147  //! Compute the left hand side block-diagonal mass matrix
148  //! \param[in] geoElem Element geometry array
149  //! \param[in,out] l Block diagonal mass matrix
150  void lhs( const tk::Fields& geoElem, tk::Fields& l ) const
151  {
152  Assert( geoElem.nunk() == l.nunk(), "Size mismatch" );
153  std::size_t nelem = geoElem.nunk();
154 
155  for (std::size_t e=0; e<nelem; ++e)
156  {
157  for (ncomp_t c=0; c<5; ++c)
158  l(e, c, m_offset) = geoElem(e,0,0);
159  }
160  }
161 
162  //! Compute right hand side
163  //! \param[in] t Physical time
164  //! \param[in] geoElem Element geometry array
165  //! \param[in] geoFace Face geometry array
166  //! \param[in] fd Face connectivity data object
167  //! \param[in] U Solution vector at recent time step
168  //! \param[in,out] R Right-hand side vector computed
169  void rhs( tk::real t,
170  const tk::Fields& geoFace,
171  const tk::Fields& geoElem,
172  const inciter::FaceData& fd,
173  const tk::Fields& U,
174  tk::Fields& R ) const
175  {
176  Assert( U.nunk() == R.nunk(), "Number of unknowns in solution "
177  "vector and right-hand side at recent time step incorrect" );
178  Assert( U.nunk() == geoElem.nunk(), "Number of unknowns in solution "
179  "vector and element-geometry at recent time step incorrect" );
180  Assert( U.nprop() == 5 && R.nprop() == 5,
181  "Number of components in solution and right-hand side vector "
182  "must equal "+ std::to_string(5) );
183 
184  // set rhs to zero
185  R.fill(0.0);
186 
187  const auto& esuf = fd.Esuf();
188 
189  // compute internal surface flux integrals
190  for (auto f=fd.Nbfac(); f<esuf.size()/2; ++f)
191  {
192  std::size_t el = static_cast< std::size_t >(esuf[2*f]);
193  std::size_t er = static_cast< std::size_t >(esuf[2*f+1]);
194  auto farea = geoFace(f,0,0);
195 
196  auto flux =
197  m_riemann.flux( f, geoFace, {{U.extract(el), U.extract(er)}} );
198 
199  for (ncomp_t c=0; c<5; ++c) {
200  R(el, c, m_offset) -= farea * flux[c];
201  R(er, c, m_offset) += farea * flux[c];
202  }
203  }
204 
205  // compute boundary surface flux integrals
206  bndIntegral< Dir >( m_bcdir, fd, geoFace, t, U, R );
207  bndIntegral< Sym >( m_bcsym, fd, geoFace, t, U, R );
208  bndIntegral< Extrapolate >( m_bcextrapolate, fd, geoFace, t, U, R );
209 
210  // Add source term to right hand side
211  for (std::size_t e=0; e<geoElem.nunk(); ++e) {
212  auto vole = geoElem(e,0,0);
213  auto xc = geoElem(e,1,0);
214  auto yc = geoElem(e,2,0);
215  auto zc = geoElem(e,3,0);
216  auto s = Problem::src(0, xc, yc, zc, t);
217  for (ncomp_t c=0; c<5; ++c)
218  R(e, c, m_offset) += vole * s[c];
219  }
220  }

236  //! \param[in] U Solution vector at recent time step
237  //! \param[in] N Element node indices

239  std::array< std::array< tk::real, 4 >, 3 >

251  std::transform( r.begin(), r.end(), v[1].begin(), v[1].begin(),

254  []( tk::real s, tk::real& d ){ return d /= s; } );

308  //! \brief State policy class providing the left and right state of a face
309  //! at Dirichlet boundaries
310  struct Dir {
311  static std::array< std::vector< tk::real >, 2 >

313  tk::real xc, tk::real yc, tk::real zc,

325  //! \brief State policy class providing the left and right state of a face
326  //! at symmetric boundaries
327  struct Sym {
328  static std::array< std::vector< tk::real >, 2 >
329  LR( const tk::Fields& U, std::size_t e,
330  tk::real /*xc*/, tk::real /*yc*/, tk::real /*zc*/,
331  std::array< tk::real, 3 > fn,
332  tk::real /*t*/ ) {
333  auto ul = U.extract( e );
334  auto ur = ul;
335  // Internal cell velocity components
336  auto v1l = ul[1]/ul[0];
337  auto v2l = ul[2]/ul[0];
338  auto v3l = ul[3]/ul[0];
339  // Normal component of velocity
340  auto vnl = v1l*fn[0] + v2l*fn[1] + v3l*fn[2];
341  // Ghost state velocity components
342  auto v1r = v1l - 2.0*vnl*fn[0];
343  auto v2r = v2l - 2.0*vnl*fn[1];
344  auto v3r = v3l - 2.0*vnl*fn[2];
345  // Boundary condition
346  ur[0] = ul[0];
347  ur[1] = ur[0] * v1r;
348  ur[2] = ur[0] * v2r;
349  ur[3] = ur[0] * v3r;
350  ur[4] = ul[4];
351  return {{ std::move(ul), std::move(ur) }};
352  }
353  };

367  //! Compute boundary surface integral for a number of faces
368  //! \param[in] faces Face IDs at which to compute surface integral
369  //! \param[in] esuf Elements surrounding face, see tk::genEsuf()
370  //! \param[in] geoFace Face geometry array
371  //! \param[in] t Physical time
372  //! \param[in] U Solution vector at recent time step
373  //! \param[in,out] R Right-hand side vector computed
374  //! \tparam State Policy class providing the left and right state at
375  //! boundaries by its member function State::LR()
376  template< class State >
377  void surfInt( const std::vector< std::size_t >& faces,
378  const std::vector< int >& esuf,
379  const tk::Fields& geoFace,
380  tk::real t,
381  const tk::Fields& U,
382  tk::Fields& R ) const
383  {
384  for (const auto& f : faces) {
385  std::size_t el = static_cast< std::size_t >(esuf[2*f]);
386  Assert( esuf[2*f+1] == -1, "outside boundary element not -1" );
387  auto farea = geoFace(f,0,0);
388  auto xc = geoFace(f,4,0);
389  auto yc = geoFace(f,5,0);
390  auto zc = geoFace(f,6,0);
391  std::array< tk::real, 3 > fn {{ geoFace(f,1,0),
392  geoFace(f,2,0),
393  geoFace(f,3,0) }};
394 
395  //--- fluxes
396  auto flux = m_riemann.flux( f, geoFace, State::LR(U,el,xc,yc,zc,fn,t) );
397 
398  for (ncomp_t c=0; c<5; ++c)
399  R(el, c, m_offset) -= farea * flux[c];
400  }
401  }

403  //! Compute boundary surface flux integrals for a given boundary type
404  //! \tparam BCType Specifies the type of boundary condition to apply
405  //! \param bcconfig BC configuration vector for multiple side sets
406  //! \param[in] fd Face connectivity data object
407  //! \param[in] geoFace Face geometry array

414  const inciter::FaceData& fd,
