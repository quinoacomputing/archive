111  //! Extract the velocity field at cell nodes
112  //! \param[in] U Solution vector at recent time step
113  //! \param[in] N Element node indices
114  //! \return Array of the four values of the velocity field
115  std::array< std::array< tk::real, 4 >, 3 >
116  velocity( const tk::Fields& U,
117  const std::array< std::vector< tk::real >, 3 >&,
118  const std::array< std::size_t, 4 >& N ) const
119  {
120  std::array< std::array< tk::real, 4 >, 3 > v;
121  v[0] = U.extract( 1, m_offset, N );
122  v[1] = U.extract( 2, m_offset, N );
123  v[2] = U.extract( 3, m_offset, N );
124  auto r = U.extract( 0, m_offset, N );
125  std::transform( r.begin(), r.end(), v[0].begin(), v[0].begin(),
126  []( tk::real s, tk::real& d ){ return d /= s; } );
127  std::transform( r.begin(), r.end(), v[1].begin(), v[1].begin(),
128  []( tk::real s, tk::real& d ){ return d /= s; } );
129  std::transform( r.begin(), r.end(), v[2].begin(), v[2].begin(),
130  []( tk::real s, tk::real& d ){ return d /= s; } );
131  return v;
132  }

149  //! increment between t+dt and t, since that is what the solution requires

159  // cg::CompFlow::dirbc() for an example search for all nodes of a side
160  // set given in sides (key=setid, value=nodeids). Instead of coord, we
161  // probably want to pass in a const-ref to DG::m_geoFace and work with the

165  std::unordered_map< std::size_t, FaceBC > bc;
