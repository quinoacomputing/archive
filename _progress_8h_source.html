23 //! Simple progress class for outputing progress indicators during a task
24 //! \details This is a helper class to abstract away the details of using
25 //! tk::Print::progress() used to output progress reports to the screen during
26 //! a task consisting of multiple sub-tasks happening at the same time. The
27 //! template argument is a compile-time integer which is the number of
28 //! independent sub-tasks the progress indicator receives messages for and
29 //! counts them independtly toward multiple maxima.
30 template< std::size_t N >
31 class Progress {
32 
33  public:
34  //! Constructor
35  //! \param[in] print Pretty printer object to use for printing progress
36  //! \param[in] feedback Whether to send sub-task feedback to host
37  //! \param[in] prefix Strings to output prefixing the progress report
38  //! \param[in] max Array of integers equaling the max number of items to be
39  //! expected per sub-task
40  //! \details Note that prefix and max are swallowed, i.e., moved in,
41  //! treating them as rvalue references.
42  explicit Progress( const tk::Print& print,
43  bool feedback,
44  std::array< std::string, N >&& prefix,
45  std::array< int, N >&& max = std::array< int, N >() )
46  : m_print( print ),
47  m_feedback( feedback ),
48  m_prefix( std::move(prefix) ),
49  m_finished( false ),
50  m_progress_size( 0 ),
51  m_max( std::move(max) )
52  {
53  m_done.fill( 0 );
54  }
55 
56  //! Start counting sub-tasks outputing an intial task message
57  //! \param[in] msg Message to output to screen. This message should be
58  //! descriptive of all the sub-tasks we are responsible for. I.e., this
59  //! is usually a list of multiple sub-tasks happening at the same time.
60  void start( const std::string& msg ) {
61  m_print.diagstart( msg );
62  m_progress_size = 0;
63  }
64 
65  //! \brief Start counting sub-tasks outputing an intial task message and set
66  //! max number of items to be expected per sub-task
67  //! \param[in] msg Message to output to screen. This message should be
68  //! descriptive of all the sub-tasks we are responsible for. I.e., this
69  //! is usually a list of multiple sub-tasks happening at the same time.
70  //! \param[in] max Array of integers equaling the max number of items to be
71  //! expected per sub-task
72  //! \details This function can be used to do the same as start( msg ) and
73  //! update/reset the max number of items per sub-task in case they are not
74  //! all yet available when the constructor is called.
75  void start( const std::string& msg, std::array< int, N >&& max ) {
76  m_max = std::move(max);
77  start( msg );
78  }
79 
80  //! Receive an update to a sub-task counter and update progress report
81  //! \details The template argument indexes the sub-task. A compile-time
82  //! assert emits an error in case it is out of bounds.
83  template< std::size_t i > void inc() {
84  static_assert( i < N, "Indexing out of bounds" );
85  ++m_done[i];
86  if (!m_finished) report();
87  }
88 
89  //! Finish progress report updating it one last time
90  //! \details When this function is called, all sub-tasks are assumed to be
91  //! finished, i.e., assumed to have reached their respective maximum
92  //! values. Thus we update our 'done' array to be equal to 'max' and output
93  //! the progress report one final time before outputing 'done'. When this
94  //! function is called it is possible that not all sub-task counters have
95  //! reached their maximum, which can happen if the final reduction (if
96  //! exists), signaling the absolute end of a task (consisting of multiple
97  //! sub-tasks we count counters for), is scheduled before (or is faster)
98  //! than as the individual sub-task counting messages arrive. Even if that
99  //! is the case, this call "officially" finishes all sub-tasks, and outputs
100  //! the progress report using the max values for all sub-tasks to leave a
101  //! consistent screen output finishing the task.
102  void end() {
103  m_finished = true;
104  m_done = m_max;
105  report();
106  m_print.diagend( "done" );
107  }
108 
109  private:
110  const tk::Print& m_print; //!< Pretty printer to use for screen output
111  bool m_feedback; //!< Whether to send sub-task feedback to host
112  const std::array< std::string, N > m_prefix; //!< Sub-task prefixes
113  bool m_finished; //!< Whether task has finished
114  std::size_t m_progress_size;//!< Size of previous progress report
115  std::array< int, N > m_max; //!< Max number of items per sub-task
116  std::array< int, N > m_done;//!< Number of items done per sub-task
117 
118  //! Output progress report to screen
119  //! \details This output contains a status on each of the multiple sub-task
120  //! counters as they all work towards their respective maxima.
121  //! \see tk::Print::progress()
122  void report() {
123  if (m_feedback)
124  m_print.progress< N >( m_prefix, m_done, m_max, m_progress_size );
125  }
126 };
