48 //! \details C0 is user-defined and we pull in a hydrodynamic timescale from an

50 //! \see kw::hydrotimescale_info

59  }

65  //! \brief Update the dissipation rate (eps)
66  //! \details Update the dissipation rate (eps) based on eps/k (from DNS) and
67  //! the turbulent kinetic energy (k) (from the SDE)
68  void update( char depvar,
69  const std::map< tk::ctr::Product, tk::real >& moments,
70  const tk::Table& hts,
71  kw::sde_c0::info::expect::type C0,
72  tk::real t,
73  tk::real& eps,
74  std::array< tk::real, 9 >& G ) const
75  {
76  using tk::ctr::lookup;
77  using tk::ctr::mean;
78 
79  const tk::ctr::Term u1( static_cast<char>(std::toupper(depvar)),
80  0, tk::ctr::Moment::CENTRAL );
81  const tk::ctr::Term u2( static_cast<char>(std::toupper(depvar)),
82  1, tk::ctr::Moment::CENTRAL );
83  const tk::ctr::Term u3( static_cast<char>(std::toupper(depvar)),
84  2, tk::ctr::Moment::CENTRAL );
85 
86  // Extract diagonal of the Reynolds stress
87  const auto R11 = tk::ctr::Product( {u1,u1} );
88  const auto R22 = tk::ctr::Product( {u2,u2} );
89  const auto R33 = tk::ctr::Product( {u3,u3} );
90  // compute turbulent kinetic energy
91  tk::real k = ( lookup(R11,moments) +
92  lookup(R22,moments) +
93  lookup(R33,moments) ) / 2.0;
94 
95  // Sample hydrodynamics timescale and prod/diss at time t
96  auto ts = hydrotimescale( t, hts ); // eps/k
97 
98  // compute turbulent kinetic energy dissipation rate
99  eps = ts * k;
100 
101  // update drift tensor based on the simplified Langevin model
102  G.fill( 0.0 );
103  G[0] = G[4] = G[8] = -(1.0/2.0 + 3.0/4.0*C0) * ts;
104  }
105 
106  //! Sample the inverse hydrodynamics time scale at time t
107  //! \param[in] t Time at which to sample inverse hydrodynamics time scale
108  //! \param[in] ts Hydro time scale table to sample
109  //! \return Sampled value from discrete table of inverse hydro time scale
110  tk::real hydrotimescale( tk::real t, const tk::Table& ts ) const
111  { return tk::sample( t, ts ); }
112 };
