38 // Constructor: register all differential equations into factory
39 //! \details This constructor consists of several blocks, each registering a
40 //! potentially large number of entries in the differential equation factory,
41 //! m_factory, which is of type walker::DiffEqFactory, a std::map. At this
42 //! time, each type of differential equation can be configured to use a unique
43 //! _initialization policy_ and a unique _coefficients policy_. (More types
44 //! of policies will most likely come in the future.) The policy classes are
45 //! template arguments to the differential equation classes and influence
46 //! their behavior in a different way, abstracting away certain functions,
47 //! e.g., how to set initial conditions and how to update their coefficients
48 //! during time integration. For more information on policy-based design, see
49 //! http://en.wikipedia.org/wiki/Policy-based_design. This abstraction allows
50 //! [separation of concerns]
51 //! (http://en.wikipedia.org/wiki/Separation_of_concerns).
52 //!
53 //! Since the functionality of the policies are orthogonal to each other,
54 //! i.e., they do not depend on each other or their host (the differential
55 //! equation class), a Cartesian product of combinations are possible,
56 //! depending on which policies are selected. _This constructor registers all
57 //! possible combinations of policies for all available differential
58 //! equations._ By _register_, we mean, an entry is recorded in an associative
59 //! container, a std::map, that associates a lightweight key of type
60 //! walker::ctr::DiffEqKey, consisting of only an enum for each policy type,
61 //! to an std::function object that holds the constructor bound to its
62 //! arguments corresponding to a particular differential equation + policies
63 //! combination. Note that registering these entries in the map does not
64 //! invoke the constructors. The mapped value simply stores how the
65 //! constructors should be invoked at a later time. At some point later,
66 //! based on user input, we then instantiate only the differential equations
67 //! (and only those configurations) that are requested by the user.
68 //!
69 //! Since all differential equation types (registered in the factory)
70 //! "inherit" from a common "base", client-code is unform and generic, and
71 //! thus immune to changes in the inner workings of the particular
72 //! differential equations as long as they fullfill certain concepts, i.e.,
73 //! implement certain member functinos, enforced by the _common base_, DiffEq.
74 //! The words "inherit and "base" are quoted here, because the common base
75 //! does not use inheritance in the normal OOP sense and does not use
76 //! reference semantics, i.e., pointers, visible to client-code either. The
77 //! relationship is more of a _models a_-type, which simplifies client-code
78 //! and allows for the benfits of runtime inheritance with value-semantics
79 //! which is less error prone and easier to read. See more about the
80 //! _models-a_ relationship and its implementation in DiffEq/DiffEq.h.
81 //!
82 //! The design discussed above allows the registration, instantiation, and
83 //! use of the differential equations to be generic, which eliminates a lot of
84 //! boiler-plate code and makes client-code uniform.
85 //!
86 //! _Details of registration using brigand::for_each and
87 //! tk::cartesian_product:_
88 //!
89 //! The template argument to brigand::for_each, as used below, requires a
90 //! list of list of types. We use brigand::list of brigand::list of types,
91 //! listing all possible policies, where the inner list must have exactly two
92 //! types, as the list of lists is constructed from two lists using the
93 //! cartesian product, and the length of the outer list (the list of lists) is
94 //! arbitrary. The constructor argument to brigand::for_each is a functor that
95 //! is to be applied to all members of the outer list. tk::cartesian_product
96 //! will create all possible combinations of these types and call the functor
97 //! with each type of the created sequence as a template parameter. The
98 //! functor here inherits from registerDiffEq, which, i.e., its constructor
99 //! call, needs a single template argument, a class templated on policy
100 //! classes. This is the differential equation class to be configured by
101 //! selecting policies and to be registered. The arguments to
102 //! registerDiffEq's constructor are the factory, the enum denoting the
103 //! differential equation type, and a reference to a variable of type
104 //! std::set< ctr::DiffEqType >, which is only used internally to DiffEqStack
105 //! for counting up the number of unique differential equation types
106 //! registered, used for diagnostics purposes.
107 // *****************************************************************************
108 {
109  registerDirichlet( m_factory, m_eqTypes );
110  registerGenDir( m_factory, m_eqTypes );
111  registerWrightFisher( m_factory, m_eqTypes );
112  registerOrnsteinUhlenbeck( m_factory, m_eqTypes );
113  registerDiagOrnsteinUhlenbeck( m_factory, m_eqTypes );
114  registerBeta( m_factory, m_eqTypes );
115  registerNumberFractionBeta( m_factory, m_eqTypes );
116  registerMassFractionBeta( m_factory, m_eqTypes );
117  registerMixNumberFractionBeta( m_factory, m_eqTypes );
118  registerMixMassFractionBeta( m_factory, m_eqTypes );
119  registerGamma( m_factory, m_eqTypes );
120  registerSkewNormal( m_factory, m_eqTypes );
121  registerVelocity( m_factory, m_eqTypes );
122  registerPosition( m_factory, m_eqTypes );
123  registerDissipation( m_factory, m_eqTypes );
124 }

126 std::vector< walker::DiffEq >
127 DiffEqStack::selected() const
128 // *****************************************************************************
129 // Instantiate all selected differential equations
130 //! \return std::vector of instantiated differential equation objects
131 // *****************************************************************************
132 {
133  std::map< ctr::DiffEqType, ncomp_t > cnt; // count DiffEqs per type
134  std::vector< DiffEq > diffeqs; // will store instantiated DiffEqs
135 
136  for (const auto& d : g_inputdeck.get< tag::selected, tag::diffeq >()) {
137  if (d == ctr::DiffEqType::DIRICHLET)
138  diffeqs.push_back( createDiffEq< tag::dirichlet >( d, cnt ) );
139  else if (d == ctr::DiffEqType::GENDIR)
140  diffeqs.push_back( createDiffEq< tag::gendir >( d, cnt ) );
141  else if (d == ctr::DiffEqType::WRIGHTFISHER)
142  diffeqs.push_back( createDiffEq< tag::wrightfisher >( d, cnt ) );
143  else if (d == ctr::DiffEqType::OU)
144  diffeqs.push_back( createDiffEq< tag::ou >( d, cnt ) );
145  else if (d == ctr::DiffEqType::DIAG_OU)
146  diffeqs.push_back( createDiffEq< tag::diagou >( d, cnt ) );
147  else if (d == ctr::DiffEqType::BETA)
148  diffeqs.push_back( createDiffEq< tag::beta >( d, cnt ) );
149  else if (d == ctr::DiffEqType::NUMFRACBETA)
150  diffeqs.push_back( createDiffEq< tag::numfracbeta >( d, cnt ) );
151  else if (d == ctr::DiffEqType::MASSFRACBETA)
152  diffeqs.push_back( createDiffEq< tag::massfracbeta >( d, cnt ) );
153  else if (d == ctr::DiffEqType::MIXNUMFRACBETA)
154  diffeqs.push_back( createDiffEq< tag::mixnumfracbeta >( d, cnt ) );
155  else if (d == ctr::DiffEqType::MIXMASSFRACBETA)
156  diffeqs.push_back( createDiffEq< tag::mixmassfracbeta >( d, cnt ) );
157  else if (d == ctr::DiffEqType::SKEWNORMAL)
158  diffeqs.push_back( createDiffEq< tag::skewnormal >( d, cnt ) );
159  else if (d == ctr::DiffEqType::GAMMA)
160  diffeqs.push_back( createDiffEq< tag::gamma >( d, cnt ) );
161  else if (d == ctr::DiffEqType::VELOCITY)
162  diffeqs.push_back( createDiffEq< tag::velocity >( d, cnt ) );
163  else if (d == ctr::DiffEqType::POSITION)
164  diffeqs.push_back( createDiffEq< tag::position >( d, cnt ) );
165  else if (d == ctr::DiffEqType::DISSIPATION)
166  diffeqs.push_back( createDiffEq< tag::dissipation >( d, cnt ) );
167  else Throw( "Can't find selected DiffEq" );
168  }
169 
170  return diffeqs;
171 }

173 std::pair< std::vector< std::string >, std::vector< tk::Table > >
174 DiffEqStack::tables() const
175 // *****************************************************************************
176 // Instantiate tables from which extra statistics data to be output sampled for
177 // all selected differential equations
178 //! \return Vector of names and tables to sample from during time stepping
179 // *****************************************************************************
180 {
181  std::map< ctr::DiffEqType, ncomp_t > cnt; // count DiffEqs per type
182  std::vector< std::string > nam; // names of instantiated tables
183  std::vector< tk::Table > tab; // instantiated tables
184 
185  for (const auto& d : g_inputdeck.get< tag::selected, tag::diffeq >()) {
186  std::pair< std::vector< std::string >, std::vector< tk::Table > > t;
187 
188  if (d == ctr::DiffEqType::MIXMASSFRACBETA)
189  t = createTables< tag::mixmassfracbeta >( d, cnt );
190  else if (d == ctr::DiffEqType::VELOCITY)
191  t = createTables< tag::velocity >( d, cnt );
192 
193  nam.insert( end(nam), begin(t.first), end(t.first) );
194  tab.insert( end(tab), begin(t.second), end(t.second) );
195  }
196 
197  return { nam, tab };
198 }
