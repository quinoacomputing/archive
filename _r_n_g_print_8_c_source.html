98 RNGPrint::MKLParams( const std::vector< ctr::RNGType >& vec,
99  const ctr::RNGMKLParameters& map ) const
100 // *****************************************************************************
101 // Print all fields of MKL RNG parameters
102 //! \param[in] vec Vector of RNG type enums to print
103 //! \param[in] map MKL RNG parameters map
104 // *****************************************************************************
105 {
106  ctr::RNG rng;
107 
108  for (auto& r : vec) {
109  if (rng.lib(r) == ctr::RNGLibType::MKL) {
110  subsection( rng.name(r) );
111  const auto& m = map.find(r);
112  if (m == map.end()) { // no parameter map entry, print defaults
113  echoMKLParams( ctr::RNGMKLParam() );
114  } else {
115  echoMKLParams( m->second );
116  }
117  }
118  }
119 }
