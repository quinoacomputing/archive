75  Assert( v.size() == 5, "Incorrect BC vector size" );
76  for (std::size_t c=0; c<v.size(); ++c)
77  if (v[c].first) unk(i,c,offset) = v[c].second;

137  //! Return field output going to file
138  //! \param[in] e Equation system index, i.e., which compressible
139  //! flow equation system we operate on among the systems of PDEs

145  output( tk::ctr::ncomp_type e,
146  tk::ctr::ncomp_type offset,
147  tk::real,
148  const std::array< std::vector< tk::real >, 3 >&,
149  const tk::Fields& U )
150  {
151  IGNORE(e);
152  std::vector< std::vector< tk::real > > out;
153  const auto r = U.extract( 0, offset );
154  const auto ru = U.extract( 1, offset );
155  const auto rv = U.extract( 2, offset );
156  const auto rw = U.extract( 3, offset );
157  const auto re = U.extract( 4, offset );
158  out.push_back( r );

160  std::transform( r.begin(), r.end(), u.begin(), u.begin(),

162  out.push_back( u );

164  std::transform( r.begin(), r.end(), v.begin(), v.begin(),
165  []( tk::real s, tk::real& d ){ return d /= s; } );
166  out.push_back( v );
167  std::vector< tk::real > w = rw;
168  std::transform( r.begin(), r.end(), w.begin(), w.begin(),
169  []( tk::real s, tk::real& d ){ return d /= s; } );
170  out.push_back( w );
171  std::vector< tk::real > E = re;
172  std::transform( r.begin(), r.end(), E.begin(), E.begin(),
173  []( tk::real s, tk::real& d ){ return d /= s; } );
174  out.push_back( E );
175  std::vector< tk::real > p = r;
176  tk::real g = g_inputdeck.get< tag::param, tag::compflow, tag::gamma >()[0];
177  for (std::size_t i=0; i<p.size(); ++i)
178  p[i] = (g-1.0)*r[i]*(E[i] - (u[i]*u[i] + v[i]*v[i] + w[i]*w[i])/2.0);
179  out.push_back( p );
180  std::vector< tk::real > T = r;
181  tk::real cv = g_inputdeck.get< tag::param, tag::compflow, tag::cv >()[0];
182  for (std::size_t i=0; i<T.size(); ++i)
183  T[i] = cv*(E[i] - (u[i]*u[i] + v[i]*v[i] + w[i]*w[i])/2.0);
184  out.push_back( p );
185  return out;
186  }

192 //! CompFlow system of PDEs problem: vortical flow
193 //! \see Waltz, et. al, "Manufactured solutions for the three-dimensional Euler
194 //! equations with relevance to Inertial Confinement Fusion", Journal of
195 //! Computational Physics 267 (2014) 196-209.
196 class CompFlowProblemVorticalFlow {
197  public:
198 
199  //! Set initial conditions
200  //! \param[in] coord Mesh node coordinates
201  //! \param[in,out] unk Array of unknowns
202  //! \param[in] e Equation system index, i.e., which compressible
203  //! flow equation system we operate on among the systems of PDEs
204  //! \param[in] offset System offset specifying the position of the system of
205  //! PDEs among other systems
206  static void init( const std::array< std::vector< tk::real >, 3 >& coord,
207  const std::vector< std::size_t >&,
208  const std::unordered_map< std::size_t,
209  std::vector< std::pair< bool, tk::real > > >&,
210  tk::Fields& unk,
211  tk::ctr::ncomp_type e,
212  tk::ctr::ncomp_type offset )
213  {
214  // manufactured solution parameters
215  const auto& a =
216  g_inputdeck.get< tag::param, tag::compflow, tag::alpha >()[e];
217  const auto& b =
218  g_inputdeck.get< tag::param, tag::compflow, tag::beta >()[e];
219  const auto& p0 =
220  g_inputdeck.get< tag::param, tag::compflow, tag::p0 >()[e];
221  // ratio of specific heats
222  tk::real g =
223  g_inputdeck.get< tag::param, tag::compflow, tag::gamma >()[e];
224  // set initial and boundary conditions
225  const auto& x = coord[0];
226  const auto& y = coord[1];
227  const auto& z = coord[2];
228  for (ncomp_t i=0; i<x.size(); ++i) {
229  auto& r = unk(i,0,offset); // rho
230  auto& ru = unk(i,1,offset); // rho * u
231  auto& rv = unk(i,2,offset); // rho * v
232  auto& rw = unk(i,3,offset); // rho * w
233  auto& re = unk(i,4,offset); // rho * e
234  r = 1.0;
235  ru = a*x[i] - b*y[i];
236  rv = b*x[i] + a*y[i];
237  rw = -2.0*a*z[i];
238  re = (ru*ru + rv*rv + rw*rw)/2.0 + (p0-2.0*a*a*z[i]*z[i])/(g-1.0);
239  }
240  }

242  //! Add source term to rhs for vortical flow manufactured solution
243  //! \param[in] coord Mesh node coordinates
244  //! \param[in] e Equation system index, i.e., which compressible
245  //! flow equation system we operate on among the systems of PDEs
246  //! \param[in] mult Multiplier differentiating the different stages in
247  //! multi-stage time stepping
248  //! \param[in] dt Size of time step
249  //! \param[in] J Element Jacobi determinant
250  //! \param[in] N Element node indices
251  //! \param[in] mass Element mass matrix, nnode*nnode [4][4]
252  //! \param[in] grad Shape function derivatives, nnode*ndim [4][3]
253  //! \param[in] r Pointers to right hand side at component and offset
254  //! \param[in,out] u Solution at element nodes at recent time step stage
255  //! \param[in,out] R Right-hand side vector contributing to
256  static void
257  sourceRhs( const std::array< std::vector< tk::real >, 3 >& coord,
258  tk::ctr::ncomp_type e,
259  tk::real mult,
260  tk::real dt,
261  tk::real J,
262  const std::array< std::size_t, 4 >& N,
263  const std::array< std::array< tk::real, 4 >, 4 >& mass,
264  const std::array< std::array< tk::real, 3 >, 4 >& grad,
265  const std::array< const tk::real*, 5 >& r,
266  std::array< std::array< tk::real, 4 >, 5 >& u,
267  tk::Fields& R )
268  {
269  // manufactured solution parameters
270  const auto& a =
271  g_inputdeck.get< tag::param, tag::compflow, tag::alpha >()[e];
272  const auto& b =
273  g_inputdeck.get< tag::param, tag::compflow, tag::beta >()[e];
274  // ratio of specific heats
275  tk::real g =
276  g_inputdeck.get< tag::param, tag::compflow, tag::gamma >()[e];
277 
278  // mesh node coordinates
279  const auto& x = coord[0];
280  const auto& y = coord[1];
281  const auto& z = coord[2];
282 
283  std::array< tk::real, 4 > ru{{ a*x[N[0]] - b*y[N[0]],
284  a*x[N[1]] - b*y[N[1]],
285  a*x[N[2]] - b*y[N[2]],
286  a*x[N[3]] - b*y[N[3]] }};
287  std::array< tk::real, 4 > rv{{ b*x[N[0]] + a*y[N[0]],
288  b*x[N[1]] + a*y[N[1]],
289  b*x[N[2]] + a*y[N[2]],
290  b*x[N[3]] + a*y[N[3]] }};
291 
292  // compute momentum source
293  std::array< std::array< tk::real, 4 >, 3 >
294  Sm{{ {{ a*ru[0] - b*rv[0],
295  a*ru[1] - b*rv[1],
296  a*ru[2] - b*rv[2],
297  a*ru[3] - b*rv[3] }},
298  {{ b*ru[0] + a*rv[0],
299  b*ru[1] + a*rv[1],
300  b*ru[2] + a*rv[2],
301  b*ru[3] + a*rv[3] }} }};
302 
303  // compute energy source
304  std::array< tk::real, 4 > Se{{
305  Sm[0][0]*ru[0] + Sm[1][0]*rv[0] + 8.0*a*a*a*z[N[0]]*z[N[0]]/(g-1.0),
306  Sm[0][1]*ru[1] + Sm[1][1]*rv[1] + 8.0*a*a*a*z[N[1]]*z[N[1]]/(g-1.0),
307  Sm[0][2]*ru[2] + Sm[1][2]*rv[2] + 8.0*a*a*a*z[N[2]]*z[N[2]]/(g-1.0),
308  Sm[0][3]*ru[3] + Sm[1][3]*rv[3] + 8.0*a*a*a*z[N[3]]*z[N[3]]/(g-1.0) }};
309 
310  // add momentum and energy source at element nodes
311  tk::real c = mult * dt;
312  for (std::size_t j=0; j<4; ++j)
313  for (std::size_t k=0; k<4; ++k) {
314  // source contribution to mass rhs
315  for (std::size_t i=0; i<3; ++i)
316  R.var(r[0],N[j]) += c * J/24.0 * grad[k][i] * u[i+1][k];
317  // source contribution to momentum rhs
318  for (std::size_t l=0; l<2; ++l)
319  R.var(r[l+1],N[j]) += c * mass[j][k] * Sm[l][k];
320  // source contribution to enerhy rhs
321  R.var(r[4],N[j]) += c * mass[j][k] * Se[k];
322  }
323  }

434  E[i] = 0.5*(u[i]*u[i] + v[i]*v[i] + w[i]*w[i]) +

436  out.push_back( E );
