22 //! \brief Keyword information bundle
23 //! \details This bundle contains the information that is used to display
24 //! on-screen help on all command-line arguments and control file keywords
25 //! for an exectuable. This struct is stored in a container that associates
26 //! keywords (used by a grammar and parser) to this struct. The container, an
27 //! runtime, std::map, is filled by the CmdLine and InputDeck objects'
28 //! constructors by one or more boost::mpl::for_each which loops through the
29 //! set of all keywords used in a grammar. The maps are stored in the CmdLine
30 //! and InputDeck objects (which are tagged tuples) and thus can be migrated
31 //! through the network, thus the Charm++ parck/unpack routines are defined.
32 //! \see Info functor used to fill the std::maps
33 struct KeywordInfo {
34  std::string shortDescription; //!< Short description
35  std::string longDescription; //!< Long description
36  boost::optional< std::string > alias; //!< Keyword alias
37  boost::optional< std::string > expt; //!< Expected type description
38  boost::optional< std::string > choices; //!< Expected choices descr.
39 
40  /** @name Pack/Unpack: Serialize KeywordInfo object for Charm++ */
41  ///@{
42  //! \brief Pack/Unpack serialize member function
43  //! \param[in,out] p Charm++'s PUP::er serializer object reference
44  void pup( PUP::er& p ) {
45  p | shortDescription;
46  p | longDescription;
47  p | alias;
48  p | expt;
49  p | choices;
50  }
51  //! \brief Pack/Unpack serialize operator|
52  //! \param[in,out] p Charm++'s PUP::er serializer object reference
53  //! \param[in,out] info KeywordInfo object reference
54  friend void operator|( PUP::er& p, KeywordInfo& info ) { info.pup(p); }
55  ///@}
56 };
