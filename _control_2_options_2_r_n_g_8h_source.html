212  //! \brief Return parameter based on Enum
213  //! \details Here 'parameter' is the library-specific identifier of the
214  //! option, i.e., as the library identifies the given option
215  //! \param[in] rng Enum value of the option requested
216  //! \return Library-specific parameter of the option
217  const ParamType& param( RNGType rng ) const {
218  using tk::operator<<;
219  auto it = brng.find( rng );
220  Assert( it != end(brng),
221  std::string("Cannot find parameter for RNG \"") << rng << "\"" );
222  return it->second;
223  }

225  //! \brief Return field from RNG parameters bundle
226  //! \details If user has specified it, return field. If user did not
227  //! specify it, return default.
228  //! \param[in] rng Enum value of the option
229  //! \param[in] def Default value of requested field if field is not found
230  //! \param[in] bundle Parameter bundle to search in
231  //! \return Field for requested enum if found, default if not found
232  template< class tag, class Param, class Field >
233  Field param( RNGType rng, const Field& def, const Param& bundle ) const {
234  auto it = bundle.find( rng );
235  if ( it != bundle.end() ) return it->second.template get< tag >();
236  else return def;
237  }

239  //! \brief Return RNG library type based on RNG options enum
240  //! \param[in] rng Enum value of the option
241  //! \return Library type enum
242  //! \see tk::ctr::RNGLibType
243  RNGLibType lib( RNGType rng ) const {
244  const auto& n = name( rng );
245  if ( found( "MKL", n ) ) return RNGLibType::MKL;
246  else if ( found( "RNGSSE", n ) ) return RNGLibType::RNGSSE;
247  else if ( found( "PRAND", n ) ) return RNGLibType::PRAND;
248  else if ( found( "Random123", n ) ) return RNGLibType::R123;
249  else return RNGLibType::NO_LIB;
250  }

258  //! \brief Return whether RNG supports sequence option given

262  template< class OptionType >
263  bool supportsOpt( RNGType rng, const OptionType& option ) const {
264  auto it = support.find( rng );
265  if ( it != end( support ) ) {
266  for (auto& o : it->second)
