34  //! Key type

37  //! Pair type

40  // Hash functor for key_type
41  struct key_hash {
42  std::size_t operator()( const key_type& key ) const {
43  return std::hash< long >()( key[0] ) ^
44  std::hash< long >()( key[1] ) ^
45  std::hash< long >()( key[2] );
46  }
47  };

49  //! \brief Joint trivariate PDF
50  //! \details The underlying container type is an unordered_map where the key
51  //! is three bin ids corresponding to the three sample space dimensions,
52  //! and the mapped value is the sample counter. The hasher functor,
53  //! defined by key_hash provides an XORed hash of the three bin ids.
54  using map_type = std::unordered_map< key_type, tk::real, key_hash >;
55 
56  //! Empty constructor for Charm++
57  explicit TriPDF() : m_binsize( {{ 0, 0, 0 }} ), m_nsample( 0 ), m_pdf() {}
58 
59  //! Constructor: Initialize joint trivariate PDF container
60  //! \param[in] bs Sample space bin size in all three directions
61  explicit TriPDF( const std::vector< tk::real >& bs ) :
62  m_binsize( {{ bs[0], bs[1], bs[2] }} ),
63  m_nsample( 0 ),
64  m_pdf() {}

66  //! Accessor to number of samples
67  //! \return Number of samples collected
68  std::size_t nsample() const noexcept { return m_nsample; }
69 
70  //! Add sample to trivariate PDF
71  //! \param[in] sample Sample to add
72  void add( std::array< tk::real, dim > sample ) {
73  ++m_nsample;
74  ++m_pdf[ {{ std::lround( sample[0] / m_binsize[0] ),
75  std::lround( sample[1] / m_binsize[1] ),
76  std::lround( sample[2] / m_binsize[2] ) }} ];
77  }
78 
79  //! Add multiple samples from a PDF
80  //! \param[in] p PDF whose samples to add
81  void addPDF( const TriPDF& p ) {
82  m_binsize = p.binsize();
83  m_nsample += p.nsample();
84  for (const auto& e : p.map()) m_pdf[ e.first ] += e.second;
85  }
86 
87  //! Zero bins

99  //! \brief Return minimum and maximum bin ids of sample space in all three
100  //! dimensions
101  //! \return {xmin,xmax,ymin,ymax,zmin,zmax} Minima and maxima of bin the ids
102  std::array< long, 2*dim > extents() const {
103  auto x = std::minmax_element( begin(m_pdf), end(m_pdf),
104  []( const pair_type& a, const pair_type& b )
105  { return a.first[0] < b.first[0]; } );
106  auto y = std::minmax_element( begin(m_pdf), end(m_pdf),
107  []( const pair_type& a, const pair_type& b )
108  { return a.first[1] < b.first[1]; } );
109  auto z = std::minmax_element( begin(m_pdf), end(m_pdf),
110  []( const pair_type& a, const pair_type& b )
111  { return a.first[2] < b.first[2]; } );
112  return {{ x.first->first[0], x.second->first[0],
113  y.first->first[1], y.second->first[1],
114  z.first->first[2], z.second->first[2] }};
115  }
