5  \brief Velocity equation coefficients policies
6  \details This file defines coefficients policy classes for the velocity
7  equation for the fluctuating velocity in variable-density turbulence,
8  defined in DiffEq/Velocity.h.
9 
10  General requirements on velocity equation coefficients policy classes:
11 
12  - Must define a _constructor_, which is used to initialize the SDE
13  coefficient, C0. Required signature:
14  \code{.cpp}
15  CoeffPolicyName(
16  kw::sde_c0::info::expect::type C0_,
17  kw::sde_c0::info::expect::type& C0,
18  std::array< tk::real, 9 >& dU )
19  \endcode
20  where
21  - C0_ denotes a real value used to initialize the velocity system.
22  - The reference C0 is to be initialized based on C0_.
23  - _dU_ is an optionally prescribed mean velocity gradient.
24 
25  - Must define the function _update()_, called from Velocity::advance(),
26  updating the model coefficients.
27  Required signature:
28  \code{.cpp}
29  void update( char depvar,
30  char dissipation_depvar,
31  const std::map< tk::ctr::Product, tk::real >& moments,
32  const tk::Table& hts,
33  ctr::DepvarType solve,
34  ctr::VelocityVariantType variant,
35  kw::sde_c0::info::expect::type C0,
36  tk::real t,
37  tk::real& eps,
38  std::array< tk::real, 9 >& G ) const
39  \endcode
40  where _depvar_ is the dependent variable of the velocity equation,
41  _dissipation_depvar_ is the dependent variable of the coupled dissipation
42  equation, _moments_ if the map of computed statistical moments, _hts_ is
43  a ctr::Table containing the inverse hydrodynamic timescale, _variant_ is
44  the velocity model variant, _solve_ is the the lable of the dependent
45  variable to solve for (full variable or fluctuation), _C0_ is the Langevin
46  eq constat to use, _t_ is the physical time, _eps_ is the dissipation rate
47  of turbulent kinetic energy to update, and _G_ is the G_{ij} tensor in the
48  Langevin model to update.
49 
50  - Must define the static function _type()_, returning the enum value of the
51  policy option. Example:
52  \code{.cpp}
53  static ctr::CoeffPolicyType type() noexcept {
54  return ctr::CoeffPolicyType::HYDROTIMESCALE;
55  }
56  \endcode
57  which returns the enum value of the option from the underlying option
58  class, collecting all possible options for coefficients policies.
59 */

74 //! Calculate the 2nd order tensor Gij based on the simplified Langevin model
75 //! \param[in] hts Inverse hydrodynamics time scale, e.g., eps/k
76 //! \param[in] C0 Coefficient C0 in SLM
77 //! \return Tensor Gij computed based on the simplified Langevin model
78 static inline std::array< tk::real, 9 >
79 slm( tk::real hts, tk::real C0 ) {
80 
81  std::array< tk::real, 9 > G;
82  G.fill( 0.0 );
83  G[0] = G[4] = G[8] = -(0.5+0.75*C0) * hts;
84 
85  return G;
86 }

88 //! Calculate the 2nd order tensor Gij based on the generalized Langevin model
89 //! \param[in] hts Inverse hydrodynamics time scale, e.g., eps/k
90 //! \param[in] C0 Coefficient C0 in SLM
91 //! \param[in] rs Reynolds stress
92 //! \param[in] dU Mean velocity gradient
93 //! \return Tensor Gij computed based on the simplified Langevin model
94 static inline std::array< tk::real, 9 >
95 glm( tk::real hts,
96  tk::real C0,
97  const std::array< tk::real, 6 >& rs,
98  const std::array< tk::real, 9 >& dU )
99 {
100  // Generalized Langevion model coefficients
101  std::array< tk::real, 2 > ALPHA{{ -(0.5 + 0.75*C0), 3.7 }};
102  std::array< tk::real, 3 > BETA{{ -0.2, 0.8, -0.2 }};
103  std::array< tk::real, 6 > GAMMA{{ -1.28, 3.01, -2.18, 0.0, 4.29, -3.09 }};
104 
105  // Compute Reynolds stress anisotropy
106  tk::real tr = rs[0] + rs[1] + rs[2];
107  std::array< tk::real, 9 > b{{ rs[0]/tr-1.0/3.0,
108  rs[3]/tr,
109  rs[4]/tr,
110  rs[3]/tr,
111  rs[1]/tr-1.0/3.0,
112  rs[5]/tr,
113  rs[4]/tr,
114  rs[5]/tr,
115  rs[2]/tr-1.0/3.0 }};
116  // Compute Gij
117  std::array< tk::real, 9 > G;
118  G.fill( 0.0 );
119  for (std::size_t i=0; i<3; ++i ) {
120  // to main diagonal: hts * ALPHA1 * delta_ij + BETA1 * deltaij * d<Ul>/dxl
121  G[i*3+i] += hts*ALPHA[0] + BETA[0]*(dU[0] + dU[4] + dU[8]);
122  // to main diagonal: GAMMA1 * deltaij * bml * d<Um>/dxl
123  tk::real dtmp = 0.0;
124  for (std::size_t m=0; m<3; ++m)
125  for (std::size_t l=0; l<3; ++l )
126  dtmp += b[m*3+l]*dU[m*3+l];
127  G[i*3+i] += GAMMA[0]*dtmp;
128  // to main and off-diagonal
129  for (std::size_t j=0; j<3; ++j) {
130  G[i*3+j] += hts*ALPHA[1]*b[i*3+j] + // eps/k * ALPHA2 * bij
131  BETA[1]*dU[i*3+j] + // BETA2 * d<Ui>/dj
132  BETA[2]*dU[j*3+i] + // BETA3 * d<Uj>/di
133  // GAMMA4 * bij * d<Ul>/dxl
134  GAMMA[3]*b[i*3+j]*(dU[0] + dU[4] + dU[8]);
135  for (std::size_t l=0; l<3; ++l)
136  G[i*3+j] += GAMMA[1]*b[j*3+l]*dU[i*3+l] + // GAMMA2 * bjl * d<Ui>/dxl
137  GAMMA[2]*b[j*3+l]*dU[l*3+i] + // GAMMA3 * bjl * d<Ul>/dxi
138  GAMMA[4]*b[i*3+l]*dU[l*3+j] + // GAMMA5 * bil * d<Ul>/dxj
139  GAMMA[5]*b[i*3+l]*dU[j*3+l]; // GAMMA6 * bil * d<Uj>/dxl
140  }
141  }
142 
143  return G;
144 }

146 //! Velocity equation coefficients policy with prescribed mean shear
147 //! \details C0 is user-defined and we prescibe a hard-coded mean shear in the x
148 //! direction
149 //! \see kw::hydrotimescale_info
150 class Velocity_ConstShear {
151 
152  public:
153  //! Constructor: initialize coefficients
154  //! \param[in] C0_ Value of C0 parameter in the Langevin model
155  //! \param[in,out] C0 Value of to set the C0 parameter in the Langevin model
156  //! \param[in,out] dU Prescribed mean velocity gradient1
157  Velocity_ConstShear( kw::sde_c0::info::expect::type C0_,
158  kw::sde_c0::info::expect::type& C0,
159  std::array< tk::real, 9 >& dU ) :
160  m_dU( {{ 0.0, 1.0, 0.0,
161  0.0, 0.0, 0.0,
162  0.0, 0.0, 0.0 }} )
163  {
164  C0 = C0_;
165  dU = m_dU;
166  }
167 
168  //! Coefficients policy type accessor
169  static ctr::CoeffPolicyType type() noexcept
170  { return ctr::CoeffPolicyType::CONST_SHEAR; }
171 
172  //! Update the model coefficients (prescribing shear)
173  //! \details Update the dissipation rate (eps) and G_{ij} based on the
174  //! turbulent kinetic energy (k) for a prescribed honmogeneous shear flow.
175  void update( char depvar,
176  char dissipation_depvar,
177  const std::map< tk::ctr::Product, tk::real >& moments,
178  const tk::Table&,
179  ctr::DepvarType solve,
180  ctr::VelocityVariantType variant,
181  kw::sde_c0::info::expect::type C0,
182  tk::real,
183  tk::real& eps,
184  std::array< tk::real, 9 >& G ) const
185  {
186  using tk::ctr::lookup;
187  using tk::ctr::mean;
188  using tk::ctr::Product;
189 
190  // Extract diagonal of the Reynolds stress
191  Product r11, r22, r33, r12, r13, r23;
192  if (solve == ctr::DepvarType::FULLVAR) {
193 
194  using tk::ctr::variance;
195  using tk::ctr::covariance;
196  r11 = variance( depvar, 0 );
197  r22 = variance( depvar, 1 );
198  r33 = variance( depvar, 2 );
199  r12 = covariance( depvar, 0, depvar, 1 );
200  r13 = covariance( depvar, 0, depvar, 2 );
201  r23 = covariance( depvar, 1, depvar, 2 );
202 
203  } else if (solve == ctr::DepvarType::FLUCTUATION) {
204 
205  // Since we are solving for the fluctuating velocity, the "ordinary"
206  // moments, e.g., <U1U1>, are really central moments, i.e., <u1u1>.
207  using tk::ctr::Term;
208  using tk::ctr::Moment;
209  auto d = static_cast< char >( std::toupper( depvar ) );
210  Term u( d, 0, Moment::ORDINARY );
211  Term v( d, 1, Moment::ORDINARY );
212  Term w( d, 2, Moment::ORDINARY );
213  r11 = tk::ctr::Product( { u, u } );
214  r22 = tk::ctr::Product( { v, v } );
215  r33 = tk::ctr::Product( { w, w } );
216  r12 = tk::ctr::Product( { u, v } );
217  r13 = tk::ctr::Product( { u, w } );
218  r23 = tk::ctr::Product( { v, w } );
219 
220  } else Throw( "Depvar type not implemented" );
221 
222  // Compute nonzero components of the Reynolds stress
223  std::array< tk::real, 6 > rs{{ lookup(r11,moments),
224  lookup(r22,moments),
225  lookup(r33,moments),
226  lookup(r12,moments),
227  lookup(r13,moments),
228  lookup(r23,moments) }};
229 
230  // Compute turbulent kinetic energy
231  tk::real k = (rs[0] + rs[1] + rs[2]) / 2.0;
232 
233  // Access mean turbulence frequency
234  tk::real O = lookup( mean(dissipation_depvar,0), moments );
235 
236  // compute turbulent kinetic energy dissipation rate
237  eps = O*k;
238 
239  // update drift tensor based on the Langevin model variant configured
240  if (variant == ctr::VelocityVariantType::SLM) // simplified
241  G = slm( O, C0 );
242  else if (variant == ctr::VelocityVariantType::GLM)// generalized
243  G = glm( O, C0, rs, m_dU );
244  else Throw( "Velocity variant type not implemented" );
245  }
246 
247  private:
248  //! Mean velocity gradient prescribed for simpled 1D homogeneous shear
249  std::array< tk::real, 9 > m_dU;
250 };

252 //! Velocity equation coefficients policy with DNS hydrodynamics time scale
253 //! \details C0 is user-defined and we pull in a hydrodynamic timescale from an
254 //! external function (from DNS).
255 //! \see kw::hydrotimescale_info
256 class Velocity_HydroTimeScale {
257 
258  public:
259  //! Constructor: initialize coefficients
260  //! \param[in] C0_ Value of C0 parameter in the Langevin model
261  //! \param[in,out] C0 Value of to set the C0 parameter in the Langevin model
262  Velocity_HydroTimeScale( kw::sde_c0::info::expect::type C0_,
263  kw::sde_c0::info::expect::type& C0,
264  std::array< tk::real, 9 >& )
265  {
266  C0 = C0_;
267  }
268 
269  //! Coefficients policy type accessor
270  static ctr::CoeffPolicyType type() noexcept
271  { return ctr::CoeffPolicyType::HYDROTIMESCALE; }
272 
273  //! \brief Update the model coefficients
274  //! \details Update the dissipation rate (eps) based on eps/k (from DNS) and
275  //! the turbulent kinetic energy (k) (from the SDE)
276  void update( char depvar,
277  char,
278  const std::map< tk::ctr::Product, tk::real >& moments,
279  const tk::Table& hts,
280  ctr::DepvarType,
281  ctr::VelocityVariantType,
282  kw::sde_c0::info::expect::type C0,
283  tk::real t,
284  tk::real& eps,
285  std::array< tk::real, 9 >& G ) const
286  {
287  using tk::ctr::lookup;
288  using tk::ctr::variance;
289 
290  // Extract diagonal of the Reynolds stress
291  const auto R11 = variance( depvar, 0 );
292  const auto R22 = variance( depvar, 1 );
293  const auto R33 = variance( depvar, 2 );
294  // compute turbulent kinetic energy
295  tk::real k = ( lookup(R11,moments) +
296  lookup(R22,moments) +
297  lookup(R33,moments) ) / 2.0;
298 
299  // Sample hydrodynamics timescale and prod/diss at time t
300  auto ts = hydrotimescale( t, hts ); // eps/k
301 
302  // compute turbulent kinetic energy dissipation rate
303  eps = ts * k;
304 
305  // update drift tensor based on the simplified Langevin model
306  G.fill( 0.0 );
307  G[0] = G[4] = G[8] = -(0.5+0.75*C0) * ts;
308  }
309 
310  //! Sample the inverse hydrodynamics time scale at time t
311  //! \param[in] t Time at which to sample inverse hydrodynamics time scale
312  //! \param[in] ts Hydro time scale table to sample
313  //! \return Sampled value from discrete table of inverse hydro time scale
314  tk::real hydrotimescale( tk::real t, const tk::Table& ts ) const
315  { return tk::sample( t, ts ); }
316 };
