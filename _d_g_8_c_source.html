83  // Output chare mesh to file
84  d->writeMesh();
85  // Output fields metadata to output file
86  d->writeElemMeta();
87 
88  // zero initial solution vector
89  // ...
90 
91  std::size_t nnpe(4);
92  std::size_t nelem = d->Inpoel().size()/nnpe;
93 
94  // Set initial conditions for all PDEs
95  for (std::size_t e=0; e<nelem; ++e)
96  {
97  auto xcc = m_geoElem(e,1,0);
98  auto ycc = m_geoElem(e,2,0);
99  auto zcc = m_geoElem(e,3,0);
100  IGNORE(zcc);
101 
102  tk::real u = 1.0 * exp( -((xcc-0.25)*(xcc-0.25)
103  + (ycc-0.25)*(ycc-0.25))/(2.0 * 0.005) );
104  m_u(e,0,0) = u; // scalar
105  }
106 
107  // Set initial conditions for all PDEs
108  std::cout << "DG::init: " << g_dgpde.size() << '\n';
109  for (const auto& eq : g_dgpde) eq.initialize( d->Coord(), m_u, d->T() );
110 
111  // Output initial conditions to file (regardless of whether it was requested)
112  if ( !g_inputdeck.get< tag::cmd, tag::benchmark >() ) writeFields( d->T() );
113 
114  // Start time stepping
115  dt();
116 }

118 void
119 DG::dt()
120 // *****************************************************************************
121 // Comppute time step size
122 // *****************************************************************************
123 {
124  tk::real mindt = std::numeric_limits< tk::real >::max();
125 
126  auto const_dt = g_inputdeck.get< tag::discr, tag::dt >();
127  auto def_const_dt = g_inputdeck_defaults.get< tag::discr, tag::dt >();
128  auto eps = std::numeric_limits< tk::real >::epsilon();
129 
130  // use constant dt if configured
131  if (std::abs(const_dt - def_const_dt) > eps) {
132 
133  mindt = const_dt;
134 
135  } else { // compute dt based on CFL
136 
137  // find the minimum dt across all PDEs integrated
138  // ...
139  mindt = 0.1; // stub for now to overwrite numeric_limits::max
140 
141  // Scale smallest dt with CFL coefficient
142  mindt *= g_inputdeck.get< tag::discr, tag::cfl >();
143 
144  }
145 
146  // Contribute to minimum dt across all chares the advance to next step
147  contribute( sizeof(tk::real), &mindt, CkReduction::min_double,
148  CkCallback(CkReductionTarget(DG,advance), thisProxy) );
149 }

204 DG::out()
205 // *****************************************************************************
206 // Output mesh field data
207 // *****************************************************************************
208 {
209  auto d = Disc();
210 
211  // Optionally output field and particle data
212  if ( !((d->It()+1) % g_inputdeck.get< tag::interval, tag::field >()) &&
213  !g_inputdeck.get< tag::cmd, tag::benchmark >() )
214  {
215  writeFields( d->T() + d->Dt() );
216  }
217 
218  // Output final field data to file (regardless of whether it was requested)
219  const auto term = g_inputdeck.get< tag::discr, tag::term >();
220  const auto eps = std::numeric_limits< tk::real >::epsilon();
221  const auto nstep = g_inputdeck.get< tag::discr, tag::nstep >();
222  if ( (std::fabs(d->T() + d->Dt() - term) < eps || (d->It()+1) >= nstep) &&
223  (!g_inputdeck.get< tag::cmd, tag::benchmark >()) )
224  {
225  writeFields( d->T()+d->Dt() );

277 void
278 DG::eval()
279 // *****************************************************************************
280 // Evaluate whether to continue with next step
281 // *****************************************************************************
282 {
283  auto d = Disc();
284 
285  const auto term = g_inputdeck.get< tag::discr, tag::term >();
286  const auto nstep = g_inputdeck.get< tag::discr, tag::nstep >();
287  const auto eps = std::numeric_limits< tk::real >::epsilon();
288 
289  // If neither max iterations nor max time reached, continue, otherwise finish
290  if (std::fabs(d->T()-term) > eps && d->It() < nstep)
291  dt();
292  else
293  contribute( CkCallback( CkReductionTarget(Transporter,finish), d->Tr() ) );
294 }
