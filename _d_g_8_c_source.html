77 DG::setup( tk::real v )
78 // *****************************************************************************
79 // Setup rows, query boundary conditions, output mesh, etc.
80 //! \param[in] v Total mesh volume
81 // *****************************************************************************
82 {
83  auto d = Disc();
84 
85  // Store total mesh volume
86  m_vol = v;
87  // Output chare mesh to file
88  d->writeMesh();
89  // Output fields metadata to output file
90  d->writeElemMeta();
91 
92  // Compute left-hand side of discrete PDEs
93  lhs();
94 
95  // zero initial solution vector
96  // ...
97 
98  // Set initial conditions for all PDEs
99  m_ax = 0.1;
100  m_ay = 0.1;
101  m_az = 0.0;
102 
103  std::size_t nelem = m_u.nunk();
104 
105  for (std::size_t e=0; e<nelem; ++e)
106  {
107  auto xcc = m_geoElem(e,1,0);
108  auto ycc = m_geoElem(e,2,0);
109  auto zcc = m_geoElem(e,3,0);
110  IGNORE(zcc);
111 
112  tk::real u = 1.0 * exp( -((xcc-0.25)*(xcc-0.25)
113  + (ycc-0.25)*(ycc-0.25))/(2.0 * 0.005) );
114  m_u(e,0,0) = u; // scalar
115  m_un(e,0,0) = u; // scalar
116  }
117 
118  // Set initial conditions for all PDEs
119  for (const auto& eq : g_dgpde) eq.initialize( d->Coord(), m_u, d->T() );
120 
121  // Output initial conditions to file (regardless of whether it was requested)
122  if ( !g_inputdeck.get< tag::cmd, tag::benchmark >() ) writeFields( d->T() );
123 
124  // Start time stepping
125  dt();
126 }

129 DG::dt()
130 // *****************************************************************************
131 // Compute time step size
132 // *****************************************************************************
133 {
134  tk::real mindt = std::numeric_limits< tk::real >::max();
135 
136  auto const_dt = g_inputdeck.get< tag::discr, tag::dt >();
137  auto def_const_dt = g_inputdeck_defaults.get< tag::discr, tag::dt >();
138  auto eps = std::numeric_limits< tk::real >::epsilon();
139 
140  // use constant dt if configured
141  if (std::abs(const_dt - def_const_dt) > eps) {
142 
143  mindt = const_dt;
144 
145  } else { // compute dt based on CFL
146 
147  // find the minimum dt across all PDEs integrated
148  // ...
149  mindt = 0.1; // stub for now to overwrite numeric_limits::max
150 
151  // Scale smallest dt with CFL coefficient
152  mindt *= g_inputdeck.get< tag::discr, tag::cfl >();
153 
154  }
155 
156  // Contribute to minimum dt across all chares the advance to next step
157  contribute( sizeof(tk::real), &mindt, CkReduction::min_double,
158  CkCallback(CkReductionTarget(DG,advance), thisProxy) );

253 void
254 DG::rhs()
255 // *****************************************************************************
256 // Compute right-hand side of discrete transport equations
257 // *****************************************************************************
258 {
259  auto& esuf = m_fd.Esuf();
260  auto& bface = m_fd.Bface();
261 
262  // set rhs to zero
263  m_rhs.fill(0.0);
264 
265  // compute internal surface flux integrals
266  for (auto f=m_fd.Nbfac(); f<m_fd.Ntfac(); ++f)
267  {
268  std::size_t el = static_cast< std::size_t >(esuf[2*f]);
269  std::size_t er = static_cast< std::size_t >(esuf[2*f+1]);
270 
271  auto farea = m_geoFace(f,0,0);
272 
273  std::array< tk::real, 3 > fn {{ m_geoFace(f,1,0),
274  m_geoFace(f,2,0),
275  m_geoFace(f,3,0) }};
276 
277  // need to use tk::Fields::extract() here somehow
278  std::vector< tk::real > ul { m_u(el,0,0) };
279  std::vector< tk::real > ur { m_u(er,0,0) };
280 
281  //--- upwind fluxes
282  auto flux = upwindFlux(ul, ur, fn);
283 
284  m_rhs(el,0,0) -= farea * flux[0];
285  m_rhs(er,0,0) += farea * flux[0];
286  }
287 
288  // compute boundary surface flux integrals
289 
290  // symmetry boundary condition
291  auto bc = bface.find(1);
292 
293  if (bc != bface.end())
294  {
295  for (const auto& f : bc->second)
296  {
297  std::size_t el = static_cast< std::size_t >(esuf[2*f]);
298 
299  Assert( esuf[2*f+1] == -1,
300  "outside boundary element not -1" );
301 
302  auto farea = m_geoFace(f,0,0);
303 
304  std::array< tk::real, 3 > fn {{ m_geoFace(f,1,0),
305  m_geoFace(f,2,0),
306  m_geoFace(f,3,0) }};
307 
308  // need to use tk::Fields::extract() here?
309  std::vector< tk::real > ul { m_u(el,0,0) };
310  std::vector< tk::real > ur { ul[0] };
311 
312  //--- upwind fluxes
313  auto flux = upwindFlux(ul, ur, fn);
314 
315  m_rhs(el,0,0) -= farea * flux[0];
316  }
317  }
318 
319  // inlet boundary condition
320  bc = bface.find(2);
321 
322  if (bc != bface.end())
323  {
324  for (const auto& f : bc->second)
325  {
326  std::size_t el = static_cast< std::size_t >(esuf[2*f]);
327 
328  Assert( esuf[2*f+1] == -1,
329  "outside boundary element not -1" );
330 
331  auto farea = m_geoFace(f,0,0);
332 
333  std::array< tk::real, 3 > fn {{ m_geoFace(f,1,0),
334  m_geoFace(f,2,0),
335  m_geoFace(f,3,0) }};
336 
337  // need to use tk::Fields::extract() here?
338  std::vector< tk::real > ul { m_u(el,0,0) };
339  std::vector< tk::real > ur { 0.0 };
340 
341  //--- upwind fluxes
342  auto flux = upwindFlux(ul, ur, fn);
343 
344  m_rhs(el,0,0) -= farea * flux[0];
345  }
346  }
347 
348  // outlet boundary condition
349  bc = bface.find(3);
350 
351  if (bc != bface.end())
352  {
353  for (const auto& f : bc->second)
354  {
355  std::size_t el = static_cast< std::size_t >(esuf[2*f]);
356 
357  Assert( esuf[2*f+1] == -1,
358  "outside boundary element not -1" );
359 
360  auto farea = m_geoFace(f,0,0);
361 
362  std::array< tk::real, 3 > fn {{ m_geoFace(f,1,0),
363  m_geoFace(f,2,0),
364  m_geoFace(f,3,0) }};
365 
366  // need to use tk::Fields::extract() here?
367  std::vector< tk::real > ul { m_u(el,0,0) };
368  std::vector< tk::real > ur { ul[0] };
369 
370  //--- upwind fluxes
371  auto flux = upwindFlux(ul, ur, fn);
372 
373  m_rhs(el,0,0) -= farea * flux[0];
374  }
375  }
376 }

378 std::vector< tk::real >
379 DG::upwindFlux( std::vector< tk::real > ul,
380  std::vector< tk::real > ur,
381  std::array< tk::real, 3 > fn )
382 // *****************************************************************************
383 // Riemann solver using upwind method
384 //! \param[in] ul Left unknown/state vector
385 //! \param[in] ur Right unknown/state vector
386 //! \param[in] fn Face unit normal vector
387 //! \return Riemann solution using upwind method
388 // *****************************************************************************
389 {
390  std::vector< tk::real > flux(ul.size(),0);
391 
392  // wave speed
393  tk::real swave = m_ax*fn[0] + m_ay*fn[1] + m_az*fn[2];
394 
395  // upwinding
396  tk::real splus = 0.5 * (swave + fabs(swave));
397  tk::real sminus = 0.5 * (swave - fabs(swave));
398 
399  flux[0] = splus * ul[0] + sminus * ur[0];
400 
401  return flux;
402 }

404 void
405 DG::solve( tk::real deltat )
406 // *****************************************************************************
407 // Explicit time-stepping using forward Euler to discretize time-derivative
408 // *****************************************************************************
409 {
410  std::size_t nelem = m_u.nunk();
411 
412  for (std::size_t e=0; e<nelem; ++e)
413  {
414  m_u(e,0,0) = m_un(e,0,0) + deltat/m_lhs[e] * m_rhs(e,0,0);
415  }
416 
417  m_un = m_u;
418 }

463 void
464 DG::eval()
465 // *****************************************************************************
466 // Evaluate whether to continue with next step
467 // *****************************************************************************
468 {
469  auto d = Disc();
470 
471  const auto term = g_inputdeck.get< tag::discr, tag::term >();
472  const auto nstep = g_inputdeck.get< tag::discr, tag::nstep >();
473  const auto eps = std::numeric_limits< tk::real >::epsilon();
474 
475  // If neither max iterations nor max time reached, continue, otherwise finish
476  if (std::fabs(d->T()-term) > eps && d->It() < nstep)
