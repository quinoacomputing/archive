69 // Constructor
70 //! \param[in] disc Discretization proxy
71 //! \param[in] Face data structures
72 // *****************************************************************************
73 {
74  // Perform leak test on mesh partition
75  Assert( !leakyPartition(), "Mesh partition leaky" );
76 
77  // Activate SDAG waits for face adjacency map (ghost data) calculation
78  wait4ghost();
79 
80  auto d = Disc();
81 
82  const auto& gid = d->Gid();
83  const auto& inpoel = d->Inpoel();
84  const auto& inpofa = fd.Inpofa();
85 
86  // Invert inpofa to enable searching for faces based on (global) node triplets
87  Assert( inpofa.size() % 3 == 0, "Inpofa must contain triplets" );
88  for (std::size_t f=0; f<inpofa.size()/3; ++f)
89  m_ipface.insert( {{{ gid[ inpofa[f*3+0] ],
90  gid[ inpofa[f*3+1] ],
91  gid[ inpofa[f*3+2] ] }}} );
92 
93  // At this point ipface has node-id-triplets (faces) on the internal
94  // chare-domain and on the physical boundary but not on chare boundaries,
95  // hence the name internal + physical boundary faces.
96 
97  // Build a set of faces (each face given by 3 global node IDs) associated to
98  // chares we potentially share boundary faces with.
99  tk::UnsMesh::FaceSet potbndface;
100  for (std::size_t e=0; e<m_esuelTet.size()/4; ++e) { // for all our tets
101  auto mark = e*4;
102  for (std::size_t f=0; f<4; ++f) // for all tet faces
103  if (m_esuelTet[mark+f] == -1) { // if face has no outside-neighbor tet
104  // if does not exist in among the internal and physical boundary faces,
105  // store as a potential chare-boundary face
106  tk::UnsMesh::Face t{{ gid[ inpoel[ mark + tk::lpofa[f][0] ] ],
107  gid[ inpoel[ mark + tk::lpofa[f][1] ] ],
108  gid[ inpoel[ mark + tk::lpofa[f][2] ] ] }};
109  if (m_ipface.find(t) == end(m_ipface)) {
110  Assert( ++m_exptNbface, "Sum up expected number of boundary faces" );
111  potbndface.insert( t );
112  }
113  }
114  }
115 
116  // In the following we assume that the size of the (potential) boundary-face
117  // adjacency map above does not necessarily equal to that of the node
118  // adjacency map (m_msumset). This is because while a node can be shared at a
119  // single corner or along an edge, that does not necessarily share a face as
120  // well (in other words, shared nodes or edges can exist that are not part of
121  // a shared face). So the chares we communicate with across faces are not
122  // necessarily the same as the chares we would communicate nodes with.
123  //
124  // Since the sizes of the node and face adjacency maps are not the same, while
125  // sending the faces on chare boundaries would be okay, however, the receiver
126  // would not necessarily know how many chares it must receive from. To solve
127  // this problem we send to chares which we share at least a single node with,
128  // i.e., rely on the node-adjacency map. Note that to all chares we share at
129  // least a single node with we send all our potential chare-boundary faces.
130  // This is the same list of faces to all chares we send.
131  //
132  // Another underlying assumption here is, of course, that the size of the face
133  // adjacency map is always smaller than or equal to that of the node adjacency
134  // map, which is always true. Since the receive side already knows how many
135  // fellow chares it must receive shared node ids from, we use that to detect
136  // completion of the number of receives in comfac(). This simplifies the
137  // communication pattern and code.
138 
139  // Send sets of faces adjacent to chare boundaries to fellow workers (if any)
140  if (m_msumset.empty()) // in serial, skip setting up ghosts altogether
141  adj();
142  else
143  for (const auto& c : m_msumset) { // for all chares we share nodes with
144  thisProxy[ c.first ].comfac( thisIndex, potbndface );
145  }

148 bool
149 DG::leakyPartition()
150 // *****************************************************************************
151 // Perform leak-test on mesh partition
152 //! \details This function computes a surface integral over the boundary of the
153 //! incoming mesh partition. A non-zero vector result indicates a leak, e.g.,
154 //! a hole in the partition, which indicates an error upstream of this code,
155 //! either in the mesh geometry, mesh partitioning, or in the data structures
156 //! that represent faces.
157 //! \return True if our chare partition leaks.
158 // *****************************************************************************
159 {
160  auto d = Disc();
161  const auto& inpoel = d->Inpoel();
162  const auto& coord = d->Coord();
163  const auto& x = coord[0];
164  const auto& y = coord[1];
165  const auto& z = coord[2];
166 
167  // Storage for surface integral over our mesh partition
168  std::array< tk::real, 3 > s{{ 0.0, 0.0, 0.0}};
169 
170  for (std::size_t e=0; e<m_esuelTet.size()/4; ++e) { // for all our tets
171  auto mark = e*4;
172  for (std::size_t f=0; f<4; ++f) // for all tet faces
173  if (m_esuelTet[mark+f] == -1) { // if face has no outside-neighbor tet
174  // 3 local node IDs of face
175  auto A = inpoel[ mark + tk::lpofa[f][0] ];
176  auto B = inpoel[ mark + tk::lpofa[f][1] ];
177  auto C = inpoel[ mark + tk::lpofa[f][2] ];
178  // Compute geometry data for face
179  auto geoface = tk::geoFaceTri( {{x[A], x[B], x[C]}},
180  {{y[A], y[B], y[C]}},
181  {{z[A], z[B], z[C]}} );
182  // Sum up face area * face unit-normal
183  s[0] += geoface(0,0,0) * geoface(0,1,0);
184  s[1] += geoface(0,0,0) * geoface(0,2,0);
185  s[2] += geoface(0,0,0) * geoface(0,3,0);
186  }
187  }
188 
189  auto eps = std::numeric_limits< tk::real >::epsilon() * 100;
190  return std::abs(s[0]) > eps || std::abs(s[1]) > eps || std::abs(s[2]) > eps;
191 }

193 bool
194 DG::leakyAdjacency()
195 // *****************************************************************************
196 // Perform leak-test on chare boundary faces
197 //! \details This function computes a surface integral over the boundary of the
198 //! faces after the face adjacency communication map is completed.. A non-zero
199 //! vector result indicates a leak, e.g., a hole in the partition (covered by
200 //! the faces of the face adjacency communication map), which indicates an
201 //! error upstream in the code that sets up the face communication data
202 //! structures.
203 //! \note Compared to leakyPartition() this function performs the leak-test on
204 //! the face geometry data structure enlarged by ghost faces on this partition
205 //! by computing a discrete surface integral considering the physical and
206 //! chare boundary faces, which should be equal to zero for a closed domain.
207 //! \return True if our chare face adjacency leaks.
208 // *****************************************************************************
209 {
210  // Storage for surface integral over our chunk of the adjacency
211  std::array< tk::real, 3 > s{{ 0.0, 0.0, 0.0}};
212 
213  // physical boundary faces
214  for (std::size_t f=0; f<m_fd.Nbfac(); ++f) {
215  s[0] += m_geoFace(f,0,0) * m_geoFace(f,1,0);
216  s[1] += m_geoFace(f,0,0) * m_geoFace(f,2,0);
217  s[2] += m_geoFace(f,0,0) * m_geoFace(f,3,0);
218  }
219 
220  // chare-boundary faces
221  for (std::size_t f=m_fd.Ntfac(); f<m_fd.Esuf().size()/2; ++f) {
222  s[0] += m_geoFace(f,0,0) * m_geoFace(f,1,0);
223  s[1] += m_geoFace(f,0,0) * m_geoFace(f,2,0);
224  s[2] += m_geoFace(f,0,0) * m_geoFace(f,3,0);
225  }
226 
227  auto eps = std::numeric_limits< tk::real >::epsilon() * 100;
228  return std::abs(s[0]) > eps || std::abs(s[1]) > eps || std::abs(s[2]) > eps;
229 }

234 // Convert chare-node adjacency map to hold sets instead of vectors
235 //! \return Chare-node adjacency map that holds sets instead of vectors

250 void
251 DG::comfac( int fromch, const tk::UnsMesh::FaceSet& infaces )
252 // *****************************************************************************
253 // Receive unique set of faces we potentially share with/from another chare
254 //! \param[in] fromch Sender chare id
255 //! \param[in] infaces Unique set of faces we potentially share with fromch
256 // *****************************************************************************
257 {
258  // Attempt to find sender chare among chares we potentially share faces with.
259  // Note that it is feasible that a sender chare called us but we do not have a
260  // set of faces associated to that chare. This can happen if we only share a
261  // single node or an edge but note a face with that chare.
262  auto& bndface = m_bndFace[ fromch ]; // will associate to sender chare
263  // Try to find incoming faces on our chare boundary with other chares. If
264  // found, generate and assign new local ID to face, associated to sender
265  // chare.
266  auto d = Disc();
267  const auto& gid = d->Gid();
268  const auto& inpoel = d->Inpoel();
269  for (std::size_t e=0; e<m_esuelTet.size()/4; ++e) { // for all our tets
270  auto mark = e*4;
271  for (std::size_t f=0; f<4; ++f) { // for all cell faces
272  if (m_esuelTet[mark+f] == -1) { // if face has no outside-neighbor tet
273  tk::UnsMesh::Face t{{ gid[ inpoel[ mark + tk::lpofa[f][0] ] ],
274  gid[ inpoel[ mark + tk::lpofa[f][1] ] ],
275  gid[ inpoel[ mark + tk::lpofa[f][2] ] ] }};
276  // if found among the incoming faces and if not one of our internal nor
277  // physical boundary faces
278  if ( infaces.find(t) != end(infaces) &&
279  m_ipface.find(t) == end(m_ipface) )
280  bndface[t][0] = m_nfac++; // assign new local face ID
281  }
282  }
283  }
284  // If at this point we have not found any face among our faces we potentially
285  // share with fromch, there is no need to keep an empty set of faces
286  // associated to fromch as we only share nodes or edges with it, but not
287  // faces.
288  if (bndface.empty()) m_bndFace.erase( fromch );
289 
290  // if we have heard from all fellow chares that we share at least a single
291  // node, edge, or face with
292  if (++m_ncomfac == m_msumset.size()) {
293 
294  tk::destroy(m_ipface);
295 
296  // Error checking on the number of expected vs received/found chare-boundary
297  // faces
298  Assert( m_exptNbface ==
299  std::accumulate( m_bndFace.cbegin(), m_bndFace.cend(),
300  std::size_t(0),
301  []( std::size_t acc, const decltype(m_bndFace)::value_type& b )
302  { return acc + b.second.size(); } ),
303  "Expected and received number of boundary faces mismatch" );
304 
305  // Basic error checking on chare-boundary-face map
306  Assert( m_bndFace.find( thisIndex ) == m_bndFace.cend(),
307  "Face-communication map should not contain data for own chare ID" );
308 
309  // Store (local) tet ID adjacent to our chare boundary from the inside
310  for (std::size_t e=0; e<m_esuelTet.size()/4; ++e) { // for all our tets
311  auto mark = e*4;
312  for (std::size_t f=0; f<4; ++f) { // for all cell faces
313  if (m_esuelTet[mark+f] == -1) { // if face has no outside-neighbor tet
314  tk::UnsMesh::Face t{{ gid[ inpoel[ mark + tk::lpofa[f][0] ] ],
315  gid[ inpoel[ mark + tk::lpofa[f][1] ] ],
316  gid[ inpoel[ mark + tk::lpofa[f][2] ] ] }};
317  auto c = findchare( t );
318  if (c > -1) {
319  auto& lbndface = tk::ref_find( m_bndFace, c );
320  auto& face = tk::ref_find( lbndface, t );
321  face[1] = e; // store (local) inner tet ID adjacent to face
322  }
323  }
324  }
325  }
326 
327  // At this point m_bndFace is complete on this PE. This means that starting
328  // from the sets of faces we potentially share with fellow chares we now
329  // only have those faces we actually share faces with (through which we need
330  // to communicate later). Also, m_bndFace not only has the unique faces
331  // associated to fellow chares, but also a newly assigned local face ID as
332  // well as the local id of the inner tet adjacent to the face. Continue by
333  // starting setting up ghost data
334  setupGhost();
335  // Besides setting up our own ghost data, we also issue requests (for ghost
336  // data) to those chares which we share faces with. Note that similar to
337  // comfac() we are calling reqGhost() by going through msumset instead,
338  // which may send requests to those chare we do not share faces with. This
339  // is so that we can test for completing by querying the size of the already
340  // complete msumset in reqGhost. Requests in sendGhost will only be
341  // fullfilled based on m_ghostData.
342  for (const auto& c : m_msumset) // for all chares we share nodes with
343  thisProxy[ c.first ].reqGhost();
344  }
345 }

347 void
348 DG::setupGhost()
349 // *****************************************************************************
350 // Setup own ghost data on this chare
351 // *****************************************************************************
352 {
353  auto d = Disc();
354  const auto& gid = d->Gid();
355  const auto& inpoel = d->Inpoel();
356 
357  // Enlarge elements surrounding faces data structure for ghosts
358  m_fd.Esuf().resize( 2*m_nfac, -1 );
359  // Enlarge face geometry data structure for ghosts
360  m_geoFace.resize( m_nfac, 0.0 );
361 
362  // Collect tet ids, their face connectivity (given by 3 global node IDs, each
363  // triplet for potentially multiple faces on the chare boundary), and their
364  // elem geometry data (see GhostData) associated to fellow chares adjacent to
365  // chare boundaries. Once received by fellow chares, these tets will become
366  // known as ghost elements and their data as ghost data.
367  for (std::size_t e=0; e<m_esuelTet.size()/4; ++e) { // for all our tets
368  auto mark = e*4;
369  for (std::size_t f=0; f<4; ++f) { // for all cell faces
370  if (m_esuelTet[mark+f] == -1) { // if face has no outside-neighbor tet
371  tk::UnsMesh::Face t{{ gid[ inpoel[ mark + tk::lpofa[f][0] ] ],
372  gid[ inpoel[ mark + tk::lpofa[f][1] ] ],
373  gid[ inpoel[ mark + tk::lpofa[f][2] ] ] }};
374  auto c = findchare( t );
375  // It is possible that we do not find the chare for this face. We are
376  // looping through all of our tets and interrogating all faces that do
377  // not have neighboring tets but we only care about chare-boundary faces
378  // here as only those need ghost data. (m_esuelTet may also contain
379  // physical boundary faces.)
380  if (c > -1) {
381  // Will store ghost data associated to neighbor chare
382  auto& ghost = m_ghostData[ c ];
383  // Store tet id adjacent to chare boundary as key for ghost data
384  auto& tuple = ghost[ e ];
385  // If tetid e has not yet been encountered, store geometry (only once)
386  auto& nodes = std::get< 0 >( tuple );
387  if (nodes.empty()) std::get< 1 >( tuple ) = m_geoElem[ e ];
388  // (Always) store face node IDs on chare boundary, even if tetid e has
389  // already been stored. Thus we store potentially multiple faces along
390  // the same chare-boundary. This happens, e.g., when the boundary
391  // between chares is zig-zaggy enough to have 2 or even 3 faces of the
392  // same tet.
393  nodes.push_back( t[0] );
394  nodes.push_back( t[1] );
395  nodes.push_back( t[2] );
396  Assert( nodes.size() <= 4*3, "Overflow of faces/tet to send" );
397  }
398  }
399  }
400  }
401 
402  // Basic error checking on local ghost data
403  Assert( m_ghostData.find( thisIndex ) == m_ghostData.cend(),
404  "Chare-node adjacency map should not contain data for own chare ID" );
405 
406  // More in-depth error checking on local ghost data
407  for (const auto& c : m_ghostData)
408  for (const auto& t : c.second) {
409  IGNORE(t);
410  Assert( !std::get< 0 >( t.second ).empty(),
411  "Emtpy face vector in ghost data" );
412  Assert( std::get< 0 >( t.second ).size() % 3 == 0,
413  "Face node IDs must be triplets" );
414  Assert( std::get< 0 >( t.second ).size() <= 4*3, // <= 4*3 (4*numfaces)
415  "Max number of faces for a single ghost tet is 4" );
416  Assert( !std::get< 1 >( t.second ).empty(),
417  "No elem geometry data for ghost" );
418  Assert( std::get< 1 >( t.second ).size() == m_geoElem.nprop(),
419  "Elem geometry data for ghost must be for single tet" );
420  }
421 
422  ownghost_complete();
423 }

461 void
462 DG::comGhost( int fromch, const GhostData& ghost )
463 // *****************************************************************************
464 // Receive ghost data on chare boundaries from fellow chare
465 //! \param[in] fromch Caller chare ID
466 //! \param[in] ghost Ghost data, see Inciter/FaceData.h for the type
467 // *****************************************************************************
468 {
469  // nodelist with fromch, currently only used for an assert
470  const auto& nl = tk::cref_find( m_msumset, fromch );
471  IGNORE(nl);
472 
473  auto& ghostelem = m_ghost[ fromch ]; // will associate to sender chare
474 
475  // Store ghost data coming from chare
476  for (const auto& g : ghost) { // loop over incoming ghost data
477  auto e = g.first; // remote/ghost tet id outside of chare boundary
478  const auto& nodes = std::get< 0 >( g.second ); // node IDs of face(s)
479  const auto& geo = std::get< 1 >( g.second ); // ghost elem geometry data
480  Assert( nodes.size() % 3 == 0, "Face node IDs must be triplets" );
481  Assert( nodes.size() <= 4*3, "Overflow of faces/tet received" );
482  Assert( geo.size() % 4 == 0, "Ghost geometry size mismatch" );
483  Assert( geo.size() == m_geoElem.nprop(), "Ghost geometry number mismatch" );
484  for (std::size_t n=0; n<nodes.size()/3; ++n) { // face(s) of ghost e
485  // node IDs of face on chare boundary
486  tk::UnsMesh::Face t{{ nodes[n*3+0], nodes[n*3+1], nodes[n*3+2] }};
487  // must find t in nodelist of chare-boundary adjacent to fromch
488  Assert( nl.find(t[0]) != end(nl) &&
489  nl.find(t[1]) != end(nl) &&
490  nl.find(t[2]) != end(nl),
491  "Ghost face not found in chare-node adjacency map on receiving end" );
492  // must find face in boundary-face adjacency map for fromch
493  Assert( tk::cref_find(m_bndFace,fromch).find( t ) !=
494  tk::cref_find(m_bndFace,fromch).cend(), "Ghost face not "
495  "found in boundary-face adjacency map on receiving end" );
496  // find local face & tet ids for t
497  auto id = tk::cref_find( tk::cref_find(m_bndFace,fromch), t );
498  // compute face geometry for chare-boundary face
499  addGeoFace( t, id );
500  // if ghost tet id not yet encountered on boundary with fromch
501  auto i = ghostelem.find( e );
502  if (i != end(ghostelem))
503  addEsuf( id, i->second ); // fill in elements surrounding face
504  else {
505  addEsuf( id, m_nunk ); // fill in elements surrounding face
506  ghostelem[e] = m_nunk; // assign new local tet id to remote ghost id
507  m_geoElem.push_back( geo );// store ghost elem geometry
508  ++m_nunk; // increase number of unknowns on this chare
509  }
510  }
511  }

517 void
518 DG::addEsuf( const std::array< std::size_t, 2 >& id, std::size_t ghostid )
519 // *****************************************************************************
520 // Fill elements surrounding a face along chare boundary
521 //! \param[in] id Local face and (inner) tet id adjacent to it
522 //! \param[in] ghostid Local ID for ghost tet
523 //! \details This function extends and fills in the elements surrounding faces
524 //! data structure (esuf) so that the left and right element id is filled
525 //! in correctly on chare boundaries to contain the correct inner tet id and
526 //! the local tet id for the outer (ghost) tet, both adjacent to the given
527 //1 chare-face boundary. Prior to this function, this data structure does not
528 //! have yet face-element connectivity adjacent to chare-boundary faces, only
529 //! for physical boundaries and internal faces that are not on the chare
530 //! boundary (this latter purely as a result of mesh partitioning). The remote
531 //! element id of the ghost is stored in a location that is local to our own
532 //! esuf. The face numbering is such that esuf stores the element-face
533 //! connectivity first for the physical-boundary faces, followed by that of
534 //! the internal faces, followed by the chare-boundary faces. As a result,
535 //! esuf can be used by physics algorithms in exactly the same way as would be
536 //! used in serial. In serial, of course, this data structure is not extended
537 //! at the end by the chare-boundaries.
538 // *****************************************************************************
539 {
540  auto& esuf = m_fd.Esuf();
541  Assert( 2*id[0]+1 < esuf.size(), "Indexing out of esuf" );
542 
543  // put in inner tet id
544  esuf[ 2*id[0]+0 ] = static_cast< int >( id[1] );
545  // put in local id for outer/ghost tet
546  esuf[ 2*id[0]+1 ] = static_cast< int >( ghostid );
547 }

721 DG::advance( tk::real newdt )
722 // *****************************************************************************
723 // Advance equations to next time step
724 //! \param[in] newdt Size of this new time step
725 // *****************************************************************************
726 {
727  auto d = Disc();
728 
729  // Set new time step size
730  d->setdt( newdt );
731 
732  // communicate solution ghost data (if any)
733  if (m_ghostData.empty())
734  comsol_complete();
735  else
736  for(const auto& n : m_ghostData) {

750 void
751 DG::comsol( int fromch,
752  const std::vector< std::size_t >& tetid,
753  const std::vector< std::vector< tk::real > >& u )
754 // *****************************************************************************
755 // Receive chare-boundary solution ghost data from neighboring chares
756 //! \param[in] tetid Ghost tet ids we receive solution data for
757 //! \param[in] u Solution ghost data
758 //! \details This function receives contributions to m_u from fellow chares.
759 // *****************************************************************************
760 {
761  Assert( u.size() == tetid.size(), "Size mismatch in DG::comsol()" );
762 
763  // Find local-to-ghost tet id map for sender chare
764  const auto& n = tk::cref_find( m_ghost, fromch );
765 
766  for (std::size_t i=0; i<tetid.size(); ++i) {
767  auto j = tk::cref_find( n, tetid[i] );
768  Assert( j >= m_esuelTet.size()/4, "Receiving solution non-ghost data" );
769  Assert( j < m_u.nunk(), "Indexing out of bounds in DG::comsol()" );
770  Assert( m_recvGhost.insert( j ).second,
771  "Failed to store local tetid of received ghost tetid" );
772  for (std::size_t c=0; c<m_u.nprop(); ++c)
773  m_u(j,c,0) = u[i][c];
774  }
775 
776  // if we have received all solution ghost contributions from those chares we
777  // communicate along chare-boundary faces with, solve the system
778  if (++m_nsol == m_ghostData.size()) {
779  Assert( m_exptGhost == m_recvGhost,
780  "Expected/received ghost tet id mismatch" );
781  m_recvGhost.clear();
782  m_nsol = 0;
783  comsol_complete();
784  }

866  eq.rhs( d->T(), m_geoFace, m_fd, m_u, m_rhs );
