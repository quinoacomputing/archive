85 //! Pack/Unpack selected partial differential equations. This Pack/Unpack method
86 //! (re-)creates the PDE factory since it needs to (re-)bind function
87 //! pointers on different processing elements. Therefore we circumvent Charm's
88 //! usual pack/unpack for this type, and thus sizing does not make sense: sizing
89 //! is a no-op. We could initialize the factory in InciterDriver's constructor
90 //! and let this function re-create the stack only when unpacking, but that
91 //! leads to repeating the same code twice: once in InciterDriver's constructor,
92 //! once here. Another option is to use this pack/unpack routine to both
93 //! initially create (when packing) and to re-create (when unpacking) the
94 //! factory, which eliminates the need for pre-creating the object in
95 //! InciterDriver's constructor and therefore eliminates the repeated code. This
96 //! explains the guard for sizing: the code below is called for packing only (in
97 //! serial) and packing and unpacking (in parallel).
98 inline
99 void operator|( PUP::er& p, std::vector< PDE >& eqs ) {
100  if (!p.isSizing()) eqs = PDEStack().selected();
101 }

105 //! \brief Charm++ main chare for the shock hydroddynamics executable, inciter.
106 //! \details In inciter the Charm++ runtime system is initialized only after the
107 //! mesh has been read in, partitioned, and the necessary data structures,
108 //! e.g., communication maps, have been generated. This delayed initialization
109 //! of the Charm++ runtime system is required since the mesh partitioning is
110 //! done by Zoltan, an MPI library. Note that this Charm++ main chare object
111 //! should not be in a namespace.
112 //! \author J. Bakosi
113 class Main : public CBase_Main {
114 
115  public:
116  //! \brief Constructor
117  //! \details The main chare constructor is the main entry point of the
118  //! Charm++ portion of inciter, called by the Charm++ runtime system. The
119  //! constructor does basic initialization steps, prints out some useful
120  //! information to screen (in verbose mode), and instantiates a driver.
121  //! Since Charm++ is fully asynchronous, the constructor usually spawns
122  //! asynchronous objects and immediately exits. Thus in the body of the
123  //! main chare constructor we fire up an 'execute' chare, which then calls
124  //! back to Main::execute(). Finishing the main chare constructor the
125  //! Charm++ runtime system then starts the network-migration of all
126  //! global-scope data (if any). The execute chare calling back to
127  //! Main::execute() signals the end of the migration of the global-scope
128  //! data. Then we are ready to execute the driver. Since inciter is
129  //! parallel and asynchronous, its driver fires up additional Charm++
130  //! chare objects which then call back to Main::finalize() at some point
131  //! in the future when all work has been finished. finalize() then exits
132  //! by calling Charm++'s CkExit(), shutting down the runtime system.
133  //! \see http://charm.cs.illinois.edu/manuals/html/charm++/manual.html
134  Main( CkArgMsg* msg )
135  try :
136  m_cmdline(),
137  // Parse command line into m_cmdline using default simple pretty printer
138  m_cmdParser( msg->argc, msg->argv, tk::Print(), m_cmdline ),
139  // Create pretty printer initializing output streams based on command line
140  m_print( m_cmdline.get< tag::verbose >() ? std::cout : std::clog ),
141  // Create Inciter driver
142  m_driver( tk::Main< inciter::InciterDriver >
143  ( msg->argc, msg->argv,
144  m_cmdline,
145  tk::HeaderType::INCITER,
146  INCITER_EXECUTABLE,
147  m_print ) ),
148  // Start new timer measuring the total runtime
149  m_timer(1),
150  m_timestamp()
151  {
152  delete msg;
153  mainProxy = thisProxy;
154  // Fire up an asynchronous execute object, which when created at some
155  // future point in time will call back to this->execute(). This is
156  // necessary so that this->execute() can access already migrated
157  // global-scope data.
158  CProxy_execute::ckNew();
159  // Start new timer measuring the migration of global-scope data
160  m_timer.emplace_back();
161  } catch (...) { tk::processExceptionCharm(); }
162 
163  //! Execute driver created and initialized by constructor
164  void execute() {
165  try {
166  m_timestamp.emplace_back("Migrate global-scope data", m_timer[1].hms());
167  m_driver.execute();
168  } catch (...) { tk::processExceptionCharm(); }
169  }
170 
171  //! Normal exit point
172  void finalize() {
173  try {
174  if (!m_timer.empty()) {
175  m_timestamp.emplace_back( "Total runtime", m_timer[0].hms() );
176  m_print.time( "Timers (h:m:s)", m_timestamp );
177  m_print.endpart();
178  }
179  } catch (...) { tk::processExceptionCharm(); }
180  // Tell the Charm++ runtime system to exit
181  CkExit();
182  }
183 
184  private:
185  inciter::ctr::CmdLine m_cmdline; //!< Command line
186  inciter::CmdLineParser m_cmdParser; //!< Command line parser
187  inciter::InciterPrint m_print; //!< Pretty printer
188  inciter::InciterDriver m_driver; //!< Driver
189  std::vector< tk::Timer > m_timer; //!< Timers
190 
191  //! Time stamps in h:m:s with labels
192  std::vector< std::pair< std::string, tk::Timer::Watch > > m_timestamp;
193 };
