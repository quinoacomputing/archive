89 //! \brief Pack/Unpack selected partial differential equations using continuous
90 //! Galerkin discretization.
91 //! \details This Pack/Unpack method (re-)creates the PDE factory since it needs
92 //! to (re-)bind function pointers on different processing elements. Therefore
93 //! we circumvent Charm's usual pack/unpack for this type, and thus sizing
94 //! does not make sense: sizing is a no-op. We could initialize the factory in
95 //! InciterDriver's constructor and let this function re-create the stack only
96 //! when unpacking, but that leads to repeating the same code twice: once in
97 //! InciterDriver's constructor, once here. Another option is to use this
98 //! pack/unpack routine to both initially create (when packing) and to
99 //! re-create (when unpacking) the factory, which eliminates the need for
100 //! pre-creating the object in InciterDriver's constructor and therefore
101 //! eliminates the repeated code. This explains the guard for sizing: the code
102 //! below is called for packing only (in serial) and packing and unpacking (in
103 //! parallel).
104 inline
105 void operator|( PUP::er& p, std::vector< CGPDE >& eqs ) {
106  try {
107  if (!p.isSizing()) eqs = PDEStack().selectedCG();
108  } catch (...) { tk::processExceptionCharm(); }
109 }

135 //! \brief Charm++ main chare for the shock hydroddynamics executable, inciter.
136 //! \details In inciter the Charm++ runtime system is initialized only after the
137 //! mesh has been read in, partitioned, and the necessary data structures,
138 //! e.g., communication maps, have been generated. This delayed initialization
139 //! of the Charm++ runtime system is required since the mesh partitioning is
140 //! done by Zoltan, an MPI library. Note that this Charm++ main chare object
141 //! should not be in a namespace.
142 class Main : public CBase_Main {
143 
144  public:
145  //! \brief Constructor
146  //! \details The main chare constructor is the main entry point of the
147  //! Charm++ portion of inciter, called by the Charm++ runtime system. The
148  //! constructor does basic initialization steps, prints out some useful
149  //! information to screen (in verbose mode), and instantiates a driver.
150  //! Since Charm++ is fully asynchronous, the constructor usually spawns
151  //! asynchronous objects and immediately exits. Thus in the body of the
152  //! main chare constructor we fire up an 'execute' chare, which then calls
153  //! back to Main::execute(). Finishing the main chare constructor the
154  //! Charm++ runtime system then starts the network-migration of all
155  //! global-scope data (if any). The execute chare calling back to
156  //! Main::execute() signals the end of the migration of the global-scope
157  //! data. Then we are ready to execute the driver. Since inciter is
158  //! parallel and asynchronous, its driver fires up additional Charm++
159  //! chare objects which then call back to Main::finalize() at some point
160  //! in the future when all work has been finished. finalize() then exits
161  //! by calling Charm++'s CkExit(), shutting down the runtime system.
162  //! \see http://charm.cs.illinois.edu/manuals/html/charm++/manual.html
163  Main( CkArgMsg* msg )
164  try :
165  m_signal( tk::setSignalHandlers() ),
166  m_cmdline(),
167  // Parse command line into m_cmdline using default simple pretty printer
168  m_cmdParser( msg->argc, msg->argv, tk::Print(), m_cmdline ),
169  // Create pretty printer initializing output streams based on command line
170  m_print( m_cmdline.get< tag::verbose >() ? std::cout : std::clog ),
171  // Create Inciter driver
172  m_driver( tk::Main< inciter::InciterDriver >
173  ( msg->argc, msg->argv,
174  m_cmdline,
175  tk::HeaderType::INCITER,
176  tk::inciter_executable(),
177  m_print ) ),
178  // Start new timer measuring the total runtime
179  m_timer(1),
180  m_timestamp()
181  {
182  delete msg;
183  mainProxy = thisProxy;
184  // Optionally enable quiscence detection
185  if (m_cmdline.get< tag::quiescence >())
186  CkStartQD( CkCallback( CkIndex_Main::quiescence(), thisProxy ) );
187  // Fire up an asynchronous execute object, which when created at some
188  // future point in time will call back to this->execute(). This is
189  // necessary so that this->execute() can access already migrated
190  // global-scope data.
191  CProxy_execute::ckNew();
192  // Start new timer measuring the migration of global-scope data
193  m_timer.emplace_back();
194  } catch (...) { tk::processExceptionCharm(); }
195 
196  //! Execute driver created and initialized by constructor
197  void execute() {
198  try {
199  m_timestamp.emplace_back("Migrate global-scope data", m_timer[1].hms());
200  m_driver.execute();
201  } catch (...) { tk::processExceptionCharm(); }
202  }
203 
204  //! Normal exit point
205  void finalize() {
206  try {
207  if (!m_timer.empty()) {
208  m_timestamp.emplace_back( "Total runtime", m_timer[0].hms() );
209  m_print.time( "Timers (h:m:s)", m_timestamp );
210  m_print.endpart();
211  }
212  } catch (...) { tk::processExceptionCharm(); }
213  // Tell the Charm++ runtime system to exit
214  CkExit();
215  }

218  [[noreturn]] void quiescence() {
219  Throw( "Quiescence detected" );
220  }
