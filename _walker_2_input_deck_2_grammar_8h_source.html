126  //! Rule used to trigger action
127  template< class eq, class vec > struct check_vector_size : pegtl::success {};
128  //! \brief Do error checking of a vector (required for a block)
129  //! \details This functor can be used to verify the correct size of an already
130  //! existing vector, specified by the user in a given block. The vector is
131  //! required to be non-empty.
132  //! \note This functor does not check existence of a vector. If the vector
133  //! does not even exist, it will throw an exception in DEBUG mode, while in

145  Message< Stack, ERROR, MsgKey::WRONGSIZE >( stack, in );

191  template< class eq >
192  struct action< check_init< eq > > {
193  template< typename Input, typename Stack >
194  static void apply( const Input& in, Stack& stack ) {
195  using walker::deck::neq;
196  const auto& init =
197  stack.template get< tag::param, eq, tag::initpolicy >();
198  // Error checks for joint delta initpolicy
199  if (init.size() == neq.get< eq >() &&
200  init.back() == walker::ctr::InitPolicyType::JOINTDELTA) {
201  // Make sure there was an icdelta...end block with at least a single

204  if (!spike.empty() && spike.back().empty())
205  Message< Stack, ERROR, MsgKey::NODELTA >( stack, in );
206  }
207  // Error checks for joint beta initpolicy
208  if (init.size() == neq.get< eq >() &&
209  init.back() == walker::ctr::InitPolicyType::JOINTBETA) {

213  stack.template get< tag::param, eq, tag::betapdf >();

230  //! Do error checking on eq coupled to another eq and compute depvar id
231  //! \tparam eq Equation tag of equation
232  //! \tparam coupledeq Equation tag of equation coupled to eq
233  //! \tparam id Equation offsets tag for coupled equation
234  //! \tparam depvar Error message key to use on missing coupled depvar
235  //! \param[in] in Parser input
236  //! \param[in,out] stack Grammar stack to wrok with
237  //! \param[in] missing Error message key to use on missing coupled equation if
238  //! the coupling is required, use missing = MsgKey::OPTIONAL is the coupling
239  //! is optional
240  template< typename eq, typename coupledeq, typename id, MsgKey depvar,
241  typename Input, typename Stack >
242  static void couple( const Input& in, Stack& stack, MsgKey missing )
243  {
244  using walker::deck::neq;
245  if (neq.get< coupledeq >() == 0) { // if no coupldeq block defined
246  if (missing != MsgKey::OPTIONAL) { // if the coupling is required
247  // error out if <eq> is configured without <coupledeq> model
248  stack.template push_back< tag::error >
249  ( std::string("Parser error: ") + tk::cref_find( message, missing ) );
250  }
251  } else { // Compute depvar id for coupled eq
252  // get coupled eq configuration
253  const auto& ceq = stack.template get< tag::param, eq, coupledeq >();
254  if (ceq.empty() && missing != MsgKey::OPTIONAL) // if !coupled & !optional
255  // error out if <coupledeq> depvar is not selected
256  Message< Stack, ERROR, depvar >( stack, in );
257  else { // find offset (local eq system index among systems) for depvar
258  // get ncomponents object from this input deck
259  const auto& ncomps = stack.template get< tag::component >();
260  // compute offset map associating offsets to dependent variables
261  auto offsetmap = ncomps.offsetmap( stack );
262  // get and save offsets for all depvars for eqs configured
263  for (auto p : ceq)
264  stack.template
265  get< tag::param, eq, id >().
266  push_back( tk::cref_find( offsetmap, p ) );
267  }
268  }
269  }
