110  //! Rule used to trigger action
111  template< class eq, class vec, MsgKey key >
112  struct check_vector_exists : pegtl::success {};
113  //! \brief Check the existence of a vector (required for a block)
114  //! \details This functor can be used to check the existence of a
115  //! user-specified vector, e.g., at the end of a diffeq ... end block, that
116  //! is required for that block. If the vector does not exist, we error out.
117  //! \note This functor only checks existence. If the vector exists, the size
118  //! of it can be checked by check_vector_size.
119  template< class eq, class vec, MsgKey Key >
120  struct action< check_vector_exists< eq, vec, Key > > {
121  template< typename Input, typename Stack >
122  static void apply( const Input& in, Stack& stack ) {
123  const auto& vv = stack.template get< tag::param, eq, vec >();
124  using walker::deck::neq;
125  if (vv.size() != neq.get< eq >())
126  Message< Stack, ERROR, Key >( stack, in );
127  }
128  };

130  //! Rule used to trigger action
131  template< class eq, class vec > struct check_vector_size : pegtl::success {};
132  //! \brief Do error checking of a vector (required for a block)
133  //! \details This functor can be used to verify the correct size of an already
134  //! existing vector, specified by the user in a given block. The vector is
135  //! required to have a specific size: ncomp/4, i.e., the number of
136  //! components in a block divided by four. This specific value is the only
137  //! way this is used at this time, thus the hard-coding of ncomp/4. However,
138  //! this could be abstracted away via a template argument if needed.
139  //! \note This functor does not check existence of a vector. If the vector
140  //! does not even exist, it will throw an exception in DEBUG mode, while in
141  //! RELEASE mode it will attempt to access unallocated memory yielding a
142  //! segfault. The existence of the vector should be checked by
143  //! check_vector_exists first.
144  template< class eq, class vec >
145  struct action< check_vector_size< eq, vec > > {
146  template< typename Input, typename Stack >
147  static void apply( const Input& in, Stack& stack ) {
148  const auto& vv = stack.template get< tag::param, eq, vec >();
149  Assert( !vv.empty(), "Vector of vectors checked must not be empty" );
150  const auto& v = vv.back();
151  const auto& ncomp = stack.template get< tag::component, eq >().back();
152  if (v.empty() || v.size() != ncomp/4)
153  Message< Stack, ERROR, MsgKey::WRONGSIZE >( stack, in );
154  }
155  };

157  //! Rule used to trigger action
158  template< class eq > struct check_eq : pegtl::success {};
159  //! \brief Do general error checking on the differential equation block
160  //! \details This is error checking that all equation types must satisfy.
161  template< class eq >
162  struct action< check_eq< eq > > {
163  template< typename Input, typename Stack >
164  static void apply( const Input& in, Stack& stack ) {
165  using walker::deck::neq;
166  // Error out if no dependent variable has been selected
167  const auto& depvar =
168  stack.template get< tag::param, eq, tag::depvar >();
169  if (depvar.empty() || depvar.size() != neq.get< eq >())
170  Message< Stack, ERROR, MsgKey::NODEPVAR >( stack, in );
171 
172  // Error out if no number of components has been selected
173  const auto& ncomp = stack.template get< tag::component, eq >();
174  if (ncomp.empty() || ncomp.size() != neq.get< eq >())
175  Message< Stack, ERROR, MsgKey::NONCOMP >( stack, in );
176 
177  // Error out if no RNG has been selected
178  const auto& rng = stack.template get< tag::param, eq, tag::rng >();
179  if (rng.empty() || rng.size() != neq.get< eq >())
180  Message< Stack, ERROR, MsgKey::NORNG >( stack, in );
181 
182  // Error out if no initialization policy has been selected
183  const auto& init =
184  stack.template get< tag::param, eq, tag::initpolicy >();
185  if (init.empty() || init.size() != neq.get< eq >())
186  Message< Stack, ERROR, MsgKey::NOINIT >( stack, in );
187 
188  // Error out if no coefficients policy has been selected
189  const auto& coeff =
190  stack.template get< tag::param, eq, tag::coeffpolicy >();
191  if (coeff.empty() || coeff.size() != neq.get< eq >())
192  Message< Stack, ERROR, MsgKey::NOCOEFF >( stack, in );
193  }
194  };

196  //! Rule used to trigger action
197  template< class eq > struct check_init : pegtl::success {};
198  //! \brief Do error checking on the selected initialization policy
199  template< class eq >
200  struct action< check_init< eq > > {
201  template< typename Input, typename Stack >
202  static void apply( const Input& in, Stack& stack ) {
203  using walker::deck::neq;
204  const auto& init =
205  stack.template get< tag::param, eq, tag::initpolicy >();
206  // Error checks for joint delta initpolicy
207  if (init.size() == neq.get< eq >() &&
208  init.back() == walker::ctr::InitPolicyType::JOINTDELTA) {
209  // Make sure there was an icdelta...end block with at least a single
210  // spike...end block
211  const auto& spike = stack.template get< tag::param, eq, tag::spike >();
212  if (!spike.empty() && spike.back().empty())
213  Message< Stack, ERROR, MsgKey::NODELTA >( stack, in );
214  }
215  // Error checks for joint beta initpolicy
216  if (init.size() == neq.get< eq >() &&
217  init.back() == walker::ctr::InitPolicyType::JOINTBETA) {
218  // Make sure there was an icbeta...end block with at least a single
219  // betapdf...end block
220  const auto& betapdf =
221  stack.template get< tag::param, eq, tag::betapdf >();
222  if (!betapdf.empty() && betapdf.back().empty())
223  Message< Stack, ERROR, MsgKey::NOBETA >( stack, in );
224  }
225 
226  }
227  };
