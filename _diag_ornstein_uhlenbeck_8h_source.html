2 /*!
3  \file src/DiffEq/DiagOrnsteinUhlenbeck.h
4  \copyright 2012-2015, J. Bakosi, 2016-2018, Los Alamos National Security, LLC.
5  \brief System of diagonal Ornstein-Uhlenbeck SDEs
6  \details This file implements the time integration of a system of stochastic
7  differential equations (SDEs), with linear drift and constant diagonal
8  diffusion, whose invariant is the joint [normal
9  distribution](http://en.wikipedia.org/wiki/Normal_distribution).
10 
11  In a nutshell, the equation integrated governs a set of scalars,
12  \f$Y_\alpha\f$, \f$\alpha\!=\!1,\dots,N\f$, as
13  \f[
14  \mathrm{d}Y_\alpha(t) = \theta_\alpha\left(\mu_\alpha - Y_\alpha\right)
15  \mathrm{d}t + \sigma_\alpha\mathrm{d}W_\alpha(t), \qquad \alpha=1,\dots,N
16  \f]
17  with parameter vectors \f$\theta_\alpha > 0\f$, \f$\mu_\alpha\f$, and
18  \f$\sigma_\alpha > 0\f$. Here \f$\mathrm{d}W_\alpha(t)\f$ is an isotropic
19  vector-valued [Wiener process](http://en.wikipedia.org/wiki/Wiener_process)
20  with independent increments. The invariant distribution is the joint normal
21  distribution. This system of SDEs consists of N independent equations, each
22  being a single-variate [Ornstein-Uhlenbeck
23  process](http://en.wikipedia.org/wiki/Ornstein%E2%80%93Uhlenbeck_process).
24 
25  From the Fokker-Planck equation, equivalent to the SDE above, the equations
26  governing the means, \f$ \langle Y_\alpha \rangle\f$, are
27  \f[
28  \newcommand{\irmean}[1]{{\langle{#1}\rangle}}
29  \frac{\partial\irmean{Y_\alpha}}{\partial t} =
30  \theta_\alpha\left(\mu_\alpha - \irmean{Y_\alpha}\right)
31  \f]
32  while the equation governing the covariance matrix, \f$ \langle y_\alpha
33  y_\beta \rangle \equiv \left\langle (Y_\alpha - \langle Y_\alpha \rangle)
34  (Y_\beta - \langle Y_\beta\rangle) \right\rangle \f$, is
35  \f[
36  \newcommand{\irmean}[1]{{\langle{#1}\rangle}}
37  \newcommand{\irv}[1]{\langle{#1^2}\rangle}
38  \frac{\partial\irmean{y_\alpha y_\beta}}{\partial t} =
39  -\left(\theta_\alpha+\theta_\beta\right)\irmean{y_\alpha y_\beta} +
40  \left\{ \begin{array}{lr}
41  \sigma_\alpha^2 & \mathrm{for} \quad \alpha = \beta,\\
42  0 & \mathrm{for} \quad \alpha \ne \beta
43  \end{array} \right..
44  \f]
45 */

110  //! \brief Advance particles according to the system of diagonal
111  //! Orsntein-Uhlenbeck SDEs
112  //! \param[in,out] particles Array of particle properties
113  //! \param[in] stream Thread (or more precisely stream) ID
114  //! \param[in] dt Time step size
115  void advance( tk::Particles& particles,
116  int stream,
117  tk::real dt,
118  tk::real,
119  const std::map< tk::ctr::Product, tk::real >& )
120  {
121  const auto npar = particles.nunk();
122  for (auto p=decltype(npar){0}; p<npar; ++p) {
123  // Generate Gaussian random numbers with zero mean and unit variance
124  std::vector< tk::real > dW( m_ncomp );
125  m_rng.gaussian( stream, m_ncomp, dW.data() );
126 
127  // Advance all m_ncomp scalars
128  for (ncomp_t i=0; i<m_ncomp; ++i) {
129  tk::real& par = particles( p, i, m_offset );
130  tk::real d = m_sigmasq[i] * dt;
131  d = (d > 0.0 ? std::sqrt(d) : 0.0);
132  par += m_theta[i]*(m_mu[i] - par)*dt + d*dW[i];
133  }
134  }
135  }
