84 void
85 Integrator::advance( tk::real dt,
86  tk::real t,
87  uint64_t it,
88  const std::map< tk::ctr::Product, tk::real >& moments )
89 // *****************************************************************************
90 // Advance all particles owned by this integrator
91 //! \param[in] dt Size of time step
92 //! \param[in] t Physical time
93 //! \param[in] it Iteration count
94 //! \param[in] moments Map of statistical moments
95 // *****************************************************************************
96 {
97  // Advance all equations one step in time. At the 0th iteration skip advance
98  // but estimate statistics and (potentially) PDFs (at the interval given by
99  // the user).
100  if (it > 0)
101  for (const auto& e : g_diffeqs)
102  e.advance( m_particles, CkMyPe(), dt, t, moments );
103 
104  if (!g_inputdeck.stat()) {// if no stats to estimate, skip to end of time step
105  contribute(
106  CkCallback(CkReductionTarget( Distributor, nostat ), m_hostproxy) );
107  } else {
108  // Accumulate sums for ordinary moments (every time step)
109  accumulateOrd( it );
110  }
111 }

113 void
114 Integrator::accumulateOrd( uint64_t it )
115 // *****************************************************************************
116 // Accumulate sums for ordinary moments and ordinary PDFs
117 //! \param[in] it Iteration count
118 // *****************************************************************************
119 {
120  // Accumulate partial sums for ordinary moments
121  m_stat.accumulateOrd();
122  // Accumulate sums for ordinary PDFs at select times
123  if ( g_inputdeck.pdf() &&
124  !((it+1) % g_inputdeck.get< tag::interval, tag::pdf >()) )
125  m_stat.accumulateOrdPDF();
126 
127  // Send accumulated ordinary moments and ordinary PDFs to collector for
128  // estimation
129  m_collproxy.ckLocalBranch()->chareOrd( m_stat.ord(),
130  m_stat.oupdf(),
131  m_stat.obpdf(),
132  m_stat.otpdf() );
133 }
