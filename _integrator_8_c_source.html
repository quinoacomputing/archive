88 void
89 Integrator::advance( tk::real dt,
90  tk::real t,
91  uint64_t it,
92  const std::map< tk::ctr::Product, tk::real >& moments )
93 // *****************************************************************************
94 // Advance all particles owned by this integrator
95 //! \param[in] dt Size of time step
96 //! \param[in] t Physical time
97 //! \param[in] it Iteration count
98 //! \param[in] moments Map of statistical moments
99 //! \author J. Bakosi
100 // *****************************************************************************
101 {
102  // Advance all equations one step in time. At the 0th iteration skip advance
103  // but estimate statistics and (potentially) PDFs (at the interval given by
104  // the user).
105  if (it > 0)
106  for (const auto& e : g_diffeqs)
107  e.advance( m_particles, CkMyPe(), dt, t, moments );
108 
109  if (!g_inputdeck.stat()) {// if no stats to estimate, skip to end of time step
110  contribute(
111  CkCallback(CkReductionTarget( Distributor, nostat ), m_hostproxy) );
112  } else {
113  // Accumulate sums for ordinary moments (every time step)
114  accumulateOrd();
115  // Accumulate sums for ordinary PDFs at select times
116  if ( g_inputdeck.pdf() &&
117  !((it+1) % g_inputdeck.get< tag::interval, tag::pdf >()) )
118  accumulateOrdPDF();
119  }
120 }
