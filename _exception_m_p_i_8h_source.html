18 //! \brief Assert macro that only throws an exception if expr fails.
19 //! \details If NDEBUG is defined (e.g. cmake's RELEASE or OPTIMIZED mode), do
20 //! nothing, expr is not evaluated. If NDEBUG is not defined, evaluate expr.
21 //! If expr is true, do nothing. If expr is false, throw Exception with
22 //! arguments passed in. The behavior is similar to libc's assert macro, but
23 //! throwing an Exception instead will also generate a nice call-trace and
24 //! will attempt to free memory. This macro should be used to detect
25 //! programmer errors.
26 #ifdef NDEBUG
27 # define AssertMPI(expr, ...) (static_cast<void>(0))
28 #else // NDEBUG
29 # define AssertMPI(expr, ...) \
30  ((expr) ? static_cast<void>(0) : Throw(__VA_ARGS__))
31 #endif // NDEBUG

33 //! \brief ErrChkMPI macro that only throws an exception if expr fails.
34 //! \details The behavior of this macro is the same whether NDEBUG is defined or
35 //! not: expr is always evaluated. If expr is true, do nothing. If expr is
36 //! false, throw Exception with arguments passed in. This macro should be
37 //! used to detect user or runtime errors. The main difference compared to
38 //! the vanilla ErrChk macro, defined in Base/Exception.h, is that this macro
39 //! always does an MPI_Allreduce on the outcome of expr. Since MPI_Allreduce
40 //! combines values of err from all MPI ranks and distributes the result back
41 //! to all ranks, the result of expr (err) appear on all ranks, thus if any
42 //! of the ranks had expr false, all will throw. Thus this macro should be
43 //! used to detect user or runtime errors from code sections that are
44 //! executed on multiple MPI ranks. Within asynchrounous Charm++ chares, the
45 //! simpler ErrChk macro suffices.
46 #ifdef QUINOA_CONFIG_MPI_ENABLED
47 #define ErrChkMPI(expr, ...) \
48 { \
49  int err = (expr) ? 0 : 1; \
50  int globalerr; \
51  MPI_Allreduce( &err, &globalerr, 1, MPI_INT, MPI_SUM, MPI_COMM_WORLD ); \
52  if ( globalerr ) Throw( __VA_ARGS__); \
53 }
54 #else
55 #define ErrChkMPI(expr, ...)\
56 {\
57  int QUINOA_ErrChkMPI_err = (expr) ? 0 : 1; \
58  if ( QUINOA_ErrChkMPI_err ) Throw( __VA_ARGS__); \
59 }
60 #endif
