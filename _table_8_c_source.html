21 // Sample a discrete y = f(x) function at x
22 //! \param[in] x Value of abscissa at which to sample y = f(x)
23 //! \param[in] table tk::Table to sample
24 //! \details If x is lower than the first x value in the function table, the
25 //! first function value is returned. If x is larger than the last x value in
26 //! the function table, the last function value is returned. In other words,
27 //! no extrapolation is performed. If x falls between the first/lowest and the
28 //! last/largest value in the table, linear interpolation is used to compute a
29 //! sample between the two closest x values of the table around the abscissa
30 //! given.
31 //! \return Sampled value from discrete table
32 //! \note The x column in the table is assumed to be in increasing order.
33 //! \see walker::invhts_eq_A005H, walker::prod_A005H for example tables
34 // *****************************************************************************
35 {
36  Assert( !table.empty(), "Empty table to sample from" );
37 
38  if (x < table.front().first) return table.front().second;
39 
40  for (std::size_t i=0; i<table.size()-1; ++i) {
41  if (table[i].first < x && x < table[i+1].first) {
42  auto t1 = table[i].first;
43  auto y1 = table[i].second;
44  auto t2 = table[i+1].first;
45  auto y2 = table[i+1].second;
46  return y1 + (y2-y1)/(t2-t1)*(x-t1);
47  }
48  }
49 
50  return table.back().second;
51 }
