66 //! Transport equation system of PDEs problem: advection + diffusion
67 class TransportPhysicsAdvDiff {
68 
69  public:
70  //! Add diffusion contribution to rhs
71  //! \param[in] e Equation system index, i.e., which transport equation
72  //! system we operate on among the systems of PDEs
73  //! \param[in] ncomp Number of components in this PDE
74  //! \param[in] mult Multiplier differentiating the different stages in
75  //! multi-stage time stepping
76  //! \param[in] dt Size of time step
77  //! \param[in] J Element Jacobi determinant
78  //! \param[in] N Element node indices
79  //! \param[in] grad Shape function derivatives, nnode*ndim [4][3]
80  //! \param[in] u Solution at element nodes at recent time step stage
81  //! \param[in] r Pointers to right hand side at component and offset
82  //! \param[in,out] R Right-hand side vector contributing to
83  static void
84  diffusionRhs( tk::ctr::ncomp_type e,
85  tk::ctr::ncomp_type ncomp,
86  tk::real mult,
87  tk::real dt,
88  tk::real J,
89  const std::array< std::size_t, 4 >& N,
90  const std::array< std::array< tk::real, 3 >, 4 >& grad,
91  const std::vector< std::array< tk::real, 4 > >& u,
92  const std::vector< const tk::real* >& r,
93  tk::Fields& R )
94  {
95  // diffusivities for all components
96  const auto& diff =
97  g_inputdeck.get< tag::param, tag::transport, tag::diffusivity >().at(e);
98  // add diffusion contribution to right hand side
99  for (ncomp_t c=0; c<ncomp; ++c) {
100  tk::real a = mult * dt * diff[c] * J;
101  for (std::size_t i=0; i<4; ++i)
102  for (std::size_t j=0; j<4; ++j)
103  for (std::size_t k=0; k<3; ++k)
104  R.var(r[c],N[j]) -= a * grad[j][k] * grad[i][k] * u[c][i];
105  }
106  }
107 
108  //! Compute the minimum time step size based on the diffusion
109  //! \param[in] e Equation system index, i.e., which transport equation
110  //! system we operate on among the systems of PDEs
111  //! \param[in] ncomp Number of components in this PDE
112  //! \param[in] L Characteristic length scale
113  //! \param[in] u Solution at element nodes at recent time step stage
114  //! \return Minimum time step size based on diffusion
115  static tk::real
116  diffusion_dt( tk::ctr::ncomp_type e,
117  tk::ctr::ncomp_type ncomp,
118  tk::real L,
119  const std::vector< std::array< tk::real, 4 > >& u )
120  {
121  // diffusivities for all components
122  const auto& diff =
123  g_inputdeck.get< tag::param, tag::transport, tag::diffusivity >().at(e);
124  // compute the minimum diffusion time step size across the four nodes
125  tk::real mindt = std::numeric_limits< tk::real >::max();
126  for (ncomp_t c=0; c<ncomp; ++c)
127  for (std::size_t j=0; j<4; ++j) {
128  auto dt = L * L * u[c][j] / diff[c]; // dt ~ dx^2/D
129  if (dt < mindt) mindt = dt;
130  }
131  return mindt;
132  }
133 
134  static ctr::PhysicsType type() noexcept
135  { return ctr::PhysicsType::ADVDIFF; }
136 };
