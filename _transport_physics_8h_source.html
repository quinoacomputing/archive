66 //! Transport equation system of PDEs problem: advection + diffusion
67 class TransportPhysicsAdvDiff {
68 
69  public:
70  //! Add diffusion contribution to rhs
71  //! \param[in] e Equation system index, i.e., which transport equation
72  //! system we operate on among the systems of PDEs
73  //! \param[in] ncomp Number of components in this PDE
74  //! \param[in] dt Size of time step
75  //! \param[in] J Element Jacobi determinant
76  //! \param[in] N Element node indices
77  //! \param[in] grad Shape function derivatives, nnode*ndim [4][3]
78  //! \param[in] u Solution at element nodes at recent time step stage
79  //! \param[in] r Pointers to right hand side at component and offset
80  //! \param[in,out] R Right-hand side vector contributing to
81  static void
82  diffusionRhs( tk::ctr::ncomp_type e,
83  tk::ctr::ncomp_type ncomp,
84  tk::real dt,
85  tk::real J,
86  const std::array< std::size_t, 4 >& N,
87  const std::array< std::array< tk::real, 3 >, 4 >& grad,
88  const std::vector< std::array< tk::real, 4 > >& u,
89  const std::vector< const tk::real* >& r,
90  tk::Fields& R )
91  {
92  // diffusivities for all components
93  const auto& diff =
94  g_inputdeck.get< tag::param, tag::transport, tag::diffusivity >().at(e);
95  // add diffusion contribution to right hand side
96  const auto a = dt * J / 6.0;
97  for (ncomp_t c=0; c<ncomp; ++c)
98  for (std::size_t k=0; k<3; ++k) {
99  const auto d = diff[ 3*c+k ];
100  for (std::size_t i=0; i<4; ++i)
101  for (std::size_t j=0; j<4; ++j)
102  R.var(r[c],N[j]) -= a * d * grad[j][k] * grad[i][k] * u[c][i];
103  }
104  }
105 
106  //! Compute the minimum time step size based on the diffusion
107  //! \param[in] e Equation system index, i.e., which transport equation
108  //! system we operate on among the systems of PDEs
109  //! \param[in] ncomp Number of components in this PDE
110  //! \param[in] L Characteristic length scale
111  //! \return Minimum time step size based on diffusion
112  static tk::real
113  diffusion_dt( tk::ctr::ncomp_type e,
114  tk::ctr::ncomp_type ncomp,
115  tk::real L,
116  const std::vector< std::array< tk::real, 4 > >& )
117  {
118  // diffusivities for all components
119  const auto& df =
120  g_inputdeck.get< tag::param, tag::transport, tag::diffusivity >().at(e);
121  // compute the minimum diffusion time step size across the four nodes
122  tk::real mindt = std::numeric_limits< tk::real >::max();
123  for (ncomp_t c=0; c<ncomp; ++c) {
124  const auto di = 3*c;
125  const auto d = std::max( df[di+2], std::max( df[di+0], df[di+1] ) );
126  const auto dt = L * L / (2.0*d); // dt ~ dx^2/(2D)
127  if (dt < mindt) mindt = dt;
128  }
129  return mindt;
130  }
131 
132  static ctr::PhysicsType type() noexcept
133  { return ctr::PhysicsType::ADVDIFF; }
134 };
