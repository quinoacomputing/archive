109  []( tk::real s, tk::real& d ){ return d /= s; } );

113  for (std::size_t i=0; i<p.size(); ++i)
114  p[i] = (g-1.0)*r[i]*(E[i] - (u[i]*u[i] + v[i]*v[i] + w[i]*w[i])/2.0);
115  out.push_back( p );
116  std::vector< tk::real > T = r;

119  T[i] = cv*(E[i] - (u[i]*u[i] + v[i]*v[i] + w[i]*w[i])/2.0);
120  out.push_back( T );

122  }
