236 // *****************************************************************************
237 {
238  // Output header for diagnostics output file
239  tk::DiagWriter dw( g_inputdeck.get< tag::cmd, tag::io, tag::diag >(),

241  g_inputdeck.get< tag::prec, tag::diag >() );
242 
243  // Collect variables names for integral/diagnostics output
244  std::vector< std::string > var;
245  for (const auto& eq : g_pdes) {
246  auto o = eq.names();
247  var.insert( end(var), begin(o), end(o) );
248  }
249 
250  const tk::ctr::Error opt;
251  auto nv = var.size();
252  std::vector< std::string > d;
253 
254  // Add 'L2(var)' for all variables as those are always computed
255  const auto& l2name = opt.name( tk::ctr::ErrorType::L2 );
256  for (std::size_t i=0; i<nv; ++i) d.push_back( l2name + '(' + var[i] + ')' );
257 
258  // Query user-requested diagnostics and augment diagnostics file header by
259  // 'err(var)', where 'err' is the error type configured, and var is the
260  // variable computed, for all variables and all error types configured.
261  const auto& err = g_inputdeck.get< tag::diag, tag::error >();
262  for (const auto& e : err) {
263  const auto& errname = opt.name( e );
264  for (std::size_t i=0; i<nv; ++i)
265  d.push_back( errname + '(' + var[i] + "-IC)" );
266  }
267 
268  // Write diagnostics header
269  dw.header( d );

362 void
363 Transporter::aveCost( tk::real c )
364 // *****************************************************************************
365 // Reduction target estimating the average communication cost of merging the
366 // linear system
367 //! \param[in] c Communication cost summed across all PEs. The cost associated
368 //! to a PE is a real number between 0 and 1, defined as the number of mesh
369 //! points the PE does not own, i.e., needs to send to some other PE, divided
370 //! by the total number of points the PE contributes to. The lower the better.
371 //! \details The average, computed here, gives an idea of the average
372 //! communication cost across all PEs, while the standard deviation, computed
373 //! by stdCost(), gives an idea on the expected load imbalance.
374 // *****************************************************************************
375 {
376  m_progReorder.end();
377  m_print.diag( "Creating workers" );
378  // Compute average and broadcast it back to all partitioners (PEs)
379  m_avcost = c / CkNumPes();
380  m_partitioner.stdCost( m_avcost );
381 }

600  Assert( d.size() == 3, "Diagnostics vector size mismatch" );
601 
602  for (std::size_t i=0; i<d.size(); ++i)
603  Assert( d[i].size() == ncomp,
604  "Size mismatch at final stage of diagnostics aggregation" );
605 
606  // Allocate storage for 'L2(var)' for all variables as those are always
607  // computed
608  m_diag.resize( g_inputdeck.get< tag::component >().nprop(), 0.0 );
609 
610  // Finish computing the L2 norm of the numerical solution
611  for (std::size_t i=0; i<d[0].size(); ++i)
612  m_diag[i] = sqrt( d[0][i] / m_V );
613 
614  // Query user-requested error types to be computed
615  const auto& error = g_inputdeck.get< tag::diag, tag::error >();
616 
617  decltype(ncomp) n = 0;
618  for (const auto& e : error) {
619  n += ncomp;
620  if (e == tk::ctr::ErrorType::L2) {
621  // Finish computing the L2 norm of the numerical - analytical solution
622  for (std::size_t i=0; i<d[1].size(); ++i)
623  m_diag.push_back( sqrt( d[1][i] / m_V ) );
624  } else if (e == tk::ctr::ErrorType::LINF) {
625  // Finish computing the Linf norm of the numerical - analytical solution
626  for (std::size_t i=0; i<d[2].size(); ++i)
627  m_diag.push_back( d[2][i] );
628  }
629  }
630 
631  diag_complete();
632 }

698 void
699 Transporter::evaluateTime()
700 // *****************************************************************************
701 // Evaluate time step and output one-liner report
702 // *****************************************************************************
703 {
704  const auto term = g_inputdeck.get< tag::discr, tag::term >();
705  const auto eps = std::numeric_limits< tk::real >::epsilon();
706  const auto nstep = g_inputdeck.get< tag::discr, tag::nstep >();
707 
708  if ( g_inputdeck.get< tag::cmd, tag::feedback >() )
709  m_progStep.end();
710 
711  if (m_stage < 1) { // if at half-stage, simply go to next one
712 
713  ++m_stage;
714 
715  } else { // if final stage of time step, finish time step just taken
716 
717  m_stage = 0;
718  // Increase number of iterations taken
719  ++m_it;
720  // Advance physical time to include time step just finished
721  m_t += m_dt;
722 
723  bool diag = false;
724 
725  // Append diagnostics file at selected times
726  if (!(m_it % g_inputdeck.get< tag::interval, tag::diag >())) {
727  tk::DiagWriter dw( g_inputdeck.get< tag::cmd, tag::io, tag::diag >(),
728  g_inputdeck.get< tag::flformat, tag::diag >(),
729  g_inputdeck.get< tag::prec, tag::diag >(),
730  std::ios_base::app );
731  if (dw.diag( m_it, m_t, m_diag )) diag = true;
732  m_diag.resize( g_inputdeck.get< tag::component >().nprop(), 0.0 );
733  }
734 
735  if (!(m_it % g_inputdeck.get< tag::interval, tag::tty >())) {
736 
737  // estimate time elapsed and time for accomplishment
738  tk::Timer::Watch ete, eta;
739  const auto& timer = tk::cref_find( m_timer, TimerTag::TIMESTEP );
740  timer.eta( g_inputdeck.get< tag::discr, tag::term >() -
741  g_inputdeck.get< tag::discr, tag::t0 >(),
742  m_t - g_inputdeck.get< tag::discr, tag::t0 >(),
743  g_inputdeck.get< tag::discr, tag::nstep >(),
744  m_it,
745  ete,
746  eta );
747 
748  // Output one-liner
749  m_print << std::setfill(' ') << std::setw(8) << m_it << " "
750  << std::scientific << std::setprecision(6)
751  << std::setw(12) << m_t << " "
752  << m_dt << " "
753  << std::setfill('0')
754  << std::setw(3) << ete.hrs.count() << ":"
755  << std::setw(2) << ete.min.count() << ":"
756  << std::setw(2) << ete.sec.count() << " "
757  << std::setw(3) << eta.hrs.count() << ":"
758  << std::setw(2) << eta.min.count() << ":"
759  << std::setw(2) << eta.sec.count() << " ";
760 
761  // Augment one-liner with output indicators
762  if (!(m_it % g_inputdeck.get<tag::interval,tag::field>())) m_print << 'F';
763  if (diag) m_print << 'D';
764 
765  m_print << std::endl;
766  }
767  }
768 
769  wait4eval();
770 
771  // if neither max iterations nor max time reached, will continue (by telling
772  // all linear system merger group elements to prepare for a new rhs),
773  // otherwise finish
774  if (std::fabs(m_t-term) > eps && m_it < nstep) {
775  m_linsysmerger.enable_wait4rhs();
776  if ( g_inputdeck.get< tag::cmd, tag::feedback >() )
777  m_progStep.start( "Time step stage ..." );
778  } else
779  finish();
780 }
