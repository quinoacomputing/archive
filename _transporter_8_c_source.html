191 void
192 Transporter::createPartitioner()
193 // *****************************************************************************
194 // Create mesh partitioner AND boundary conditions group
195 // *****************************************************************************
196 {
197  // Create mesh partitioner Charm++ chare group and start partitioning mesh
198  m_progGraph.start( "Creating partitioners and reading mesh graph" );
199 
200  // Start timing mesh read
201  m_timer[ TimerTag::MESHREAD ];
202 
203  // Create ExodusII reader for reading side sets from file.
204  tk::ExodusIIMeshReader er(g_inputdeck.get< tag::cmd, tag::io, tag::input >());
205 
206  // Read in side sets associated to mesh node IDs from file
207  m_print.diag( "Reading side sets" );
208  auto sidenodes = er.readSidesets();
209 
210  // Read side sets for boundary faces
211  m_print.diag( "Reading side set faces" );
212  std::map< int, std::vector< std::size_t > > bface;
213  auto nbfac = er.readSidesetFaces( bface );
214 
215  std::vector< std::size_t > triinpoel;
216  const auto scheme = g_inputdeck.get< tag::selected, tag::scheme >();
217 
218  if (nbfac<1 && scheme == ctr::SchemeType::DG)
219  {
220  Throw( "Boundary faces not specified using side-sets in ExodusII input file" );
221  }
222  else if (nbfac>0)
223  {
224  // Read triangle boundary-face connectivity
225  er.readFaces( nbfac, triinpoel );
226  }
227 
228  // Verify that side sets to which boundary conditions are assigned by user
229  // exist in mesh file
230  std::unordered_set< int > conf;
231  for (const auto& eq : g_pdes) eq.side( conf );
232  for (auto i : conf)
233  {
234  if (sidenodes.find(i) == end(sidenodes)) {
235  m_print.diag( "WARNING: Boundary conditions specified on side set " +
236  std::to_string(i) + " which does not exist in mesh file" );
237  break;
238  }
239  }
240 
241  // Create boundary conditions Charm++ chare group
242  m_bc = inciter::CProxy_BoundaryConditions::ckNew( sidenodes );
243 
244  // Create partitioner callbacks
245  std::vector< CkCallback > cbp {{
246  CkCallback( CkReductionTarget(Transporter,part), thisProxy )
247  , CkCallback( CkReductionTarget(Transporter,distributed), thisProxy )
248  , CkCallback( CkReductionTarget(Transporter,flattened), thisProxy )
249  , CkCallback( CkReductionTarget(Transporter,load), thisProxy )
250  , CkCallback( CkReductionTarget(Transporter,aveCost), thisProxy )
251  , CkCallback( CkReductionTarget(Transporter,stdCost), thisProxy )
252  , CkCallback( CkReductionTarget(Transporter,coord), thisProxy )
253  }};
254 
255  // Create mesh partitioner Charm++ chare group
256  m_partitioner =
257  CProxy_Partitioner::ckNew( cbp, thisProxy, m_solver, m_bc, m_scheme,
258  nbfac, bface, triinpoel );
259 }

263 // *****************************************************************************

265 // *****************************************************************************
266 {
267  // Output header for diagnostics output file
268  tk::DiagWriter dw( g_inputdeck.get< tag::cmd, tag::io, tag::diag >(),
269  g_inputdeck.get< tag::flformat, tag::diag >(),
270  g_inputdeck.get< tag::prec, tag::diag >() );
271 
272  // Collect variables names for integral/diagnostics output
273  std::vector< std::string > var;
274  for (const auto& eq : g_pdes) {
275  auto o = eq.names();
276  var.insert( end(var), begin(o), end(o) );
277  }
278 
279  const tk::ctr::Error opt;
280  auto nv = var.size();
281  std::vector< std::string > d;
282 
283  // Add 'L2(var)' for all variables as those are always computed
284  const auto& l2name = opt.name( tk::ctr::ErrorType::L2 );
285  for (std::size_t i=0; i<nv; ++i) d.push_back( l2name + '(' + var[i] + ')' );
286 
287  // Query user-requested diagnostics and augment diagnostics file header by
288  // 'err(var)', where 'err' is the error type configured, and var is the
289  // variable computed, for all variables and all error types configured.
290  const auto& err = g_inputdeck.get< tag::diag, tag::error >();
291  for (const auto& e : err) {
292  const auto& errname = opt.name( e );
293  for (std::size_t i=0; i<nv; ++i)
294  d.push_back( errname + '(' + var[i] + "-IC)" );
295  }
296 
297  // Write diagnostics header
298  dw.header( d );

301 void
302 Transporter::load( uint64_t nelem )
303 // *****************************************************************************
304 // Reduction target indicating that all Partitioner chare groups have finished
305 // reading their part of the contiguously-numbered computational mesh graph and
306 // we are ready to compute the computational load
307 //! \param[in] nelem Total number of mesh elements (summed across all PEs)
308 // *****************************************************************************
309 {
310  m_progGraph.end();
311 
312  // Compute load distribution given total work (nelem) and user-specified
313  // virtualization
314  uint64_t chunksize, remainder;
315  m_nchare = static_cast<int>(
316  tk::linearLoadDistributor(
317  g_inputdeck.get< tag::cmd, tag::virtualization >(),
318  nelem, CkNumPes(), chunksize, remainder ) );
319 
320  // Send total number of chares to all linear solver PEs, if they exist
321  const auto scheme = g_inputdeck.get< tag::selected, tag::scheme >();
322  if (scheme == ctr::SchemeType::MatCG || scheme == ctr::SchemeType::DiagCG)
323  m_solver.nchare( m_nchare );
324 
325  // signal to runtime system that m_nchare is set
326  load_complete();
327 
328  // Print out mesh graph stats
329  m_print.section( "Input mesh graph statistics" );
330  m_print.item( "Number of tetrahedra", nelem );
331  tk::ExodusIIMeshReader er(g_inputdeck.get< tag::cmd, tag::io, tag::input >());
332  m_npoin = er.readHeader();
333  m_print.item( "Number of nodes", m_npoin );
334 
335  // Print out info on load distribution
336  const auto ir = g_inputdeck.get< tag::selected, tag::initialamr >();
337  if (ir == ctr::InitialAMRType::UNIFORM)
338  m_print.section( "Load distribution (before initial mesh refinement)" );
339  else
340  m_print.section( "Load distribution" );
341 
342  m_print.item( "Virtualization [0.0...1.0]",
343  g_inputdeck.get< tag::cmd, tag::virtualization >() );
344  m_print.item( "Load (number of tetrahedra)", nelem );
345  m_print.item( "Number of processing elements", CkNumPes() );
346  m_print.item( "Number of work units",
347  std::to_string( m_nchare ) + " (" +
348  std::to_string( m_nchare-1 ) + "*" +
349  std::to_string( chunksize ) + "+" +
350  std::to_string( chunksize+remainder ) + ')' );
351 
352  // Print out mesh partitioning configuration
353  m_print.section( "Initial mesh partitioning" );
354  m_print.Item< tk::ctr::PartitioningAlgorithm,
355  tag::selected, tag::partitioner >();
356 
357  // Print out mesh refinement configuration and new mesh statistics
358  if (ir == ctr::InitialAMRType::UNIFORM) {
359  m_print.section( "Mesh refinement" );
360  m_print.Item< ctr::InitialAMR, tag::selected, tag::initialamr >();
361  m_print.item( "Final number of tetrahedra",
362  std::to_string(nelem*8) + " (8*" + std::to_string(nelem) + ')' );
363  }
364 
365  m_print.endsubsection();
366 }

407 void
408 Transporter::aveCost( tk::real c )
409 // *****************************************************************************
410 // Reduction target estimating the average communication among all PEs
411 //! \param[in] c Communication cost summed across all PEs. The cost associated
412 //! to a PE is a real number between 0 and 1, defined as the number of mesh
413 //! points the PE does not own, i.e., needs to send to some other PE, divided
414 //! by the total number of points the PE contributes to. The lower the better.
415 //! \details The average, computed here, gives an idea of the average
416 //! communication cost across all PEs, while the standard deviation, computed
417 //! by stdCost(), gives an idea on the expected load imbalance.
418 // *****************************************************************************
419 {
420  m_progReorder.end();
421  // Compute average and broadcast it back to all partitioners (PEs)
422  m_avcost = c / CkNumPes();
423  m_partitioner.stdCost( m_avcost );
424 }

631 void
632 Transporter::diagnostics( CkReductionMsg* msg )
633 // *****************************************************************************
634 // Reduction target optionally collecting diagnostics, e.g., residuals
635 //! \param[in] msg Serialized diagnostics vector aggregated across all PEs
636 // *****************************************************************************
637 {
638  std::vector< std::vector< tk::real > > d;
639 
640  // Deserialize diagnostics vector
641  PUP::fromMem creator( msg->getData() );
642  creator | d;
643  delete msg;
644 
645  auto ncomp = g_inputdeck.get< tag::component >().nprop();
646 
647  Assert( d.size() == NUMDIAG, "Diagnostics vector size mismatch" );
648 
649  for (std::size_t i=0; i<d.size(); ++i)
650  Assert( d[i].size() == ncomp,
651  "Size mismatch at final stage of diagnostics aggregation" );
652 
653  // Allocate storage for those diagnostics that are always computed
654  std::vector< tk::real > diag( ncomp, 0.0 );
655 
656  // Finish computing diagnostics
657  for (std::size_t i=0; i<d[L2SOL].size(); ++i)
658  diag[i] = sqrt( d[L2SOL][i] / m_V );
659 
660  // Query user-requested error types to output
661  const auto& error = g_inputdeck.get< tag::diag, tag::error >();
662 
663  decltype(ncomp) n = 0;
664  for (const auto& e : error) {
665  n += ncomp;
666  if (e == tk::ctr::ErrorType::L2) {
667  // Finish computing the L2 norm of the numerical - analytical solution
668  for (std::size_t i=0; i<d[L2ERR].size(); ++i)
669  diag.push_back( sqrt( d[L2ERR][i] / m_V ) );
670  } else if (e == tk::ctr::ErrorType::LINF) {
671  // Finish computing the Linf norm of the numerical - analytical solution
672  for (std::size_t i=0; i<d[LINFERR].size(); ++i)
673  diag.push_back( d[LINFERR][i] );
674  }
675  }
676 
677  // Append diagnostics file at selected times
678  tk::DiagWriter dw( g_inputdeck.get< tag::cmd, tag::io, tag::diag >(),
679  g_inputdeck.get< tag::flformat, tag::diag >(),
680  g_inputdeck.get< tag::prec, tag::diag >(),
681  std::ios_base::app );
682  dw.diag( static_cast<uint64_t>(d[ITER][0]), d[TIME][0], d[DT][0], diag );
683 
684  // Evaluate whther to continue with next step
685  m_scheme.eval< tag::bcast >();
686 }
