307 void
308 Transporter::aveCost( tk::real c )
309 // *****************************************************************************
310 // Reduction target estimating the average communication cost of merging the
311 // linear system
312 //! \param[in] c Communication cost summed across all PEs. The cost associated
313 //! to a PE is a real number between 0 and 1, defined as the number of mesh
314 //! points the PE does not own, i.e., needs to send to some other PE, divided
315 //! by the total number of points the PE contributes to. The lower the better.
316 //! \details The average, computed here, gives an idea of the average
317 //! communication cost across all PEs, while the standard deviation, computed
318 //! by stdCost(), gives an idea on the expected load imbalance.
319 //! \author J. Bakosi
320 // *****************************************************************************
321 {
322  m_progReorder.end();
323  m_print.diag( "Creating workers" );
324  // Compute average and broadcast it back to all partitioners (PEs)
325  m_avcost = c / CkNumPes();
326  m_partitioner.stdCost( m_avcost );
327 }

437 void
438 Transporter::dt( tk::real* d, std::size_t n )
439 // *****************************************************************************
440 // Reduction target yielding a single minimum time step size across all workers
441 //! \param[in] d Minimum time step size collected over all chares
442 //! \param[in] n Size of data behind d
443 //! \author J. Bakosi
444 // *****************************************************************************
445 {
446  #ifdef NDEBUG
447  IGNORE(n);
448  #endif
449 
450  Assert( n == 1, "Size of min(dt) must be 1" );
451 
452  if (m_stage == 1 || m_it == 0) {
453  // Use newly computed time step size
454  m_dt = *d;
455  // Truncate the size of last time step
456  const auto term = g_inputdeck.get< tag::discr, tag::term >();
457  if (m_t+m_dt > term) m_dt = term - m_t;;
458  }
459 
460  // Advance to next time step stage
461  m_carrier.advance( m_stage, m_dt, m_it, m_t );
462 }

504 void
505 Transporter::evaluateTime()
506 // *****************************************************************************
507 // Evaluate time step and output one-liner report
508 //! \author J. Bakosi
509 // *****************************************************************************
510 {
511  const auto term = g_inputdeck.get< tag::discr, tag::term >();
512  const auto eps = std::numeric_limits< tk::real >::epsilon();
513  const auto nstep = g_inputdeck.get< tag::discr, tag::nstep >();
514 
515  if ( g_inputdeck.get< tag::cmd, tag::feedback >() )
516  m_progStep.end();
517 
518  if (m_stage < 1) { // if at half-stage, simply go to next one
519 
520  ++m_stage;
521 
522  } else { // if final stage of time step, finish time step just taken
523 
524  m_stage = 0;
525  // Increase number of iterations taken
526  ++m_it;
527  // Advance physical time to include time step just finished
528  m_t += m_dt;
529 
530  bool diag = false;
531 
532  // Append diagnostics file at selected times
533  if (!(m_it % g_inputdeck.get< tag::interval, tag::diag >())) {
534  tk::DiagWriter dw( g_inputdeck.get< tag::cmd, tag::io, tag::diag >(),
535  g_inputdeck.get< tag::flformat, tag::diag >(),
536  g_inputdeck.get< tag::prec, tag::diag >(),
537  std::ios_base::app );
538  if (dw.diag( m_it, m_t, m_diag )) diag = true;
539  }
540 
541  if (!(m_it % g_inputdeck.get< tag::interval, tag::tty >())) {
542 
543  // estimate time elapsed and time for accomplishment
544  tk::Timer::Watch ete, eta;
545  const auto& timer = tk::cref_find( m_timer, TimerTag::TIMESTEP );
546  timer.eta( g_inputdeck.get< tag::discr, tag::term >() -
547  g_inputdeck.get< tag::discr, tag::t0 >(),
548  m_t - g_inputdeck.get< tag::discr, tag::t0 >(),
549  g_inputdeck.get< tag::discr, tag::nstep >(),
550  m_it,
551  ete,
552  eta );
553 
554  // Output one-liner
555  m_print << std::setfill(' ') << std::setw(8) << m_it << " "
556  << std::scientific << std::setprecision(6)
557  << std::setw(12) << m_t << " "
558  << m_dt << " "
559  << std::setfill('0')
560  << std::setw(3) << ete.hrs.count() << ":"
561  << std::setw(2) << ete.min.count() << ":"
562  << std::setw(2) << ete.sec.count() << " "
563  << std::setw(3) << eta.hrs.count() << ":"
564  << std::setw(2) << eta.min.count() << ":"
565  << std::setw(2) << eta.sec.count() << " ";
566 
567  // Augment one-liner with output indicators
568  if (!(m_it % g_inputdeck.get<tag::interval,tag::field>())) m_print << 'F';
569  if (diag) m_print << 'D';
570 
571  m_print << std::endl;
572  }
573  }
574 
575  wait4eval();
576 
577  // if neither max iterations nor max time reached, will continue (by telling
578  // all linear system merger group elements to prepare for a new rhs),
579  // otherwise finish
580  if (std::fabs(m_t-term) > eps && m_it < nstep) {
581  m_linsysmerger.enable_wait4rhs();
582  if ( g_inputdeck.get< tag::cmd, tag::feedback >() )
583  m_progStep.start( "Time step stage ..." );
584  } else
585  finish();
586 }
