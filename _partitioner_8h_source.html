199  //! Receive number of uniquely assigned global mesh node IDs from lower PEs
200  //! \details This function computes the offset each PE will need to start
201  //! assigning its new node IDs from (for those nodes that are not assigned
202  //! new IDs by any PEs with lower indices). The offset for a PE is the
203  //! offset for the previous PE plus the number of node IDs the previous PE
204  //! (uniquely) assigns new IDs for minus the number of node IDs the
205  //! previous PE receives from others (lower PEs). This is computed here in
206  //! a parallel/distributed fashion by each PE sending its number of node
207  //! IDs (that it uniquely assigns) to all PEs. Note that each PE would
208  //! only need to send this information to higher PEs, but instead this
209  //! function is called in a broadcast fashion, because that is more
210  //! efficient than individual calls to only the higher PEs. Therefore when
211  //! computing the offsets, we only count the lower PEs. When this is done,
212  //! we have the precise communication map as well as the start offset on
213  //! all PEs and so we can start the distributed global mesh node ID
214  //! reordering.
215  void offset( int p, std::size_t u ) {
216  if (p < CkMyPe()) m_start += u;
217  if (++m_noffset == static_cast<std::size_t>(CkNumPes())) reorder();
218  }

303  //! \brief Query our global node IDs by other PEs so they know if they
304  //! receive IDs for those from during reordering
305  //! \param[in] p Querying PE
306  //! \param[in] id Vector of global mesh node IDs to query
307  //! \details Note that every PE calls this function in a broadcast fashion,
308  //! including our own. However, to compute the correct result, this would
309  //! only be necessary for PEs whose ID is higher than ours. However, the
310  //! broadcast (calling everyone) is more efficient. This also results in a
311  //! simpler logic, because every PE goes through this single call path.
312  //! The returned mask is simply a boolean array signaling if the node ID
313  //! is found (owned).
314  void query( int p, const std::set< std::size_t >& id ) const {
315  std::unordered_map< std::size_t, std::vector< int > > ch;
316  for (auto j : id) {
317  const auto it = m_id.find( j );
318  if (it != end(m_id)) {
319  const auto& c = tk::cref_find( m_ch, j );
320  auto& chares = ch[j];
321  chares.insert( end(chares), begin(c), end(c) );
322  }
323  }
324  Group::thisProxy[ p ].mask( CkMyPe(), ch );
325  }

327  //! Receive mask of to-be-received global mesh node IDs
328  //! \param[in] p The PE uniquely assigns the node IDs marked listed in ch
329  //! \param[in] ch Vector containing the set of potentially multiple chare
330  //! IDs that we own (i.e., contribute to) for all of our node IDs.
331  //! \details Note that every PE will call this function, since query() was
332  //! called in a broadcast fashion and query() answers to every PE once.
333  //! This is more efficient than calling only the PEs from which we would
334  //! have to receive results from. Thus the incoming results are only
335  //! interesting from PEs with lower IDs than ours.
336  void mask( int p, const std::unordered_map< std::size_t,
337  std::vector< int > >& ch )
338  {
339  // Store the old global mesh node IDs associated to chare IDs bordering
340  // the mesh chunk held by and associated to chare IDs we own
341  for (const auto& h : ch) {
342  const auto& chares = tk::ref_find( m_ch, h.first );
343  for (auto c : chares) { // surrounded chares
344  auto& sch = m_msum[c];
345  for (auto s : h.second) // surrounding chares
346  if (s != c) sch[s].push_back( h.first );
347  }
348  }
349  // Associate global mesh node IDs to lower PEs we will need to receive
350  // from during node reordering. The choice of associated container is
351  // std::map, which is ordered (vs. unordered, hash-map). This is required
352  // by the following operation that makes the mesh node IDs unique in the
353  // communication map. (We are called in an unordered fashion, so we need
354  // to collect from all PEs and then we need to make the node IDs unique,
355  // keeping only the lowest PEs a node ID is associated with.)
356  if (p < CkMyPe()) {
357  auto& id = m_comm[ p ];
358  for (const auto& h : ch) id.insert( h.first );
359  }
360  if (++m_nquery == static_cast<std::size_t>(CkNumPes())) {
361  // Make sure we have received all we need
362  Assert( m_comm.size() == static_cast<std::size_t>(CkMyPe()),
363  "Communication map size on PE " +
364  std::to_string(CkMyPe()) + " must equal " +
365  std::to_string(CkMyPe()) );
366  // Fill new hash-map, keeping only unique node IDs obtained from the
367  // lowest possible PEs
368  for (auto c=begin(m_comm); c!=end(m_comm); ++c) {
369  auto& u = m_communication[ c->first ];
370  for (auto j : c->second)
371  if (std::none_of( begin(m_comm), c,
372  [j](const typename decltype(m_comm)::value_type& s)
373  { return s.second.find(j) != end(s.second); } )) {
374  u.insert(j);
375  }
376  if (u.empty()) m_communication.erase( c->first );
377  }
378  // Free storage of temporary communication map used to receive global
379  // mesh node IDs as it is no longer needed once the final communication
380  // map is generated.
381  tk::destroy( m_comm );
382  // Count up total number of nodes we will need receive during reordering
383  std::size_t nrecv = 0;
384  for (const auto& u : m_communication) nrecv += u.second.size();
385  // send progress report to host
386  if ( g_inputdeck.get< tag::cmd, tag::feedback >() )
387  m_host.pemask();
388  // Start computing PE offsets for node reordering
389  Group::thisProxy.offset( CkMyPe(), m_id.size()-nrecv );
390  }
391  }

479  //! Read our contiguously-numbered chunk of the mesh graph from file
480  //! \param[in] er ExodusII mesh reader
481  void readGraph( tk::ExodusIIMeshReader& er ) {
482  // Get number of mesh points and number of tetrahedron elements in file
483  er.readElemBlockIDs();
484  auto nel = er.nelem( tk::ExoElemType::TET );
485  // Read our contiguously-numbered chunk of tetrahedron element
486  // connectivity from file and also generate and store the list of global
487  // element indices for our chunk of the mesh
488  auto npes = static_cast< std::size_t >( CkNumPes() );
489  auto mype = static_cast< std::size_t >( CkMyPe() );
490  auto chunk = nel / npes;
491  auto from = mype * chunk;
492  auto till = from + chunk;
493  if (mype == npes-1) till += nel % npes;
494  er.readElements( {{from, till-1}}, tk::ExoElemType::TET, m_tetinpoel );
495  m_gelemid.resize( till-from );
496  std::iota( begin(m_gelemid), end(m_gelemid), from );
497  // send progress report to host
498  if ( g_inputdeck.get< tag::cmd, tag::feedback >() )
499  m_host.pegraph();
500  signal2host_graph_complete( m_host, m_gelemid.size() );
501  }

503  // Compute element centroid coordinates
504  //! \param[in] er ExodusII mesh reader
505  void computeCentroids( tk::ExodusIIMeshReader& er) {
506  // Construct unique global mesh point indices of our chunk
507  auto gid = m_tetinpoel;
508  tk::unique( gid );
509  // Read node coordinates of our chunk of the mesh elements from file
510  auto ext = tk::extents( gid );
511  auto coord = er.readNodes( ext );
512  const auto& x = std::get< 0 >( coord );
513  const auto& y = std::get< 1 >( coord );
514  const auto& z = std::get< 2 >( coord );
515  // Make room for element centroid coordinates
516  auto& cx = m_centroid[0];
517  auto& cy = m_centroid[1];
518  auto& cz = m_centroid[2];
519  auto num = m_tetinpoel.size()/4;
520  cx.resize( num );
521  cy.resize( num );
522  cz.resize( num );
523  // Compute element centroids for our chunk of the mesh elements
524  for (std::size_t e=0; e<num; ++e) {
525  auto A = m_tetinpoel[e*4+0] - ext[0];
526  auto B = m_tetinpoel[e*4+1] - ext[0];
527  auto C = m_tetinpoel[e*4+2] - ext[0];
528  auto D = m_tetinpoel[e*4+3] - ext[0];
529  cx[e] = (x[A] + x[B] + x[C] + x[D]) / 4.0;
530  cy[e] = (y[A] + y[B] + y[C] + y[D]) / 4.0;
531  cz[e] = (z[A] + z[B] + z[C] + z[D]) / 4.0;
532  }
533  signal2host_setup_complete( m_host );
534  }

536  //! Construct global mesh node ids for each chare
537  //! \param[in] che Chares of elements: array of chare ownership IDs mapping
538  //! graph elements to Charm++ chares. Size: number of elements in the
539  //! chunk of the mesh graph on this PE.
540  //! \return Vector of global mesh node ids connecting elements owned by each
541  //! chare on this PE
542  //! \note The chare IDs, as keys in the map constructed here, are simply the
543  //! chare IDs returned by the partitioner assigning mesh elements to these
544  //! chares. It does not mean that these chare IDs are owned on this PE.
545  std::unordered_map< int, std::vector< std::size_t > >
546  chareNodes( const std::vector< std::size_t >& che ) const
547  {
548  Assert( che.size() == m_gelemid.size(), "The size of the global element "
549  "index and the chare element arrays must equal" );
550  Assert( che.size() == m_tetinpoel.size()/4, "The size of the mesh "
551  "connectivity / 4 and the chare element arrays must equal" );
552  // Categorize global mesh node ids of elements by chares
553  std::unordered_map< int, std::vector< std::size_t > > nodes;
554  for (std::size_t e=0; e<che.size(); ++e) {
555  auto& c = nodes[ static_cast<int>(che[e]) ];
556  for (std::size_t n=0; n<4; ++n) c.push_back( m_tetinpoel[e*4+n] );
557  }
558  // Make sure all PEs have chares assigned
559  Assert( !nodes.empty(), "No nodes have been assigned to chares on PE " +
560  std::to_string(CkMyPe()) );
561  // This check should always be done, hence ErrChk and not Assert, as it
562  // can result from particular pathological combinations of (1) too large
563  // degree of virtualization, (2) too many PEs, and/or (3) too small of a
564  // mesh and not due to programmer error.
565  for(const auto& c : nodes)
566  ErrChk( !c.second.empty(),
567  "Overdecomposition of the mesh is too large compared to the "
568  "number of work units computed based on the degree of "
569  "virtualization desired. As a result, there would be at least "
570  "one work unit with no mesh elements to work on, i.e., nothing "
571  "to do. Solution 1: decrease the virtualization to a lower "
572  "value using the command-line argument '-u'. Solution 2: "
573  "decrease the number processing elements (PEs) using the "
574  "charmrun command-line argument '+pN' where N is the number of "
575  "PEs, which implicitly increases the size (and thus decreases "
576  "the number) of work units.)" );
577  return nodes;
578  }

580  //! Distribute global mesh node IDs to their owner PEs
581  //! \param[in] n Global mesh node IDs connecting elements associated to
582  //! chare IDs on this PE resulting from partitioning the mesh elements.
583  //! Note that this data is moved in.
584  //! \details Chare ids are distributed to PEs in a linear continguous order
585  //! with the last PE taking the remainder if the number of PEs is not
586  //! divisible by the number chares. For example, if nchare=7 and npe=3,
587  //! the chare distribution is PE0: 0 1, PE1: 2 3, and PE2: 4 5 6. As a
588  //! result of this distribution, all PEs will have their m_node map filled
589  //! with the global mesh node IDs associated to the Charm++ chare IDs each
590  //! PE owns.
591  void distribute( std::unordered_map< int, std::vector< std::size_t > >&& n )
592  {
593  int chunksize, mynchare;
594  std::tie( chunksize, mynchare ) = chareDistribution();
595  for (int c=0; c<mynchare; ++c) {
596  auto chid = CkMyPe() * chunksize + c; // compute owned chare ID
597  const auto it = n.find( chid ); // attempt to find its nodes
598  if (it != end(n)) { // if found
599  m_node.insert( *it ); // move over owned key-value pairs
600  n.erase( it ); // remove chare ID and nodes
601  }
602  Assert( n.find(chid) == end(n), "Not all owned node IDs stored" );
603  }
604  // Construct export map associating those map entries (mesh node indices
605  // associated to chare IDs) owned by chares we do not own. Outer key: PE
606  // to export to, inner key: chare ID, value: vector of global node IDs
607  std::unordered_map< int,
608  std::unordered_map< int, std::vector< std::size_t > > > exp;
609  for (auto&& c : n) exp[ pe(c.first) ].insert( std::move(c) );
610  // Export chare IDs and node IDs we do not own to fellow PEs
611  m_npe = exp.size();
612  for (const auto& p : exp)
613  Group::thisProxy[ p.first ].add( CkMyPe(), p.second );
614  // send progress report to host
615  if ( g_inputdeck.get< tag::cmd, tag::feedback >() )
616  m_host.pedistributed();
617  if (m_npe == 0) signal2host_distributed( m_host );
618  }

620  //! Compute chare distribution
621  //! \return Chunksize, i.e., number of chares per all PEs except the last
622  //! one, and the number of chares for my PE
623  //! \details This computes a simple contiguous chare distribution across
624  //! PEs.
625  std::pair< int, int > chareDistribution() const {
626  auto chunksize = m_nchare / CkNumPes();
627  auto mynchare = chunksize;
628  if (CkMyPe() == CkNumPes()-1) mynchare += m_nchare % CkNumPes();
629  return { chunksize, mynchare };

632  //! Reorder global mesh node IDs
633  void reorder() {
634  // Activate SDAG waits for completing the reordering of our node IDs and
635  // for having requests arrive from other PEs for some of our node IDs; and
636  // for computing/receiving lower and upper bounds of global node IDs our
637  // PE operates on after reordering
638  wait4prep();
639  wait4bounds();
640  // In serial signal to the runtime system that we have participated in
641  // reordering. This is required here because this is only triggered if
642  // communication is required during mesh node reordering. See also
643  // particioner.ci.
644  if (CkNumPes() == 1) participated_complete();
645  // Send out request for new global node IDs for nodes we do not reorder
646  for (const auto& c : m_communication)
647  Group::thisProxy[ c.first ].request( CkMyPe(), c.second );
648  // Lambda to decide if node ID is being assigned a new ID by us
649  auto own = [ this ]( std::size_t p ) {
650  using Set = typename std::remove_reference<
651  decltype(m_communication) >::type::value_type;
652  return !std::any_of( m_communication.cbegin(), m_communication.cend(),
653  [&](const Set& s)
654  { return s.second.find(p) != s.second.cend(); } );
655  };
656  // Reorder our chunk of the mesh node IDs by looping through all of our
657  // node IDs (resulting from reading our chunk of the mesh cells). We test
658  // if we are to assign a new ID to a node ID, and if so, we assign new ID,
659  // i.e., reorder, by constructing a map associating new to old IDs. We
660  // also count up the reordered nodes.
661  for (auto p : m_id) if (own(p)) m_newid[ p ] = m_start++;
662  // Trigger SDAG wait, indicating that reordering own node IDs are complete
663  reorderowned_complete();
664  // If we have reordered all our nodes, compute and send result to host
665  if (m_newid.size() == m_id.size()) reordered();
666  }

668  //! Return processing element for chare id
669  //! \param[in] id Chare id
670  //! \return PE that creates the chare
671  //! \details This is computed based on a simple contiguous linear
672  //! distribution of chare ids to PEs.
673  int pe( int id ) const {
674  auto p = id / (m_nchare / CkNumPes());
675  if (p >= CkNumPes()) p = CkNumPes()-1;
676  Assert( p < CkNumPes(), "Assigning to nonexistent PE" );
677  return p;
678  }

680  //! Associate new node IDs to old ones and return them to the requestor(s)
681  void prepare() {
682  // Signal to the runtime system that we have participated in reordering
683  participated_complete();
684  for (const auto& r : m_req) {
685  std::unordered_map< std::size_t, std::size_t > n;
686  for (auto p : r.second) n[ p ] = tk::cref_find( m_newid, p );
687  Group::thisProxy[ r.first ].neworder( n );
688  tk::destroy( n );
689  }
690  tk::destroy(m_req); // Clear queue of requests just fulfilled
691  wait4prep(); // Re-enable SDAG wait for preparing new node requests
692  // Re-enable trigger signaling that reordering of owned node IDs are
693  // complete right away
694  reorderowned_complete();
695  }

697  //! Compute final result of reordering
698  //! \details This member function is called when both those node IDs that we
699  //! assign a new ordering to as well as those assigned new IDs by other
700  //! PEs have been reordered (and we contribute to) and we are ready (on
701  //! this PE) to compute our final result of the reordering.
702  void reordered() {
703  // Free storage of communication map used for distributed mesh node
704  // reordering as it is no longer needed after reordering.
705  tk::destroy( m_communication );
706  // Construct maps associating old node IDs (as in file) to new node IDs
707  // (as in producing contiguous-row-id linear system contributions)
708  // associated to chare IDs (outer key). This is basically the inverse of
709  // m_newid and categorized by chares. Note that m_node at this point still
710  // contains the old global node IDs the chares contribute to.
711  for (const auto& c : m_node) {
712  auto& m = m_chcid[ c.first ];
713  for (auto p : c.second) m[ tk::cref_find(m_newid,p) ] = p;
714  }
715  // Update our chare ID maps to now contain the new global node IDs
716  // instead of the old ones
717  for (auto& c : m_node)
718  for (auto& p : c.second)
719  p = tk::cref_find( m_newid, p );
720  // Update old global mesh node IDs to new ones associated to chare IDs
721  // bordering the mesh chunk held by and associated to chare IDs we own
722  for (auto& c : m_msum)
723  for (auto& s : c.second) {
724  for (auto& p : s.second) p = tk::cref_find( m_newid, p );
725  std::sort( begin(s.second), end(s.second) );
726  }
727  // Update unique global node IDs of chares our PE will contribute to to
728  // now contain the new IDs resulting from reordering
729  decltype(m_id) nid;
730  for (auto p : m_id) nid.insert( tk::cref_find( m_newid, p ) );
731  m_id = std::move( nid );
732  tk::destroy( nid );
733  // send progress report to host
734  if ( g_inputdeck.get< tag::cmd, tag::feedback >() )
735  m_host.pereordered();
736  // Compute lower and upper bounds of reordered node IDs our PE operates on
737  bounds();
738  }

740  // Compute lower and upper bounds of reordered node IDs our PE operates on
741  // \details This function computes the global row IDs at which the linear
742  // system will have a PE boundary. We simply find the largest node ID
743  // assigned on each PE by the reordering and use that as the upper global
744  // row index. Note that while this rarely results in equal number of rows
745  // assigned to PEs, potentially resulting in some load-imbalance, it
746  // yields a pretty good division reducing communication costs during the
747  // assembly of the linear system, which is more important than a slight
748  // (FLOP) load imbalance. Since the upper index for PE 1 is the same as
749  // the lower index for PE 2, etc. We build the upper indices and then the
750  // lower indices for all PEs are communicated.
751  void bounds() {
752  m_upper = 0;
753  using pair_type = std::pair< const std::size_t, std::size_t >;
754  for (const auto& c : m_chcid) {
755  auto x = std::max_element( begin(c.second), end(c.second),
756  []( const pair_type& a, const pair_type& b )
757  { return a.first < b.first; } );
758  if (x->first > m_upper) m_upper = x->first;
759  }
760  // The bounds are the dividers (global mesh point indices) at which the
761  // linear system assembly is divided among PEs. However, Hypre and thus
762  // LinSysMerger expect exclusive upper indices, so we increase the last
763  // one by one here. Note that the cost calculation, Partitioner::cost()
764  // also expects exclusive upper indices.
765  if (CkMyPe() == CkNumPes()-1) ++m_upper;
766  // Tell the runtime system that the upper bound has been computed
767  upper_complete();
768  // Set lower index for PE 0 as 0
769  if (CkMyPe() == 0) lower(0);
770  // All PEs except the last one send their upper indices as the lower index
771  // for PE+1
772  if (CkMyPe() < CkNumPes()-1)
773  Group::thisProxy[ CkMyPe()+1 ].lower( m_upper );
774  }

776  //! \brief Create chare array elements on this PE and assign the global mesh
777  //! element IDs they will operate on
778  //! \details We create chare array elements by calling the insert() member
779  //! function, which allows specifying the PE on which the array element is
780  //! created and we send each chare array element the global mesh element
781  //! connectivity, i.e., node IDs, it contributes to and the old->new node

835  //! Compute communication cost of linear system merging for our PE
836  //! \param[in] l Lower global row ID of linear system this PE works on
837  //! \param[in] u Upper global row ID of linear system this PE works on
838  //! \return Communication cost of merging the linear system for our PE
839  //! \details The cost is a real number between 0 and 1, defined as the
840  //! number of mesh points we do not own, i.e., need to send to some other
841  //! PE, divided by the total number of points we contribute to. The lower
842  //! the better.
843  tk::real cost( std::size_t l, std::size_t u ) {
844  std::size_t ownpts = 0, compts = 0;
845  for (auto p : m_id) if (p >= l && p < u) ++ownpts; else ++compts;
846  // Free storage of unique global node IDs chares on our PE will contribute
847  // to in a linear system as it is no longer needed after computing the
848  // communication cost.
849  tk::destroy( m_id );
850  return static_cast<tk::real>(compts) /
851  static_cast<tk::real>(ownpts + compts);
852  }
