208  //! Receive number of uniquely assigned global mesh node IDs from lower PEs
209  //! \details This function computes the offset each PE will need to start
210  //! assigning its new node IDs from (for those nodes that are not assigned
211  //! new IDs by any PEs with lower indices). The offset for a PE is the
212  //! offset for the previous PE plus the number of node IDs the previous PE
213  //! (uniquely) assigns new IDs for minus the number of node IDs the
214  //! previous PE receives from others (lower PEs). This is computed here in
215  //! a parallel/distributed fashion by each PE sending its number of node
216  //! IDs (that it uniquely assigns) to all PEs. Note that each PE would
217  //! only need to send this information to higher PEs, but instead this
218  //! function is called in a broadcast fashion, because that is more
219  //! efficient than individual calls to only the higher PEs. Therefore when
220  //! computing the offsets, we only count the lower PEs. When this is done,
221  //! we have the precise communication map as well as the start offset on
222  //! all PEs and so we can start the distributed global mesh node ID
223  //! reordering.
224  void offset( int p, std::size_t u ) {
225  if (p < CkMyPe()) m_start += u;
226  if (++m_noffset == static_cast<std::size_t>(CkNumPes())) reorder();
227  }

291  //! \details The 'flatten' is used here as a concatenation of a data
292  //! structure that stores date categorized by chares owned on this PE. The
293  //! result of the flattening is thus a simpler data structure that is no
294  //! longer categorized by (or associated to) chares.
295  void flatten() {
296  // Optionally refine mesh if requested
297  const auto ir = g_inputdeck.get< tag::selected, tag::initialamr >();
298  if (ir == tk::ctr::InitialAMRType::UNIFORM) refine();
299  // Make sure we are not fed garbage

301  static_cast< std::size_t >( chareDistribution()[1] ),
302  "Global mesh nodes ids associated to chares on PE " +
303  std::to_string( CkMyPe() ) + " is incomplete" );
304  // Collect chare IDs we own associated to old global mesh node IDs
305  for (const auto& c : m_chinpoel)
306  for (auto p : c.second)
307  m_nodechares[p].push_back( c.first );
308  // Make chare IDs (associated to old global mesh node IDs) unique
309  for (auto& c : m_nodechares) tk::unique( c.second );
310  // Collect chare IDs we own associated to edges
311  for (const auto& c : m_chedgenodes)
312  for (const auto& e : c.second)

314  // Make chare IDs (associated to edges) unique
315  for (auto& c : m_edgechares) tk::unique( c.second );

317  for (const auto& c : m_chinpoel)

321  for (const auto& c : m_chedgenodes)

326  m_host.peflattened();
327  // Signal host that we are ready for computing the communication map,
328  // required for parallel distributed global mesh node reordering
329  signal2host_flattened( m_host );

352  //! \brief Query our global node IDs and edges by other PEs so they know if
353  //! they are to receive IDs for those from during reordering
354  //! \param[in] p Querying PE
355  //! \param[in] nodes List of global mesh node IDs to query
356  //! \param[in] edges List of edges to query
357  //! \details Note that every PE calls this function in a broadcast fashion,
358  //! including our own. However, to compute the correct result, this would
359  //! only be necessary for PEs whose ID is higher than ours. However, the
360  //! broadcast (calling everyone) is more efficient. This also results in a
361  //! simpler logic, because every PE goes through this single call path.
362  //! The returned mask is simply a boolean array signaling if the node ID
363  //! is found (owned).
364  void query( int p,
365  const std::set< std::size_t >& nodes,
366  const tk::UnsMesh::Edges& edges ) const
367  {
368  std::unordered_map< std::size_t, std::vector< int > > cn;
369  for (auto j : nodes) {
370  const auto it = m_nodeset.find( j );
371  if (it != end(m_nodeset)) {
372  const auto& c = tk::cref_find( m_nodechares, j );
373  auto& chares = cn[j];
374  chares.insert( end(chares), begin(c), end(c) );
375  }
376  }
377  tk::UnsMesh::EdgeChares ce;
378  for (auto j : edges) {
379  const auto it = m_edgeset.find( j );
380  if (it != end(m_edgeset)) {
381  const auto& c = tk::cref_find( m_edgechares, j );
382  auto& chares = ce[j];
383  chares.insert( end(chares), begin(c), end(c) );
384  }
385  }
386  Group::thisProxy[ p ].mask( CkMyPe(), cn, ce );
387  }

389  //! Receive mask of to-be-received global mesh node IDs
390  //! \param[in] p The PE uniquely assigns the node IDs marked listed in ch
391  //! \param[in] cn Vector containing the set of potentially multiple chare
392  //! IDs that we own (i.e., contribute to) for all of our node IDs.
393  //! \param[in] ce Map associating a vector of chare IDs to edges (at which
394  //! we added nodes during initial mesh refinement)
395  //! \details Note that every PE will call this function, since query() was
396  //! called in a broadcast fashion and query() answers to every PE once.
397  //! This is more efficient than calling only the PEs from which we would
398  //! have to receive results from. Thus the incoming results are only
399  //! interesting from PEs with lower IDs than ours.
400  void mask( int p,
401  const std::unordered_map< std::size_t, std::vector< int > >& cn,
402  const tk::UnsMesh::EdgeChares& ce )
403  {
404  // Store the old global mesh node IDs associated to chare IDs bordering
405  // the mesh chunk held by and associated to chare IDs we own
406  for (const auto& h : cn) {
407  const auto& chares = tk::ref_find( m_nodechares, h.first );
408  for (auto c : chares) { // surrounded chares
409  auto& sch = m_msum[c];
410  for (auto s : h.second) // surrounding chares
411  if (s != c) sch[ s ].insert( h.first );
412  }
413  }
414  // Store the edges associated to chare IDs bordering the mesh chunk held
415  // by and associated to chare IDs we own
416  for (const auto& h : ce) {
417  const auto& chares = tk::ref_find( m_edgechares, h.first );
418  for (auto c : chares) { // surrounded chares
419  auto& sch = m_msumed[c];
420  for (auto s : h.second) // surrounding chares
421  if (s != c) sch[ s ].insert( h.first );
422  }
423  }
424  // Associate global mesh node IDs to lower PEs we will need to receive
425  // from during node reordering. The choice of associated container is
426  // std::map, which is ordered (vs. unordered, hash-map). This is required
427  // by the following operation that makes the mesh node IDs unique in the
428  // communication map. (We are called in an unordered fashion, so we need
429  // to collect from all PEs and then we need to make the node IDs unique,
430  // keeping only the lowest PEs a node ID is associated with.)
431  if (p < CkMyPe()) {
432  auto& id = m_ncomm[ p ];
433  for (const auto& h : cn) id.insert( h.first );
434  auto& ed = m_ecomm[ p ];
435  for (const auto& h : ce) ed.insert( h.first );
436  }
437  if (++m_nquery == static_cast<std::size_t>(CkNumPes())) {
438  // Make sure we have received all we need
439  Assert( m_ncomm.size() == static_cast<std::size_t>(CkMyPe()),
440  "Communication map size on PE " +
441  std::to_string(CkMyPe()) + " must equal " +
442  std::to_string(CkMyPe()) );
443  Assert( m_ecomm.size() == static_cast<std::size_t>(CkMyPe()),
444  "Communication map size on PE " +
445  std::to_string(CkMyPe()) + " must equal " +
446  std::to_string(CkMyPe()) );
447  // Fill new hash-map, keeping only unique node IDs obtained from the
448  // lowest possible PEs
449  for (auto c=m_ncomm.cbegin(); c!=m_ncomm.cend(); ++c) {
450  auto& n = m_ncommunication[ c->first ];
451  for (auto j : c->second)
452  if (std::none_of( m_ncomm.cbegin(), c,
453  [ j ]( const typename decltype(m_ncomm)::value_type& s )
454  { return s.second.find(j) != end(s.second); } )) {
455  n.insert(j);
456  }
457  if (n.empty()) m_ncommunication.erase( c->first );
458  }
459  for (auto c=m_ecomm.cbegin(); c!=m_ecomm.cend(); ++c) {
460  auto& e = m_ecommunication[ c->first ];
461  for (const auto& j : c->second)
462  if (std::none_of( m_ecomm.cbegin(), c,
463  [ &j ]( const typename decltype(m_ecomm)::value_type& s )
464  { return s.second.find(j) != end(s.second); } )) {
465  e.insert(j);
466  }
467  if (e.empty()) m_ecommunication.erase( c->first );
468  }
469  // Free storage of temporary communication map used to receive global
470  // mesh node IDs as it is no longer needed once the final communication
471  // map is generated.
472  tk::destroy( m_ncomm );
473  // Free storage of temporary communication map used to receive global
474  // mesh edge-node IDs as it is no longer needed once the final
475  // communication map is generated.
476  tk::destroy( m_ecomm );
477  // Count up total number of nodes and (nodes associated to edges) we
478  // will need receive during reordering
479  std::size_t nrecv = 0, erecv = 0;
480  for (const auto& u : m_ncommunication) nrecv += u.second.size();
481  for (const auto& e : m_ecommunication) erecv += e.second.size();
482  // send progress report to host
483  if ( g_inputdeck.get< tag::cmd, tag::feedback >() )
484  m_host.pemask();
485  // Compute number of mesh node IDs we will assign IDs to
486  auto nuniq = m_nodeset.size() - nrecv + m_edgeset.size() - erecv;
487  // Start computing PE offsets for node reordering
488  Group::thisProxy.offset( CkMyPe(), nuniq );
489  }

777  //! Reorder global mesh node IDs
778  void reorder() {
779  // Activate SDAG waits for having requests arrive from other PEs for some
780  // of our node IDs; and for computing/receiving lower and upper bounds of
781  // global node IDs our PE operates on after reordering
782  wait4prep();
783  wait4bounds();
784  wait4reorder();
785  // In serial signal to the runtime system that we have participated in
786  // reordering. This is required here because this is only triggered if
787  // communication is required during mesh node reordering. See also
788  // particioner.ci.
789  if (CkNumPes() == 1) participated_complete();
790  // Send out request for new global node IDs for nodes we do not reorder
791  for (const auto& c : m_ncommunication)
792  Group::thisProxy[ c.first ].request( CkMyPe(), c.second );
793  // Send out request for new global node IDs for edges we do not reorder
794  for (const auto& e : m_ecommunication)
795  Group::thisProxy[ e.first ].request( CkMyPe(), e.second );
796  // Lambda to decide if node ID is being assigned a new ID by us
797  auto ownnode = [ this ]( std::size_t p ) {
798  using Set = typename std::remove_reference<
799  decltype(m_ncommunication) >::type::value_type;
800  return !std::any_of( m_ncommunication.cbegin(), m_ncommunication.cend(),
801  [&](const Set& s)
802  { return s.second.find(p) != s.second.cend(); } );
803  };
804  // Lambda to decide if edge-node ID is being assigned a new ID by us
805  auto ownedge = [ this ]( const tk::UnsMesh::Edge& e ) {
806  using Set = typename std::remove_reference<
807  decltype(m_ecommunication) >::type::value_type;
808  return !std::any_of( m_ecommunication.cbegin(), m_ecommunication.cend(),
809  [&](const Set& s)
810  { return s.second.find(e) != s.second.cend(); } );
811  };
812  // Reorder our chunk of the mesh node IDs by looping through all of our
813  // node IDs (resulting from reading our chunk of the mesh cells). We test
814  // if we are to assign a new ID to a node ID, and if so, we assign new ID,
815  // i.e., reorder, by constructing a map associating new to old IDs. We
816  // also count up the reordered nodes, which also serves as the new node
817  // id.
818  for (auto p : m_nodeset)
819  if (ownnode(p)) m_linnodes[ p ] = m_start++;
820  // Reorder our chunk of the mesh edges by looping through all of our edges
821  // (resulting from initial uniform refinement of our chunk of the mesh
822  // cells). We test if we are to assign a new ID to an edge, and if so, we
823  // assign new ID, i.e., reorder, by constructing a map associating new
824  // node IDs to edges. We also count up the reordered edge-nodes, which
825  // also serves as the new node id.
826  for (const auto& e : m_edgeset)
827  if (ownedge(e)) m_linedges[ e ] = m_start++;
828  // Trigger SDAG wait indicating that reordering own node IDs are complete
829  reorderowned_complete();
830  // If all our nodes have new IDs assigned, signal that to the runtime
831  if (m_linnodes.size() == m_nodeset.size()) nodesreorder_complete();
832  // If all our edges have new IDs assigned, signal that to the runtime
833  if (m_linedges.size() == m_edgeset.size()) edgesreorder_complete();
834  }
