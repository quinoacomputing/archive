43 // Generate derived data structure, elements surrounding points
44 //! \param[in] inpoel Inteconnectivity of points and elements. These are the
45 //! node ids of each element of an unstructured mesh. Example:
46 //! \code{.cpp}
47 //! std::vector< std::size_t > inpoel { 12, 14, 9, 11,
48 //! 10, 14, 13, 12 };
49 //! \endcode
50 //! specifies two tetrahedra whose vertices (node ids) are { 12, 14, 9, 11 },
51 //! and { 10, 14, 13, 12 }.
52 //! \param[in] nnpe Number of nodes per element
53 //! \return Linked lists storing elements surrounding points
54 //! \warning It is not okay to call this function with an empty container or a
55 //! non-positive number of nodes per element; it will throw an exception.
56 //! \details The data generated here is stored in a linked list, more precisely,
57 //! two linked arrays (vectors), _esup1_ and _esup2_, where _esup2_ holds the
58 //! indices at which _esup1_ holds the element ids surrounding points. Looping
59 //! over all elements surrounding all points can then be accomplished by the
60 //! following loop:
61 //! \code{.cpp}
62 //! for (std::size_t p=0; p<npoin; ++p)
63 //! for (auto i=esup.second[p]+1; i<=esup.second[p+1]; ++i)
64 //! use element id esup.first[i]
65 //! \endcode
66 //! To find out the number of points, _npoin_, the mesh connectivity,
67 //! _inpoel_, can be queried:
68 //! \code{.cpp}
69 //! auto minmax = std::minmax_element( begin(inpoel), end(inpoel) );
70 //! Assert( *minmax.first == 0, "node ids should start from zero" );
71 //! auto npoin = *minmax.second + 1;
72 //! \endcode
73 //! \note In principle, this function *should* work for any positive nnpe,
74 //! however, only nnpe = 4 (tetrahedra) and nnpe = 3 (triangles) are tested.
75 //! \see Lohner, An Introduction to Applied CFD Techniques, Wiley, 2008
76 // *****************************************************************************
77 {
78  Assert( !inpoel.empty(), "Attempt to call genEsup() on empty container" );
79  Assert( nnpe > 0, "Attempt to call genEsup() with zero nodes per element" );
80  Assert( inpoel.size()%nnpe == 0, "Size of inpoel must be divisible by nnpe" );
81 
82  // find out number of points in mesh connectivity
83  auto minmax = std::minmax_element( begin(inpoel), end(inpoel) );
84  Assert( *minmax.first == 0, "node ids should start from zero" );
85  auto npoin = *minmax.second + 1;
86 
87  // allocate one of the linked lists storing elements surrounding points: esup2
88  // fill with zeros
89  std::vector< std::size_t > esup2( npoin+1, 0 );
90 
91  // element pass 1: count number of elements connected to each point
92  for (auto n : inpoel) ++esup2[ n + 1 ];
93 
94  // storage/reshuffling pass 1: update storage counter and store
95  // also find out the maximum size of esup1 (mesup)
96  auto mesup = esup2[0]+1;
97  for (std::size_t i=1; i<npoin+1; ++i) {
98  esup2[i] += esup2[i-1];
99  if (esup2[i]+1 > mesup) mesup = esup2[i]+1;
100  }
101 
102  // now we know mesup, so allocate the other one of the linked lists storing
103  // elements surrounding points: esup1
104  std::vector< std::size_t > esup1( mesup );
105 
106  // store the elements in esup1
107  std::size_t e = 0;
108  for (auto n : inpoel) {
109  auto j = esup2[n]+1;
110  esup2[n] = j;
111  esup1[j] = e/nnpe;
112  ++e;
113  }
114 
115  // storage/reshuffling pass 2
116  for (auto i=npoin; i>0; --i) esup2[i] = esup2[i-1];
117  esup2[0] = 0;
118 
119  // Return (move out) linked lists
120  return std::make_pair( std::move(esup1), std::move(esup2) );
121 }

131 //! node ids of each element of an unstructured mesh. Example:
132 //! \code{.cpp}
133 //! std::vector< std::size_t > inpoel { 12, 14, 9, 11,
134 //! 10, 14, 13, 12 };
135 //! \endcode
136 //! specifies two tetrahedra whose vertices (node ids) are { 12, 14, 9, 11 },
137 //! and { 10, 14, 13, 12 }.
138 //! \param[in] nnpe Number of nodes per element
139 //! \param[in] esup Elements surrounding points as linked lists, see tk::genEsup
140 //! \return Linked lists storing points surrounding points
141 //! \warning It is not okay to call this function with an empty container for
142 //! inpoel or esup.first or esup.second or a non-positive number of nodes per
143 //! element; it will throw an exception.
144 //! \details The data generated here is stored in a linked list, more precisely,
145 //! two linked arrays (vectors), _psup1_ and _psup2_, where _psup2_ holds the
146 //! indices at which _psup1_ holds the point ids surrounding points. Looping
147 //! over all points surrounding all points can then be accomplished by the
148 //! following loop:
149 //! \code{.cpp}
150 //! for (std::size_t p=0; p<npoin; ++p)
151 //! for (auto i=psup.second[p]+1; i<=psup.second[p+1]; ++i)
152 //! use point id psup.first[i]
153 //! \endcode
154 //! To find out the number of points, _npoin_, the mesh connectivity,
155 //! _inpoel_, can be queried:
156 //! \code{.cpp}
157 //! auto minmax = std::minmax_element( begin(inpoel), end(inpoel) );
158 //! Assert( *minmax.first == 0, "node ids should start from zero" );
159 //! auto npoin = *minmax.second + 1;
160 //! \endcode
161 //! or the length-1 of the generated index list:
162 //! \code{.cpp}
163 //! auto npoin = psup.second.size()-1;
164 //! \endcode
165 //! \note In principle, this function *should* work for any positive nnpe,
166 //! however, only nnpe = 4 (tetrahedra) and nnpe = 3 (triangles) are tested.
167 //! \see Lohner, An Introduction to Applied CFD Techniques, Wiley, 2008
168 // *****************************************************************************
169 {
170  Assert( !inpoel.empty(), "Attempt to call genPsup() on empty container" );
171  Assert( nnpe > 0, "Attempt to call genPsup() with zero nodes per element" );
172  Assert( inpoel.size()%nnpe == 0, "Size of inpoel must be divisible by nnpe" );
173  Assert( !esup.first.empty(), "Attempt to call genPsup() with empty esup1" );
174  Assert( !esup.second.empty(), "Attempt to call genPsup() with empty esup2" );
175 
176  // find out number of points in mesh connectivity
177  auto minmax = std::minmax_element( begin(inpoel), end(inpoel) );
178  Assert( *minmax.first == 0, "node ids should start from zero" );
179  auto npoin = *minmax.second + 1;
180 
181  auto& esup1 = esup.first;
182  auto& esup2 = esup.second;
183 
184  // allocate both of the linked lists storing points surrounding points, we
185  // only know the size of psup2, put in a single zero in psup1
186  std::vector< std::size_t > psup2( npoin+1 ), psup1( 1, 0 );
187 
188  // allocate and fill with zeros a temporary array, only used locally
189  std::vector< std::size_t > lpoin( npoin, 0 );
190 
191  // fill both psup1 and psup2
192  psup2[0] = 0;
193  std::size_t j = 0;
194  for (std::size_t p=0; p<npoin; ++p) {
195  for (std::size_t i=esup2[p]+1; i<=esup2[p+1]; ++i ) {
196  for (std::size_t n=0; n<nnpe; ++n) {
197  auto q = inpoel[ esup1[i] * nnpe + n ];
198  if (q != p && lpoin[q] != p+1) {
199  ++j;
200  psup1.push_back( q );
201  lpoin[q] = p+1;
202  }
203  }
204  }
205  psup2[p+1] = j;
206  }
207 
208  // sort point ids for each point in psup1
209  for (std::size_t p=0; p<npoin; ++p)
210  std::sort(
211  std::next( begin(psup1), static_cast<std::ptrdiff_t>(psup2[p]+1) ),
212  std::next( begin(psup1), static_cast<std::ptrdiff_t>(psup2[p+1]+1) ) );
213 
214  // Return (move out) linked lists
215  return std::make_pair( std::move(psup1), std::move(psup2) );
216 }

228 //! std::vector< std::size_t > inpoel { 12, 14, 9, 11,

230 //! \endcode
231 //! specifies two tetrahedra whose vertices (node ids) are { 12, 14, 9, 11 },
232 //! and { 10, 14, 13, 12 }.
233 //! \param[in] nnpe Number of nodes per element (3 or 4)
234 //! \param[in] esup Elements surrounding points as linked lists, see tk::genEsup
235 //! \return Linked lists storing edges (point ids p < q) emanating from points
236 //! \warning It is not okay to call this function with an empty container for
237 //! inpoel or esup.first or esup.second or a non-positive number of nodes per
238 //! element; it will throw an exception.
239 //! \details The data generated here is stored in a linked list, more precisely,
240 //! two linked arrays (vectors), _edsup1_ and _edsup2_, where _edsup2_ holds
241 //! the indices at which _edsup1_ holds the edge-end point ids emanating from
242 //! points for all points. The generated data structure, linked lists edsup1
243 //! and edsup2, are very similar to psup1 and psup2, generated by genPsup(),
244 //! except here only unique edges are stored, i.e., for edges with point ids
245 //! p < q, only ids q are stored that are still associated to point p. Looping
246 //! over all unique edges can then be accomplished by the following loop:
247 //! \code{.cpp}
248 //! for (std::size_t p=0; p<npoin; ++p)
249 //! for (auto i=edsup.second[p]+1; i<=edsup.second[p+1]; ++i)
250 //! use edge with point ids p < edsup.first[i]
251 //! \endcode
252 //! To find out the number of points, _npoin_, the mesh connectivity,
253 //! _inpoel_, can be queried:
254 //! \code{.cpp}
255 //! auto minmax = std::minmax_element( begin(inpoel), end(inpoel) );
256 //! Assert( *minmax.first == 0, "node ids should start from zero" );
257 //! auto npoin = *minmax.second + 1;
258 //! \endcode
259 //! \note At first sight, this function seems to work for elements with more
260 //! vertices than that of tetrahedra. However, that is not the case since the
261 //! algorithm for nnpe > 4 would erronously identify any two combination of
262 //! vertices as a valid edge of an element. Since only triangles and
263 //! tetrahedra have no internal edges, this algorithm only works for triangle
264 //! and tetrahedra element connectivity.
265 //! \see tk::genInpoed for similar data that sometimes may be more advantageous
266 //! \see Lohner, An Introduction to Applied CFD Techniques, Wiley, 2008
267 // *****************************************************************************
268 {
269  Assert( !inpoel.empty(), "Attempt to call genEdsup() on empty container" );
270  Assert( nnpe > 0, "Attempt to call genEdsup() with zero nodes per element" );
271  Assert( nnpe == 3 || nnpe == 4,
272  "Attempt to call genEdsup() with nodes per element, nnpe, that is "
273  "neither 4 (tetrahedra) nor 3 (triangles)." );
274  Assert( inpoel.size()%nnpe == 0, "Size of inpoel must be divisible by nnpe" );
275  Assert( !esup.first.empty(), "Attempt to call genEdsup() with empty esup1" );
276  Assert( !esup.second.empty(), "Attempt to call genEdsup() with empty esup2" );
277 
278  // find out number of points in mesh connectivity
279  auto minmax = std::minmax_element( begin(inpoel), end(inpoel) );
280  Assert( *minmax.first == 0, "node ids should start from zero" );
281  auto npoin = *minmax.second + 1;
282 
283  auto& esup1 = esup.first;
284  auto& esup2 = esup.second;
285 
286  // allocate and fill with zeros a temporary array, only used locally
287  std::vector< std::size_t > lpoin( npoin, 0 );
288 
289  // map to contain stars, a point associated to points connected with edges
290  // storing only the end-point id, q, of point ids p < q
291  std::map< std::size_t, std::vector< std::size_t > > star;
292 
293  // generate edge connectivity and store as stars where center id < spike id
294  for (std::size_t p=0; p<npoin; ++p)
295  for (std::size_t i=esup2[p]+1; i<=esup2[p+1]; ++i )
296  for (std::size_t n=0; n<nnpe; ++n) {
297  auto q = inpoel[ esup1[i] * nnpe + n ];
298  if (q != p && lpoin[q] != p+1) {
299  if (p < q) star[p].push_back(q);
300  lpoin[q] = p+1;
301  }
302  }
303 
304  // linked lists (vectors) to store edges surrounding points and their indices
305  std::vector< std::size_t > edsup1( 1, 0 ), edsup2( 1, 0 );
306 
307  // sort non-center points of each star and store nodes and indices in vectors
308  for (auto p : star) {
309  std::sort( begin(p.second), end(p.second) );
310  edsup2.push_back( edsup2.back() + p.second.size() );
311  for (auto e : p.second) edsup1.push_back( e );
312  }
313  // fill up index array with the last index for points with no new edges
314  for (std::size_t i=0; i<npoin-star.size(); ++i)
315  edsup2.push_back( edsup2.back() );
316 
317  // Return (move out) linked lists
318  return std::make_pair( std::move(edsup1), std::move(edsup2) );
319 }

483  for (auto i=esup2[p]+1; i<=esup2[p+1]; ++i) esuel.insert( esup1[i] );
484  if (++e%nnpe == 0) { // when finished checking all nodes of element
485  // erase element whose surrounding elements are considered
486  esuel.erase( e/nnpe-1 );
487  // store unique element ids in esupel1
488  for (auto i : esuel) esupel1.push_back( i );
489  // store end-index for element used to address into esupel1
490  esupel2.push_back( esupel2.back() + esuel.size() );
491  esuel.clear();
492  }
493  }
494 
495  // Return (move out) linked lists
496  return std::make_pair( std::move(esupel1), std::move(esupel2) );
497 }

499 std::pair< std::vector< std::size_t >, std::vector< std::size_t > >
500 genEsuel( const std::vector< std::size_t >& inpoel,
501  std::size_t nnpe,
502  const std::pair< std::vector< std::size_t >,
503  std::vector< std::size_t > >& esup )
504 // *****************************************************************************
505 // Generate derived data structure, elements surrounding elements
506 //! \param[in] inpoel Inteconnectivity of points and elements. These are the
507 //! node ids of each element of an unstructured mesh. Example:
508 //! \code{.cpp}
509 //! std::vector< std::size_t > inpoel { 12, 14, 9, 11,
510 //! 10, 14, 13, 12 };
511 //! \endcode
512 //! specifies two tetrahedra whose vertices (node ids) are { 12, 14, 9, 11 },
513 //! and { 10, 14, 13, 12 }.
514 //! \param[in] nnpe Number of nodes per element
515 //! \param[in] esup Elements surrounding points as linked lists, see tk::genEsup
516 //! \return Linked lists storing elements surrounding elements
517 //! \warning It is not okay to call this function with an empty container for
518 //! inpoel or esup.first or esup.second; it will throw an exception.
519 //! \details The data generated here is stored in a linked list, more precisely,
520 //! two linked arrays (vectors), _esuel1_ and _esuel2_, where _esuel2_ holds
521 //! the indices at which _esuel1_ holds the element ids surrounding elements.
522 //! Looping over elements surrounding elements can then be accomplished by the
523 //! following loop:
524 //! \code{.cpp}
525 //! for (std::size_t e=0; e<nelem; ++e)
526 //! for (auto i=esuel.second[e]+1; i<=esuel.second[e+1]; ++i)
527 //! use element id esuel.first[i]
528 //! \endcode
529 //! To find out the number of elements, _nelem_, the size of the mesh
530 //! connectivity vector, _inpoel_, can be devided by the number of nodes per
531 //! elements, _nnpe_:
532 //! \code{.cpp}
533 //! auto nelem = inpoel.size()/nnpe;
534 //! \endcode
535 //! \note In principle, this function *should* work for any positive nnpe,
536 //! however, only nnpe = 4 (tetrahedra) and nnpe = 3 (triangles) are tested.
537 //! \see Lohner, An Introduction to Applied CFD Techniques, Wiley, 2008
538 // *****************************************************************************
539 {
540  Assert( !inpoel.empty(), "Attempt to call genEsuel() on empty container" );
541  Assert( nnpe > 0, "Attempt to call genEsuel() with zero nodes per element" );
542  Assert( inpoel.size()%nnpe == 0, "Size of inpoel must be divisible by four" );
543  Assert( !esup.first.empty(), "Attempt to call genEsuel() with empty esuel1" );
544  Assert( !esup.second.empty(),
545  "Attempt to call genEsuel() with empty esuel2" );
546 
547  auto& esup1 = esup.first;
548  auto& esup2 = esup.second;
549 
550  auto nelem = inpoel.size()/nnpe;
551 
552  // lambda that returns true if elements hel and gel share a face
553  auto adj = [ &inpoel, nnpe ]( std::size_t hel, std::size_t gel ) -> bool {
554  std::vector< bool > sp;
555  for (std::size_t h=0; h<nnpe; ++h)
556  for (std::size_t g=0; g<nnpe; ++g)
557  if (inpoel[hel*nnpe+h] == inpoel[gel*nnpe+g]) sp.push_back( true );
558  if (sp.size() == nnpe-1) return true; else return false;
559  };
560 
561  // map to associate unique elements and their surrounding elements
562  std::map< std::size_t, std::vector< std::size_t > > es;
563 
564  for (std::size_t e=0; e<nelem; ++e) {
565  std::set< std::size_t > faces; // will collect elem ids of shared faces
566  for (std::size_t n=0; n<nnpe; ++n) {
567  auto i = inpoel[ e*nnpe+n ];
568  for (auto j=esup2[i]+1; j<=esup2[i+1]; ++j)
569  if (adj( e, esup1[j] )) faces.insert( esup1[j] );
570  }
571  // store element ids of shared faces
572  for (auto j : faces) es[e].push_back(j);
573  }
574 
575  // storing elements surrounding elements
576  std::vector< std::size_t > esuel1( 1, 0 ), esuel2( 1, 0 );
577 
578  // store elements surrounding elements in linked lists
579  for (auto e : es) {
580  esuel2.push_back( esuel2.back() + e.second.size() );
581  for (auto s : e.second) esuel1.push_back( s );
582  }
583 
584  // Return (move out) linked lists
585  return std::make_pair( std::move(esuel1), std::move(esuel2) );
586 }

588 std::vector< std::size_t >
589 genInedel( const std::vector< std::size_t >& inpoel,
590  std::size_t nnpe,
591  const std::vector< std::size_t >& inpoed )
592 // *****************************************************************************
593 // Generate derived data structure, edges of elements
594 //! \param[in] inpoel Inteconnectivity of points and elements. These are the
595 //! node ids of each element of an unstructured mesh. Example:
596 //! \code{.cpp}
597 //! std::vector< std::size_t > inpoel { 12, 14, 9, 11,
598 //! 10, 14, 13, 12 };
599 //! \endcode
600 //! specifies two tetrahedra whose vertices (node ids) are { 12, 14, 9, 11 },
601 //! and { 10, 14, 13, 12 }.
602 //! \param[in] nnpe Number of nodes per element
603 //! \param[in] inpoed Edge connectivity as linear vector, see tk::genInpoed
604 //! \return Linear vector storing all edge ids * 2 of all elements
605 //! \warning It is not okay to call this function with an empty container for
606 //! inpoel or inpoed or a non-positive number of nodes per element; it will
607 //! throw an exception.
608 //! \details The data generated here is stored in a linear vector with all
609 //! edge ids (as defined by inpoed) of all elements. The edge ids stored in
610 //! inedel can be directly used to index the vector inpoed. Because the
611 //! derived data structure generated here, inedel, is intended to be used in
612 //! conjunction with the linear vector inpoed and not with the linked lists
613 //! edsup1 and edsup2, this function takes inpoed as an argument. Accessing
614 //! the edges of element e using the edge of elements data structure, inedel,
615 //! generated here can be accomplished by
616 //! \code{.cpp}
617 //! for (std::size_t e=0; e<nelem; ++e) {
618 //! for (std::size_t i=0; i<nepe; ++i) {
619 //! use edge id inedel[e*nepe+i] of element e, or
620 //! use point ids p < q of edge id inedel[e*nepe+i] of element e as
621 //! p = inpoed[ inedel[e*nepe+i]*2 ]
622 //! q = inpoed[ inedel[e*nepe+i]*2+1 ]
623 //! }
624 //! }
625 //! \endcode
626 //! where _nepe_ denotes the number of edges per elements: 3 for triangles, 6
627 //! for tetrahedra. To find out the number of elements, _nelem_, the size of
628 //! the mesh connectivity vector, _inpoel_, can be devided by the number of
629 //! nodes per elements, _nnpe_:
630 //! \code{.cpp}
631 //! auto nelem = inpoel.size()/nnpe;
632 //! \endcode
633 //! \note At first sight, this function seems to work for elements with more
634 //! vertices than that of tetrahedra. However, that is not the case since the
635 //! algorithm for nnpe > 4 would erronously identify any two combination of
636 //! vertices as a valid edge of an element. Since only triangles and
637 //! tetrahedra have no internal edges, this algorithm only works for triangle
638 //! and tetrahedra element connectivity.
639 //! \see Lohner, An Introduction to Applied CFD Techniques, Wiley, 2008
640 // *****************************************************************************
641 {
642  Assert( !inpoel.empty(), "Attempt to call genInedel() on empty container" );
643  Assert( nnpe > 0, "Attempt to call genInedel() with zero nodes per element" );
644  Assert( nnpe == 3 || nnpe == 4,
645  "Attempt to call genInedel() with nodes per element, nnpe, that is "
646  "neither 4 (tetrahedra) nor 3 (triangles)." );
647  Assert( inpoel.size()%nnpe == 0, "Size of inpoel must be divisible by nnpe" );
648  Assert( !inpoed.empty(), "Attempt to call genInedel() with empty inpoed" );
649 
650  // find out number of points in mesh connectivity
651  auto minmax = std::minmax_element( begin(inpoel), end(inpoel) );
652  Assert( *minmax.first == 0, "node ids should start from zero" );
653  auto npoin = *minmax.second + 1;
654 
655  // First, generate index of star centers. This is necessary to avoid a
656  // brute-force search for point ids of edges when searching for element edges.
657  // Note that this is the same as edsup2, generated by genEdsup(). However,
658  // because the derived data structure generated here, inedel, is intended to
659  // be used in conjunction with the linear vector inpoed and not with the
660  // linked lists edsup1 and edsup2, this function takes inpoed as an argument,
661  // and so edsup2 is temporarily generated here to avoid a brute-force search.
662 
663  // map to contain stars, a point associated to points connected with edges
664  // storing only the end-point id, q, of point ids p < q
665  std::map< std::size_t, std::vector< std::size_t > > star;
666 
667  // generate stars from inpoed; starting with zero, every even is a star
668  // center, every odd is a spike
669  for (std::size_t i=0; i<inpoed.size()/2; ++i)
670  star[ inpoed[i*2] ].push_back( inpoed[i*2+1] );
671 
672  // store index of star centers in vector; assume non-center points of each
673  // star have already been sorted
674  std::vector< std::size_t > edsup2( 1, 0 );
675  for (auto p : star) edsup2.push_back( edsup2.back() + p.second.size() );
676  // fill up index array with the last index for points with no new edges
677  for (std::size_t i=0; i<npoin-star.size(); ++i)
678  edsup2.push_back( edsup2.back() );
679  star.clear();
680 
681  // Second, generate edges of elements
682 
683  auto nelem = inpoel.size()/nnpe;
684 
685  // map associating elem id with vector of edge ids
686  std::map< std::size_t, std::vector< std::size_t > > edges;
687 
688  // generate map of elements associated to edge ids
689  for (std::size_t e=0; e<nelem; ++e)
690  for (std::size_t n=0; n<nnpe; ++n) {
691  auto p = inpoel[e*nnpe+n];
692  for (auto i=edsup2[p]+1; i<=edsup2[p+1]; ++i)
693  for (std::size_t j=0; j<nnpe; ++j)
694  if (inpoed[(i-1)*2+1] == inpoel[e*nnpe+j])
695  edges[e].push_back( i-1 );
696  }
697 
698  // linear vector to store the edge ids of all elements
699  std::vector< std::size_t > inedel;
700 
701  // store edge ids of elements in linear vector
702  for (auto e : edges) for (auto p : e.second) inedel.push_back( p );
703 
704  // Return (move out) vector
705  return inedel;
706 }

708 std::pair< std::vector< std::size_t >, std::vector< std::size_t > >
709 genEsued( const std::vector< std::size_t >& inpoel,
710  std::size_t nnpe,
711  const std::pair< std::vector< std::size_t >,
712  std::vector< std::size_t > >& esup )
713 // *****************************************************************************
714 // Generate derived data structure, elements surrounding edges
715 //! \param[in] inpoel Inteconnectivity of points and elements. These are the
716 //! node ids of each element of an unstructured mesh. Example:
717 //! \code{.cpp}
718 //! std::vector< std::size_t > inpoel { 12, 14, 9, 11,
719 //! 10, 14, 13, 12 };
720 //! \endcode
721 //! specifies two tetrahedra whose vertices (node ids) are { 12, 14, 9, 11 },
722 //! and { 10, 14, 13, 12 }.
723 //! \param[in] nnpe Number of nodes per element (3 or 4)
724 //! \param[in] esup Elements surrounding points as linked lists, see tk::genEsup
725 //! \return Linked lists storing elements surrounding edges
726 //! \warning It is not okay to call this function with an empty container for
727 //! inpoel or esup.first or esup.second or a non-positive number of nodes per
728 //! element; it will throw an exception.
729 //! \details The data generated here is stored in a linked list, more precisely,
730 //! two linked arrays (vectors), _esued1_ and _esued2_, where _esued2_ holds
731 //! the indices at which _esued1_ holds the element ids surrounding edges.
732 //! Looping over all elements surrounding edges can then be accomplished by
733 //! the following loop:
734 //! \code{.cpp}
735 //! for (std::size_t e=0; e<nedge; ++e)
736 //! for (auto i=esued.second[e]+1; i<=esued.second[e+1]; ++i)
737 //! use element id esued.first[i]
738 //! \endcode
739 //! To find out the number of edges, _nedge_, the edge connectivity, _inpoed_,
740 //! can be queried:
741 //! \code{.cpp}
742 //! auto esup = tk::genEsup(inpoel,nnpe);
743 //! auto nedge = tk::genInpoed(inpoel,nnpe,esup).size()/2;
744 //! \endcode
745 //! where _nnpe_ is the number of nodes per element (4 for tetrahedra, 3 for
746 //! triangles).
747 //! \note At first sight, this function seems to work for elements with more
748 //! vertices than that of tetrahedra. However, that is not the case since the
749 //! algorithm for nnpe > 4 would erronously identify any two combination of
750 //! vertices as a valid edge of an element. Since only triangles and
751 //! tetrahedra have no internal edges, this algorithm only works for triangle
752 //! and tetrahedra element connectivity.
753 //! \see Lohner, An Introduction to Applied CFD Techniques, Wiley, 2008
754 // *****************************************************************************
755 {
756  Assert( !inpoel.empty(), "Attempt to call genEsued() on empty container" );
757  Assert( nnpe > 0, "Attempt to call genEsued() with zero nodes per element" );
758  Assert( nnpe == 3 || nnpe == 4,
759  "Attempt to call genEsued() with nodes per element, nnpe, that is "
760  "neither 4 (tetrahedra) nor 3 (triangles)." );
761  Assert( inpoel.size()%nnpe == 0, "Size of inpoel must be divisible by nnpe" );
762  Assert( !esup.first.empty(), "Attempt to call genEsued() with empty esup1" );
763  Assert( !esup.second.empty(), "Attempt to call genEsued() with empty esup2" );
764 
765  // find out number of points in mesh connectivity
766  auto minmax = std::minmax_element( begin(inpoel), end(inpoel) );
767  Assert( *minmax.first == 0, "node ids should start from zero" );
768  auto npoin = *minmax.second + 1;
769 
770  auto& esup1 = esup.first;
771  auto& esup2 = esup.second;
772 
773  // allocate and fill with zeros a temporary array, only used locally
774  std::vector< std::size_t > lpoin( npoin, 0 );
775 
776  // lambda that returns true if element e contains edge (p < q)
777  auto has = [ &inpoel, nnpe ]( std::size_t e, std::size_t p, std::size_t q )
778  -> bool {
779  std::vector< bool > sp;
780  for (std::size_t n=0; n<nnpe; ++n)
781  if (inpoel[e*nnpe+n] == p || inpoel[e*nnpe+n] == q)
782  sp.push_back( true );
783  if (sp.size() == 2) return true; else return false;
784  };
785 
786  // map to associate edges to unique surrounding element ids
787  std::map< std::size_t, std::vector< std::size_t > > revolver;
788 
789  // generate edges and associated vector of unique surrounding element ids
790  std::size_t ed = 0;
791  for (std::size_t p=0; p<npoin; ++p)
792  for (std::size_t i=esup2[p]+1; i<=esup2[p+1]; ++i )
793  for (std::size_t n=0; n<nnpe; ++n) {
794  auto q = inpoel[ esup1[i] * nnpe + n ];
795  if (q != p && lpoin[q] != p+1) {
796  if (p < q) { // for edge given point ids p < q
797  for (std::size_t j=esup2[p]+1; j<=esup2[p+1]; ++j ) {
798  auto e = esup1[j];
799  if (has(e,p,q)) revolver[ed].push_back(e);
800  }
801  ++ed;
802  }
803  lpoin[q] = p+1;
804  }
805  }
806 
807  // linked lists (vectors) to store elements surrounding edges
808  std::vector< std::size_t > esued1( 1, 0 ), esued2( 1, 0 );
809 
810  // sort and store elements surrounding edges and their indices in vectors
811  for (auto p : revolver) {
812  std::sort( begin(p.second), end(p.second) );
813  esued2.push_back( esued2.back() + p.second.size() );
814  for (auto e : p.second) esued1.push_back( e );
815  }
816 
817  // Return (move out) linked lists
818  return std::make_pair( std::move(esued1), std::move(esued2) );
819 }

821 std::size_t
822 genNbfacTet( std::size_t tnbfac,
823  const std::vector< std::size_t >& inpoel,
824  const std::vector< std::size_t >& triinpoel_complete,
825  const std::map< int, std::vector< std::size_t > >& bface_complete,
826  const std::unordered_map< std::size_t, std::size_t >& lid,
827  std::vector< std::size_t >& triinpoel,
828  std::map< int, std::vector< std::size_t > >& bface )
829 // *****************************************************************************
830 // Generate the number of boundary-faces and the triangle boundary-face
831 // connectivity for a chunk of a full mesh.
832 // \warning This is for Triangular face-elements only.
833 //! \param[in] tnbfac Total number of boundary faces in the entire mesh.
834 //! \param[in] inpoel Inteconnectivity of points and elements. These are the
835 //! node ids of each element of an unstructured mesh.
836 //! \param[in] triinpoel_complete Interconnectivity of points and boundary-face
837 //! in the entire mesh.
838 //! \param[in] bface_complete Map of boundary-face lists mapped to corresponding
839 //! side set ids for the entire mesh.
840 //! \param[in] lid Mapping between the node indices used in the smaller inpoel
841 //! connectivity (a subset of the entire triinpoel_complete connectivity),
842 //! e.g., after mesh partitioning.
843 //! \param[inout] triinpoel Interconnectivity of points and boundary-face in
844 //! this mesh-partition.
845 //! \param[inout] bface Map of boundary-face lists mapped to corresponding
846 //! side set ids for this mesh-partition
847 //! \return Number of boundary-faces on this chare/mesh-partition.
848 //! \details This function takes a mesh by its domain-element
849 //! (tetrahedron-connectivity) in inpoel and a boundary-face (triangle)
850 //! connectivity in triinpoel_complete. Based on these two arrays, it
851 //! searches for those faces of triinpoel_complete that are also in inpoel
852 //! and as a result it generates (1) the number of boundary faces shared with
853 //! the mesh in inpoel and (2) the intersection of the triangle element
854 //! connectivity whose faces are shared with inpoel. An example use case is
855 //! where triinpoel_complete contains the connectivity for the boundary of the
856 //! full problem/mesh and inpoel contains the connectivity for only a chunk of
857 //! an already partitioned mesh. This function then intersects
858 //! triinpoel_complete with inpoel and returns only those faces that share
859 //! nodes with inpoel.
860 // *****************************************************************************
861 {
862  std::size_t nbfac(0), nnpf(3);
863 
864  if (tnbfac > 0)
865  {
866 
867  Assert( !inpoel.empty(),
868  "Attempt to call genNbfacTet() on empty inpoel container" );
869  Assert( !triinpoel_complete.empty(),
870  "Attempt to call genNbfacTet() on empty triinpoel_complete container" );
871  Assert( triinpoel_complete.size()/nnpf == tnbfac,
872  "Incorrect size of triinpoel in genNbfacTet()" );
873 
874  auto nptet = inpoel;
875  auto nptri = triinpoel_complete;
876 
877  tk::unique( nptet );
878  tk::unique( nptri );
879 
880  std::unordered_set< std::size_t > snptet;
881 
882  // getting the reduced inpoel as a set for quick searches
883  snptet.insert( begin(nptet), end(nptet));
884 
885  // vector to store boundary-face-nodes in this chunk
886  std::vector< std::size_t > nptri_chunk;
887 
888  // getting the nodes of the boundary-faces in this chunk
889  for (auto i : nptri)
890  if (snptet.find(i) != end(snptet))
891  nptri_chunk.push_back(i);
892 
893  std::size_t tag, icoun;
894 
895  // matching nodes in nptri_chunk with nodes in inpoel and
896  // triinpoel_complete to get the number of faces in this chunk
897  for (const auto& ss : bface_complete)
898  {
899  for (auto f : ss.second)
900  {
901  icoun = f*nnpf;
902  tag = 0;
903  for (std::size_t i=0; i<nnpf; ++i)
904  {
905  for (auto j : nptri_chunk)
906  {
907  if (triinpoel_complete[icoun+i] == j) ++tag;
908  }
909  }
910  if (tag == nnpf)
911  // this is a boundary face
912  {
913  for (std::size_t i=0; i<nnpf; ++i)
914  {
915  auto ip = triinpoel_complete[icoun+i];
916 
917  // find local renumbered node-id to store in triinpoel
918  triinpoel.push_back( tk::cref_find(lid,ip) );
919  }
920 
921  bface[ss.first].push_back(nbfac);
922  ++nbfac;
923  }
924  }
925  }
926 
927  }
928 
929  return nbfac;
930 }

937 // Generate derived data structure, elements surrounding elements

941 //! \param[in] inpoel Inteconnectivity of points and elements. These are the

956 //! esuelTet stores value -1 indicating that this is outside the domain. The

958 //! important, e.g., in generating the inpofa array later. This node ordering
959 //! convention is stored in tk::lpofa. Thus function is specific to
960 //! tetrahedra, which is reflected in the fact that nnpe and nfpe are being
961 //! set here in the function rather than being input arguments. To find out
962 //! the number of elements, _nelem_, the size of the mesh connectivity vector,
963 //! _inpoel_, can be devided by the number of nodes per elements, _nnpe_:
964 //! \code{.cpp}
