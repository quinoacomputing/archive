21 //! Serialize mesh node indices categorized by chares to raw memory stream
22 //! \param[in] m Chare mesh node indices to serialize
23 //! \return Pair of the length and the raw stream containing the serialized data
24 template< typename T >
25 std::pair< int, std::unique_ptr<char[]> >
26 serialize( const std::vector< std::pair< int, T > >& m ) {
27  // Prepare for serializing node indices to a raw binary stream, compute size
28  PUP::sizer sizer;
29  sizer |
30  const_cast< std::vector< std::pair< int, T > >& >( m );
31 
32  // Create raw character stream to store the serialized node indices
33  std::unique_ptr<char[]> flatData = tk::make_unique<char[]>( sizer.size() );
34 
35  // Serialize, each message will contain a list of node indices per chare
36  PUP::toMem packer( flatData.get() );
37  packer |
38  const_cast< std::vector< std::pair< int, T > >& >( m );
39 
40  // Return size of and raw stream
41  return { sizer.size(), std::move(flatData) };
42 }

44 //! \brief Charm++ custom reducer for merging mesh node indices categorized by
45 //! chares during reduction across PEs
46 //! \param[in] nmsg Number of messages in msgs
47 //! \param[in] msgs Charm++ reduction message containing the serialized mesh
48 //! node indices categorized by chares
49 //! \return Aggregated mesh node indices categorized by chares built for further
50 //! aggregation if needed
51 //! \details During aggregation the outer vector is simply concatenated.
52 template< typename T >
53 CkReductionMsg*
54 mergeFields( int nmsg, CkReductionMsg **msgs ) {
55  // Will store deserialized mesh node indices categorized by chares
56  std::vector< std::pair< int, T > > p;
57 
58  // Create PUP deserializer based on message passed in
59  PUP::fromMem creator( msgs[0]->getData() );
60 
61  // Deserialize from raw stream
62  creator | p;
63 
64  for (int m=1; m<nmsg; ++m) {
65  // Unpack mesh node indices
66  std::vector< std::pair< int, T > > u;
67  PUP::fromMem curCreator( msgs[m]->getData() );
68  curCreator | u;
69  // Concatenate mesh node indices categorized by chares
70  p.insert( end(p), begin(u), end(u) );
71  }
72 
73  // Serialize concatenated node indices categorized by chares to raw stream
74  auto stream = serialize( p );
75 
76  // Forward serialized mesh node indices categorized by chares
77  return CkReductionMsg::buildNew( stream.first, stream.second.get() );
78 }
