41  m_print( rngtest::g_inputdeck.get< tag::cmd, tag::verbose >() ?

70  // Collect number of results from all tests (one per RNG)
71  for (std::size_t i=0; i<ntest(); ++i) m_tests[i].npval();

92 void
93 TestU01Suite::names( std::vector< std::string > n )
94 // *****************************************************************************
95 // Collect test names from a statistical test
96 //! \param[in] n Vector of test names (there can be more than one from one test)
97 //! \author J. Bakosi
98 // *****************************************************************************
99 {
100  m_print.names( n );
101 
102  if ( ++m_ntest == ntest() ) {
103  const auto& rngs = g_inputdeck.get< tag::selected, tag::rng >();
104  std::stringstream ss;
105  ss << "RNGs tested (" << rngs.size() << ")";
106  m_print.section( ss.str() );
107  #ifdef HAS_MKL
108  m_print.MKLParams( rngs, g_inputdeck.get< tag::param, tag::rngmkl >() );
109  #endif
110  m_print.RNGSSEParams( rngs, g_inputdeck.get< tag::param, tag::rngsse >() );
111  m_print.Random123Params( rngs,
112  g_inputdeck.get< tag::param, tag::rng123 >() );
113  m_print.endpart();
114  m_print.part( m_name );
115  m_print.statshead( "Statistics computed",
116  m_npval*rngs.size(),
117  m_ctrs.size() );
118 
119  // Run battery of RNG tests
120  for (const auto& t : m_tests) t.run();
121 
122  // Initialize space for counting the number failed tests per RNG. Note
123  // that we could use tk::ctr::RNGType as the map-key here instead of the
124  // RNG name (RNGType would be an enum vs the name which is a string),
125  // but then the RNGType would have to be part of the status (instead of
126  // the RNG name) returning from a completed test by TestU01Props::run().
127  // That would require a return type that is less generic than the
128  // current vector of vector of strings. Since the RNG name is already
129  // part of the status, we just match the RNG name for counting failed
130  // tests per RNG and keep the status more generic. See also the
131  // discussion on the return type in TestU01Props::run().
132  tk::ctr::RNG rng;
133  for (const auto& r : rngs )
134  m_nfail[ rng.name(r) ] = 0;
135  }
136 }

138 void
139 TestU01Suite::evaluate( std::vector< std::vector< std::string > > status )
140 // *****************************************************************************
141 // Evaluate statistical test
142 //! \param[in] status Status vectors of strings for a test
143 //! \author J. Bakosi
144 // *****************************************************************************
145 {
146  m_print.test( ++m_ncomplete, m_ctrs.size(), m_nfail, status );
147 
148  // Store information on failed test for final assessment
149  for (std::size_t p=0; p<status[1].size(); ++p)
150  if (status[1][p].size() > 4)
151  m_failed.emplace_back( status[0][p], status[2][0], status[1][p] );
152 
153  if ( m_ncomplete == m_ctrs.size() ) {
154  // Collect measured test run times
155  m_ncomplete = 0;
156  for (const auto& t : m_tests) t.time();
157  }
158 }

162 // *****************************************************************************

173 void
174 TestU01Suite::assess()
175 // *****************************************************************************
176 // Output final assessment
177 //! \author J. Bakosi
178 // *****************************************************************************
179 {
180  // Output summary of failed tests for all RNGs tested
181  if ( !m_failed.empty() ) {
182  const auto& rngs = g_inputdeck.get< tag::selected, tag::rng >();
183  m_print.failed( "Failed statistics", m_npval*rngs.size(), m_failed );
184  } else m_print.note< tk::QUIET >( "All tests passed" );
185 
186  // Cost and quality assessment only for more than one RNG
187  if (m_time.size() > 1) {
188  // Output measured times per RNG in order of computational cost
189  m_print.cost( "Generator cost",
190  "Measured times in seconds in increasing order (low is good)",
191  m_time );
192  // Output number of failed tests per RNG in order of decreasing quality
193  m_print.rank( "Generator quality",
194  "Number of failed tests in increasing order (low is good)",
195  m_nfail );
196  }
197 
198  // Quit
199  mainProxy.finalize();
200 }
