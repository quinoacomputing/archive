26 //! Transport PDE problem: rotation of Zalesak's slotted cylinder
27 //! \see Zalesak, S. (1978) Fully Multidimensional Flux-Corrected Transport
28 //! Algorithms for Fluids. Journal of Computational Physics 31, 335-362.
29 //! \see Leveque R.J. (1996) High-Resolution Conservative Algorithms for
30 //! Advection in Incompressible Flow. SIAM Journal on Numerical Analyis 33,
31 //! 627-665.
32 class TransportProblemSlotCyl {
33 
34  public:
35  //! Evaluate analytical solution at (x,y,t) for all components
36  //! \param[in] ncomp Number of components in this transport equation system
37  //! \param[in] x X coordinate where to evaluate the solution
38  //! \param[in] y Y coordinate where to evaluate the solution
39  //! \param[in] t Time where to evaluate the solution
40  //! \return Values of all components evaluated at (x,y,t)
41  static std::vector< tk::real >
42  solution( ncomp_t, ncomp_t ncomp,
43  tk::real x, tk::real y, tk::real, tk::real t )
44  {
45  using std::sin; using std::cos;
46  std::vector< tk::real > s( ncomp, 0.0 );
47  for (ncomp_t c=0; c<ncomp; ++c) {
48  auto T = t + 2.0*M_PI/ncomp * c;
49  const tk::real R0 = 0.15;
50  // center of the cone
51  tk::real x0 = 0.5;
52  tk::real y0 = 0.25;
53  tk::real r = std::sqrt((x0-0.5)*(x0-0.5) + (y0-0.5)*(y0-0.5));
54  tk::real kx = 0.5 + r*sin( T );
55  tk::real ky = 0.5 - r*cos( T );
56  // center of the hump
57  x0 = 0.25;
58  y0 = 0.5;
59  r = std::sqrt((x0-0.5)*(x0-0.5) + (y0-0.5)*(y0-0.5));
60  tk::real hx = 0.5 + r*sin( T-M_PI/2.0 ),
61  hy = 0.5 - r*cos( T-M_PI/2.0 );
62  // center of the slotted cylinder
63  x0 = 0.5;
64  y0 = 0.75;
65  r = std::sqrt((x0-0.5)*(x0-0.5) + (y0-0.5)*(y0-0.5));
66  tk::real cx = 0.5 + r*sin( T+M_PI ),
67  cy = 0.5 - r*cos( T+M_PI );
68  // end points of the cylinder slot
69  tk::real i1x = 0.525, i1y = cy - r*cos( std::asin(0.025/r) ),
70  i2x = 0.525, i2y = 0.8,
71  i3x = 0.475, i3y = 0.8;
72  // rotate end points of cylinder slot
73  tk::real ri1x = 0.5 + cos(T)*(i1x-0.5) - sin(T)*(i1y-0.5),
74  ri1y = 0.5 + sin(T)*(i1x-0.5) + cos(T)*(i1y-0.5),
75  ri2x = 0.5 + cos(T)*(i2x-0.5) - sin(T)*(i2y-0.5),
76  ri2y = 0.5 + sin(T)*(i2x-0.5) + cos(T)*(i2y-0.5),
77  ri3x = 0.5 + cos(T)*(i3x-0.5) - sin(T)*(i3y-0.5),
78  ri3y = 0.5 + sin(T)*(i3x-0.5) + cos(T)*(i3y-0.5);
79  // direction of slot sides
80  tk::real v1x = ri2x-ri1x, v1y = ri2y-ri1y,
81  v2x = ri3x-ri2x, v2y = ri3y-ri2y;
82  // lengths of direction of slot sides vectors
83  tk::real v1 = std::sqrt(v1x*v1x + v1y*v1y),
84  v2 = std::sqrt(v2x*v2x + v2y*v2y);
85  // cone
86  r = std::sqrt((x-kx)*(x-kx) + (y-ky)*(y-ky)) / R0;
87  if (r<1.0) s[c] = 0.6*(1.0-r);
88  // hump
89  r = std::sqrt((x-hx)*(x-hx) + (y-hy)*(y-hy)) / R0;
90  if (r<1.0) s[c] = 0.2*(1.0+cos(M_PI*std::min(r,1.0)));
91  // cylinder
92  r = std::sqrt((x-cx)*(x-cx) + (y-cy)*(y-cy)) / R0;
93  const std::array< tk::real, 2 > r1{{ v1x, v1y }},
94  r2{{ x-ri1x, y-ri1y }};
95  const auto d1 = (r1[0]*r2[1] - r2[0]*r1[1]) / v1;
96  const std::array< tk::real, 2 > r3{{ v2x, v2y }},
97  r4{{ x-ri2x, y-ri2y }};
98  const auto d2 = (r3[0]*r4[1] - r4[0]*r3[1]) / v2;
99  if (r<1.0 && (d1>0.05 || d1<0.0 || d2<0.0)) s[c] = 0.6;
100  }
101  return s;
102  }
103 
104  //! \brief Evaluate the increment from t to t+dt of the analytical solution
105  //! at (x,y,z) for all components
106  //! \param[in] ncomp Number of components in this transport equation system
107  //! \param[in] x X coordinate where to evaluate the solution
108  //! \param[in] y Y coordinate where to evaluate the solution
109  //! \param[in] t Time where to evaluate the solution increment starting from
110  //! \param[in] dt Time increment at which evaluate the solution increment to
111  //! \return Increment in values of all components evaluated at (x,y,t+dt)
112  static std::vector< tk::real >
113  solinc( ncomp_t, ncomp_t ncomp, tk::real x, tk::real y, tk::real,
114  tk::real t, tk::real dt )
115  {
116  auto st1 = solution( 0, ncomp, x, y, 0.0, t );
117  auto st2 = solution( 0, ncomp, x, y, 0.0, t+dt );
118  std::transform( begin(st1), end(st1), begin(st2), begin(st2),
119  []( tk::real s, tk::real& d ){ return d -= s; } );
120  return st2;
121  }

137  //! \param[in] x X coordinate at which to assign velocity
138  //! \param[in] y y coordinate at which to assign velocity
