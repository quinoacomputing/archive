21 //! HLLC approximate Riemann solver
22 //! \details This class is used polymorphically with inciter::RiemannSolver
23 struct HLLC {
24  //! HLLC approximate Riemann solver flux function
25  //! \param[in] f Face ID
26  //! \param[in] geoFace Face geometry array
27  //! \param[in] u Left and right unknown/state vector
28  //! \return Riemann solution using a central difference method
29  std::vector< tk::real >
30  flux( std::size_t f,
31  const tk::Fields& geoFace,
32  const std::array< std::vector< tk::real >, 2 >& u ) const
33  {
34  std::vector< tk::real > flx( u[0].size(), 0 );
35 
36  std::array< tk::real, 3 > fn {{ geoFace(f,1,0),
37  geoFace(f,2,0),
38  geoFace(f,3,0) }};
39 
40  // ratio of specific heats
41  auto g = g_inputdeck.get< tag::param, tag::compflow, tag::gamma >()[0];
42 
43  // Primitive variables
44  auto rhol = u[0][0];
45  auto rhor = u[1][0];
46 
47  auto pl = (g-1.0)*(u[0][4] - (u[0][1]*u[0][1] +
48  u[0][2]*u[0][2] +
49  u[0][3]*u[0][3]) / (2.0*rhol));
50 
51  auto pr = (g-1.0)*(u[1][4] - (u[1][1]*u[1][1] +
52  u[1][2]*u[1][2] +
53  u[1][3]*u[1][3]) / (2.0*rhor));
54 
55  auto al = sqrt(g * pl / rhol);
56  auto ar = sqrt(g * pr / rhor);
57 
58  // Face-normal velocities
59  auto ul = u[0][1]/rhol;
60  auto vl = u[0][2]/rhol;
61  auto wl = u[0][3]/rhol;
62 
63  tk::real vnl = ul*fn[0] + vl*fn[1] + wl*fn[2];
64 
65  auto ur = u[1][1]/rhor;
66  auto vr = u[1][2]/rhor;
67  auto wr = u[1][3]/rhor;
68 
69  tk::real vnr = ur*fn[0] + vr*fn[1] + wr*fn[2];
70 
71  // Roe-averaged variables
72  auto rlr = sqrt(rhor/rhol);
73  auto rlr1 = 1.0 + rlr;
74 
75  auto vnroe = (vnr*rlr + vnl)/rlr1 ;
76  auto aroe = (ar*rlr + al)/rlr1 ;
77 
78  // Signal velocities
79  auto Sl = fmin(vnl-al, vnroe-aroe);
80  auto Sr = fmax(vnr+ar, vnroe+aroe);
81  auto Sm = ( rhor*vnr*(Sr-vnr) - rhol*vnl*(Sl-vnl) + pl-pr )
82  /( rhor*(Sr-vnr) - rhol*(Sl-vnl) );
83 
84  // Middle-zone (star) variables
85  auto pStar = rhol*(vnl-Sl)*(vnl-Sm) + pl;
86  auto uStar = u;
87 
88  uStar[0][0] = (Sl-vnl) * rhol/ (Sl-Sm);
89  uStar[0][1] = ((Sl-vnl) * u[0][1] + (pStar-pl)*fn[0]) / (Sl-Sm);
90  uStar[0][2] = ((Sl-vnl) * u[0][2] + (pStar-pl)*fn[1]) / (Sl-Sm);
91  uStar[0][3] = ((Sl-vnl) * u[0][3] + (pStar-pl)*fn[2]) / (Sl-Sm);
92  uStar[0][4] = ((Sl-vnl) * u[0][4] - pl*vnl + pStar*Sm) / (Sl-Sm);
93 
94  uStar[1][0] = (Sr-vnr) * rhor/ (Sr-Sm);
95  uStar[1][1] = ((Sr-vnr) * u[1][1] + (pStar-pr)*fn[0]) / (Sr-Sm);
96  uStar[1][2] = ((Sr-vnr) * u[1][2] + (pStar-pr)*fn[1]) / (Sr-Sm);
97  uStar[1][3] = ((Sr-vnr) * u[1][3] + (pStar-pr)*fn[2]) / (Sr-Sm);
98  uStar[1][4] = ((Sr-vnr) * u[1][4] - pr*vnr + pStar*Sm) / (Sr-Sm);
99 
100  // Numerical fluxes
101  if (Sl > 0.0) {
102  flx[0] = u[0][0] * vnl;
103  flx[1] = u[0][1] * vnl + pl*fn[0];
104  flx[2] = u[0][2] * vnl + pl*fn[1];
105  flx[3] = u[0][3] * vnl + pl*fn[2];
106  flx[4] = ( u[0][4] + pl ) * vnl;
107  }
108  else if (Sl <= 0.0 && Sm > 0.0) {
109  flx[0] = uStar[0][0] * Sm;
110  flx[1] = uStar[0][1] * Sm + pStar*fn[0];
111  flx[2] = uStar[0][2] * Sm + pStar*fn[1];
112  flx[3] = uStar[0][3] * Sm + pStar*fn[2];
113  flx[4] = ( uStar[0][4] + pStar ) * Sm;
114  }
115  else if (Sm <= 0.0 && Sr >= 0.0) {
116  flx[0] = uStar[1][0] * Sm;
117  flx[1] = uStar[1][1] * Sm + pStar*fn[0];
118  flx[2] = uStar[1][2] * Sm + pStar*fn[1];
119  flx[3] = uStar[1][3] * Sm + pStar*fn[2];
120  flx[4] = ( uStar[1][4] + pStar ) * Sm;
121  }
122  else {
123  flx[0] = u[1][0] * vnr;
124  flx[1] = u[1][1] * vnr + pr*fn[0];
125  flx[2] = u[1][2] * vnr + pr*fn[1];
126  flx[3] = u[1][3] * vnr + pr*fn[2];
127  flx[4] = ( u[1][4] + pr ) * vnr;
128  }
129 
130  return flx;
131  }
132 
133  //! Flux type accessor
134  //! \return Flux type
135  static ctr::FluxType type() noexcept { return ctr::FluxType::HLLC; }
136 };
