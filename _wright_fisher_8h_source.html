96  IGNORE( stream );
97  //! Set initial conditions using initialization policy
98  //Init::template
99  // init< tag::wrightfisher >

102  const auto npar = particles.nunk();
103  for (auto p=decltype(npar){0}; p<npar; ++p) {
104  // Initialize the first m_ncomp (N-1) scalars
105  ncomp_t i;

119  //! \brief Advance particles according to the Wright-Fisher SDE
120  //! \param[in,out] particles Array of particle properties
121  //! \param[in] stream Thread (or more precisely stream) ID
122  //! \param[in] dt Time step size
123  //! \author J. Bakosi
124  void advance( tk::Particles& particles,
125  int stream,
126  tk::real dt,
127  tk::real,
128  const std::map< tk::ctr::Product, tk::real >& )
129  {
130  // Compute sum of coefficients
131  const auto omega = std::accumulate( begin(m_omega), end(m_omega), 0.0 );
132  const auto npar = particles.nunk();
133 
134  #if defined(__clang__)
135  #pragma clang diagnostic push
136  #pragma clang diagnostic ignored "-Wvla"
137  #pragma clang diagnostic ignored "-Wvla-extension"
138  #elif defined(__GNUC__)
139  #pragma GCC diagnostic push
140  #pragma GCC diagnostic ignored "-Wvla"
141  #endif
142 
143  for (auto p=decltype(npar){0}; p<npar; ++p) {
144  // Need to build the square-root of the Wright-Fisher diffusion matrix:
145  // B_ij = y_i * ( delta_ij - y_j ). If the matrix is positive definite,
146  // the Cholesky decomposition would work, however, B_ij is only positive
147  // semi-definite, so Cholesky may fail and considering floating-point
148  // errors it will definitely fail at some point.
149  //
150  // A stable square-root for B_ij is not yet implemented. Here is some
151  // details and points for further development:
152  //
153  // An interesting fact from http://math.stackexchange.com/a/332465, on
154  // gauging the eigen values of a matrix:
155  //
156  // "Choosing each diagonal entry to be greater than the sum of the
157  // absolute values of the other entries in the same row will immediately
158  // imply that all of the eigenvalues of A are positive, and therefore
159  // that A is positive definite."
160  //
161  // The WF diffusion matrix seems to be positive semi-definite with at
162  // least one very small eigenvalue O(10e-18). The row and column sums
163  // are zero. It does not appear that the matrix has negative
164  // eigenvalues.
165  //
166  // Using Cholesky decomposition to compute the square-root may fail.
167  // Other methods, such as diagonalizing the matrix or the LDL
168  // decomposition, may work, but have not tried. The former requires
169  // finding the eigenvalues, with e.g., LAPACK's DSPEVD, thereby
170  // diagonalizing the matrix, and taking the square-root of the diagonal
171  // elements to find the square-root of the matrix. See
172  // http://en.wikipedia.org/wiki/Square_root_of_a_matrix. This may be
173  // okay, but probably overkill, as we don't really need the eigenvalues,
174  // only the square-root of the matrix.
175  //
176  // The latter (LDL) is similar to Cholesky, but does not involve square
177  // root. See http://en.wikipedia.org/wiki/Cholesky_decomposition. See
178  // also Golub van Loan: Sec.4.1.2 LDL. It appears that LDL would compare
179  // in perfromance to Cholesky and would work even on large indefinite
180  // matrices. LAPACK probably has LDL.
181  //
182  // I'm putting this aside for now. The next step should probably be
183  // idintifying the LAPACK for LDL and use it to advance the WF system
184  // using LD^{1/2}.
185  //
186  // See also:
187  // http://en.wikipedia.org/wiki/Positive-definite_matrix
188  // file:///opt/intel/composerxe/Documentation/en_US/mkl/mklman/index.htm
189  // http://icl.cs.utk.edu/lapack-forum/archives/lapack/msg01497.html
190  // http://yarchive.net/comp/sqrtm.html
191 
192  tk::real B[m_ncomp][m_ncomp];
193  tk::real Bo[m_ncomp][m_ncomp];
194  for (ncomp_t i=0; i<m_ncomp; ++i) {
195  const tk::real& pari = particles( p, i, m_offset );
196  for (ncomp_t j=0; j<m_ncomp; ++j) {
197  const tk::real& parj = particles( p, j, m_offset );
198  if (i == j) {
199  B[i][i] = std::abs( pari * (1.0 - pari) );
200  if (B[i][i] < 1.0e-10) B[i][i] = 1.0;
201  } else {
202  B[i][j] = -pari*parj;
203  }
204  }
205  }
206  std::memcpy( Bo, B, m_ncomp*m_ncomp*sizeof(tk::real) );
207  // Compute diffusion matrix (lower triangle of Cholesky-decomposition)
208  lapack_int n = static_cast< lapack_int >( m_ncomp );
209  lapack_int info = LAPACKE_dpotrf( LAPACK_ROW_MAJOR, 'L', n, B[0], n );
210  if (info != 0 ) {
211  print_matrix( "=======\nOriginal Matrix", Bo[0], n, n );
212  std::cout << "info: " << info << std::endl;
213  print_matrix( "Result of Cholesky factorization", B[0], n, n );
214  for (ncomp_t i=0; i<m_ncomp; ++i) {
215  std::cout <<
216  std::setprecision( std::numeric_limits< tk::real >::digits10 )
217  << i << " par: " << particles( p, i, m_offset) << std::endl;
218  }
219  }
220  //ErrChk( info == 0, "Wright-Fisher Cholesky decomposition unsuccessful, "
221  // "info = " + std::to_string( info ) + ", particle: " +
222  // std::to_string( p ) );
223 
224  // Advance the first m_ncomp (N-1) scalars
225  if (info == 0) {
226  ncomp_t i = 0;
227  for (i=0; i<m_ncomp-1; ++i) {
228  tk::real& par = particles( p, i, m_offset );
229  // Advance first m_ncomp (K=N-1) scalars due to drift
230  par += 0.5*(m_omega[i] - omega*par)*dt;
231  // Advance first m_ncomp (K=N-1) particles with Cholesky-decomposed
232  // lower triangle (diffusion matrix)
233  for (ncomp_t j=0; j<m_ncomp-1; ++j)
234  if (j<=i) {
235  tk::real dW;
236  m_rng.gaussian( stream, 1, &dW );
237  par += B[i][j] * sqrt(dt) * dW;
238  }
239  }
240  // Compute the (N-1)th scalar from unit-sum
241  tk::real& par = particles( p, i, m_offset );
242  par = 1.0 - particles( p, 0, m_offset );
243  for (i=1; i<m_ncomp-1; ++i) {
244  par -= particles( p, i, m_offset );
245  }
246  }
247  }
248

250  #pragma clang diagnostic pop
251  #elif defined(__GNUC__)
252  #pragma GCC diagnostic pop
253  #endif
254  }
