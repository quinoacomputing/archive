39  //! \brief Univariate PDF
40  //! \details The underlying container type is an unordered_map where the key

58  //! Add sample to univariate PDF
59  //! \param[in] sample Sample to insert
60  void add( tk::real sample ) {
61  Assert( m_binsize > 0, "Bin size must be positive" );
62  ++m_nsample;
63  ++m_pdf[ std::lround( sample / m_binsize ) ];
64  }
65 
66  //! Add multiple samples from a PDF
67  //! \param[in] p PDF whose samples to add
68  void addPDF( const UniPDF& p ) {
69  m_binsize = p.binsize();
70  m_nsample += p.nsample();
71  for (const auto& e : p.map()) m_pdf[ e.first ] += e.second;
72  }
73 
74  //! Zero bins

85  //! Return minimum and maximum bin ids of sample space

87  std::array< long, 2*dim > extents() const {
88  auto x = std::minmax_element( begin(m_pdf), end(m_pdf),
89  []( const pair_type& a, const pair_type& b )
