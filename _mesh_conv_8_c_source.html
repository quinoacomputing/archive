46  public:
47  //! \brief Constructor
48  //! \details The main chare constructor is the main entry point of the
49  //! program, called by the Charm++ runtime system. The constructor does
50  //! basic initialization steps, e.g., parser the command-line, prints out
51  //! some useful information to screen (in verbose mode), and instantiates
52  //! a driver. Since Charm++ is fully asynchronous, the constructor
53  //! usually spawns asynchronous objects and immediately exits. Thus in the
54  //! body of the main chare constructor we fire up an 'execute' chare,
55  //! which then calls back to Main::execute(). Finishing the main chare
56  //! constructor the Charm++ runtime system then starts the
57  //! network-migration of all global-scope data (if any). The execute chare
58  //! calling back to Main::execute() signals the end of the migration of
59  //! the global-scope data. Then we are ready to execute the driver which
60  //! calls back to Main::finalize() when it finished. Then finalize() exits
61  //! by calling Charm++'s CkExit(), shutting down the runtime system.
62  //! \see http://charm.cs.illinois.edu/manuals/html/charm++/manual.html
63  Main( CkArgMsg* msg )
64  try :
65  m_cmdline(),
66  // Parse command line into m_cmdline using default simple pretty printer
67  m_cmdParser( msg->argc, msg->argv, tk::Print(), m_cmdline ),
68  // Create pretty printer initializing output streams based on command line
69  m_print( m_cmdline.get< tag::verbose >() ? std::cout : std::clog ),
70  // Create MeshConv driver
71  m_driver( tk::Main< meshconv::MeshConvDriver >
72  ( msg->argc, msg->argv,
73  m_cmdline,
74  tk::HeaderType::MESHCONV,
75  MESHCONV_EXECUTABLE,
76  m_print ) ),
77  m_timer(1), // Start new timer measuring the total runtime
78  m_timestamp()
79  {
80  delete msg;
81  mainProxy = thisProxy;
82  // Fire up an asynchronous execute object, which when created at some
83  // future point in time will call back to this->execute(). This is
84  // necessary so that this->execute() can access already migrated
85  // global-scope data.
86  CProxy_execute::ckNew();

88  m_timer.emplace_back();
89  } catch (...) { tk::processExceptionCharm(); }
