30 //! CompFlow system of PDEs problem: Navier-Stokes (viscous flow)
31 //! \details This class adds the viscous force contributions to the momentum and
32 //! energy conservation equations governing compressible flow.

35  public:
36  //! Add viscous stress contribution to momentum and energy rhs
37  //! \param[in] dt Size of time step
38  //! \param[in] J Element Jacobi determinant
39  //! \param[in] N Element node indices
40  //! \param[in] grad Shape function derivatives, nnode*ndim [4][3]
41  //! \param[in] u Solution at element nodes at recent time step
42  //! \param[in] r Pointers to right hand side at component and offset
43  //! \param[in,out] R Right-hand side vector contributing to
44  static void
45  viscousRhs( tk::real dt,
46  tk::real J,
47  const std::array< std::size_t, 4 >& N,
48  const std::array< std::array< tk::real, 3 >, 4 >& grad,
49  const std::array< std::array< tk::real, 4 >, 5 >& u,
50  const std::array< const tk::real*, 5 >& r,
51  tk::Fields& R )
52  {
53  // dynamic viscosity
54  auto mu = g_inputdeck.get< tag::param, tag::compflow, tag::mu >()[0];
55  // add deviatoric viscous stress contribution to momentum rhs
56  auto c = dt * J/6.0 * mu;
57  for (std::size_t i=0; i<3; ++i)
58  for (std::size_t j=0; j<3; ++j)
59  for (std::size_t k=0; k<4; ++k) {
60  R.var(r[i+1],N[0]) -= c * grad[0][j]*(grad[k][j]*u[i+1][k] +
61  grad[k][i]*u[j+1][k])/u[0][k];
62  R.var(r[i+1],N[1]) -= c * grad[1][j]*(grad[k][j]*u[i+1][k] +
63  grad[k][i]*u[j+1][k])/u[0][k];
64  R.var(r[i+1],N[2]) -= c * grad[2][j]*(grad[k][j]*u[i+1][k] +
65  grad[k][i]*u[j+1][k])/u[0][k];
66  R.var(r[i+1],N[3]) -= c * grad[3][j]*(grad[k][j]*u[i+1][k] +
67  grad[k][i]*u[j+1][k])/u[0][k];
68  }
69  // add isotropic viscous stress contribution to momentum rhs
70  c = dt * J/6.0 * mu * 2.0/3.0;
71  for (std::size_t i=0; i<3; ++i)
72  for (std::size_t j=0; j<3; ++j)
73  for (std::size_t k=0; k<4; ++k) {
74  R.var(r[i+1],N[0]) += c * grad[0][i]*grad[k][j]*u[j+1][k]/u[0][k];
75  R.var(r[i+1],N[1]) += c * grad[1][i]*grad[k][j]*u[j+1][k]/u[0][k];
76  R.var(r[i+1],N[2]) += c * grad[2][i]*grad[k][j]*u[j+1][k]/u[0][k];
77  R.var(r[i+1],N[3]) += c * grad[3][i]*grad[k][j]*u[j+1][k]/u[0][k];
78  }
79  // add deviatoric viscous stress contribution to energy rhs
80  c = dt * J/24.0 * mu;
81  for (std::size_t i=0; i<3; ++i)
82  for (std::size_t j=0; j<3; ++j)
83  for (std::size_t k=0; k<4; ++k) {
84  R.var(r[4],N[0]) -= c * u[i+1][k]/u[0][k] *
85  grad[0][j]*(grad[k][j]*u[i+1][k] +
86  grad[k][i]*u[j+1][k])/u[0][k];
87  R.var(r[4],N[1]) -= c * u[i+1][k]/u[0][k] *
88  grad[1][j]*(grad[k][j]*u[i+1][k] +
89  grad[k][i]*u[j+1][k])/u[0][k];
90  R.var(r[4],N[2]) -= c * u[i+1][k]/u[0][k] *
91  grad[2][j]*(grad[k][j]*u[i+1][k] +
92  grad[k][i]*u[j+1][k])/u[0][k];
93  R.var(r[4],N[3]) -= c * u[i+1][k]/u[0][k] *
94  grad[3][j]*(grad[k][j]*u[i+1][k] +
95  grad[k][i]*u[j+1][k])/u[0][k];
96  }
97  // add isotropic viscous stress contribution to energy rhs
98  c = dt * J/24.0 * mu * 2.0/3.0;
99  for (std::size_t i=0; i<3; ++i)
100  for (std::size_t j=0; j<3; ++j)
101  for (std::size_t k=0; k<4; ++k) {
102  R.var(r[4],N[0]) += c * u[i+1][k]/u[0][k] *
103  grad[0][i]*grad[k][j]*u[j+1][k]/u[0][k];
104  R.var(r[4],N[1]) += c * u[i+1][k]/u[0][k] *
105  grad[1][i]*grad[k][j]*u[j+1][k]/u[0][k];
106  R.var(r[4],N[2]) += c * u[i+1][k]/u[0][k] *
107  grad[2][i]*grad[k][j]*u[j+1][k]/u[0][k];
108  R.var(r[4],N[3]) += c * u[i+1][k]/u[0][k] *
109  grad[3][i]*grad[k][j]*u[j+1][k]/u[0][k];
110  }
111  }

113  //! Compute the minimum time step size based on the viscous force
114  //! \param[in] L Characteristic length scale
115  //! \param[in] u Solution at element nodes at recent time step
116  //! \return Minimum time step size based on viscous force
117  static tk::real
118  viscous_dt( tk::real L,
119  const std::array< std::array< tk::real, 4 >, 5 >& u )
120  {
121  // dynamic viscosity
122  auto mu = g_inputdeck.get< tag::param, tag::compflow, tag::mu >()[0];
123  // compute the minimum viscous time step size across the four nodes
124  tk::real mindt = std::numeric_limits< tk::real >::max();
125  for (std::size_t j=0; j<4; ++j) {
126  auto& r = u[0][j]; // rho
127  auto dt = L * L * r / (2.0*mu); // dt ~ dx^2/nu = dx^2*rho/(2mu)
128  if (dt < mindt) mindt = dt;
129  }
130  return mindt;
131  }

133  //! Add heat conduction contribution to energy rhs
134  //! \param[in] dt Size of time step
135  //! \param[in] J Element Jacobi determinant
136  //! \param[in] N Element node indices
137  //! \param[in] grad Shape function derivatives, nnode*ndim [4][3]
138  //! \param[in] u Solution at element nodes at recent time step
139  //! \param[in] r Pointers to right hand side at component and offset
140  //! \param[in,out] R Right-hand side vector contributing to
141  static void
142  conductRhs( tk::real dt,
143  tk::real J,
144  const std::array< std::size_t, 4 >& N,
145  const std::array< std::array< tk::real, 3 >, 4 >& grad,
146  const std::array< std::array< tk::real, 4 >, 5 >& u,
147  const std::array< const tk::real*, 5 >& r,
148  tk::Fields& R )
149  {
150  // specific heat at constant volume
151  auto cv = g_inputdeck.get< tag::param, tag::compflow, tag::cv >()[0];
152  // thermal conductivity
153  auto kc = g_inputdeck.get< tag::param, tag::compflow, tag::k >()[0];
154  // compute temperature
155  std::array< tk::real, 4 > T;
156  for (std::size_t i=0; i<4; ++i)
157  T[i] = cv*(u[4][i] - (u[1][i]*u[1][i] +
158  u[2][i]*u[2][i] +
159  u[3][i]*u[3][i])/2.0/u[0][i]) / u[0][i];
160  // add heat conduction contribution to energy rhs
161  auto c = dt * J/24.0 * kc;
162  for (std::size_t i=0; i<3; ++i)
163  for (std::size_t k=0; k<4; ++k) {
164  R.var(r[4],N[0]) += c * grad[k][i] * T[k];
165  R.var(r[4],N[1]) += c * grad[k][i] * T[k];
166  R.var(r[4],N[2]) += c * grad[k][i] * T[k];
167  R.var(r[4],N[3]) += c * grad[k][i] * T[k];
168  }
169  }

171  //! Compute the minimum time step size based on thermal diffusion
172  //! \param[in] L Characteristic length scale
173  //! \param[in] u Solution at element nodes at recent time step
174  //! \return Minimum time step size based on thermal diffusion
175  static tk::real
176  conduct_dt( tk::real L,
177  const std::array< std::array< tk::real, 4 >, 5 >& u )
178  {
179  // ratio of specific heats
180  auto g = g_inputdeck.get< tag::param, tag::compflow, tag::gamma >()[0];
181  // specific heat at constant volume
182  auto cv = g_inputdeck.get< tag::param, tag::compflow, tag::cv >()[0];
183  // thermal conductivity
184  auto k = g_inputdeck.get< tag::param, tag::compflow, tag::k >()[0];
185  // specific heat at constant pressure
186  auto cp = g * cv;
187  // compute the minimum conduction time step size across the four nodes
188  tk::real mindt = std::numeric_limits< tk::real >::max();
189  for (std::size_t j=0; j<4; ++j) {
190  auto& r = u[0][j]; // rho
191  auto dt = L * L * r * cp / k; // dt ~ dx^2/alpha = dx^2*rho*C_p/k
192  if (dt < mindt) mindt = dt;
193  }
194  return mindt;
195  }
