24 //! CompFlow system of PDEs problem: Rayleigh-Taylor
25 //! \see Waltz, et. al, "Manufactured solutions for the three-dimensional Euler
26 //! equations with relevance to Inertial Confinement Fusion", Journal of
27 //! Computational Physics 267 (2014) 196-209.

31  //! Set initial conditions
32  //! \param[in] coord Mesh node coordinates
33  //! \param[in,out] unk Array of unknowns
34  //! \param[in] e Equation system index, i.e., which compressible
35  //! flow equation system we operate on among the systems of PDEs
36  //! \param[in] offset System offset specifying the position of the system of
37  //! PDEs among other systems
38  static void init( const std::array< std::vector< tk::real >, 3 >& coord,
39  const std::vector< std::size_t >&,
40  tk::Fields& unk,
41  tk::ctr::ncomp_type e,
42  tk::ctr::ncomp_type offset,
43  tk::real )
44  {
45  Assert( coord[0].size() == unk.nunk(), "Size mismatch" );
46  // manufactured solution parameters
47  const auto& a =
48  g_inputdeck.get< tag::param, tag::compflow, tag::alpha >()[e];
49  const auto& bx =
50  g_inputdeck.get< tag::param, tag::compflow, tag::betax >()[e];
51  const auto& by =
52  g_inputdeck.get< tag::param, tag::compflow, tag::betay >()[e];
53  const auto& bz =
54  g_inputdeck.get< tag::param, tag::compflow, tag::betaz >()[e];
55  const auto& p0 =
56  g_inputdeck.get< tag::param, tag::compflow, tag::p0 >()[e];
57  const auto& r0 =
58  g_inputdeck.get< tag::param, tag::compflow, tag::r0 >()[e];
59  // ratio of specific heats
60  tk::real g =
61  g_inputdeck.get< tag::param, tag::compflow, tag::gamma >()[e];
62 
63  // set initial and boundary conditions
64  const auto& x = coord[0];
65  const auto& y = coord[1];
66  const auto& z = coord[2];
67  for (ncomp_t i=0; i<x.size(); ++i) {
68  auto& r = unk(i,0,offset); // rho
69  auto& ru = unk(i,1,offset); // rho * u
70  auto& rv = unk(i,2,offset); // rho * v
71  auto& rw = unk(i,3,offset); // rho * w
72  auto& re = unk(i,4,offset); // rho * e
73  // pressure field
74  tk::real p = p0 + a*(bx*x[i]*x[i] + by*y[i]*y[i] + bz*z[i]*z[i]);
75  r = r0-(bx*x[i]*x[i] + by*y[i]*y[i] + bz*z[i]*z[i]);
76  ru = r*(z[i]*std::sin(M_PI*x[i]));
77  rv = r*(z[i]*std::cos(M_PI*y[i]));
78  rw = r*(-M_PI*z[i]*z[i]*(std::cos(M_PI*x[i])-std::sin(M_PI*y[i]))/2.0);
79  re = p/(g-1) + (ru*ru + rv*rv + rw*rw)/2.0;
80  }
81  }

329  rho[2]*f*f*(gx[0][2]*dg[2][0][2] + gx[1][2]*dg[2][1][2] +
330  gx[2][2]*dg[2][2][2]),
331  rho[3]*gx[2][3]*df + f*gx[2][3]*Sr[3] +
332  rho[3]*f*f*(gx[0][3]*dg[2][0][3] + gx[1][3]*dg[2][1][3] +
333  gx[2][3]*dg[2][2][3]) }} }};
334 
335  // add momentum and energy source at element nodes
336  for (std::size_t j=0; j<4; ++j)
337  for (std::size_t k=0; k<4; ++k) {
338  // source contribution to mass rhs
339  R.var(r[0],N[j]) += dt * mass[j][k] * Sr[k];
340  // source contribution to momentum rhs
341  for (std::size_t l=0; l<3; ++l)
342  R.var(r[l+1],N[j]) += dt * mass[j][k] * Sm[l][k];
343  // source contribution to enerhy rhs
344  R.var(r[4],N[j]) += dt * mass[j][k] * Se[k];
345  }
346  }

412  //! Return field output going to file
413  //! \param[in] e Equation system index, i.e., which compressible
414  //! flow equation system we operate on among the systems of PDEs
415  //! \param[in] offset System offset specifying the position of the system of
416  //! PDEs among other systems
417  //! \param[in] t Physical time
418  //! \param[in] coord Mesh node coordinates
419  //! \param[in] U Solution vector at recent time step stage
420  //! \return Vector of vectors to be output to file
421  static std::vector< std::vector< tk::real > >
422  fieldOutput( tk::ctr::ncomp_type e,
423  tk::ctr::ncomp_type offset,
424  tk::real t,
425  tk::real,
426  const std::vector< tk::real >&,
427  const std::array< std::vector< tk::real >, 3 >& coord,
428  const tk::Fields& U )
429  {
430  // manufactured solution parameters
431  const auto& a =
432  g_inputdeck.get< tag::param, tag::compflow, tag::alpha >()[e];
433  const auto& bx =
434  g_inputdeck.get< tag::param, tag::compflow, tag::betax >()[e];
435  const auto& by =
436  g_inputdeck.get< tag::param, tag::compflow, tag::betay >()[e];
437  const auto& bz =
438  g_inputdeck.get< tag::param, tag::compflow, tag::betaz >()[e];
439  const auto& kappa =
440  g_inputdeck.get< tag::param, tag::compflow, tag::kappa >()[e];
441  const auto& p0 =
442  g_inputdeck.get< tag::param, tag::compflow, tag::p0 >()[e];
443  const auto& r0 =
444  g_inputdeck.get< tag::param, tag::compflow, tag::r0 >()[e];
445  // ratio of specific heats
446  tk::real g =
447  g_inputdeck.get< tag::param, tag::compflow, tag::gamma >()[e];
448 
449  std::vector< std::vector< tk::real > > out;
450  const auto r = U.extract( 0, offset );
451  const auto ru = U.extract( 1, offset );
452  const auto rv = U.extract( 2, offset );
453  const auto rw = U.extract( 3, offset );
454  const auto re = U.extract( 4, offset );
455 
456  // mesh node coordinates
457  const auto& x = coord[0];
458  const auto& y = coord[1];
459  const auto& z = coord[2];
460 
461  std::vector< tk::real > rho = r;
462  out.push_back( rho );
463  for (std::size_t i=0; i<rho.size(); ++i) {
464  rho[i] = r0-(bx*x[i]*x[i] + by*y[i]*y[i] + bz*z[i]*z[i]);
465  }
466  out.push_back( rho);
467 
468  std::vector< tk::real > u = ru;
469  std::transform( r.begin(), r.end(), u.begin(), u.begin(),
470  []( tk::real s, tk::real& d ){ return d /= s; } );
471  out.push_back( u );
472  for (std::size_t i=0; i<u.size(); ++i) {
473  u[i] = rho[i]*std::cos(kappa*M_PI*t)*(z[i]*std::sin(M_PI*x[i]));
474  }
475  out.push_back( u );
476 
477  std::vector< tk::real > v = rv;
478  std::transform( r.begin(), r.end(), v.begin(), v.begin(),
479  []( tk::real s, tk::real& d ){ return d /= s; } );
480  out.push_back( v );
481  for (std::size_t i=0; i<v.size(); ++i) {
482  v[i] = rho[i]*std::cos(kappa*M_PI*t)*(z[i]*std::cos(M_PI*y[i]));
483  }
484  out.push_back( v );
485 
486  std::vector< tk::real > w = rw;
487  std::transform( r.begin(), r.end(), w.begin(), w.begin(),
488  []( tk::real s, tk::real& d ){ return d /= s; } );
489  out.push_back( w );
490  for (std::size_t i=0; i<w.size(); ++i) {
491  w[i] = rho[i]*std::cos(kappa*M_PI*t)*(-M_PI*z[i]*z[i]*
492  (std::cos(M_PI*x[i])-std::sin(M_PI*y[i]))/2.0);
493  }
494  out.push_back( w );
495 
496  std::vector< tk::real > E = re;
497  std::transform( r.begin(), r.end(), E.begin(), E.begin(),
498  []( tk::real s, tk::real& d ){ return d /= s; } );
499  out.push_back( E );
500  for (std::size_t i=0; i<E.size(); ++i) {
501  tk::real p = p0 + a*(bx*x[i]*x[i] + by*y[i]*y[i] + bz*z[i]*z[i]);
502  E[i] = p/(g-1) + (u[i]*u[i] + v[i]*v[i] + w[i]*w[i])/2.0;
503  }
504  out.push_back( E );
505 
506  return out;
507  }
