24 //! CompFlow system of PDEs problem: Rayleigh-Taylor
25 //! \see Waltz, et. al, "Manufactured solutions for the three-dimensional Euler
26 //! equations with relevance to Inertial Confinement Fusion", Journal of
27 //! Computational Physics 267 (2014) 196-209.
28 class CompFlowProblemRayleighTaylor {
29 
30  public:
31  //! Evaluate analytical solution at (x,y,z,t) for all components
32  //! \param[in] e Equation system index, i.e., which compressible
33  //! flow equation system we operate on among the systems of PDEs
34  //! \param[in] x X coordinate where to evaluate the solution
35  //! \param[in] y Y coordinate where to evaluate the solution
36  //! \param[in] z Z coordinate where to evaluate the solution
37  //! \param[in] t Time where to evaluate the solution
38  //! \return Values of all components evaluated at (x,y,z,t)
39  static std::array< tk::real, 5 >
40  solution( tk::ctr::ncomp_type e,
41  tk::real x, tk::real y, tk::real z, tk::real t )
42  {
43  using tag::param; using tag::compflow; using std::sin; using std::cos;
44  // manufactured solution parameters
45  const auto a = g_inputdeck.get< param, compflow, tag::alpha >()[e];
46  const auto bx = g_inputdeck.get< param, compflow, tag::betax >()[e];
47  const auto by = g_inputdeck.get< param, compflow, tag::betay >()[e];
48  const auto bz = g_inputdeck.get< param, compflow, tag::betaz >()[e];
49  const auto p0 = g_inputdeck.get< param, compflow, tag::p0 >()[e];
50  const auto r0 = g_inputdeck.get< param, compflow, tag::r0 >()[e];
51  const auto k = g_inputdeck.get< param, compflow, tag::kappa >()[e];
52  // ratio of specific heats
53  const tk::real g = g_inputdeck.get< param, compflow, tag::gamma >()[e];
54  // spatial component of density and pressure fields
55  const tk::real gx = bx*x*x + by*y*y + bz*z*z;
56  // density
57  const tk::real r = r0 - gx;
58  // pressure
59  const tk::real p = p0 + a*gx;
60  // velocity
61  const tk::real ft = cos(k*M_PI*t);
62  const tk::real u = ft*z*sin(M_PI*x);
63  const tk::real v = ft*z*cos(M_PI*y);
64  const tk::real w = ft*(-0.5*M_PI*z*z*(cos(M_PI*x)-sin(M_PI*y)));
65  // total specific energy
66  const tk::real rE = p/(g-1.0) + 0.5*r*(u*u + v*v + w*w);
67  return {{ r, r*u, r*v, r*w, rE }};
68  }

70  //! \brief Evaluate the increment from t to t+dt of the analytical solution
71  //! at (x,y,z) for all components
72  //! \param[in] e Equation system index, i.e., which compressible
73  //! flow equation system we operate on among the systems of PDEs
74  //! \param[in] x X coordinate where to evaluate the solution
75  //! \param[in] y Y coordinate where to evaluate the solution
76  //! \param[in] z Z coordinate where to evaluate the solution
77  //! \param[in] t Time where to evaluate the solution increment starting from

80  static std::array< tk::real, 5 >
81  solinc( tk::ctr::ncomp_type e,

88  return st2;

91  //! Compute and return source term for Rayleigh-Taylor manufactured solution
92  //! \param[in] e Equation system index, i.e., which compressible
93  //! flow equation system we operate on among the systems of PDEs
94  //! \param[in] x X coordinate where to evaluate the solution
95  //! \param[in] y Y coordinate where to evaluate the solution
96  //! \param[in] z Z coordinate where to evaluate the solution
97  //! \param[in] t Physical time at which to evaluate the source
98  //! \return Array of reals containing the source for all components
99  static std::array< tk::real, 5 >
100  src( tk::ctr::ncomp_type e, tk::real x, tk::real y, tk::real z, tk::real t )
101  {
102  using tag::param; using tag::compflow; using std::sin; using std::cos;
103 
104  // manufactured solution parameters
105  auto a = g_inputdeck.get< param, compflow, tag::alpha >()[e];
106  auto bx = g_inputdeck.get< param, compflow, tag::betax >()[e];
107  auto by = g_inputdeck.get< param, compflow, tag::betay >()[e];
108  auto bz = g_inputdeck.get< param, compflow, tag::betaz >()[e];
109  auto k = g_inputdeck.get< param, compflow, tag::kappa >()[e];
110  auto p0 = g_inputdeck.get< param, compflow, tag::p0 >()[e];
111  // ratio of specific heats
112  tk::real g = g_inputdeck.get< param, compflow, tag::gamma >()[e];
113 
114  // evaluate solution at x,y,z,t
115  auto s = solution( e, x, y, z, t );
116 
117  // density, velocity, energy, pressure
118  auto rho = s[0];
119  auto u = s[1]/s[0];
120  auto v = s[2]/s[0];
121  auto w = s[3]/s[0];
122  auto E = s[4]/s[0];
123  auto p = p0 + a*(bx*x*x + by*y*y + bz*z*z);
124 
125  // spatial gradients
126  std::array< tk::real, 3 > drdx{{ -2.0*bx*x, -2.0*by*y, -2.0*bz*z }};
127  std::array< tk::real, 3 > dpdx{{ 2.0*a*bx*x, 2.0*a*by*y, 2.0*a*bz*z }};
128  tk::real ft = cos(k*M_PI*t);
129  std::array< tk::real, 3 > dudx{{ ft*M_PI*z*cos(M_PI*x),
130  0.0,
131  ft*sin(M_PI*x) }};
132  std::array< tk::real, 3 > dvdx{{ 0.0,
133  -ft*M_PI*z*sin(M_PI*y),
134  ft*cos(M_PI*y) }};
135  std::array< tk::real, 3 > dwdx{{ ft*M_PI*0.5*M_PI*z*z*sin(M_PI*x),
136  ft*M_PI*0.5*M_PI*z*z*cos(M_PI*y),
137  -ft*M_PI*z*(cos(M_PI*x) - sin(M_PI*y)) }};
138  std::array< tk::real, 3 > dedx{{
139  dpdx[0]/rho/(g-1.0) - p/(g-1.0)/rho/rho*drdx[0]
140  + u*dudx[0] + v*dvdx[0] + w*dwdx[0],
141  dpdx[1]/rho/(g-1.0) - p/(g-1.0)/rho/rho*drdx[1]
142  + u*dudx[1] + v*dvdx[1] + w*dwdx[1],
143  dpdx[2]/rho/(g-1.0) - p/(g-1.0)/rho/rho*drdx[2]
144  + u*dudx[2] + v*dvdx[2] + w*dwdx[2] }};
145 
146  // time derivatives
147  auto dudt = -k*M_PI*sin(k*M_PI*t)*z*sin(M_PI*x);
148  auto dvdt = -k*M_PI*sin(k*M_PI*t)*z*cos(M_PI*y);
149  auto dwdt = k*M_PI*sin(k*M_PI*t)/2*M_PI*z*z*(cos(M_PI*x) - sin(M_PI*y));
150  auto dedt = u*dudt + v*dvdt + w*dwdt;
151 
152  std::array< tk::real, 5 > r;
153  // density source
154  r[0] = u*drdx[0] + v*drdx[1] + w*drdx[2];
155  // momentum source
156  r[1] = rho*dudt+u*r[0]+dpdx[0] + s[1]*dudx[0]+s[2]*dudx[1]+s[3]*dudx[2];
157  r[2] = rho*dvdt+v*r[0]+dpdx[1] + s[1]*dvdx[0]+s[2]*dvdx[1]+s[3]*dvdx[2];
158  r[3] = rho*dwdt+w*r[0]+dpdx[2] + s[1]*dwdx[0]+s[2]*dwdx[1]+s[3]*dwdx[2];
159  // energy source
160  r[4] = rho*dedt + E*r[0] + s[1]*dedx[0]+s[2]*dedx[1]+s[3]*dedx[2]
161  + u*dpdx[0]+v*dpdx[1]+w*dpdx[2];
162 
163  return r;
164  }

201  //! Return field output going to file
202  //! \param[in] e Equation system index, i.e., which compressible
203  //! flow equation system we operate on among the systems of PDEs
204  //! \param[in] offset System offset specifying the position of the system of
205  //! PDEs among other systems
206  //! \param[in] t Physical time
207  //! \param[in] coord Mesh node coordinates
208  //! \param[in] U Solution vector at recent time step
209  //! \return Vector of vectors to be output to file
210  static std::vector< std::vector< tk::real > >
211  fieldOutput( tk::ctr::ncomp_type e,
212  tk::ctr::ncomp_type offset,
213  tk::real t,
214  tk::real V,
215  const std::vector< tk::real >& vol,
216  const std::array< std::vector< tk::real >, 3 >& coord,
217  tk::Fields& U )
218  {
219  // ratio of specific heats
220  tk::real g =
221  g_inputdeck.get< tag::param, tag::compflow, tag::gamma >()[e];
222 
223  std::vector< std::vector< tk::real > > out;
224  auto r = U.extract( 0, offset );
225  auto u = U.extract( 1, offset );
226  auto v = U.extract( 2, offset );
227  auto w = U.extract( 3, offset );
228  auto E = U.extract( 4, offset );
229 
230  // mesh node coordinates
231  const auto& x = coord[0];
232  const auto& y = coord[1];
233  const auto& z = coord[2];
234 
235  out.push_back( r );
236  std::transform( r.begin(), r.end(), u.begin(), u.begin(),
237  []( tk::real s, tk::real& d ){ return d /= s; } );
238  out.push_back( u );
239  std::transform( r.begin(), r.end(), v.begin(), v.begin(),
240  []( tk::real s, tk::real& d ){ return d /= s; } );
241  out.push_back( v );
242  std::transform( r.begin(), r.end(), w.begin(), w.begin(),
243  []( tk::real s, tk::real& d ){ return d /= s; } );
244  out.push_back( w );
245  std::transform( r.begin(), r.end(), E.begin(), E.begin(),
246  []( tk::real s, tk::real& d ){ return d /= s; } );
247  out.push_back( E );
248 
249  auto p = r;
250  for (std::size_t i=0; i<r.size(); ++i)
251  p[i] = (g-1.0)*r[i]*(E[i] - (u[i]*u[i] + v[i]*v[i] + w[i]*w[i])/2.0);
252  out.push_back( p );
253 
254  auto er = r, ee = r, ep = r, eu = r, ev = r, ew = r;
255  for (std::size_t i=0; i<r.size(); ++i) {
256  auto s = solution( e, x[i], y[i], z[i], t );
257  er[i] = std::pow( r[i] - s[0], 2.0 ) * vol[i] / V;
258  ee[i] = std::pow( E[i] - s[4]/s[0], 2.0 ) * vol[i] / V;
259  eu[i] = std::pow( u[i] - s[1]/s[0], 2.0 ) * vol[i] / V;
260  ev[i] = std::pow( v[i] - s[2]/s[0], 2.0 ) * vol[i] / V;
261  ew[i] = std::pow( w[i] - s[3]/s[0], 2.0 ) * vol[i] / V;
262  auto ap = (g-1.0)*(s[4] - (s[1]*s[1] + s[2]*s[2] + s[3]*s[3])/2.0/s[0]);
263  r[i] = s[0];
264  u[i] = s[1]/s[0];
265  v[i] = s[2]/s[0];
266  w[i] = s[3]/s[0];
267  E[i] = s[4]/s[0];
268  p[i] = (g-1.0)*r[i]*(E[i] - (u[i]*u[i] + v[i]*v[i] + w[i]*w[i])/2.0);
269  ep[i] = std::pow( ap - p[i], 2.0 ) * vol[i] / V;
270  }
271 
272  out.push_back( r );
273  out.push_back( u );
274  out.push_back( v );
275  out.push_back( w );
276  out.push_back( E );
277  out.push_back( p );
278 
279  out.push_back( er );
280  out.push_back( ee );
281  out.push_back( ep );
282  out.push_back( eu );
283  out.push_back( ev );
284  out.push_back( ew );
285 
286  return out;
287  }
