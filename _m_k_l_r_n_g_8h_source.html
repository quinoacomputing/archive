18 //! MKL-based random number generator used polymorphically with tk::RNG

21  using ncomp_t = kw::ncomp::info::expect::type;
22 
23  public:
24  //! Constructor
25  //! \param[in] n Initialize RNG using this many independent streams
26  //! \param[in] brng Index of the basic generator to initialize the stream
27  //! \param[in] seed RNG seed
28  //! \param[in] uniform_method MKL ID of the method to use for uniform RNGs
29  //! \param[in] gaussian_method MKL ID of the method to use for Gaussian RNGs
30  //! \param[in] beta_method MKL ID of the method to use for beta RNGs
31  //! \param[in] gamma_method MKL ID of the method to use for gamma RNGs
32  explicit MKLRNG( int n = 1,
33  int brng = VSL_BRNG_MCG59,
34  unsigned int seed = 0,
35  int uniform_method = VSL_RNG_METHOD_UNIFORM_STD,
36  int gaussian_method = VSL_RNG_METHOD_GAUSSIAN_BOXMULLER,
37  int beta_method = VSL_RNG_METHOD_BETA_CJA,
38  int gamma_method = VSL_RNG_METHOD_GAMMA_GNORM ) :
39  m_brng( brng ),
40  m_seed( seed ),
41  m_uniform_method( uniform_method ),
42  m_gaussian_method( gaussian_method ),
43  m_beta_method( beta_method ),
44  m_gamma_method( gamma_method ),
45  m_nthreads( n ),
46  m_stream()
47  {
48  Assert( n > 0, "Need at least one thread" );
49  Assert( brng > 0, "Basic RNG MKL parameter must be positive" );
50  // Allocate array of stream-pointers for threads
51  m_stream = tk::make_unique< VSLStreamStatePtr[] >(
52  static_cast<std::size_t>(n) );
53  // Initialize thread-streams for block-splitting. These MKL VSL functions
54  // dynamically allocate memory, so these calls being in a constructor are
55  // a potential memory leak hazard in the presence of exceptions. However,
56  // thankfully, the MKL functions below only emit warnings if they
57  // encounter errors and always continue. As a result, the constructor
58  // finishes, the MKLRNG object gets created, so the destructor will also
59  // get called when leaving scope.
60  if (n == 1)
61  errchk( vslNewStream( &m_stream[0], brng, seed ) );
62  else
63  for (int i=0; i<n; ++i) {
64  auto I = static_cast< std::size_t >( i );
65  errchk( vslNewStream( &m_stream[I], brng, seed ) );
66  errchk( vslLeapfrogStream( m_stream[I], i, n ) );
67  }
68  }
69 
70  //! Destructor
71  ~MKLRNG() noexcept { deleteStreams(); }

73  //! Uniform RNG: Generate uniform random numbers
74  //! \param[in] tid Thread (or more precisely stream) ID
75  //! \param[in] num Number of RNGs to generate
76  //! \param[in,out] r Pointer to memory to write the random numbers to
77  void uniform( int tid, ncomp_t num, double* r ) const {
78  vdRngUniform( m_uniform_method,
79  m_stream[ static_cast<std::size_t>(tid) ],
80  static_cast< long long >( num ),
81  r,
82  0.0, 1.0 );
83  }
84

88  //! \param[in,out] r Pointer to memory to write the random numbers to

95  }

98  //! \param[in] tid Thread (or more precisely stream) ID
99  //! \param[in] num Number of RNGs to generate
100  //! \param[in] p First beta shape parameter

116  //! \param[in] tid Thread (or more precisely stream) ID
117  //! \param[in] num Number of RNGs to generate
118  //! \param[in] a Gamma shape parameter
119  //! \param[in] b Gamma scale factor

121  void gamma( int tid, ncomp_t num, double a, double b, double* r ) const

123  vdRngGamma( m_beta_method,
124  m_stream[ static_cast<std::size_t>(tid) ],
125  static_cast< long long >( num ),
126  r,
127  a, 0.0, b ); // displacement = 0.0
128  }

147  errchk( vslLeapfrogStream( m_stream[I], i, x.m_nthreads ) );
148  }

181  }

202  for (int i=0; i<m_nthreads; ++i) {
203  auto I = static_cast< std::size_t >( i );
204  if (m_stream[I]) {

211  //! MKL VSL error check
212  //! \param[in] err MKL VSL error code as returned from MKL VSL functions

218  std::to_string(err) + ", see mkl_vsl_defines.h for more info" );
219  }
