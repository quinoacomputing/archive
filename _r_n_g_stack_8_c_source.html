80 std::map< tk::ctr::RawRNGType, tk::RNG >
81 RNGStack::selected( const std::vector< tk::ctr::RNGType >& sel ) const
82 // *****************************************************************************
83 // Instantiate selected RNGs from factory and place them in map
84 //! \param[in] sel Vector of selected RNGs to instantiate (selected by the user)
85 //! \return A std::map of keys and their associated instantiated RNG objects
86 // *****************************************************************************
87 {
88  using tk::ctr::RawRNGType;
89  std::map< RawRNGType, tk::RNG > rng;
90  for (const auto& s : sel) {
91  const auto r = m_factory.find(s);
92  if (r != end(m_factory))
93  rng.emplace( static_cast< RawRNGType >( s ), r->second() );
94  else Throw( "RNG not found in factory" );
95  }
96  return rng;
97 }

102 // *****************************************************************************
103 // Register MKL random number generators into factory
104 //! \details Note that registering these entries in the map does not
105 //! invoke the constructors. The mapped value simply stores how the
106 //! constructors should be invoked at a later time. At some point later,
107 //! based on user input, we then instantiate only the RNGs (correctly
108 //! configured by the pre-bound constructor arguments) that are requested by
109 //! the user.
110 //! \param[in] nstreams Register MKL RNG using this many independent streams
111 //! \param[in] param MKL RNG parameters to use to configure the RNGs
112 // *****************************************************************************
113 {
114  using tk::ctr::RNGType;
115  using tk::ctr::MKLUniformMethodType;
116  using tk::ctr::MKLGaussianMethodType;
117  using tk::ctr::MKLBetaMethodType;
118  using tag::uniform_method;
119  using tag::gaussian_method;
120  using tag::beta_method;
121 
122  // Defaults for MKL RNGs
123  unsigned int s_def = 0;
124  MKLUniformMethodType u_def = MKLUniformMethodType::STANDARD;
125  MKLGaussianMethodType g_def = MKLGaussianMethodType::BOXMULLER;
126  MKLBetaMethodType b_def = MKLBetaMethodType::CJA;
127 
128  tk::ctr::RNG opt;
129  tk::ctr::MKLUniformMethod um_opt;
130  tk::ctr::MKLGaussianMethod gm_opt;
131  tk::ctr::MKLBetaMethod bm_opt;
132 
133  //! Lambda to register a MKL random number generator into factory
134  auto regMKLRNG = [&]( RNGType rng ) {
135  recordModel< tk::RNG, tk::MKLRNG >
136  ( m_factory, rng,
137  nstreams,
138  opt.param( rng ),
139  opt.param< tag::seed >( rng, s_def, param ),
140  um_opt.param( opt.param< uniform_method >( rng, u_def, param ) ),
141  gm_opt.param( opt.param< gaussian_method >( rng, g_def, param) ),
142  bm_opt.param( opt.param< beta_method >( rng, b_def, param) ) );
143  };
