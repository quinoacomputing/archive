2 /*!
3  \file src/Inciter/MatCG.C
4  \copyright 2012-2015, J. Bakosi, 2016-2018, Los Alamos National Security, LLC.
5  \brief MatCG for a PDE system with continuous Galerkin with a matrix
6  \details MatCG advances a system of partial differential equations (PDEs)
7  using continuous Galerkin (CG) finite element (FE) spatial discretization
8  (using linear shapefunctions on tetrahedron elements) combined with a time
9  stepping scheme that is equivalent to the Lax-Wendroff (LW) scheme within
10  the unstructured-mesh FE context and treats discontinuities with
11  flux-corrected transport (FCT). The left-hand side matrix is stored in a
12  compressed sparse row (CSR) storage and thus uses a matrix-based linear
13  solver.
14  \see The documentation in MatCG.h.
15 */

95 void
96 MatCG::setup( tk::real v )
97 // *****************************************************************************
98 // Setup rows, query boundary conditions, output mesh, etc.
99 //! \param[in] v Total mesh volume
100 // *****************************************************************************
101 {
102  auto d = Disc();
103 
104  m_solver.ckLocalBranch()->comfinal();
105 
106  // Store total mesh volume
107  m_vol = v;
108  // Output chare mesh to file
109  d->writeMesh();
110  // Output fields metadata to output file
111  d->writeNodeMeta();
112 
113  // Compute left-hand side of PDEs
114  lhs();
115 
116  // zero initial solution vector
117  m_du.fill( 0.0 );
118 
119  // Set initial conditions for all PDEs
120  for (const auto& eq : g_cgpde) eq.initialize( d->Coord(), m_u, d->T() );
121 
122  // Send off initial guess for assembly
123  m_solver.ckLocalBranch()->charesol( thisIndex, d->Gid(), m_du );
124 
125  // Output initial conditions to file (regardless of whether it was requested)
126  if ( !g_inputdeck.get< tag::cmd, tag::benchmark >() ) writeFields( d->T() );
127 
128  // Start time stepping
129  contribute( CkCallback( CkReductionTarget(Transporter,start), d->Tr()) );
130 
131  // Start timer measuring time stepping wall clock time
132  d->Timer().zero();
133 }

135 void
136 MatCG::dt()
137 // *****************************************************************************
138 // Comppute time step size
139 // *****************************************************************************
140 {
141  tk::real mindt = std::numeric_limits< tk::real >::max();
142 
143  auto const_dt = g_inputdeck.get< tag::discr, tag::dt >();
144  auto def_const_dt = g_inputdeck_defaults.get< tag::discr, tag::dt >();
145  auto eps = std::numeric_limits< tk::real >::epsilon();
146 
147  auto d = Disc();
148 
149  // use constant dt if configured
150  if (std::abs(const_dt - def_const_dt) > eps) {
151 
152  mindt = const_dt;
153 
154  } else { // compute dt based on CFL
155 
156  // find the minimum dt across all PDEs integrated
157  for (const auto& eq : g_cgpde) {
158  auto eqdt = eq.dt( d->Coord(), d->Inpoel(), m_u );
159  if (eqdt < mindt) mindt = eqdt;
160  }
161 
162  // Scale smallest dt with CFL coefficient
163  mindt *= g_inputdeck.get< tag::discr, tag::cfl >();
164 
165  }
166 
167  // Contribute to minimum dt across all chares the advance to next step
168  contribute( sizeof(tk::real), &mindt, CkReduction::min_double,
169  CkCallback(CkReductionTarget(MatCG,advance), thisProxy) );
170 }

172 void
173 MatCG::lhs()
174 // *****************************************************************************
175 // Compute left-hand side of transport equations
176 // *****************************************************************************
177 {
178  auto d = Disc();
179 
180  // Compute left-hand side matrix for all equations
181  for (const auto& eq : g_cgpde)
182  eq.lhs( d->Coord(), d->Inpoel(), d->Psup(), m_lhsd, m_lhso );
183 
184  // Send off left hand side for assembly
185  m_solver.ckLocalBranch()->
186  charelhs( thisIndex, d->Gid(), d->Psup(), m_lhsd, m_lhso );
187 
188  // Compute lumped mass lhs required for the low order solution
189  auto lump = d->FCT()->lump( *d );
190  // Send off lumped mass lhs for assembly
191  m_solver.ckLocalBranch()->charelowlhs( thisIndex, d->Gid(), lump );
192 }

194 void
195 MatCG::rhs()
196 // *****************************************************************************
197 // Compute right-hand side of transport equations
198 // *****************************************************************************
199 {
200  auto d = Disc();
201 
202  // Compute right-hand side and query Dirichlet BCs for all equations
203  tk::Fields r( d->Gid().size(), g_inputdeck.get< tag::component >().nprop() );
204  for (const auto& eq : g_cgpde)
205  eq.rhs( d->T(), d->Dt(), d->Coord(), d->Inpoel(), m_u, m_ue, r );
206 
207  // Query and match user-specified boundary conditions to side sets
208  bc();
209 
210  // Send off right-hand sides for assembly
211  m_solver.ckLocalBranch()->charerhs( thisIndex, d->Gid(), r );
212 
213  // Compute mass diffusion rhs contribution required for the low order solution
214  auto diff = d->FCT()->diff( *d, m_u );
215  // Send off mass diffusion rhs contribution for assembly
216  m_solver.ckLocalBranch()->charelowrhs( thisIndex, d->Gid(), diff );
217 }

235 void
236 MatCG::updateLowSol( const std::vector< std::size_t >& gid,
237  const std::vector< tk::real >& du )
238 // *****************************************************************************
239 // Update low order solution vector
240 //! \param[in] gid Global row indices of the vector updated
241 //! \param[in] du Portion of the unknown/solution vector update
242 // *****************************************************************************
243 {
244  auto ncomp = g_inputdeck.get< tag::component >().nprop();
245  Assert( gid.size() * ncomp == du.size(),
246  "Size of row ID vector times the number of scalar components and the "
247  "size of the low order solution vector must equal" );
248 
249  auto d = Disc();
250 
251  // Receive update of solution vector
252  for (std::size_t i=0; i<gid.size(); ++i) {
253  auto id = tk::cref_find( d->Lid(), gid[i] );
254  for (ncomp_t c=0; c<ncomp; ++c) m_dul( id, c, 0 ) = du[ i*ncomp+c ];
255  }
256 
257  // Count number of solution nodes updated
258  m_nlsol += gid.size();
259 
260  // If all contributions we own have been received, continue with FCT
261  if (m_nlsol == d->Gid().size()) {
262  m_nlsol = 0;
263  m_ul = m_u + m_dul;
264  d->FCT()->alw( m_u, m_ul, m_dul, thisProxy );
265  }
266 }

300 void
301 MatCG::writeFields( tk::real time )
302 // *****************************************************************************
303 // Output mesh-based fields to file
304 //! \param[in] time Physical time
305 // *****************************************************************************
306 {
307  auto d = Disc();
308 
309  // Only write if the last time is different than the current one
310  if (std::abs(d->LastFieldWriteTime() - time) <
311  std::numeric_limits< tk::real >::epsilon() )
312  return;
313 
314  // Save time stamp at which the last field write happened
315  d->LastFieldWriteTime() = time;
316 
317  // Increase field output iteration count
318  ++m_itf;
319 
320  // Lambda to collect node fields output from all PDEs
321  auto nodefields = [&]() {
322  auto u = m_u; // make a copy as eq::output() may overwrite its arg
323  std::vector< std::vector< tk::real > > output;
324  for (const auto& eq : g_cgpde) {
325  auto o = eq.fieldOutput( time, m_vol, d->Coord(), d->V(), u );
326  output.insert( end(output), begin(o), end(o) );
327  }
328  return output;
329  };
330 
331  #ifdef HAS_ROOT
332  auto filetype = g_inputdeck.get< tag::selected, tag::filetype >();
333 
334  if (filetype == tk::ctr::FieldFileType::ROOT) {
335 
336  // Create Root writer
337  tk::RootMeshWriter rmw( d->OutFilename(), 1 );
338  // Write time stamp
339  rmw.writeTimeStamp( m_itf, time );
340  // Write node fields to file
341  d->writeNodeSolution( rmw, m_itf, nodefields() );
342 
343  } else
344  #endif
345  {
346 
347  // Create ExodusII writer
348  tk::ExodusIIMeshWriter ew( d->OutFilename(), tk::ExoWriter::OPEN );
349  // Write time stamp
350  ew.writeTimeStamp( m_itf, time );
351  // Write node fields to file
352  d->writeNodeSolution( ew, m_itf, nodefields() );
353 
354  }

357 void
358 MatCG::out()
359 // *****************************************************************************
360 // Output mesh field data
361 // *****************************************************************************
362 {
363  auto d = Disc();
364 
365  // Optionally output field and particle data
366  if ( !((d->It()+1) % g_inputdeck.get< tag::interval, tag::field >()) &&
367  !g_inputdeck.get< tag::cmd, tag::benchmark >() )
368  {
369  writeFields( d->T()+d->Dt() );
370  }
371 
372  // Output final field data to file (regardless of whether it was requested)
373  const auto term = g_inputdeck.get< tag::discr, tag::term >();
374  const auto eps = std::numeric_limits< tk::real >::epsilon();
375  const auto nstep = g_inputdeck.get< tag::discr, tag::nstep >();
376  if ( (std::fabs(d->T() + d->Dt()-term) < eps || (d->It()+1) >= nstep) &&
377  (!g_inputdeck.get< tag::cmd, tag::benchmark >()) )
378  {
379  writeFields( d->T()+d->Dt() );
380  }
381 }

451 void
452 MatCG::eval()
453 // *****************************************************************************
454 // Evaluate whether to continue with next step
455 // *****************************************************************************
456 {
457  auto d = Disc();
458 
459  const auto term = g_inputdeck.get< tag::discr, tag::term >();
460  const auto nstep = g_inputdeck.get< tag::discr, tag::nstep >();
461  const auto eps = std::numeric_limits< tk::real >::epsilon();
462 
463  // If neither max iterations nor max time reached, continue, otherwise finish
464  if (std::fabs(d->T()-term) > eps && d->It() < nstep)
465  contribute( CkCallback( CkReductionTarget(Transporter,next), d->Tr() ) );
466  else
467  contribute( CkCallback( CkReductionTarget(Transporter,finish), d->Tr() ) );
468 }
