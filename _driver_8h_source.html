20 //! Driver. The class below uses runtime polymorphism without client-side
21 //! inheritance: inheritance is confined to the internals of the class below,
22 //! inivisble to client-code. The class exclusively deals with ownership
23 //! enabling client-side value semantics. Credit goes to Sean Parent at Adobe:
24 //! https://github.com/sean-parent/sean-parent.github.com/wiki/
25 //! Papers-and-Presentations

30  template< typename T >
31  explicit Driver( T x ) : self( make_unique< Model<T> >( std::move(x) ) ) {}
32 
33  //! Constructor taking a function pointer to a constructor of an object
34  //! modeling Concept (see below). Passing std::function allows late
35  //! execution of the constructor, i.e., as late as inside this class'
36  //! constructor, and thus usage from a factory.
37  template< typename T >
38  explicit Driver( std::function<T()> x ) :
39  self( make_unique< Model<T> >( std::move(x()) ) ) {}
