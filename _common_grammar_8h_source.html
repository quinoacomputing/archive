267  //! \brief Parser error and warning message handler.
268  //! \details This function is used to associated and dispatch an error or a
269  //! warning during parsing. After finding the error message corresponding to
270  //! a key, it pushes back the message to a std::vector of std::string, which
271  //! then will be diagnosed later by tk::FileParser::diagnostics. The
272  //! template arguments define (1) the grammar stack (Stack, a tagged tuple)
273  //! to operate on, (2) the message type (error or warning), and (3) the
274  //! message key used to look up the error message associated with the key.
275  //! \param[in,out] stack Grammar stack (a tagged tuple) to operate on
276  //! \param[in] in Last parsed PEGTL input token (can be empty, depending on
277  //! what context this function gets called.
278  template< class Stack, MsgType type, MsgKey key, class Input >
279  static void Message( Stack& stack, const Input& in ) {
280  const auto& msg = message.find(key);
281  if (msg != message.end()) {
282  std::stringstream ss;
283  const std::string typestr( type == MsgType::ERROR ? "Error" : "Warning" );
284  auto pos = in.position();
285  if (!in.empty()) {
286  ss << typestr << " while parsing '" << in.string() << "' at "
287  << pos.line << ',' << pos.byte_in_line << ". " << msg->second;
288  } else {
289  ss << typestr << " while parsing at " << pos.line << ','
290  << pos.byte_in_line << ". " << msg->second;
291  }
292  stack.template push_back< tag::error >( ss.str() );
293  } else {
294  stack.template push_back< tag::error >
295  ( std::string("Unknown parser ") +
296  (type == MsgType::ERROR ? "error" : "warning" ) +
297  " with no location information." );
298  }
299  }

332  //! \brief Put option (i.e., a tk::Toggle) in grammar state (or stack) at a
333  //! position given by tags
334  //! \details This function is used to store an option (an object deriving from
335  //! tk::Toggle) into the grammar stack. See walker::ctr::DiffEq for an
336  //! example specialization of tk::Toggle to see how an option is created
337  //! from tk::Toggle.) The grammar stack is a hiearchical tagged tuple and
338  //! the variadic list of template arguments, tags..., are used to specify
339  //! a series tags (empty structs, see Control/Tags.h) addressing a
340  //! particular field of the tagged tuple, i.e., one tag for every additional
341  //! depth level.
342  //! \param[in,out] stack Grammar stack (a tagged tuple) to operate on
343  //! \param[in] in Last parsed PEGTL input token
344  //! \param[in] defaults Reference to a copy of the full grammar stack at the
345  //! initial state, i.e., containing the defaults for all of its fields. This
346  //! is used to detect if the user wants to overwrite an option value that
347  //! has already been set differently from the default
348  template< class Stack, template< class > class use, class Option,
349  class DefaultStack, class Input, class... tags >
350  static void store_option( Stack& stack,
351  const Input& in,
352  const DefaultStack& defaults ) {
353  Option opt;
354  auto value = in.string();
355  if (opt.exist(value)) {
356  auto pos = in.position();
357  // Emit warning on overwriting a non-default option value. This is
358  // slightly inelegant. To be more elegant, we could simply call Message()
359  // here, but the warning message can be more customized here (inside of
360  // this function) and thus produces a more user-friendly message, compared
361  // to a static message that Message() operates with due to its generic
362  // nature. Instead, we emit this more user-friendly message here
363  // (during parsing), instead of after parsing as part of the final parser-
364  // diagnostics. We still provide location information here though.
365  if (stack.template get< tags... >() != opt.value( value ) &&
366  stack.template get< tags... >() != defaults.template get< tags... >())
367  g_print << "\n>>> WARNING: Multiple definitions for '"
368  << opt.group() << "' option. Overwriting '"
369  << opt.name( stack.template get< tags... >() ) << "' with '"
370  << opt.name( opt.value( value ) ) << "' at "
371  << pos.line << ',' << pos.byte_in_line << ".\n\n";
372  stack.template set< tags... >( opt.value( value ) );
373  } else {
374  Message< Stack, ERROR, MsgKey::NOOPTION >( stack, in );
375  }
376  // trigger error at compile-time if any of the expected option values
377  // is not in the keywords pool of the grammar
378  boost::mpl::for_each< typename Option::keywords >( is_keyword< use >() );
379  }

1547  //! \brief Ensures that a grammar only uses keywords from a pool of
1548  //! pre-defined keywords
1549  //! \details In grammar definitions, every keyword should be wrapped around
1550  //! this use template, which conditionally inherits the keyword type its
1551  //! templated on (as defined in Control/Keywords.h) if the keyword is listed
1552  //! in any of the pools of keywords passed in as the required pool template
1553  //! argument and zero or more additional pools. If the keyword is not in any
1554  //! of the pools, a compiler error is generated, since the struct cannot
1555  //! inherit from base class 'char'. In that case, simply add the new keyword
1556  //! into one of the pools of keywords corresponding to the given grammar.
1557  //! The rationale behind this wrapper is to force the developer to maintain
1558  //! the keywords pool for a grammar. The pools are boost::mpl::sets and are
1559  //! used to provide help on command line arguments for a given executable.
1560  //! They allow compile-time iteration with boost::mpl::for_each or
1561  //! generating a run-time std::map associating, e.g., keywords to their help
1562  //! strings.
1563  //! \warning Since the default maximum number of elements in a
1564  //! boost::mpl::set is 20, and increasing this number would require custom
1565  //! boost::preprocessor-generated boost::mpl headers, which is cumbersome
1566  //! and error-prone, instead we work with several pools here that can each
1567  //! hold a maximum 20 items and use boost::mpl::or_ and boost::mpl::has_key
1568  //! to search for the keyword in either of the pools. Note that
1569  //! boost::mpl::or_ allows a maximum 5 template arguments by definition,
1570  //! i.e., max 5 OR'd pools, which corresponds to a maximum of 5x20=100
1571  //! keywords. If you need more than that, increase
1572  //! BOOST_MPL_LIMIT_METAFUNCTION_ARITY in src/CMakeLitst.txt.
1573  //! \warning Note that an even more elegant solution to the problem this
1574  //! wrapper is intended to solve is to use a metaprogram that collects all
1575  //! occurrences of the keywords in a grammar. However, that does not seem to
1576  //! be possible without extensive macro-trickery. Instead, if all keywords
1577  //! in all grammar definitions are wrapped inside this use template (or one
