253  //! Extract moment names of requested statistics
254  std::vector< std::string > momentNames( std::function<
255  bool ( const std::vector< tk::ctr::Term >& ) > momentType ) const
256  {
257  std::vector< std::string > names;
258  for (const auto& product : get< tag::stat >()) {
259  if (momentType( product )) {
260  names.emplace_back( std::string() );
261  for (const auto& term : product)
262  names.back() += tk::ctr::Term( term.var, term.field );
263  }
264  }
265  return names;
266  }

268  // Count number requested PDFs with given number of sample space dimensions

271  std::size_t n = 0;

273  if (bs.size() == d) ++n;
