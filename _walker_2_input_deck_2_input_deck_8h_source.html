208  //! \brief Constructor: set all defaults
209  //! \param[in] cl Previously parsed and store command line
210  //! \details Anything not set here is initialized by the compiler using the
211  //! default constructor for the corresponding type.
212  explicit InputDeck( const CmdLine& cl = {} ) {
213  // Set previously parsed command line
214  set< tag::cmd >( cl );
215  // Default discretization parameters
216  set< tag::discr, tag::npar >( 1 );
217  set< tag::discr, tag::nstep >
218  ( std::numeric_limits< kw::nstep::info::expect::type >::max() );
219  set< tag::discr, tag::term >( 1.0 );
220  set< tag::discr, tag::dt >( 0.5 );
221  // Default txt floating-point output precision in digits
222  set< tag::prec, tag::stat >( std::cout.precision() );
223  set< tag::prec, tag::pdf >( std::cout.precision() );
224  // Default intervals
225  set< tag::interval, tag::tty >( 1 );
226  set< tag::interval, tag::stat >( 1 );
227  set< tag::interval, tag::pdf >( 1 );
228  // Default requested statistics
229  set< tag::stat >( std::vector< tk::ctr::Product >() );
230  // Initialize help
231  const auto& ctrinfoFill = tk::ctr::Info( get< tag::cmd, tag::ctrinfo >() );
232  brigand::for_each< keywords >( ctrinfoFill );
233  }

235  //! Extract moment names of requested statistics
236  std::vector< std::string > momentNames( std::function<
237  bool ( const std::vector< tk::ctr::Term >& ) > momentType ) const
238  {
239  std::vector< std::string > names;
240  for (const auto& product : get< tag::stat >()) {
241  if (momentType( product )) {
242  names.emplace_back( std::string() );
243  for (const auto& term : product)
244  names.back() += tk::ctr::Term( term.var, term.field );
245  }
246  }
247  return names;
248  }

250  // Count number requested PDFs with given number of sample space dimensions
251  template< std::size_t d >
252  std::size_t npdf() const {
253  std::size_t n = 0;
254  for (const auto& bs : get< tag::discr, tag::binsize >())
255  if (bs.size() == d) ++n;
