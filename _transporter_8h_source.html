212  //! Verify boundarty condition (BC) side sets used exist in mesh file
213  //! \details This function verifies that the side sets to which boundary
214  //! conditions B(C) are assigned by the user in the input file all exist
215  //! in the mesh file and warns if at least one does not.
216  //! \tparam Eq Equation type, e.g., CG, DG, we are using to solver PDEs
217  //! \param[in] pde List of PDE system solved
218  //! \param[in] bnd Node lists mapped to side set ids
219  //! \note Failure here is not an error, bot only a wanring, i.e., the user
220  //! must check the screen output for this warning. Would this be more user
221  //! friendly to make it an error, i.e., abort with an error message?
222  //! \note If the input vector is empty, no checking is done.
223  template< class Eq >
224  void verifyBCsExist(const std::vector< Eq >& pde,
225  const std::map< int, std::vector< std::size_t > >& bnd)
226  {
227  // Query BC assigned to all side sets for all PDEs
228  std::unordered_set< int > conf;
229  for (const auto& eq : pde) eq.side( conf );
230  for (auto i : conf)
231  if (bnd.find(i) == end(bnd))
232  Throw( "WARNING: Boundary conditions specified on side set " +
233  std::to_string(i) + " which does not exist in mesh file" );
234  }
235 };
