228  //! Verify boundary condition (BC) side sets used exist in mesh file
229  //! \details This function does two things: (1) it verifies that the side
230  //! sets to which boundary conditions (BC) are assigned by the user in the
231  //! input file all exist among the side sets read from the input mesh
232  //! file and errors out if at least one does not, and (2) it matches the
233  //! side set ids at which the user has configured BCs to side set ids read
234  //! from the mesh file and removes those face and node lists associated
235  //! to side sets that the user did not set BCs on (as they will not need
236  //! processing further since they will not be used).
237  //! \tparam Eq Equation type, e.g., CG, DG, used to solve PDEs
238  //! \param[in] pde List of PDE system solved
239  //! \param[in,out] bnd Node or face lists mapped to side set ids
240  template< class Eq >
241  void matchBCs( const std::vector< Eq >& pde,
242  std::map< int, std::vector< std::size_t > >& bnd )
243  {
244  // Query side set ids at which BCs assigned for all PDEs
245  std::unordered_set< int > userbc;
246  for (const auto& eq : pde) eq.side( userbc );
247  // Find user-configured side set ids among side sets read from mesh file
248  std::unordered_set< int > sidesets_as_bc;
249  for (auto i : userbc) { // for all side sets at which BCs are assigned
250  if (bnd.find(i) != end(bnd)) // user BC found among side sets in file
251  sidesets_as_bc.insert( i ); // store side set id configured as BC
252  else {
253  m_print << "\n>>> ERROR: Boundary conditions specified on side set " +
254  std::to_string(i) + " which does not exist in mesh file";
255  finish();
256  }
257  }
258  // Remove sidesets not configured as BCs (will not process those further)
259  using Pair = std::pair< const int, std::vector< std::size_t > >;
260  tk::erase_if( bnd, [&]( Pair& item ) {
261  return sidesets_as_bc.find( item.first ) == end(sidesets_as_bc);
262  });
263  // Warn on no BCs
264  if (bnd.empty())
265  m_print << "\n>>> WARNING: No boundary conditions set\n\n";
266  }
267 };
