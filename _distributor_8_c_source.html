53 Distributor::Distributor( const ctr::CmdLine& cmdline ) :

63  m_nameOrdinary( g_inputdeck.momentNames( tk::ctr::ordinary ) ),
64  m_nameCentral( g_inputdeck.momentNames( tk::ctr::central ) ),
65  m_ordinary( m_nameOrdinary.size(), 0.0 ),
66  m_central( m_nameCentral.size(), 0.0 ),

75 // *****************************************************************************
76 // Constructor
77 //! \param[in] cmdline Data structure storing data from the command-line parser
78 //! \author J. Bakosi
79 // *****************************************************************************
80 {
81  // Compute load distribution given total work (= number of particles) and
82  // user-specified virtualization
83  uint64_t chunksize, remainder;
84  auto nchare = tk::linearLoadDistributor(
85  g_inputdeck.get< tag::cmd, tag::virtualization >(),
86  g_inputdeck.get< tag::discr, tag::npar >(),
87  CkNumPes(),
88  chunksize,
89  remainder );
90 
91  // Compute total number of particles distributed over all workers. Note that
92  // this number will not necessarily be the same as given by the user, coming
93  // from g_inputdeck.get< tag::discr, tag::npar >(), since each Charm++ chare
94  // array element constructor takes this chunksize argument, which equals the
95  // number of particles the array element (worker) will work on.
96  m_npar = static_cast< tk::real >( nchare * chunksize );
97 
98  // Print out info on what will be done and how
99  info( chunksize, nchare );
100 
101  // Output header for statistics output file
102  tk::TxtStatWriter sw( !m_nameOrdinary.empty() || !m_nameCentral.empty() ?
103  g_inputdeck.get< tag::cmd, tag::io, tag::stat >() :
104  std::string(),
105  g_inputdeck.get< tag::flformat, tag::stat >(),
106  g_inputdeck.get< tag::prec, tag::stat >() );
107  sw.header( m_nameOrdinary, m_nameCentral, m_tables.first );
108 
109  // Print out time integration header
110  m_print.endsubsection();
111  m_print.diag( "Starting time stepping ..." );
112  header();
113 
114  // Start timer measuring total integration time
115  m_timer.emplace_back();
116 
117  // Construct and initialize map of statistical moments
118  for (const auto& product : g_inputdeck.get< tag::stat >())
119  m_moments[ product ] = 0.0;
120 
121  // Activate SDAG-wait for estimation of ordinary statistics
122  wait4ord();
123  // Activate SDAG-wait for estimation of central moments
124  wait4cen();
125  // Activate SDAG-wait for estimation of PDFs at select times
126  if ( !(m_it % g_inputdeck.get< tag::interval, tag::pdf >()) ) wait4pdf();
127 
128  // Create statistics merger chare group collecting chare contributions
129  CProxy_Collector collproxy = CProxy_Collector::ckNew( thisProxy );
130 
131  // Fire up asynchronous differential equation integrators
132  m_intproxy =
133  CProxy_Integrator::ckNew( thisProxy, collproxy, chunksize,
134  static_cast<int>( nchare ) );
135 }

231 void
232 Distributor::estimateOrd( tk::real* ord, int n )
233 // *****************************************************************************
234 // Estimate ordinary moments
235 //! \param[in] ord Ordinary moments (sum) collected over all chares
236 //! \param[in] n Number of ordinary moments in array ord
237 //! \author J. Bakosi
238 // *****************************************************************************
239 {
240  #ifdef NDEBUG
241  IGNORE(n);
242  #endif
243 
244  Assert( static_cast<std::size_t>(n) == m_ordinary.size(),
245  "Number of ordinary moments contributed not equal to expected" );
246 
247  // Add contribution from PE to total sums, i.e., u[i] += v[i] for all i
248  for (std::size_t i=0; i<m_ordinary.size(); ++i) m_ordinary[i] += ord[i];
249 
250  // Finish computing moments, i.e., divide sums by the number of samples
251  for (auto& m : m_ordinary) m /= m_npar;
252 
253  // Activate SDAG trigger signaling that ordinary moments have been estimated
254  estimateOrdDone();
255 }

583 void
584 Distributor::evaluateTime()
585 // *****************************************************************************
586 // Evaluate time step, compute new time step size, decide if it is time to quit
587 //! \author J. Bakosi
588 // *****************************************************************************
589 {
590  const auto term = g_inputdeck.get< tag::discr, tag::term >();
591  const auto eps = std::numeric_limits< tk::real >::epsilon();
592  const auto nstep = g_inputdeck.get< tag::discr, tag::nstep >();
593 
594  // Increase number of iterations taken
595  ++m_it;
596  // Advance physical time
597  m_t += m_dt;
598  // Truncate the size of last time step
599  if (m_t > term) m_t = term;
600  // Compute size of next time step
601  m_dt = computedt();
602  // Echo one-liner info on time step
603  report();
604 
605  // Finish if either max iterations or max time reached
606  if ( std::fabs(m_t-term) > eps && m_it < nstep ) {
607 
608  if (g_inputdeck.stat()) {
609  // Update map of statistical moments
610  std::size_t ord = 0;
611  std::size_t cen = 0;
612  for (const auto& product : g_inputdeck.get< tag::stat >())
613  if (tk::ctr::ordinary( product ))
614  m_moments[ product ] = m_ordinary[ ord++ ];
615  else
616  m_moments[ product ] = m_central[ cen++ ];
617 
618  // Zero statistics counters and accumulators
619  std::fill( begin(m_ordinary), end(m_ordinary), 0.0 );
620  std::fill( begin(m_central), end(m_central), 0.0 );
621 
622  // Re-activate SDAG-wait for estimation of ordinary stats for next step
623  wait4ord();
624  // Re-activate SDAG-wait for estimation of central moments for next step
625  wait4cen();
626 
627  // Selectively re-activate SDAG-wait for estimation of PDFs for next step
628  if ( !(m_it % g_inputdeck.get< tag::interval, tag::pdf >()) ) wait4pdf();
629  }
630 
631  // Continue with next time step with all integrators
632  m_intproxy.advance( m_dt, m_t, m_it, m_moments );
633 
634  } else finish();
635 }
