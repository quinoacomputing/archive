98  //! Test uniform distribution of a random number generator
99  //! \param[in] r RNG to test
100  //! \details In real code the member function used to generate random numbers
101  //! are called by different threads, but here we pass a different thread ID
102  //! (first argument to the member function called) will exercise multiple
103  //! streams in serial, i.e., emulate multi-threaded RNG on a single thread.
104  //! This is done here this way because the unit test harness deals out each
105  //! test to a PE, thus calling the generators from real threads would create
106  //! contention.
107  template< class rng >
108  static void test_uniform( const rng& r ) {
109  std::size_t num = 100000;
110  std::vector< double > numbers( num );
111  auto n = r.nthreads();
112  for (std::size_t i=0; i<n; ++i)
113  r.uniform( static_cast<int>(i), num/n, &numbers[i*num/n] );
114  for (auto m : numbers) ensure( "sample space incorrect", 0.0<m && m<1.0 );
115  test_stats( numbers, 0.5, 1.0/12.0, 0.0, -6.0/5.0 );
116  }

137  //! Test beta distribution of a random number generator
138  //! \param[in] r RNG to test
139  //! \details In real code the member function used to generate random numbers
140  //! are called by different threads, but here we pass a different thread ID
141  //! (first argument to the member function called) will exercise multiple
142  //! streams in serial, i.e., emulate multi-threaded RNG on a single thread.
143  //! This is done here this way because the unit test harness deals out each
144  //! test to a PE, thus calling the generators from real threads would create
145  //! contention.
146  template< class rng >
147  static void test_beta( const rng& r ) {
148  std::size_t num = 100000;
149  double a = 0.2;
150  double b = 0.6;
151  std::vector< double > numbers( num );
152  auto n = r.nthreads();
153  for (std::size_t i=0; i<n; ++i)
154  r.beta( static_cast<int>(i), num/n, a, b, 0.0, 1.0, &numbers[i*num/n] );
155  // test sample space
156  for (auto m : numbers)
157  ensure( "sample space incorrect", -eps<m && m<1.0+eps );
158  // test first four moments
159  test_stats( numbers, a/(a+b), a*b/(a+b)/(a+b)/(a+b+1.0),
160  2.0*(b-a)*std::sqrt(a+b+1.0)/(a+b+2.0)/std::sqrt(a*b),
161  6.0*((a-b)*(a-b)*(a+b+1.0)-a*b*(a+b+2.0))/a/b/(a+b+2.0)/(a+b+3.0) );
162  // test scaled sample space
163  for (std::size_t i=0; i<n; ++i)
164  r.beta( static_cast<int>(i), num/n, a, b, 0.5, 2.5, &numbers[i*num/n] );
165  for (auto m : numbers)
166  ensure( "sample space incorrect", 0.5-eps<m && m<3.0+eps );
167  }

208  // Test the first four moments of random numbers passed in
209  static void test_stats( const std::vector< double >& numbers,
210  double correct_mean,
211  double correct_variance,
212  double correct_skewness,
213  double correct_excess_kurtosis )
214  {
215  const double precision = 0.05;
216  std::array< double, 4 > s{{ 0.0, 0.0, 0.0, 0.0 }};
217  for (auto n : numbers) s[0] += n;
218  s[0] /= static_cast< double >( numbers.size() );
219  ensure_equals( "mean inaccurate", s[0], correct_mean, precision );
220  for (auto n : numbers) {
221  s[1] += (n-s[0])*(n-s[0]);
222  s[2] += (n-s[0])*(n-s[0])*(n-s[0]);
223  s[3] += (n-s[0])*(n-s[0])*(n-s[0])*(n-s[0]);
224  }
225  s[1] /= static_cast< double >( numbers.size() );
226  s[2] /= static_cast< double >( numbers.size() );
227  s[3] /= static_cast< double >( numbers.size() );
228  ensure_equals( "variance inaccurate", s[1], correct_variance, precision );
229  ensure_equals( "skewness inaccurate",
230  s[2]/std::pow(s[1],1.5), correct_skewness, precision );
231  ensure_equals( "excess kurtosis inaccurate",
232  s[3]/s[1]/s[1]-3.0, correct_excess_kurtosis, precision );
233  }
