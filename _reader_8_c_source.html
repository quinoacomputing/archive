23 Reader::Reader( const std::string& filename, std::ios_base::openmode mode ) :
24  m_filename( filename ), m_inFile()
25 // *****************************************************************************
26 // Constructor: Acquire file handle
27 //! \param[in] filename Name of file to open for reading
28 //! \param[in] mode Open mode, see
29 //! http://en.cppreference.com/w/cpp/io/ios_base/openmode
30 //! \author J. Bakosi
31 // *****************************************************************************
32 {
33  // Make sure there is a filename
34  Assert( !filename.empty(), "No filename specified" );
35 
36  // Check if file exists, throw exception if it does not
37  m_inFile.open( filename, mode );
38  ErrChk( m_inFile.good(), "Failed to open file: " + filename );
39 
40  // Attempt to read a character, throw if it fails
41  // It is curious that on some systems opening a directory instead of a file
42  // with the above ifstream::open() call does not set the failbit. Thus we get
43  // here fine, so we try to read a character from it. If it is a directory or
44  // an empty file the read will fail, so we throw. Read more at: http://
45  // stackoverflow.com/questions/9591036/
46  // ifstream-open-doesnt-set-error-bits-when-argument-is-a-directory.
47  m_inFile.get();
48  ErrChk( m_inFile.good(), "Failed to read from file: " + filename );
49 
50  // Close it
51  m_inFile.close();
52  ErrChk( !m_inFile.fail(), "Failed to close file: " + filename );
53 
54  // Re-open
55  m_inFile.open( filename, std::ifstream::in );
56  ErrChk( m_inFile.good(), "Failed to open file: " + filename );
57 }
