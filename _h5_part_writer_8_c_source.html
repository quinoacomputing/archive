21 // *****************************************************************************
22 // Constructor: create/open H5Part file
23 //! \param[in] filename File to open as H5Part file
24 //! \details It is okay to call this constructor with empty filename. In that
25 //! case no IO will be performed. This is basically a punt to enable skipping
26 //! H5Part I/O. Particles are a highly experimental feature at this point.
27 //! \note If the file exists, it will be truncated.
28 //! \author J. Bakosi
29 // *****************************************************************************
30 {
31  if (m_filename.empty()) return;
32 
33  #if defined(__clang__)
34  #pragma clang diagnostic push
35  #pragma clang diagnostic ignored "-Wold-style-cast"
36  #endif
37 
38  auto f =
39  H5PartOpenFileParallel(filename.c_str(), H5PART_WRITE, MPI_COMM_WORLD);
40 
41  #if defined(__clang__)
42  #pragma clang diagnostic pop
43  #endif
44 
45  ErrChk( f, "Failed to create/open H5Part file: " + filename );
46 
47  ErrChk( H5PartWriteFileAttribString( f, "Origin", "Written by Quinoa" ) ==
48  H5PART_SUCCESS, "Failed to write file attribute to " + filename );
49 
50  ErrChk( H5PartCloseFile( f ) == H5PART_SUCCESS,
51  "Failed to close file " + filename );
52 }

77

80 
81  #if defined(__clang__)

84

93
