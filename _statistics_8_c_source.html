77 void
78 Statistics::setupOrdinary( const ctr::OffsetMap& offset,
79  const std::vector< ctr::Product >& stat )
80 // *****************************************************************************
81 // Prepare for computing ordinary moments
82 //! \param[in] offset Map of offsets in memory to address variable fields
83 //! \param[in] stat List of requested statistical moments
84 // *****************************************************************************
85 {
86  for (const auto& product : stat)
87  if (ordinary(product)) {
88 
89  m_instOrd.emplace_back( std::vector< const tk::real* >() );
90 
91  int i = 0;
92  for (const auto& term : product) {
93  auto o = offset.find( term.var );
94  Assert( o != end( offset ), "No such depvar" );
95  // Put in starting address of instantaneous variable
96  m_instOrd.back().push_back( m_particles.cptr( term.field, o->second ) );
97  // Collect all means of estimated statistics in a linear vector; this
98  // will be used to find means for fluctuations. Thus only collect single
99  // terms, i.e., <Y1>, <Y2>, etc., but not <Y1Y2>, etc.
100  if (i==0) m_ordTerm.push_back( term );
101  ++i;
102  }
103 
104  // Increase number of ordinary moments by one
105  m_ordinary.push_back( 0.0 );
106  // Count up orindary moments
107  ++m_nord;
108  }
109 
110  // Put in a zero as the last ordinary moment. This will be used as the center
111  // about which central moments are computed. If this is not needed, e.g.,
112  // because there is no central moments or not central PDFs are requested, this
113  // is small, unused, and harmless.
114  if (m_nord) {
115  // Storage for all the required ordinary moments
116  // +1 for 0 as center for ordinary moments in computing central moments
117  m_ordinary.resize( m_nord + 1 );
118  // Put in zero as center for ordinary moments in central products
119  m_ordinary[ m_nord ] = 0.0;
120  }
121 }

123 void
124 Statistics::setupCentral( const ctr::OffsetMap& offset,
125  const std::vector< ctr::Product >& stat )
126 // *****************************************************************************
127 // Prepare for computing central moments
128 //! \param[in] offset Map of offsets in memory to address variable fields
129 //! \param[in] stat List of requested statistical moments
130 // *****************************************************************************
131 {
132  // Central moments can only be estimated about ordinary moments
133  if (m_nord)
134  for (const auto& product : stat) {
135  if (central(product)) {
136 
137  m_instCen.emplace_back( std::vector< const tk::real* >() );
138  m_ctr.emplace_back( std::vector< const tk::real* >() );
139 
140  for (const auto& term : product) {
141  auto o = offset.find( term.var );
142  Assert( o != end( offset ), "No such depvar" );
143  // Put in starting address of instantaneous variable
144  m_instCen.back().push_back( m_particles.cptr(term.field, o->second) );
145  // Put in index of center for central, m_nord for ordinary moment
146  m_ctr.back().push_back(
147  m_ordinary.data() + (std::islower(term.var) ? mean(term) : m_nord) );
148  }
149 
150  // Increase number of central moments by one
151  m_central.push_back( 0.0 );
152  // Count up central moments
153  ++m_ncen;
154  }
155  }

158 void
159 Statistics::setupPDF( const ctr::OffsetMap& offset,
160  const std::vector< ctr::Probability >& pdf,
161  const std::vector< std::vector< tk::real > >& binsize )
162 // *****************************************************************************
163 // Prepare for computing PDFs
164 //! \param[in] offset Map of offsets in memory to address variable fields
165 //! \param[in] pdf List of requested probability density functions (PDF)
166 //! \param[in] binsize List of binsize vectors configuring the PDF estimators
167 // *****************************************************************************
168 {
169  std::size_t i = 0;
170  for (const auto& probability : pdf) {
171  if (ordinary(probability)) {
172 
173  // Detect number of sample space dimensions and create ordinary PDFs
174  const auto& bs = binsize[i++];
175  if (bs.size() == 1) {
176  m_ordupdf.emplace_back( bs[0] );
177  m_instOrdUniPDF.emplace_back( std::vector< const tk::real* >() );
178  } else if (bs.size() == 2) {
179  m_ordbpdf.emplace_back( bs );
180  m_instOrdBiPDF.emplace_back( std::vector< const tk::real* >() );
181  } else if (bs.size() == 3) {
182  m_ordtpdf.emplace_back( bs );
183  m_instOrdTriPDF.emplace_back( std::vector< const tk::real* >() );
184  }
185 
186  // Put in starting addresses of instantaneous variables
187  for (const auto& term : probability) {
188  auto o = offset.find( term.var );
189  Assert( o != end( offset ), "No such depvar" );
190  const tk::real* iptr = m_particles.cptr( term.field, o->second );
191  if (bs.size() == 1) m_instOrdUniPDF.back().push_back( iptr );
192  else if (bs.size() == 2) m_instOrdBiPDF.back().push_back( iptr );
193  else if (bs.size() == 3) m_instOrdTriPDF.back().push_back( iptr );
194  }
195 
196  } else { // if central PDF
197 
198  // Detect number of sample space dimensions and create central PDFs,
199  // create new storage for instantaneous variable pointer, create new
200  // storage for center pointer
201  const auto& bs = binsize[i++];
202  if (bs.size() == 1) {
203  m_cenupdf.emplace_back( bs[0] );
204  m_instCenUniPDF.emplace_back( std::vector< const tk::real* >() );
205  m_ctrUniPDF.emplace_back( std::vector< const tk::real* >() );
206  } else if (bs.size() == 2) {
207  m_cenbpdf.emplace_back( bs );
208  m_instCenBiPDF.emplace_back( std::vector< const tk::real* >() );
209  m_ctrBiPDF.emplace_back( std::vector< const tk::real* >() );
210  } else if (bs.size() == 3) {
211  m_centpdf.emplace_back( bs );
212  m_instCenTriPDF.emplace_back( std::vector< const tk::real* >() );
213  m_ctrTriPDF.emplace_back( std::vector< const tk::real* >() );
214  }
215 
216  // Put in starting address of instantaneous variables
217  for (const auto& term : probability) {
218  auto o = offset.find( term.var );
219  Assert( o != end( offset ), "No such depvar" );
220  // Put in starting address of instantaneous variable as well as index
221  // of center for central, m_nord for ordinary moment
222  const tk::real* iptr = m_particles.cptr( term.field, o->second );
223  const tk::real* cptr =
224  m_ordinary.data() + (std::islower(term.var) ? mean(term) : m_nord);
225  if (bs.size() == 1) {
226  m_instCenUniPDF.back().push_back( iptr );
227  m_ctrUniPDF.back().push_back( cptr );
228  } else if (bs.size() == 2) {
229  m_instCenBiPDF.back().push_back( iptr );
230  m_ctrBiPDF.back().push_back( cptr );
231  } else if (bs.size() == 3) {
232  m_instCenTriPDF.back().push_back( iptr );
233  m_ctrTriPDF.back().push_back( cptr );
234  }
235  }
236  }
237  }

240 std::size_t
241 Statistics::mean( const tk::ctr::Term& term ) const
242 // *****************************************************************************
243 // Return mean for fluctuation
244 //! \param[in] term Term (a fluctuation) whose mean to search for
245 //! \return Index to mean
246 // *****************************************************************************
247 {
248  const auto size = m_ordTerm.size();
249  for (auto i=decltype(size){0}; i<size; ++i) {
250  if (m_ordTerm[i].var == std::toupper(term.var) &&
251  m_ordTerm[i].field == term.field) {
252  return i;
253  }
254  }
255  Throw( std::string("Cannot find mean for variable ") + term );
256 }

258 void
259 Statistics::accumulateOrd()
260 // *****************************************************************************
261 // Accumulate (i.e., only do the sum for) ordinary moments
262 // *****************************************************************************
263 {
264  if (m_nord) {
265  // Zero ordinary moment accumulators
266  std::fill( begin(m_ordinary), end(m_ordinary), 0.0 );
267 
268  // Accumulate sum for ordinary moments. This is a partial sum, so no
269  // division by the number of samples.
270  const auto npar = m_particles.nunk();
271  for (auto p=decltype(npar){0}; p<npar; ++p) {
272  for (std::size_t i=0; i<m_nord; ++i) {
273  auto prod = m_particles.var( m_instOrd[i][0], p );
274  const auto s = m_instOrd[i].size();
275  for (auto j=decltype(s){1}; j<s; ++j) {
276  prod *= m_particles.var( m_instOrd[i][j], p );
277  }
278  m_ordinary[i] += prod;
279  }
280  }
281  }
282 }

284 void
285 Statistics::accumulateCen( const std::vector< tk::real >& om )
286 // *****************************************************************************
287 // Accumulate (i.e., only do the sum for) central moments
288 //! \details The ordinary moments container, m_ordinary, is overwritten here
289 //! with the argument om, because each of multiple Statistics class objects
290 //! (residing on different PEs) only collect their partial sums when
291 //! accumulateOrd() is run. By the time the accumulation of the central
292 //! moments is started, the ordinary moments have been collected from all
293 //! PEs and thus are the same to be passed here on all PEs. For example
294 //! client-code, see walker::Distributor.
295 //! \param[in] om Ordinary moments
296 // *****************************************************************************
297 {
298  if (m_ncen) {
299  // Overwrite ordinary moments by those computed across all PEs
300  for (std::size_t i=0; i<om.size(); ++i) m_ordinary[i] = om[i];
301 
302  // Zero central moment accumulators
303  std::fill( begin(m_central), end(m_central), 0.0 );
304 
305  // Accumulate sum for central moments. This is a partial sum, so no division
306  // by the number of samples.
307  const auto npar = m_particles.nunk();
308  for (auto p=decltype(npar){0}; p<npar; ++p) {
309  for (std::size_t i=0; i<m_ncen; ++i) {
310  auto prod = m_particles.var( m_instCen[i][0], p ) - *(m_ctr[i][0]);
311  const auto s = m_instCen[i].size();
312  for (auto j=decltype(s){1}; j<s; ++j) {
313  prod *= m_particles.var( m_instCen[i][j], p ) - *(m_ctr[i][j]);
314  }
315  m_central[i] += prod;
316  }
317  }
318  }

321 void
322 Statistics::accumulateOrdPDF()
323 // *****************************************************************************
324 // Accumulate (i.e., only do the sum for) ordinary PDFs
325 // *****************************************************************************
326 {
327  if (!m_ordupdf.empty() || !m_ordbpdf.empty() || !m_ordtpdf.empty()) {
328  // Zero PDF accumulators
329  for (auto& pdf : m_ordupdf) pdf.zero();
330  for (auto& pdf : m_ordbpdf) pdf.zero();
331  for (auto& pdf : m_ordtpdf) pdf.zero();
332 
333  // Accumulate partial sum for PDFs
334  const auto npar = m_particles.nunk();
335  for (auto p=decltype(npar){0}; p<npar; ++p) {
336  std::size_t i = 0;
337  // Accumulate partial sum for univariate PDFs
338  for (auto& pdf : m_ordupdf) {
339  pdf.add( m_particles.var( m_instOrdUniPDF[i++][0], p ) );
340  }
341  // Accumulate partial sum for bivariate PDFs
342  i = 0;
343  for (auto& pdf : m_ordbpdf) {
344  const auto inst = m_instOrdBiPDF[i++];
345  pdf.add( {{ m_particles.var( inst[0], p ),
346  m_particles.var( inst[1], p ) }} );
347  }
348  // Accumulate partial sum for trivariate PDFs
349  i = 0;
350  for (auto& pdf : m_ordtpdf) {
351  const auto inst = m_instOrdTriPDF[i++];
352  pdf.add( {{ m_particles.var( inst[0], p ),
353  m_particles.var( inst[1], p ),
354  m_particles.var( inst[2], p ) }} );
355  }
356  }
357  }
