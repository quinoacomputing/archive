30  //! Edge: IDs of two end-points
31  using Edge = std::array< std::size_t, 2 >;
32  //! Hash functor for Edge (node end-point order does not matter)
33  struct EdgeHash {
34  std::size_t operator()( const Edge& key ) const {
35  return std::hash< std::size_t >()( key[0] ) ^
36  std::hash< std::size_t >()( key[1] );
37  }
38  };
39  //! Key-equal function for Edge (node end-point order does not matter)
40  struct EdgeEq {
41  bool operator()( const Edge& left, const Edge& right ) const {
42  return (left[0] == right[0] && left[1] == right[1]) ||
43  (left[0] == right[1] && left[1] == right[0]);
44  }
45  };
46  //! Map associating IDs of mesh nodes to edges
47  using EdgeNodes = std::unordered_map< Edge, std::size_t, EdgeHash, EdgeEq >;
48  using EdgeChares = std::unordered_map< Edge,
49  std::vector< int >,
50  EdgeHash,
51  EdgeEq >;
52  //! Unique set of edges
53  using Edges = std::unordered_set< Edge, EdgeHash, EdgeEq >;

55  //! Node ID triplet denoting a tetrahedron face
56  using Face = std::array< std::size_t, 3 >;
57  // Hash functor for node triplet (Face)
58  struct FaceHasher {
59  std::size_t operator()( const Face& key ) const {
60  return std::hash< std::size_t >()( key[0] ) ^
61  std::hash< std::size_t >()( key[1] ) ^
62  std::hash< std::size_t >()( key[2] );
63  }
64  };
65  //! Key-equal function for node triplet (Face)
66  //! \note This requires all 3 node IDs be present in both the left and right
67  //! hand side but ignores ordering. For example, for A,B,C it will match
68  //! B,A,C, C,B,A, etc.
69  struct FaceEq {
70  bool operator()( const Face& l, const Face& r ) const {
71  return (l[0] == r[0] || l[0] == r[1] || l[0] == r[2]) &&
72  (l[1] == r[0] || l[1] == r[1] || l[1] == r[2]) &&
73  (l[2] == r[0] || l[2] == r[1] || l[2] == r[2]);
74  }
75  };
76  //! Unique set of node ID triplets representing unique tetrahedron faces
77  using FaceSet = std::unordered_set< Face, FaceHasher, FaceEq >;
