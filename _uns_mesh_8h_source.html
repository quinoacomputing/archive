35  //! Edge: IDs of two end-points

37  //! Hash functor for Edge (node end-point order does not matter)
38  struct EdgeHash {
39  std::size_t operator()( const Edge& key ) const {
40  const highwayhash::HH_U64 key2[2] HH_ALIGNAS(16) = {key[0], key[1]};
41  char in[8] = {1};
42  return highwayhash::SipHash( key2, in, 8 );
43  }
44  };
45  //! Key-equal function for Edge (node end-point order does not matter)
46  struct EdgeEq {
47  bool operator()( const Edge& left, const Edge& right ) const {
48  return (left[0] == right[0] && left[1] == right[1]) ||
49  (left[0] == right[1] && left[1] == right[0]);
50  }
51  };

69  //! Node ID triplet denoting a tetrahedron face
70  using Face = std::array< std::size_t, 3 >;
71  // Hash functor for node triplet (Face)
72  struct FaceHasher {
73  std::size_t operator()( const Face& key ) const {
74  return std::hash< std::size_t >()( key[0] ) ^
75  std::hash< std::size_t >()( key[1] ) ^
76  std::hash< std::size_t >()( key[2] );
77  }
78  };
79  //! Key-equal function for node triplet (Face)
80  //! \note This requires all 3 node IDs be present in both the left and right
81  //! hand side but ignores ordering. For example, for A,B,C it will match
82  //! B,A,C, C,B,A, etc.
83  struct FaceEq {
84  bool operator()( const Face& l, const Face& r ) const {
85  return (l[0] == r[0] || l[0] == r[1] || l[0] == r[2]) &&
86  (l[1] == r[0] || l[1] == r[1] || l[1] == r[2]) &&
87  (l[2] == r[0] || l[2] == r[1] || l[2] == r[2]);
88  }
89  };
90  //! Unique set of node ID triplets representing unique tetrahedron faces
91  using FaceSet = std::unordered_set< Face, FaceHasher, FaceEq >;
