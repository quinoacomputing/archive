61  //! Hash functor for element primitives, given by node IDs
62  //! \tparam N Number of nodes describing element primitive. E.g., Edge:2,
63  //! Face:3, Tet:4.
64  template< std::size_t N >
65  struct Hash {
66  //! Function call operator computing hash of node IDs
67  //! \param[in] p Array of node IDs of element primitive
68  //! \return Unique hash value for the same array of node IDs
69  //! \note The order of the nodes does not matter: the IDs are sorted
70  //! before the hash is computed.
71  std::size_t operator()( const std::array< std::size_t, N >& p ) const {
72  using highwayhash::SipHash;
73  Shaper< N > shaper;
74  for (std::size_t i=0; i<N; ++i) shaper.sizets[i] = p[i];
75  std::sort( std::begin(shaper.sizets), std::end(shaper.sizets) );
76  return SipHash( hh_key, shaper.bytes, N*sizeof(std::size_t) );
77  }
78  };

80  //! Comparitor functor for element primitives, given by node IDs
81  //! \tparam N Number of nodes describing element primitive. E.g., Edge:2,
82  //! Face:3, Tet:4.
83  template< std::size_t N >
84  struct Eq {
85  //! Function call operator computing equality of element primitives
86  //! \param[in] l Left element primitive given by array of node IDs
87  //! \param[in] r Right element primitive given by array of node IDs
88  //! \return True if l = r, false otherwise
89  //! \note The order of the nodes does not matter: the IDs are sorted
90  //! before the primitives, given by the node IDs, is computed.
91  bool operator()( const std::array< std::size_t, N >& l,
92  const std::array< std::size_t, N >& r ) const
93  {
94  std::array< std::size_t, N > s = l, p = r;
95  std::sort( begin(s), end(s) );
96  std::sort( begin(p), end(p) );
97  return s == p;
98  }
99  };
