27  public:
28  //! Add diffusion contribution to rhs at 2nd time step stage
29  //! \param[in] e Equation system index, i.e., which transport equation
30  //! system we operate on among the systems of PDEs
31  //! \param[in] ncomp Number of components in this PDE
32  //! \param[in] deltat Size of time step
33  //! \param[in] J Element Jacobi determinant
34  //! \param[in] grad Shape function derivatives, nnode*ndim [4][3]
35  //! \param[in] N Element node indices
36  //! \param[in] u Solution at element nodes at recent time step
37  //! \param[in] r Pointers to right hand side at component and offset
38  //! \param[in,out] R Right-hand side vector contributing to
39  static void
40  diffusionRhs( tk::ctr::ncomp_type e,
41  tk::ctr::ncomp_type ncomp,
42  tk::real deltat,
43  tk::real J,
44  const std::array< std::array< tk::real, 3 >, 4 >& grad,
45  const std::array< std::size_t, 4 >& N,
46  const std::vector< std::array< tk::real, 4 > >& u,
47  const std::vector< const tk::real* >& r,
48  tk::Fields& R )
49  {
50  // diffusivities for all components
51  const auto& diff =
52  g_inputdeck.get< tag::param, tag::transport, tag::diffusivity >().at(e);
53  // add diffusion contribution to right hand side
54  const auto d = deltat * J/6.0;
55  for (ncomp_t c=0; c<ncomp; ++c)
56  for (std::size_t k=0; k<3; ++k) {
57  const auto D = diff[ 3*c+k ];
58  for (std::size_t a=0; a<4; ++a)
59  for (std::size_t b=0; b<4; ++b)
60  R.var(r[c],N[a]) -= d * D * grad[a][k] * grad[b][k] * u[c][b];
61  }
62  }
63 
64  //! Compute the minimum time step size based on the diffusion
65  //! \param[in] e Equation system index, i.e., which transport equation
66  //! system we operate on among the systems of PDEs
67  //! \param[in] ncomp Number of components in this PDE
68  //! \param[in] L Characteristic length scale
69  //! \return Minimum time step size based on diffusion
70  static tk::real
71  diffusion_dt( tk::ctr::ncomp_type e,
72  tk::ctr::ncomp_type ncomp,
73  tk::real L,
74  const std::vector< std::array< tk::real, 4 > >& )
75  {
76  // diffusivities for all components
77  const auto& df =
78  g_inputdeck.get< tag::param, tag::transport, tag::diffusivity >().at(e);
79  // compute the minimum diffusion time step size across the four nodes
80  tk::real mindt = std::numeric_limits< tk::real >::max();
81  for (ncomp_t c=0; c<ncomp; ++c) {
82  const auto di = 3*c;
83  const auto d = std::max( df[di+2], std::max( df[di+0], df[di+1] ) );
84  const auto dt = L * L / (2.0*d); // dt ~ dx^2/(2D)
85  if (dt < mindt) mindt = dt;
86  }
87  return mindt;
88  }
