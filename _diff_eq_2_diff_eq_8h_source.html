40  public:
41  //! \brief Constructor taking an object modeling Concept.
42  //! \details The object of class T comes pre-constructed.
43  //! \param[in] x Instantiated object of type T given by the template
44  //! argument.
45  template< typename T > explicit DiffEq( T x ) :
46  self( tk::make_unique< Model<T> >( std::move(x) ) ) {}
47 
48  //! \brief Constructor taking a function pointer to a constructor of an
49  //! object modeling Concept.
50  //! \details Passing std::function allows late execution of the constructor,
51  //! i.e., as late as inside this class' constructor, and thus usage from
52  //! a factory. Note that there are at least two different ways of using
53  //! this constructor:
54  //! - Bind T's constructor arguments and place it in std::function<T()>
55  //! and passing no arguments as args.... This case then instantiates the
56  //! model via its constructor and stores it in here.
57  //! - Bind a single placeholder argument to T's constructor and pass it in
58  //! as host's args..., which then forwards it to model's constructor. This
59  //! allows late binding, i.e., binding the argument only here.
60  //! \see See also the wrapper tk::recordModel() which does the former and
61  //! tk::recordModelLate() which does the latter, both defined in
62  //! src/Base/Factory.h.
63  //! \param[in] x Function pointer to a constructor of an object modeling
64  //! Concept.
65  //! \param[in] args Zero or more constructor arguments
66  template< typename T, typename...Args >
67  explicit DiffEq( std::function<T(Args...)> x, Args&&... args ) :
68  self( tk::make_unique< Model<T> >(
69  std::move( x( std::forward<Args>(args)... ) ) ) ) {}
