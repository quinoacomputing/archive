27 //! \brief Function object for registering a differential equation into the
28 //! differential equation factory
29 //! \details This functor is repeatedly called by brigand's cartesian_product,
30 //! sweeping all combinations of the differential equation policies. The
31 //! purpose of template template is to simplify client code as that will
32 //! not have to specify the template arguments of the template argument
33 //! (the policies of Eq), since we can figure it out here. See also
34 //! http://stackoverflow.com/a/214900
35 template< template< class, class > class Eq >
36 struct registerDiffEq {
37  //! Need to store the reference to factory we are registering into
38  DiffEqFactory& factory;
39  //! Need to store which differential equation we are registering
40  const ctr::DiffEqType type;
41  //! Constructor, also count number of unique equation types registered
42  explicit registerDiffEq( DiffEqFactory& f,
43  ctr::DiffEqType t,
44  std::set< ctr::DiffEqType >& eqTypes ) :
45  factory( f ), type( t ) { eqTypes.insert( t ); }
46  //! \brief Function call operator called with tk::cartesian_product for
47  //! each unique sequence of policy combinations
48  template< typename U > void operator()( brigand::type_<U> ) {
49  // Get Initialization policy: first type of brigand::list U
50  using InitPolicy = typename brigand::front< U >;
51  // Get coefficients policy: last type of brigand::list U
52  using CoeffPolicy = typename brigand::back< U >;
53  // Build differential equation key
54  ctr::DiffEqKey key{ type, InitPolicy::type(), CoeffPolicy::type() };
55  // Register equation (with policies given by brigand::list U) into
56  // factory
57  tk::recordModelLate< DiffEq, Eq< InitPolicy, CoeffPolicy > >
58  ( factory, key, static_cast<tk::ctr::ncomp_type>(0) );
59  }
60 };
