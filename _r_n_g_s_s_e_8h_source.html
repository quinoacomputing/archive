33  //! Adaptor to use a std distribution with the RNGSSE generator
34  //! \see C++ concepts: UniformRandomNumberGenerator
35  struct Adaptor {

71  //! Uniform RNG: Generate uniform random numbers
72  //! \param[in] tid Thread (or more precisely) stream ID
73  //! \param[in] num Number of RNGs to generate
74  //! \param[in,out] r Pointer to memory to write the random numbers to
75  void uniform( int tid, ncomp_t num, double* r ) const {
76  for (ncomp_t i=0; i<num; ++i)
77  r[i] = static_cast<double>(
78  Generate( &m_stream[ static_cast<std::size_t>(tid) ] ) )
79  / 4294967296.0;
80  }

82  //! Gaussian RNG: Generate Gaussian random numbers
83  //! \param[in] tid Thread (or more precisely stream) ID
84  //! \param[in] num Number of RNGs to generate
85  //! \param[in,out] r Pointer to memory to write the random numbers to
86  //! \details Generating Gaussian random numbers is implemented via an
87  //! adaptor, modeling std::UniformRandomNumberGenerator, outsourcing the
88  //! transformation of uniform random numbers to Gaussian ones, to the
89  //! standard library. The adaptor is instantiated here because a standard
90  //! distribution, such as e.g., std::normal_distribution, generates
91  //! numbers using operator() with no arguments, thus the RNG state and the
92  //! thread ID (this latter only known here) must be stored in the adaptor
93  //! functor's state. Even though creating the adaptor seems like a
94  //! potentially costly operation for every call, using the standard
95  //! library implementation is still faster than a hand-coded
96  //! implementation of the Box-Muller algorithm. Note that libc++ uses a
97  //! cache, as Box-Muller, implemented using the polar algorithm generates
98  //! 2 Gaussian numbers for each pair of uniform ones, caching every 2nd.
99  void gaussian( int tid, ncomp_t num, double* r ) const {
100  Adaptor generator( m_stream, tid );
101  std::normal_distribution<> gauss_dist( 0.0, 1.0 );
102  for (ncomp_t i=0; i<num; ++i) r[i] = gauss_dist( generator );
103  }

105  //! Beta RNG: Generate beta random numbers
106  //! \param[in] tid Thread (or more precisely stream) ID
107  //! \param[in] num Number of RNGs to generate
108  //! \param[in] p First beta shape parameter
109  //! \param[in] q Second beta shape parameter
110  //! \param[in] a Beta displacement parameter
111  //! \param[in] b Beta scale factor
112  //! \param[in,out] r Pointer to memory to write the random numbers to
113  //! \details Generating beta-distributed random numbers is implemented via
114  //! an adaptor, modeling boost::UniformRandomNumberGenerator, outsourcing
115  //! the transformation of uniform random numbers to beta-distributed ones,
116  //! to boost::random. The adaptor is instantiated here because a boost
117  //! random number distribution, such as e.g.,
118  //! boost::random::beta_distribution, generates numbers using operator()
119  //! with no arguments, thus the RNG state and the thread ID (this latter
120  //! only known here) must be stored in the adaptor functor's state.
121  void beta( int tid, ncomp_t num, double p, double q, double a, double b,
122  double* r ) const {
123  Adaptor generator( m_stream, tid );
124  boost::random::beta_distribution<> beta_dist( p, q );
125  for (ncomp_t i=0; i<num; ++i) r[i] = beta_dist( generator ) * b + a;
126  }

130  //! \param[in] num Number of RNGs to generate
131  //! \param[in] a Gamma shape parameter
132  //! \param[in] b Gamma scale factor
133  //! \param[in,out] r Pointer to memory to write the random numbers to
134  //! \details Generating gamma-distributed random numbers is implemented via
135  //! an adaptor, modeling boost::UniformRandomNumberGenerator, outsourcing
136  //! the transformation of uniform random numbers to gamma-distributed
137  //! ones, to boost::random. The adaptor is instantiated here because a
138  //! boost random number distribution, such as e.g.,
139  //! boost::random::gamma_distribution, generates numbers using operator()
140  //! with no arguments, thus the RNG state and the thread ID (this latter
141  //! only known here) must be stored in the adaptor functor's state.
142  void gamma( int tid, ncomp_t num, double a, double b, double* r ) const {
143  Adaptor generator( m_stream, tid );
144  boost::random::gamma_distribution<> gamma_dist( a, b );
145  for (ncomp_t i=0; i<num; ++i) r[i] = gamma_dist( generator );
146  }
