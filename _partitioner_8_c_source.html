131 void
132 Partitioner::offset( int p, std::size_t u )
133 // *****************************************************************************
134 // Receive number of uniquely assigned global mesh node IDs from lower PEs
135 //! \param[in] p PE ID
136 //! \param[in] u Number of mesh node IDs PE p will assign IDs to
137 //! \details This function computes the offset each PE will need to start
138 //! assigning its new node IDs from (for those nodes that are not assigned
139 //! new IDs by any PEs with lower indices). The offset for a PE is the
140 //! offset for the previous PE plus the number of node IDs the previous PE
141 //! (uniquely) assigns new IDs for minus the number of node IDs the
142 //! previous PE receives from others (lower PEs). This is computed here in
143 //! a parallel/distributed fashion by each PE sending its number of node
144 //! IDs (that it uniquely assigns) to all PEs. Note that each PE would
145 //! only need to send this information to higher PEs, but instead this
146 //! function is called in a broadcast fashion, because that is more
147 //! efficient than individual calls to only the higher PEs. Therefore when
148 //! computing the offsets, we only count the lower PEs. When this is done,
149 //! we have the precise communication map as well as the start offset on
150 //! all PEs and so we can start the distributed global mesh node ID
151 //! reordering.
152 // *****************************************************************************
153 {
154  if (p < CkMyPe()) m_start += u;
155  if (++m_noffset == static_cast<std::size_t>(CkNumPes())) reorder();
156 }

246 void
247 Partitioner::flatten()
248 // *****************************************************************************
249 // Prepare owned mesh node IDs for reordering
250 //! \details The 'flatten' is used here as a concatenation of a data
251 //! structure that stores date categorized by chares owned on this PE. The
252 //! result of the flattening is thus a simpler data structure that is no
253 //! longer categorized by (or associated to) chares.
254 // *****************************************************************************
255 {
256  // Optionally refine mesh if requested
257  const auto ir = g_inputdeck.get< tag::selected, tag::initialamr >();
258  if (ir == ctr::InitialAMRType::UNIFORM) refine();
259 
260  // Make sure we are not fed garbage
261  Assert( m_chinpoel.size() ==
262  static_cast< std::size_t >( chareDistribution()[1] ),
263  "Global mesh nodes ids associated to chares on PE " +
264  std::to_string( CkMyPe() ) + " is incomplete" );
265 
266  // Collect chare IDs we own associated to old global mesh node IDs
267  for (const auto& c : m_chinpoel)
268  for (auto p : c.second)
269  m_nodechares[p].push_back( c.first );
270 
271  // Make chare IDs (associated to old global mesh node IDs) unique
272  for (auto& c : m_nodechares) tk::unique( c.second );
273 
274  // Collect chare IDs we own associated to edges
275  for (const auto& c : m_chedgenodes)
276  for (const auto& e : c.second)
277  m_edgechares[ e.first ].push_back( c.first );
278 
279  // Make chare IDs (associated to edges) unique
280  for (auto& c : m_edgechares) tk::unique( c.second );
281 
282  // Flatten node IDs of elements our chares operate on
283  for (const auto& c : m_chinpoel)
284  for (auto i : c.second)
285  m_nodeset.insert( i );
286 
287  // Flatten edges of elements our chares operate on
288  for (const auto& c : m_chedgenodes)
289  for (const auto& e : c.second)
290  m_edgeset.insert( e.first );
291 
292  // send progress report to host
293  if ( g_inputdeck.get< tag::cmd, tag::feedback >() )
294  m_host.peflattened();
295 
296  // Signal host that we are ready for computing the communication map,
297  // required for parallel distributed global mesh node reordering
298  contribute( m_cb.get< tag::flattened >() );
299 }

327 tk::real
328 Partitioner::cost( std::size_t l, std::size_t u )
329 // *****************************************************************************
330 // Compute communication cost on our PE
331 //! \param[in] l Lower global node ID this PE works on
332 //! \param[in] u Upper global node ID this PE works on
333 //! \return Communication cost for our PE
334 //! \details The cost is a real number between 0 and 1, defined as the
335 //! number of mesh points we do not own, i.e., need to send to some other
336 //! PE, divided by the total number of points we contribute to. The lower
337 //! the better.
338 // *****************************************************************************
339 {
340  std::size_t ownpts = 0, compts = 0;
341  for (auto p : m_nodeset) if (p >= l && p < u) ++ownpts; else ++compts;
342 
343  // Free storage of unique global node IDs chares on our PE will contribute to
344  // as it is no longer needed after computing the communication cost.
345  tk::destroy( m_nodeset );
346 
347  return static_cast<tk::real>(compts) / static_cast<tk::real>(ownpts + compts);
348 }

360 void
361 Partitioner::query( int p,
362  const std::set< std::size_t >& nodes,
363  const tk::UnsMesh::Edges& edges ) const
364 // *****************************************************************************
365 // Query our global node IDs and edges by other PEs so they know if they are to
366 // receive IDs for those from during reordering
367 //! \param[in] p Querying PE
368 //! \param[in] nodes List of global mesh node IDs to query
369 //! \param[in] edges List of edges to query
370 //! \details Note that every PE calls this function in a broadcast fashion,
371 //! including our own. However, to compute the correct result, this would
372 //! only be necessary for PEs whose ID is higher than ours. However, the
373 //! broadcast (calling everyone) is more efficient. This also results in a
374 //! simpler logic, because every PE goes through this single call path.
375 //! The returned mask is simply a boolean array signaling if the node ID
376 //! is found (owned).
377 // *****************************************************************************
378 {
379  std::unordered_map< std::size_t, std::vector< int > > cn;
380  for (auto j : nodes) {
381  const auto it = m_nodeset.find( j );
382  if (it != end(m_nodeset)) {
383  const auto& c = tk::cref_find( m_nodechares, j );
384  auto& chares = cn[j];
385  chares.insert( end(chares), begin(c), end(c) );
386  }
387  }
388  tk::UnsMesh::EdgeChares ce;
389  for (auto j : edges) {
390  const auto it = m_edgeset.find( j );
391  if (it != end(m_edgeset)) {
392  const auto& c = tk::cref_find( m_edgechares, j );
393  auto& chares = ce[j];
394  chares.insert( end(chares), begin(c), end(c) );
395  }
396  }
397  thisProxy[ p ].mask( CkMyPe(), cn, ce );
398 }

400 void
401 Partitioner::mask(
402  int p,
403  const std::unordered_map< std::size_t, std::vector< int > >& cn,
404  const tk::UnsMesh::EdgeChares& ce )
405 // *****************************************************************************
406 // Receive mask of to-be-received global mesh node IDs
407 //! \param[in] p The PE uniquely assigns the node IDs marked listed in ch
408 //! \param[in] cn Vector containing the set of potentially multiple chare
409 //! IDs that we own (i.e., contribute to) for all of our node IDs.
410 //! \param[in] ce Map associating a vector of chare IDs to edges (at which
411 //! we added nodes during initial mesh refinement)
412 //! \details Note that every PE will call this function, since query() was
413 //! called in a broadcast fashion and query() answers to every PE once.
414 //! This is more efficient than calling only the PEs from which we would
415 //! have to receive results from. Thus the incoming results are only
416 //! interesting from PEs with lower IDs than ours.
417 // *****************************************************************************
418 {
419  // Store the old global mesh node IDs associated to chare IDs bordering
420  // the mesh chunk held by and associated to chare IDs we own
421  for (const auto& h : cn) {
422  const auto& chares = tk::ref_find( m_nodechares, h.first );
423  for (auto c : chares) { // surrounded chares
424  auto& sch = m_msum[c];
425  for (auto s : h.second) // surrounding chares
426  if (s != c) sch[ s ].insert( h.first );
427  }
428  }
429 
430  // Store the edges associated to chare IDs bordering the mesh chunk held
431  // by and associated to chare IDs we own
432  for (const auto& h : ce) {
433  const auto& chares = tk::ref_find( m_edgechares, h.first );
434  for (auto c : chares) { // surrounded chares
435  auto& sch = m_msumed[c];
436  for (auto s : h.second) // surrounding chares
437  if (s != c) sch[ s ].insert( h.first );
438  }
439  }
440 
441  // Associate global mesh node IDs to lower PEs we will need to receive
442  // from during node reordering. The choice of associated container is
443  // std::map, which is ordered (vs. unordered, hash-map). This is required
444  // by the following operation that makes the mesh node IDs unique in the
445  // communication map. (We are called in an unordered fashion, so we need
446  // to collect from all PEs and then we need to make the node IDs unique,
447  // keeping only the lowest PEs a node ID is associated with.)
448  if (p < CkMyPe()) {
449  auto& id = m_ncomm[ p ];
450  for (const auto& h : cn) id.insert( h.first );
451  auto& ed = m_ecomm[ p ];
452  for (const auto& h : ce) ed.insert( h.first );
453  }
454 
455  if (++m_nquery == static_cast<std::size_t>(CkNumPes())) {
456  // Make sure we have received all we need
457  Assert( m_ncomm.size() == static_cast<std::size_t>(CkMyPe()),
458  "Communication map size on PE " +
459  std::to_string(CkMyPe()) + " must equal " +
460  std::to_string(CkMyPe()) );
461  Assert( m_ecomm.size() == static_cast<std::size_t>(CkMyPe()),
462  "Communication map size on PE " +
463  std::to_string(CkMyPe()) + " must equal " +
464  std::to_string(CkMyPe()) );
465  // Fill new hash-map, keeping only unique node IDs obtained from the
466  // lowest possible PEs
467  for (auto c=m_ncomm.cbegin(); c!=m_ncomm.cend(); ++c) {
468  auto& n = m_ncommunication[ c->first ];
469  for (auto j : c->second)
470  if (std::none_of( m_ncomm.cbegin(), c,
471  [ j ]( const typename decltype(m_ncomm)::value_type& s )
472  { return s.second.find(j) != end(s.second); } )) {
473  n.insert(j);
474  }
475  if (n.empty()) m_ncommunication.erase( c->first );
476  }
477  for (auto c=m_ecomm.cbegin(); c!=m_ecomm.cend(); ++c) {
478  auto& e = m_ecommunication[ c->first ];
479  for (const auto& j : c->second)
480  if (std::none_of( m_ecomm.cbegin(), c,
481  [ &j ]( const typename decltype(m_ecomm)::value_type& s )
482  { return s.second.find(j) != end(s.second); } )) {
483  e.insert(j);
484  }
485  if (e.empty()) m_ecommunication.erase( c->first );
486  }
487  // Free storage of temporary communication map used to receive global
488  // mesh node IDs as it is no longer needed once the final communication
489  // map is generated.
490  tk::destroy( m_ncomm );
491  // Free storage of temporary communication map used to receive global
492  // mesh edge-node IDs as it is no longer needed once the final
493  // communication map is generated.
494  tk::destroy( m_ecomm );
495  // Count up total number of nodes and (nodes associated to edges) we
496  // will need receive during reordering
497  std::size_t nrecv = 0, erecv = 0;
498  for (const auto& u : m_ncommunication) nrecv += u.second.size();
499  for (const auto& e : m_ecommunication) erecv += e.second.size();
500 
501  // send progress report to host
502  //if ( g_inputdeck.get< tag::cmd, tag::feedback >() )
503  // m_host.pemask();
504 
505  // Compute number of mesh node IDs we will assign IDs to
506  auto nuniq = m_nodeset.size() - nrecv + m_edgeset.size() - erecv;
507 
508  // Start computing PE offsets for node reordering
509  thisProxy.offset( CkMyPe(), nuniq );
510  }

514 Partitioner::readGraph( tk::ExodusIIMeshReader& er )
515 // *****************************************************************************
516 // Read our contiguously-numbered chunk of the mesh graph from file
517 //! \param[in] er ExodusII mesh reader
518 // *****************************************************************************
519 {
520  // Get number of mesh points and number of tetrahedron elements in file
521  er.readElemBlockIDs();
522  auto nel = er.nelem( tk::ExoElemType::TET );
523 
524  // Read our contiguously-numbered chunk of tetrahedron element
525  // connectivity from file and also generate and store the list of global
526  // element indices for our chunk of the mesh
527  auto npes = static_cast< std::size_t >( CkNumPes() );
528  auto mype = static_cast< std::size_t >( CkMyPe() );
529  auto chunk = nel / npes;
530  auto from = mype * chunk;
531  auto till = from + chunk;
532  if (mype == npes-1) till += nel % npes;
533  er.readElements( {{from, till-1}}, tk::ExoElemType::TET, m_tetinpoel );
534  m_gelemid.resize( till-from );
535  std::iota( begin(m_gelemid), end(m_gelemid), from );

537  // send progress report to host
538  //if ( g_inputdeck.get< tag::cmd, tag::feedback >() )
539  // m_host.pegraph();

542  contribute( sizeof(uint64_t), &nelem, CkReduction::sum_int,

546 void
547 Partitioner::computeCentroids( tk::ExodusIIMeshReader& er )
548 // *****************************************************************************
549 // Compute element centroid coordinates
550 //! \param[in] er ExodusII mesh reader
551 // *****************************************************************************
552 {
553  // Construct unique global mesh point indices of our chunk
554  auto gid = m_tetinpoel;
555  tk::unique( gid );
556 
557  // Read node coordinates of our chunk of the mesh elements from file
558  auto ext = tk::extents( gid );
559  auto coord = er.readNodes( ext );
560  const auto& x = std::get< 0 >( coord );
561  const auto& y = std::get< 1 >( coord );
562  const auto& z = std::get< 2 >( coord );
563 
564  // Make room for element centroid coordinates
565  auto& cx = m_centroid[0];
566  auto& cy = m_centroid[1];
567  auto& cz = m_centroid[2];
568  auto num = m_tetinpoel.size()/4;
569  cx.resize( num );
570  cy.resize( num );
571  cz.resize( num );
572 
573  // Compute element centroids for our chunk of the mesh elements
574  for (std::size_t e=0; e<num; ++e) {
575  auto A = m_tetinpoel[e*4+0] - ext[0];
576  auto B = m_tetinpoel[e*4+1] - ext[0];
577  auto C = m_tetinpoel[e*4+2] - ext[0];
578  auto D = m_tetinpoel[e*4+3] - ext[0];
579  cx[e] = (x[A] + x[B] + x[C] + x[D]) / 4.0;
580  cy[e] = (y[A] + y[B] + y[C] + y[D]) / 4.0;
581  cz[e] = (z[A] + z[B] + z[C] + z[D]) / 4.0;
582  }
583 
584  contribute( m_cb.get< tag::part >() );
585 }

678 std::array< int, 2 >
679 Partitioner::chareDistribution() const
680 // *****************************************************************************
681 // Compute chare distribution
682 //! \return Chunksize, i.e., number of chares per all PEs except the last
683 //! one, and the number of chares for my PE
684 //! \details Chare ids are distributed to PEs in a linear continguous order
685 //! with the last PE taking the remainder if the number of PEs is not
686 //! divisible by the number chares. For example, if nchare=7 and npe=3,
687 //! the chare distribution is PE0: 0 1, PE1: 2 3, and PE2: 4 5 6. As a
688 //! result of this distribution, all PEs will have their chare-categorized
689 //! element connectivity filled with the global mesh node IDs associated
690 //! to the Charm++ chare IDs each PE owns.
691 // *****************************************************************************
692 {
693  auto chunksize = m_nchare / CkNumPes();
694  auto mynchare = chunksize;
695  if (CkMyPe() == CkNumPes()-1) mynchare += m_nchare % CkNumPes();
696  return {{ chunksize, mynchare }};
697 }

699 void
700 Partitioner::reorder()
701 // *****************************************************************************
702 // Reorder global mesh node IDs
703 // *****************************************************************************
704 {
705  // Activate SDAG waits for having requests arrive from other PEs for some
706  // of our node IDs; and for computing/receiving lower and upper bounds of
707  // global node IDs our PE operates on after reordering
708  wait4prep();
709  wait4bounds();
710  wait4reorder();
711 
712  // In serial signal to the runtime system that we have participated in
713  // reordering. This is required here because this is only triggered if
714  // communication is required during mesh node reordering. See also
715  // particioner.ci.
716  if (CkNumPes() == 1) participated_complete();
717 
718  // Send out request for new global node IDs for nodes we do not reorder
719  for (const auto& c : m_ncommunication)
720  thisProxy[ c.first ].request( CkMyPe(), c.second );
721 
722  // Send out request for new global node IDs for edges we do not reorder
723  for (const auto& e : m_ecommunication)
724  thisProxy[ e.first ].request( CkMyPe(), e.second );
725 
726  // Lambda to decide if node ID is being assigned a new ID by us
727  auto ownnode = [ this ]( std::size_t p ) {
728  using Set = typename std::remove_reference<
729  decltype(m_ncommunication) >::type::value_type;
730  return !std::any_of( m_ncommunication.cbegin(), m_ncommunication.cend(),
731  [&](const Set& s)
732  { return s.second.find(p) != s.second.cend(); } );
733  };
734 
735  // Lambda to decide if edge-node ID is being assigned a new ID by us
736  auto ownedge = [ this ]( const tk::UnsMesh::Edge& e ) {
737  using Set = typename std::remove_reference<
738  decltype(m_ecommunication) >::type::value_type;
739  return !std::any_of( m_ecommunication.cbegin(), m_ecommunication.cend(),
740  [&](const Set& s)
741  { return s.second.find(e) != s.second.cend(); } );
742  };
743 
744  // Reorder our chunk of the mesh node IDs by looping through all of our
745  // node IDs (resulting from reading our chunk of the mesh cells). We test
746  // if we are to assign a new ID to a node ID, and if so, we assign new ID,
747  // i.e., reorder, by constructing a map associating new to old IDs. We
748  // also count up the reordered nodes, which also serves as the new node
749  // id.
750  for (auto p : m_nodeset)
751  if (ownnode(p)) m_linnodes[ p ] = m_start++;
752 
753  // Reorder our chunk of the mesh edges by looping through all of our edges
754  // (resulting from initial uniform refinement of our chunk of the mesh
755  // cells). We test if we are to assign a new ID to an edge, and if so, we
756  // assign new ID, i.e., reorder, by constructing a map associating new
757  // node IDs to edges. We also count up the reordered edge-nodes, which
758  // also serves as the new node id.
759  for (const auto& e : m_edgeset)
760  if (ownedge(e)) m_linedges[ e ] = m_start++;
761 
762  // Trigger SDAG wait indicating that reordering own node IDs are complete
763  reorderowned_complete();
764 
765  // If all our nodes have new IDs assigned, signal that to the runtime
766  if (m_linnodes.size() == m_nodeset.size()) nodesreorder_complete();
767 
768  // If all our edges have new IDs assigned, signal that to the runtime
769  if (m_linedges.size() == m_edgeset.size()) edgesreorder_complete();
770 }

1124 void
1125 Partitioner::bounds()
1126 // *****************************************************************************
1127 // Compute lower and upper bounds of reordered node IDs our PE operates on
1128 // \details This function computes the global row IDs at which the linear
1129 // system will have a PE boundary. We simply find the largest node ID
1130 // assigned on each PE by the reordering and use that as the upper global
1131 // row index. Note that while this rarely results in equal number of rows
1132 // assigned to PEs, potentially resulting in some load-imbalance, it
1133 // yields a pretty good division reducing communication costs during the
1134 // assembly of the linear system, which is more important than a slight
1135 // (FLOP) load imbalance. Since the upper index for PE 1 is the same as
1136 // the lower index for PE 2, etc., we find the upper indices and then the
1137 // lower indices for all PEs are communicated.
1138 // *****************************************************************************
1139 {
