127 void
128 Partitioner::partref()
129 // *****************************************************************************
130 // Partition the mesh to NumPes partitions (before an initial refinement step)
131 //! \details This function calls the mesh partitioner to partition (or
132 //! re-partition) the (current) mesh as a first step for an initial mesh
133 //! refinement step. The number of partitions always equals the numher of PEs.
134 // *****************************************************************************
135 {
136  // Move mesh connectivity to under a new name. This also clears the original
137  // one. The new one (rinpoel) is treated during communication as a source, and
138  // the old one (ginpoel) will serve as the destination for the new one.
139  m_rinpoel = std::move( m_ginpoel );
140 
141  // Generate element IDs for Zoltan
142  std::vector< long > gelemid( m_rinpoel.size()/4 );
143  std::iota( begin(gelemid), end(gelemid), 0 );
144 
145  // Partition the mesh using Zoltan to number of PEs parts
146  const auto alg = g_inputdeck.get< tag::selected, tag::partitioner >();
147  const auto pel = tk::zoltan::geomPartMesh( alg,
148  centroids( m_inpoel, m_coord ),
149  gelemid,
150  CkNumPes() );
151 
152  Assert( pel.size() == gelemid.size(), "Size of ownership array (PE of "
153  "elements) after mesh partitioning does not equal the number of mesh "
154  "graph elements" );
155 
156  // Prepare for a step of initial mesh refinement
157  m_bndEdges.clear();
158  m_pe.clear();
159  m_edgeNodeCoord.clear();
160  m_edgenode.clear();
161  m_coordmap.clear();
162 
163  // Categorize mesh elements (given by their gobal node IDs) by target PE and
164  // distribute to their PEs based on mesh partitioning.
165  distributePE( categorize( pel, m_rinpoel ) );
166 }

207 void
208 Partitioner::offset( int p, std::size_t u )
209 // *****************************************************************************
210 // Receive number of uniquely assigned global mesh node IDs from lower PEs
211 //! \param[in] p PE ID
212 //! \param[in] u Number of mesh node IDs PE p will assign IDs to
213 //! \details This function computes the offset each PE will need to start
214 //! assigning its new node IDs from (for those nodes that are not assigned
215 //! new IDs by any PEs with lower indices). The offset for a PE is the
216 //! offset for the previous PE plus the number of node IDs the previous PE
217 //! (uniquely) assigns new IDs for minus the number of node IDs the
218 //! previous PE receives from others (lower PEs). This is computed here in
219 //! a parallel/distributed fashion by each PE sending its number of node
220 //! IDs (that it uniquely assigns) to all PEs. Note that each PE would
221 //! only need to send this information to higher PEs, but instead this
222 //! function is called in a broadcast fashion, because that is more
223 //! efficient than individual calls to only the higher PEs. Therefore when
224 //! computing the offsets, we only count the lower PEs. When this is done,
225 //! we have the precise communication map as well as the start offset on
226 //! all PEs and so we can start the distributed global mesh node ID
227 //! reordering.
228 // *****************************************************************************
229 {
230  if (p < CkMyPe()) m_start += u;
231  if (++m_noffset == static_cast<std::size_t>(CkNumPes())) reorder();
232 }

351 void
352 Partitioner::flatten()
353 // *****************************************************************************
354 // Prepare owned mesh node IDs for reordering
355 //! \details The 'flatten' is used here as a concatenation of a data
356 //! structure that stores date categorized by chares owned on this PE. The
357 //! result of the flattening is thus a simpler data structure that is no
358 //! longer categorized by (or associated to) chares.
359 // *****************************************************************************
360 {
361  // Make sure we are not fed garbage
362  Assert( m_chinpoel.size() ==
363  static_cast< std::size_t >( distribution(m_nchare)[1] ),
364  "Global mesh nodes ids associated to chares on PE " +
365  std::to_string( CkMyPe() ) + " is incomplete" );
366 
367  // Flatten node IDs of elements our chares operate on and associate global
368  // node ID to chare IDs.
369  for (const auto& c : m_chinpoel)
370  for (auto i : c.second) {
371  m_nodeset.insert( i );
372  m_nodech[ i ].push_back( c.first );
373  }
374 
375  m_chcoordmap.clear();
376 
377  // Find chare-boundary nodes of all chares on this PE
378  for (const auto& c : m_chinpoel) { // for all chare connectivities
379  // generate local ids and connectivity from global connectivity
380  auto el = tk::global2local( c.second );
381  const auto& inpoel = std::get< 0 >( el ); // local connectivity
382  const auto& gid = std::get< 1 >( el ); // local->global node ids
383  auto esup = tk::genEsup( inpoel, 4 ); // elements surrounding points
384  auto esuel = tk::genEsuelTet( inpoel, esup ); // elems surrounding elements
385  // collect chare boundary nodes
386  for (std::size_t e=0; e<esuel.size()/4; ++e) {
387  auto mark = e*4;
388  for (std::size_t f=0; f<4; ++f) {
389  if (esuel[mark+f] == -1) {
390  auto A = gid[ inpoel[ mark+tk::lpofa[f][0] ] ];
391  auto B = gid[ inpoel[ mark+tk::lpofa[f][1] ] ];
392  auto C = gid[ inpoel[ mark+tk::lpofa[f][2] ] ];
393  m_bnodechares[ A ].push_back( c.first );
394  m_bnodechares[ B ].push_back( c.first );
395  m_bnodechares[ C ].push_back( c.first );
396  }
397  }
398  }
399  }
400  // Make node chare lists unique
401  for (auto& n : m_bnodechares) tk::unique( n.second );
402 
403  // send progress report to host
404  if ( g_inputdeck.get< tag::cmd, tag::feedback >() ) m_host.peflattened();
405 
406  // Signal host that we are ready for computing the communication map,
407  // required for parallel distributed global mesh node reordering
408  contribute( m_cb.get< tag::flattened >() );
409 }

437 tk::real
438 Partitioner::cost( std::size_t l, std::size_t u )
439 // *****************************************************************************
440 // Compute communication cost on our PE
441 //! \param[in] l Lower global node ID this PE works on
442 //! \param[in] u Upper global node ID this PE works on
443 //! \return Communication cost for our PE
444 //! \details The cost is a real number between 0 and 1, defined as the
445 //! number of mesh points we do not own, i.e., need to send to some other
446 //! PE, divided by the total number of points we contribute to. The lower
447 //! the better.
448 // *****************************************************************************
449 {
450  std::size_t ownpts = 0, compts = 0;
451  for (auto p : m_nodeset) if (p >= l && p < u) ++ownpts; else ++compts;
452 
453  // Free storage of unique global node IDs chares on our PE will contribute to
454  // as it is no longer needed after computing the communication cost.
455  tk::destroy( m_nodeset );
456 
457  return static_cast<tk::real>(compts) / static_cast<tk::real>(ownpts + compts);
458 }

479 void
480 Partitioner::query( int p, const std::vector< std::size_t >& bnodes )
481 // *****************************************************************************
482 // Query our global node IDs by other PEs so they know if they are to receive
483 // IDs for those from during reordering
484 //! \param[in] p Querying PE
485 //! \param[in] nodes List of global mesh node IDs to query
486 //! \details Note that every PE calls this function in a broadcast fashion,
487 //! including our own. However, to compute the correct result, this would
488 //! only be necessary for PEs whose ID is higher than ours. However, the
489 //! broadcast (calling everyone) is more efficient. This also results in a
490 //! simpler logic, because every PE goes through this single call path.
491 //! The returned mask is simply a boolean array signaling if the node ID
492 //! is found (owned).
493 // *****************************************************************************
494 {
495  std::unordered_map< std::size_t, std::vector< int > > cn;
496  for (auto j : bnodes) {
497  const auto it = m_nodeset.find( j );
498  if (it != end(m_nodeset)) {
499  const auto& c = tk::cref_find( m_bnodechares, j );
500  auto& chares = cn[j];
501  chares.insert( end(chares), begin(c), end(c) );
502  }
503  }
504 
505  // when we have heard from all PEs, send progress report to host
506  if ( g_inputdeck.get< tag::cmd, tag::feedback >() &&
507  ++m_nquery == static_cast<std::size_t>(CkNumPes()) )
508  m_host.pequery();
509 
510  thisProxy[ p ].mask( CkMyPe(), cn );
511 }

513 void
514 Partitioner::mask( int p, const std::unordered_map< std::size_t,
515  std::vector< int > >& cn )
516 // *****************************************************************************
517 // Receive mask of to-be-received global mesh node IDs
518 //! \param[in] p The PE uniquely assigns the node IDs marked listed in ch
519 //! \param[in] cn Vector containing the set of potentially multiple chare
520 //! IDs that we own (i.e., contribute to) for all of our node IDs.
521 //! \details Note that every PE will call this function, since query() was
522 //! called in a broadcast fashion and query() answers to every PE once.
523 //! This is more efficient than calling only the PEs from which we would
524 //! have to receive results from. Thus the incoming results are only
525 //! interesting from PEs with lower IDs than ours.
526 // *****************************************************************************
527 {
528  // Store the old global mesh node IDs associated to chare IDs bordering the
529  // mesh chunk held by and associated to chare IDs we own. This loop computes
530  // m_msum, a symmetric chare-node communication map, that associates (in its
531  // inner map) a unique set of global node IDs to areceiving chare ID, both
532  // associated (in its outer map) to a sending chare ID.
533  for (const auto& h : cn) {
534  const auto& chares = tk::ref_find( m_bnodechares, h.first );
535  for (auto c : chares) { // surrounded chares
536  auto& sch = m_msum[c];
537  for (auto s : h.second) // surrounding chares
538  if (s != c) sch[ s ].insert( h.first );
539  }
540  }
541 
542  // Associate global mesh node IDs to lower PEs we will need to receive from
543  // during node reordering. The choice of associated container is std::map,
544  // which is ordered (vs. unordered, hash-map). This is required by the
545  // following operation that makes the mesh node IDs unique in the
546  // communication map. (We are called in an unordered fashion, so we need to
547  // collect from all PEs and then we need to make the node IDs unique, keeping
548  // only the lowest PEs a node ID is associated with.) Note that m_ncomm is an
549  // asymmetric PE-node communication map, associating a set of unique global
550  // node IDs to PE IDs from which we (this PE) will need to receive newly
551  // assigned node IDs during global mesh node reordering. This map is
552  // asymmetric, beacuse of the agreement of the reordering that if a mesh node
553  // is shared by multiple PEs, the PE with the lowest ID gets to assign a new
554  // ID to it and all others must receive it instead of assigning it.
555 
556  if (p < CkMyPe()) {
557  auto& id = m_ncomm[ p ];
558  for (const auto& h : cn) id.insert( h.first );
559  }
560 
561  if (++m_nmask == static_cast<std::size_t>(CkNumPes())) {
562  // Make sure we have received all we need
563  Assert( m_ncomm.size() == static_cast<std::size_t>(CkMyPe()),
564  "Communication map size on PE " +
565  std::to_string(CkMyPe()) + " must equal " +
566  std::to_string(CkMyPe()) );
567  // Fill new hash-map, keeping only unique node IDs obtained from the
568  // lowest possible PEs
569  for (auto c=m_ncomm.cbegin(); c!=m_ncomm.cend(); ++c) {
570  auto& n = m_ncommunication[ c->first ];
571  for (auto j : c->second)
572  if (std::none_of( m_ncomm.cbegin(), c,
573  [ j ]( const typename decltype(m_ncomm)::value_type& s )
574  { return s.second.find(j) != end(s.second); } )) {
575  n.insert(j);
576  }
577  if (n.empty()) m_ncommunication.erase( c->first );
578  }
579  // Free storage of temporary communication map used to receive global
580  // mesh node IDs as it is no longer needed once the final communication
581  // map is generated.

598 std::array< std::vector< tk::real >, 3 >
599 Partitioner::centroids( const std::vector< std::size_t >& inpoel,
600  const tk::UnsMesh::Coords& coord )
601 // *****************************************************************************
602 // Compute element centroid coordinates
603 //! \param[in] inpoel Mesh connectivity with local ids
604 //! \param[ib] coord Node coordinates
605 //! \return Centroids for all cells on this PE
606 // *****************************************************************************
607 {
608  Assert( tk::uniquecopy(inpoel).size() == coord[0].size(), "Size mismatch" );
609 
610  const auto& x = coord[0];
611  const auto& y = coord[1];
612  const auto& z = coord[2];
613 
614  // Make room for element centroid coordinates
615  std::array< std::vector< tk::real >, 3 > cent;
616  auto& cx = cent[0];
617  auto& cy = cent[1];
618  auto& cz = cent[2];
619  auto num = inpoel.size()/4;
620  cx.resize( num );
621  cy.resize( num );
622  cz.resize( num );
623 
624  // Compute element centroids for mesh passed in
625  for (std::size_t e=0; e<num; ++e) {
626  auto A = inpoel[e*4+0];
627  auto B = inpoel[e*4+1];
628  auto C = inpoel[e*4+2];
629  auto D = inpoel[e*4+3];
630  cx[e] = (x[A] + x[B] + x[C] + x[D]) / 4.0;
631  cy[e] = (y[A] + y[B] + y[C] + y[D]) / 4.0;
632  cz[e] = (z[A] + z[B] + z[C] + z[D]) / 4.0;
633  }
634 
635  return cent;
636 }

641 // *****************************************************************************
642 // Categorize mesh elements (given by their gobal node IDs) by target
643 //! \param[in] target Targets (chares or PEs) of mesh elements, size: number of
644 //! elements in the chunk of the mesh graph on this PE.

786 std::array< int, 2 >
787 Partitioner::distribution( int npart ) const
788 // *****************************************************************************
789 // Compute chare (partition) distribution
790 //! \param[in] npart Total number of chares (partitions) to distribute
791 //! \return Chunksize, i.e., number of chares per all PEs except the last
792 //! one, and the number of chares for my PE
793 //! \details Chare ids are distributed to PEs in a linear continguous order
794 //! with the last PE taking the remainder if the number of PEs is not
795 //! divisible by the number chares. For example, if nchare=7 and npe=3,
796 //! the chare distribution is PE0: 0 1, PE1: 2 3, and PE2: 4 5 6. As a
797 //! result of this distribution, all PEs will have their chare-categorized
798 //! element connectivity filled with the global mesh node IDs associated
799 //! to the Charm++ chare IDs each PE owns.
800 // *****************************************************************************
801 {
802  auto chunksize = npart / CkNumPes();
803  auto mynchare = chunksize;
804  if (CkMyPe() == CkNumPes()-1) mynchare += npart % CkNumPes();
805  return {{ chunksize, mynchare }};
806 }

808 void
809 Partitioner::reorder()
810 // *****************************************************************************
811 // Reorder global mesh node IDs
812 // *****************************************************************************
813 {
814  // Activate SDAG waits for having requests arrive from other PEs for some
815  // of our node IDs; and for computing/receiving lower and upper bounds of
816  // global node IDs our PE operates on after reordering
817  thisProxy[ CkMyPe() ].wait4prep();
818  thisProxy[ CkMyPe() ].wait4bounds();
819 
820  // In serial signal to the runtime system that we have participated in
821  // reordering. This is required here because this is only triggered if
822  // communication is required during mesh node reordering. See also
823  // particioner.ci.
824  if (CkNumPes() == 1) participated_complete();
825 
826  // Send out request for new global node IDs for nodes we do not reorder
827  for (const auto& c : m_ncommunication)
828  thisProxy[ c.first ].request( CkMyPe(), c.second );
829 
830  // Lambda to decide if node ID is being assigned a new ID by us
831  auto ownnode = [ this ]( std::size_t p ) {
832  using Set = typename std::remove_reference<
833  decltype(m_ncommunication) >::type::value_type;
834  return !std::any_of( m_ncommunication.cbegin(), m_ncommunication.cend(),
835  [&](const Set& s)
836  { return s.second.find(p) != s.second.cend(); } );
837  };
838 
839  // Reorder our chunk of the mesh node IDs by looping through all of our node
840  // IDs. We test if this PE is to assign a new ID to a node ID, and if so, we
841  // assign a new ID, i.e., reorder, by constructing a map associating new to
842  // old IDs. We also count up the reordered nodes, which also serves as the new
843  // node id. Also, we store the coordinates associated to the new node ID for
844  // each chare on this PE. Since multiple chares can contribute to a single
845  // node, we store such shared node coordinates for all chares that contribute.
846  for (auto p : m_nodeset)
847  if (ownnode(p)) {
848  m_linnodes[ p ] = m_start;
849  auto coord = tk::cref_find( m_coordmap, p );
850  for (auto c : tk::cref_find(m_nodech,p))
851  m_chcoordmap[ c ].emplace( m_start, coord );
852  ++m_start;
853  }
854 
855  // Trigger SDAG wait indicating that reordering own node IDs are complete
856  reorderowned_complete();
857 
858  // If all our nodes have new IDs assigned, reordering complete on this PE
859  if (m_linnodes.size() == m_nodeset.size()) reordered();
860 }

881 // *****************************************************************************
882 // Associate new node IDs to old ones and return them to the requestor(s)
883 // *****************************************************************************
884 {

906  reorderowned_complete();

909 void
910 Partitioner::bndEdges()
911 // *****************************************************************************
912 // Generate boundary edges and send them to all PEs
913 //! \details This step happens when the mesh chunk on this PE has been
914 //! distributed after partitioning during an initial mesh refinement step. At
915 //! this point we have a contiguous chunk of the mesh on this PE as
916 //! determined by the partitioner. The next step is to extract the edges on
917 //! the boundary only. The boundary edges (shared by multiple PEs) will be
918 //! agreed on a refinement that yields a conforming mesh across PE boundaries.
919 // *****************************************************************************
920 {
921  // Compute local data from global mesh connectivity (m_inpoel, m_gid, m_lid)
922  m_el = tk::global2local( m_ginpoel );
923 
924  // Generate boundary edges of our mesh chunk
925  tk::UnsMesh::EdgeSet bnded;
926  auto esup = tk::genEsup( m_inpoel, 4 ); // elements surrounding points
927  auto esuel = tk::genEsuelTet( m_inpoel, esup ); // elems surrounding elements
928  for (std::size_t e=0; e<esuel.size()/4; ++e) {
929  auto mark = e*4;
930  for (std::size_t f=0; f<4; ++f) {
931  if (esuel[mark+f] == -1) {
932  auto A = m_gid[ m_inpoel[ mark+tk::lpofa[f][0] ] ];
933  auto B = m_gid[ m_inpoel[ mark+tk::lpofa[f][1] ] ];
934  auto C = m_gid[ m_inpoel[ mark+tk::lpofa[f][2] ] ];
935  bnded.insert( {{{A,B}}} );
936  bnded.insert( {{{B,C}}} );
937  bnded.insert( {{{C,A}}} );
938  Assert( m_lid.find( A ) != end(m_lid), "Local node ID not found" );
939  Assert( m_lid.find( B ) != end(m_lid), "Local node ID not found" );
940  Assert( m_lid.find( C ) != end(m_lid), "Local node ID not found" );
941  }
942  }
943  }
944 
945  // Export boundary edges to all PEs
946  thisProxy.addBndEdges( CkMyPe(), bnded );
947 }

974 void
975 Partitioner::refine()
976 // *****************************************************************************
977 // Optionally refine mesh
978 //! \details This is a single step in a potentially multiple-entry list of
979 //! initial adaptive mesh refinement steps. Distribution of the PE-boundary
980 //! edges has preceded this step, so that boundary edges (shared by multiple
981 //! PEs) can agree on a refinement that yields a conforming mesh across PE
982 //! boundaries.
983 // *****************************************************************************
984 {
985  // Convert node coordinates associated to global node IDs to a flat vector
986  auto npoin = m_coordmap.size();
987  Assert( m_gid.size() == npoin, "Size mismatch" );
988  m_coord[0].resize( npoin );
989  m_coord[1].resize( npoin );
990  m_coord[2].resize( npoin );
991  for (const auto& c : m_coordmap) {
992  auto i = tk::cref_find( m_lid, c.first );
993  m_coord[0][i] = c.second[0];
994  m_coord[1][i] = c.second[1];
995  m_coord[2][i] = c.second[2];
996  }
997 
998  // Query user input for initial mesh refinement type list
999  auto initref = g_inputdeck.get< tag::amr, tag::init >();
1000  Assert( !initref.empty(), "No initial mesh refinement steps configured" );
1001  // Determine which level this is
1002  auto level = initref.size() - m_initref.size();
1003 
1004  // Output mesh before this initial refinement step
1005  tk::UnsMesh refmesh( m_inpoel, m_coord );
1006  tk::ExodusIIMeshWriter mw( "initref." + std::to_string(level) + '.' +
1007  std::to_string(CkMyPe()),
1008  tk::ExoWriter::CREATE );
1009  mw.writeMesh( refmesh );
1010 
1011  for (const auto& e : tk::cref_find(m_bndEdges,CkMyPe())) {
1012  IGNORE(e);
1013  Assert( m_lid.find( e[0] ) != end( m_lid ) &&
1014  m_lid.find( e[1] ) != end( m_lid ),
1015  "Boundary edge not found before refinement" );
1016  }
1017 
1018  // Refine mesh based on next initial refinement type
1019  //std::cout << CkMyPe() << " ref start " << level << '\n';
1020  auto r = m_initref.back(); // consume (reversed) list from back
1021  if (r == ctr::AMRInitialType::UNIFORM)
1022  uniformRefine();
1023  else if (r == ctr::AMRInitialType::INITIAL_CONDITIONS)
1024  errorRefine();
1025  else Throw( "Initial AMR type not implemented" );
1026  std::cout << CkMyPe() << " ref finish " << level << '\n';
1027 
1028  for (const auto& e : tk::cref_find(m_bndEdges,CkMyPe())) {
1029  IGNORE(e);
1030  Assert( m_lid.find( e[0] ) != end( m_lid ) &&
1031  m_lid.find( e[1] ) != end( m_lid ),
1032  "Boundary edge not found after refinement" );
1033  }
1034 
1035  // Ensure valid mesh after refinement
1036  Assert( tk::positiveJacobians( m_inpoel, m_coord ),
1037  "Refined mesh Jacobian non-positive" );
1038 
1039  // Export added nodes on our mesh chunk boundary to other PEs
1040  if (m_pe.empty())
1041  contribute( m_cb.get< tag::matched >() );
1042  else
1043  for (auto p : m_pe) { // for all PEs we share at least an edge with
1044  // For all boundary edges of PE p, find out if we have added a new
1045  // node to it, and if so, export parents->(newid,coords) to p.
1046  tk::UnsMesh::EdgeNodeCoord exp;
1047  for (const auto& e : tk::cref_find(m_bndEdges,p)) {
1048  auto i = m_edgenode.find(e);
1049  if (i != end(m_edgenode)) exp[ e ] = i->second;
1050  }
1051  thisProxy[ p ].addRefBndEdges( CkMyPe(), exp );
1052  }
1053 }

1080 void
1081 Partitioner::nextref()
1082 // *****************************************************************************
1083 // Decide wether to continue with another step of initial mesh refinement
1084 //! \details At this point the mesh has been refined and all PEs have received
1085 //! a map associating the global IDs and the coordinates of a node added to
1086 //! an edge during initial mesh refinement associated to all other PEs the
1087 //! edges are shared with. Now the mesh is corrected so that it conforms
1088 //! across PE-boundaries by tagging those edges for refinement that have been
1089 //! refined by at least a PE. This concludes this initial mesh refinement
1090 //! step, and we continue if there are more steps configured by the user.
1091 // *****************************************************************************
1092 {
1093  // Storage for edges that still need a new node to yield a conforming mesh
1094  tk::UnsMesh::EdgeSet extra;
1095 
1096  // Ensure that the same global node ID has been assigned by all PEs and that
1097  // the new nodes have the same coordinates generated by potentially multiple
1098  // PEs sharing the refined edge. This is done by searching for all edges that
1099  // we share with and refined by other PEs: (1) If the incoming edge is found
1100  // among our refined ones, we ensure the newly assigned global IDs equal
1101  // (independently assigned PEs) and also that the new node coordinates equal
1102  // to machine precision. (2) If the incoming edge is not found among our
1103  // refined ones, we need to correct the mesh to make it conforming since the
1104  // edge has been refined by the remote PE. We collect these extra edges, and
1105  // run a correction refinement.
1106  for (const auto& p : m_edgeNodeCoord) // for all PEs we share edges with
1107  for (const auto& e : p.second) { // for all refined edges on p.first
1108  auto i = m_edgenode.find( e.first ); // find refined edge given parents
1109  if (i != end(m_edgenode)) { // found same added node on edge
1110  // locally assigned added node ID and coordinates: i->second
1111  // remotely assigned added node ID and coordinates: e.second
1112  // Ensure global IDs of newly added nodes are the same
1113  Assert( std::get< 0 >( i->second ) == std::get< 0 >( e.second ),

1392 void
1393 Partitioner::bounds()
1394 // *****************************************************************************
1395 // Compute lower and upper bounds of reordered node IDs our PE operates on
1396 // \details This function computes the global row IDs at which the linear
1397 // system will have a PE boundary. We simply find the largest node ID
1398 // assigned on each PE by the reordering and use that as the upper global
1399 // row index. Note that while this rarely results in equal number of rows
1400 // assigned to PEs, potentially resulting in some load-imbalance, it
1401 // yields a pretty good division reducing communication costs during the
1402 // assembly of the linear system, which is more important than a slight
1403 // (FLOP) load imbalance. Since the upper index for PE 1 is the same as
1404 // the lower index for PE 2, etc., we find the upper indices and then the
1405 // lower indices for all PEs are communicated.
1406 // *****************************************************************************
1407 {
1408  m_upper = 0;
1409 
1410  using P1 = std::pair< const std::size_t, std::size_t >;
1411  for (const auto& c : m_chfilenodes) {
1412  auto x = std::max_element( begin(c.second), end(c.second),
1413  [](const P1& a, const P1& b){ return a.first < b.first; } );
1414  if (x->first > m_upper) m_upper = x->first;
1415  }
1416 
1417  // The bounds are the dividers (global mesh point indices) at which the
1418  // linear system assembly is divided among PEs. However, Hypre and thus
1419  // Solver expect exclusive upper indices, so we increase the last one by
1420  // one here. Note that the cost calculation, Partitioner::cost() also
1421  // expects exclusive upper indices.
1422  if (CkMyPe() == CkNumPes()-1) ++m_upper;
1423 
1424  // Tell the runtime system that the upper bound has been computed
1425  upper_complete();
1426 
1427  // Set lower index for PE 0 as 0
1428  if (CkMyPe() == 0) lower(0);
1429 
1430  // All PEs except the last one send their upper indices as the lower index for
1431  // PE+1
1432  if (CkMyPe() < CkNumPes()-1)
1433  thisProxy[ CkMyPe()+1 ].lower( m_upper );
1434 }
