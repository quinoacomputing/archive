114 void
115 Partitioner::partition( int nchare )
116 // *****************************************************************************
117 // Partition the computational mesh into a number of chares
118 //! \param[in] nchare Number of parts the mesh will be partitioned into
119 //! \details This function calls the mesh partitioner to partition the mesh. The
120 //! number of partitions equals the number nchare argument which must be no
121 //! lower than the number of PEs.
122 // *****************************************************************************
123 {
124  Assert( nchare >= CkNumPes(),
125  "Number of chares must not be lower than the number of PEs" );
126 
127  // Generate element IDs for Zoltan
128  std::vector< long > gelemid( m_ginpoel.size()/4 );
129  std::iota( begin(gelemid), end(gelemid), 0 );
130 
131  m_nchare = nchare;
132  const auto alg = g_inputdeck.get< tag::selected, tag::partitioner >();
133  const auto che = tk::zoltan::geomPartMesh( alg,
134  centroids( m_inpoel, m_coord ),
135  gelemid,
136  nchare );
137 
138  if ( g_inputdeck.get< tag::cmd, tag::feedback >() ) m_host.pepartitioned();
139 
140  Assert( che.size() == gelemid.size(), "Size of ownership array (chare ID "
141  "of elements) after mesh partitioning does not equal the number of "
142  "mesh graph elements" );
143 
144  m_chcoordmap.clear();
145  m_chinpoel.clear();
146 
147  // Categorize mesh elements (given by their gobal node IDs) by target chare
148  // and distribute to their PEs based on mesh partitioning.
149  distribute( categorize( che, m_ginpoel ) );
150 }

157 // *****************************************************************************
158 // Receive mesh associated to chares we own after refinement

183 int
184 Partitioner::pe( int id ) const
185 // *****************************************************************************
186 // Return processing element for chare id
187 //! \param[in] id Chare id
188 //! \return PE that creates the chare
189 //! \details This is computed based on a simple contiguous linear
190 //! distribution of chare ids to PEs.
191 // *****************************************************************************
192 {
193  Assert( m_nchare > 0, "Number of chares must be a positive number" );
194  auto p = id / (m_nchare / CkNumPes());
195  if (p >= CkNumPes()) p = CkNumPes()-1;
196  Assert( p < CkNumPes(), "Assigning to nonexistent PE" );
197  return p;
198 }

245 std::array< std::vector< tk::real >, 3 >
246 Partitioner::centroids( const std::vector< std::size_t >& inpoel,
247  const tk::UnsMesh::Coords& coord )
248 // *****************************************************************************
249 // Compute element centroid coordinates
250 //! \param[in] inpoel Mesh connectivity with local ids
251 //! \param[ib] coord Node coordinates
252 //! \return Centroids for all cells on this PE
253 // *****************************************************************************
254 {
255  Assert( tk::uniquecopy(inpoel).size() == coord[0].size(), "Size mismatch" );
256 
257  const auto& x = coord[0];
258  const auto& y = coord[1];
259  const auto& z = coord[2];
260 
261  // Make room for element centroid coordinates
262  std::array< std::vector< tk::real >, 3 > cent;
263  auto& cx = cent[0];
264  auto& cy = cent[1];
265  auto& cz = cent[2];
266  auto num = inpoel.size()/4;
267  cx.resize( num );
268  cy.resize( num );
269  cz.resize( num );
270 
271  // Compute element centroids for mesh passed in
272  for (std::size_t e=0; e<num; ++e) {
273  auto A = inpoel[e*4+0];
274  auto B = inpoel[e*4+1];
275  auto C = inpoel[e*4+2];
276  auto D = inpoel[e*4+3];
277  cx[e] = (x[A] + x[B] + x[C] + x[D]) / 4.0;
278  cy[e] = (y[A] + y[B] + y[C] + y[D]) / 4.0;
279  cz[e] = (z[A] + z[B] + z[C] + z[D]) / 4.0;
280  }
281 
282  return cent;
283 }

285 std::unordered_map< int, std::vector< std::size_t > >
286 Partitioner::categorize( const std::vector< std::size_t >& target,
287  const std::vector< std::size_t >& inpoel ) const
288 // *****************************************************************************
289 // Categorize mesh elements (given by their gobal node IDs) by target
290 //! \param[in] target Targets (chares or PEs) of mesh elements, size: number of
291 //! elements in the chunk of the mesh graph on this PE.
292 //! \param[in] inpoel Mesh connectivity to distribute elemets from
293 //! \return Vector of global mesh node ids connecting elements owned by each
294 //! target (chare or PE)
295 // *****************************************************************************
296 {
297  Assert( target.size() == inpoel.size()/4, "Size mismatch");
298 
299  // Categorize global mesh node ids of elements by target
300  std::unordered_map< int, std::vector< std::size_t > > nodes;
301  for (std::size_t e=0; e<target.size(); ++e) {
302  auto& c = nodes[ static_cast<int>(target[e]) ];
303  for (std::size_t n=0; n<4; ++n) c.push_back( inpoel[e*4+n] );
304  }
305 
306  // Make sure all PEs have targets assigned
307  Assert( !nodes.empty(), "No nodes have been assigned to chares on PE " );
308 
309  // This check should always be done, hence ErrChk and not Assert, as it
310  // can result from particular pathological combinations of (1) too large
311  // degree of virtualization, (2) too many PEs, and/or (3) too small of a
312  // mesh and not due to programmer error.
313  for(const auto& c : nodes)
314  ErrChk( !c.second.empty(),
315  "Overdecomposition of the mesh is too large compared to the "
316  "number of work units computed based on the degree of "
317  "virtualization desired. As a result, there would be at least "
318  "one work unit with no mesh elements to work on, i.e., nothing "
319  "to do. Solution 1: decrease the virtualization to a lower "
320  "value using the command-line argument '-u'. Solution 2: "
321  "decrease the number processing elements (PEs) using the "
322  "charmrun command-line argument '+pN' where N is the number of "
323  "PEs, which implicitly increases the size (and thus decreases "
324  "the number) of work units.)" );
325 
326  return nodes;
327 }

329 tk::UnsMesh::CoordMap
330 Partitioner::coordmap( const std::vector< std::size_t >& inpoel )
331 // *****************************************************************************
332 // Extract coordinates associated to global nodes of a mesh chunk
333 //! \param[in] inpoel Mesh connectivity
334 //! \return Map storing the coordinates of unique nodes associated to global
335 //! node IDs in mesh given by inpoel
336 // *****************************************************************************
337 {
338  Assert( inpoel.size() % 4 == 0, "Incomplete mesh connectivity" );
339 
340  tk::UnsMesh::CoordMap map;
341 
342  for (auto g : tk::uniquecopy(inpoel)) {
343  auto i = tk::cref_find( m_lid, g );
344  auto& c = map[g];
345  c[0] = m_coord[0][i];
346  c[1] = m_coord[1][i];
347  c[2] = m_coord[2][i];
348  }
349 
350  Assert( tk::uniquecopy(inpoel).size() == map.size(), "Size mismatch" );
351 
352  return map;
353 }

355 void
356 Partitioner::distribute(
357  std::unordered_map< int, std::vector< std::size_t > >&& elems )
358 // *****************************************************************************
359 // Distribute mesh to target PEs after mesh partitioning
360 //! \param[in] elems Mesh cells (with global node IDs) categorized by target
361 //! chares
362 // *****************************************************************************
363 {
364  auto dist = distribution( m_nchare );
365 
366  // Extract those mesh connectivities whose chares live on this PE
367  for (int c=0; c<dist[1]; ++c) {
368  auto chid = CkMyPe() * dist[0] + c; // compute owned chare ID
369  const auto it = elems.find( chid ); // attempt to find its nodes
370  if (it != end(elems)) { // if found
371  auto& inp = m_chinpoel[ chid ]; // store own mesh
372  inp.insert( end(inp), begin(it->second), end(it->second) );
373  auto& chcm = m_chcoordmap[ chid ]; // store own node coordinates
374  auto cm = coordmap( it->second ); // extract node coordinates
375  chcm.insert( begin(cm), end(cm) ); // concatenate node coords
376  elems.erase( it ); // remove chare ID and nodes
377  }
378  Assert( elems.find(chid) == end(elems), "Not all owned node IDs stored" );
379  }
380 
381  // Construct export map associating mesh connectivities with global node
382  // indices and node coordinates for mesh chunks associated to chare IDs (inner
383  // key) owned by chares we do not own.
384  std::unordered_map< int, // target PE
385  std::unordered_map< int, // chare ID
386  std::tuple< std::vector< std::size_t >, // mesh connectivity
387  tk::UnsMesh::CoordMap > > > exp; // node ID & coords
388  for (const auto& c : elems)
389  exp[ pe(c.first) ][ c.first ] =
390  std::make_tuple( c.second, coordmap(c.second) );
391 
392  // Export chare IDs and mesh we do not own to fellow PEs
393  if (exp.empty()) {
394  if ( g_inputdeck.get< tag::cmd, tag::feedback >() ) m_host.pedistributed();
395  contribute( m_cbp.get< tag::distributed >() );
396  } else {
397  m_ndist = exp.size();
398  for (const auto& p : exp)
399  thisProxy[ p.first ].addMesh( CkMyPe(), p.second );
400  }
401 }

403 std::array< int, 2 >
404 Partitioner::distribution( int npart ) const
405 // *****************************************************************************
406 // Compute chare (partition) distribution
407 //! \param[in] npart Total number of chares (partitions) to distribute
408 //! \return Chunksize, i.e., number of chares per all PEs except the last
409 //! one, and the number of chares for my PE
410 //! \details Chare ids are distributed to PEs in a linear continguous order
411 //! with the last PE taking the remainder if the number of PEs is not
412 //! divisible by the number chares. For example, if nchare=7 and npe=3,
413 //! the chare distribution is PE0: 0 1, PE1: 2 3, and PE2: 4 5 6. As a
414 //! result of this distribution, all PEs will have their chare-categorized
415 //! element connectivity filled with the global mesh node IDs associated
416 //! to the Charm++ chare IDs each PE owns.
417 // *****************************************************************************
418 {
419  auto chunksize = npart / CkNumPes();
420  auto mynchare = chunksize;
421  if (CkMyPe() == CkNumPes()-1) mynchare += npart % CkNumPes();
422  return {{ chunksize, mynchare }};
423 }
