81  //! Verify MeshReader by invoking various member functions yielding tetrahedra
82  //! \param[in,out] mr MeshReader object to verify
83  void verifyTets( tk::MeshReader& mr ) {
84  // Read mesh graph in serial
85  std::vector< std::size_t > ginpoel, inpoel, gid;
86  std::unordered_map< std::size_t, std::size_t > lid;
87  tk::UnsMesh::Coords coord;
88  mr.readMeshPart( ginpoel, inpoel, gid, lid, coord );
89 
90  // Test if the number of elements is correct
91  ensure_equals( "number of elements incorrect",
92  inpoel.size()/4, box24_inpoel.size()/4 );
93 
94  // Test if the mesh element connectivity is correct
95  ensure( "element connectivity incorrect", inpoel == box24_inpoel );
96 
97  // Test if the number of coordinates is correct
98  ensure_equals( "number of x coordinates incorrect",
99  coord[0].size(), box24_coord.size()/3 );
100  ensure_equals( "number of y coordinates incorrect",
101  coord[1].size(), box24_coord.size()/3 );
102  ensure_equals( "number of z coordinates incorrect",
103  coord[2].size(), box24_coord.size()/3 );
104 
105  // Test if the mesh node coordinates are correct
106  std::vector< tk::real > x, y, z;
107  for (std::size_t p=0; p<box24_coord.size()/3; ++p) {
108  x.push_back( box24_coord[p*3] );
109  y.push_back( box24_coord[p*3+1] );
110  z.push_back( box24_coord[p*3+2] );
111  }
112  ensure( "nodes' x coordinates incorrect", coord[0] == x );
113  ensure( "nodes' y coordinates incorrect", coord[1] == y );
114  ensure( "nodes' z coordinates incorrect", coord[2] == z );
115  }

117  //! Verify MeshReader by invoking various member functions yielding triangles
118  //! \param[in,out] mr MeshReader object to verify
119  void verifyTris( tk::MeshReader& mr ) {
120  // Read side set faces
121  std::map< int, std::vector< std::size_t > > bface;
122  std::map< int, std::vector< int > > faceid;
123  auto nbfac = mr.readSidesetFaces( bface, faceid );
124  // Test if the number of boundary faces is correct
125  ensure_equals( "number of boundary faces incorrect", nbfac, 2398 );
126  // Test if the number of boundary face element ids is correct
127  ensure_equals( "number of boundary face elements incorrect",
128  tk::sumvalsize(bface), nbfac );
129  // Test if the number of side set sides is correct
130  ensure_equals( "number of side set sides incorrect",
131  tk::sumvalsize(faceid), nbfac );
132 
133  // Read face connectivity
134  std::vector< std::size_t > triinpoel;
135  mr.readFaces( nbfac, triinpoel );
136  // Test if the number of faces is correct
137  ensure_equals( "number of faces in triangle connectivity incorrect",
138  triinpoel.size()/3, 2398 );
139 
140  // Read node lists associated to side sets
141  auto bnode = mr.readSidesets();
142  // Test if the number of nodes is correct
143  ensure_equals( "number of nodes of sidesets incorrect",
144  tk::sumvalsize(bnode), 1365 );
145  }
