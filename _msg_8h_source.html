30 //! Operator << for writing std::vector< T > to output streams; for debugging.

53 //! Charm++ message type for sending a string of strings separated by ';'.
54 struct StringsMsg : public CMessage_StringsMsg {
55  using value_type = std::vector< std::string >;
56  //! Constructor
57  explicit StringsMsg( const std::string& v )
58  { strncpy( strings, v.c_str(), CSTYLE_STRLEN ); }
59  //! Value accessor splitting the migrated C-style string at semi-colons
60  value_type get() {
61  std::string s( strings );
62  boost::char_separator<char> sep(";");
63  boost::tokenizer< boost::char_separator<char> > tokens( s, sep );
64  value_type v;
65  for (const auto& t : tokens) v.push_back( t );
66  return v;
67  }
68  //! Data
69  char strings[CSTYLE_STRLEN];
70 };

72 //! Charm++ message type for sending a vector of strings of strings separated by
73 //! semi-colons. The size of the vector must be determined at compile-time and
74 //! comes in via the template argument. This avoids using variable-size
75 //! messages, which is more complicated, more error-prone, and most likely less
76 //! amenable to optimizations. The constructor and the value accessor interface
77 //! with std::vector< std::string >, but converted to POD so that the Charm++
78 //! runtime system easily serialize the data and migrate it over the network.
79 template< std::size_t Size = 1 >
80 struct VecStrsMsg : public CMessage_VecStrsMsg< Size > {
81  using value_type = std::vector< std::vector< std::string > >;
82  //! Constructor
83  explicit VecStrsMsg( const std::vector< std::string >& v ) {
84  Assert( Size <= v.size(), "Size of input vector to VecStrsMsg is larger "
85  "than specified in template argument" );
86  for (std::size_t i=0; i<Size; ++i)
87  strncpy( strings[i], v[i].c_str(), CSTYLE_STRLEN );
88  }
89  //! Value accessor
90  value_type get() {
91  value_type v( Size );
92  for (std::size_t i=0; i<Size; ++i) {
93  std::string s( strings[i] );
94  boost::char_separator<char> sep(";");
95  boost::tokenizer< boost::char_separator<char> > tokens( s, sep );
96  for (const auto& t : tokens) v[i].push_back( t );
97  }
98  return v;
99  }
100  //! Data
101  char strings[Size][CSTYLE_STRLEN];
102 };
