2 /*!
3  \file src/DiffEq/OrnsteinUhlenbeck.h
4  \copyright 2012-2015, J. Bakosi, 2016-2018, Los Alamos National Security, LLC.
5  \brief System of Ornstein-Uhlenbeck SDEs
6  \details This file implements the time integration of a system of stochastic
7  differential equations (SDEs), with linear drift and constant diffusion,
8  whose invariant is the [joint normal
9  distribution](http://en.wikipedia.org/wiki/Multivariate_normal_distribution).
10 
11  In a nutshell, the equation integrated governs a set of scalars,
12  \f$Y_\alpha\f$, \f$\alpha\!=\!1,\dots,N\f$, as
13  \f[
14  \mathrm{d}Y_\alpha(t) = \theta_\alpha\left(\mu_\alpha - Y_\alpha\right)
15  \mathrm{d}t + \sum_{\beta=1}^N \sigma_{\alpha\beta}\mathrm{d}W_\beta(t),
16  \qquad \alpha=1,\dots,N
17  \f]
18  with parameter vectors \f$\theta_\alpha > 0\f$, \f$\mu_\alpha\f$, and
19  symmetric positive semi-definite diffusion matrix
20  \f$\sigma_{\alpha\beta}\f$. Here \f$\mathrm{d}W_\beta(t)\f$ is an isotropic
21  vector-valued [Wiener process](http://en.wikipedia.org/wiki/Wiener_process)
22  with independent increments. The invariant distribution is the joint normal
23  distribution. This system of SDEs consists of N coupled equations, each
24  being a single-variate [Ornstein-Uhlenbeck
25  process](http://en.wikipedia.org/wiki/Ornstein%E2%80%93Uhlenbeck_process).
26 
27  From the Fokker-Planck equation, equivalent to the SDE above, the equations
28  governing the means, \f$ \langle Y_\alpha \rangle\f$, are
29  \f[
30  \newcommand{\irmean}[1]{{\langle{#1}\rangle}}
31  \frac{\partial\irmean{Y_\alpha}}{\partial t} =
32  \theta_\alpha\left(\mu_\alpha - \irmean{Y_\alpha}\right)
33  \f]
34  while the equation governing the covariance matrix, \f$ \langle y_\alpha
35  y_\beta \rangle \equiv \left\langle (Y_\alpha - \langle Y_\alpha \rangle)
36  (Y_\beta - \langle Y_\beta\rangle) \right\rangle \f$, is
37  \f[
38  \newcommand{\irmean}[1]{{\langle{#1}\rangle}}
39  \newcommand{\irv}[1]{\langle{#1^2}\rangle}
40  \frac{\partial\irmean{y_\alpha y_\beta}}{\partial t} =
41  -\left(\theta_\alpha+\theta_\beta\right)\irmean{y_\alpha y_\beta}
42  +\sum_{\gamma=1}^N \sigma_{\alpha\gamma} \sigma_{\gamma\beta}.
43  \f]
44 */

125  //! \brief Advance particles according to the system of Orsntein-Uhlenbeck
126  //! SDEs
127  //! \param[in,out] particles Array of particle properties
128  //! \param[in] stream Thread (or more precisely stream) ID
129  //! \param[in] dt Time step size
130  void advance( tk::Particles& particles,
131  int stream,
132  tk::real dt,
133  tk::real,
134  const std::map< tk::ctr::Product, tk::real >& )
135  {
136  const auto npar = particles.nunk();
137  for (auto p=decltype(npar){0}; p<npar; ++p) {
138  // Generate Gaussian random numbers with zero mean and unit variance
139  std::vector< tk::real > dW( m_ncomp );
140  m_rng.gaussian( stream, m_ncomp, dW.data() );
141 
142  // Advance all m_ncomp scalars
143  for (ncomp_t i=0; i<m_ncomp; ++i) {
144  tk::real& par = particles( p, i, m_offset );
145  par += m_theta[i]*(m_mu[i] - par)*dt;
146  for (ncomp_t j=0; j<m_ncomp; ++j) {
147  tk::real d = m_sigma[ j*m_ncomp+i ] * sqrt(dt); // use transpose
148  par += d*dW[j];
149  }
150  }
151  }
152  }
